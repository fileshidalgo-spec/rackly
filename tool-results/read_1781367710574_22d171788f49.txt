   287вЖТ   287вЖТexport async function listarMovimientos(
   288вЖТ   288вЖТ  sectorId?: string,
   289вЖТ   289вЖТ  columnaId?: string,
   290вЖТ   290вЖТ  bloqueId?: string,
   291вЖТ   291вЖТ  desde?: string,
   292вЖТ   292вЖТ  hasta?: string
   293вЖТ   293вЖТ): Promise<MovimientoConDetalles[]> {
   294вЖТ   294вЖТ  let query = dataClient
   295вЖТ   295вЖТ    .from('piso_movimientos')
   296вЖТ   296вЖТ    .select('*')
   297вЖТ   297вЖТ    .order('fecha', { ascending: false })
   298вЖТ   298вЖТ    .order('numero_operacion', { ascending: false })
   299вЖТ   299вЖТ
   300вЖТ   300вЖТ  if (desde) query = query.gte('fecha', desde)
   301вЖТ   301вЖТ  if (hasta) query = query.lte('fecha', hasta + 'T23:59:59')
   302вЖТ   302вЖТ  const PAGE_SIZE = 500
   303вЖТ   303вЖТ  const MAX_PAGES = 100
   304вЖТ   304вЖТ  let all: PisoMovimiento[] = []
   305вЖТ   305вЖТ  let from = 0
   306вЖТ   306вЖТ  let pages = 0
   307вЖТ   307вЖТ  while (pages++ < MAX_PAGES) {
   308вЖТ   308вЖТ    const to = from + PAGE_SIZE - 1
   309вЖТ   309вЖТ    const { data, error } = await query.range(from, to)
   310вЖТ   310вЖТ    if (error) throw error
   311вЖТ   311вЖТ    const rows = (data ?? []) as PisoMovimiento[]
   312вЖТ   312вЖТ    all.push(...rows)
   313вЖТ   313вЖТ    if (rows.length < PAGE_SIZE) break
   314вЖТ   314вЖТ    from += PAGE_SIZE
   315вЖТ   315вЖТ  }
   316вЖТ   316вЖТ
   317вЖТ   317вЖТ  if (all.length === 0) return []
   318вЖТ   318вЖТ
   319вЖТ   319вЖТ  const movIds = all.map((m) => m.id)
   320вЖТ   320вЖТ  const { data: detData, error: detErr } = await dataClient
   321вЖТ   321вЖТ    .from('piso_movimiento_detalles')
   322вЖТ   322вЖТ    .select('*')
   323вЖТ   323вЖТ    .in('movimiento_id', movIds)
   324вЖТ   324вЖТ  if (detErr) throw detErr
   325вЖТ   325вЖТ  const detalles = (detData ?? []) as MovimientoDetalle[]
   326вЖТ   326вЖТ
   327вЖТ   327вЖТ  const detalleMap = new Map<
   328вЖТ   328вЖТ    string,
   329вЖТ   329вЖТ    (MovimientoDetalle & { bloque_codigo?: string; nivel_codigo?: string })[]
   330вЖТ   330вЖТ  >()
   331вЖТ   331вЖТ  for (const d of detalles) {
   332вЖТ   332вЖТ    if (!detalleMap.has(d.movimiento_id)) detalleMap.set(d.movimiento_id, [])
   333вЖТ   333вЖТ    detalleMap.get(d.movimiento_id)!.push(d)
   334вЖТ   334вЖТ  }
   335вЖТ   335вЖТ
   336вЖТ   336вЖТ  const blockIds = [...new Set(detalles.map((d) => d.bloque_id))]
   337вЖТ   337вЖТ  const nivelIds = [...new Set(detalles.map((d) => d.nivel_id))]
   338вЖТ   338вЖТ
   339вЖТ   339вЖТ  const [bloquesRes, nivelesRes] = await Promise.all([
   340вЖТ   340вЖТ    blockIds.length > 0
   341вЖТ   341вЖТ      ? dataClient
   342вЖТ   342вЖТ          .from('piso_bloques')
   343вЖТ   343вЖТ          .select('id, codigo')
   344вЖТ   344вЖТ          .in('id', blockIds)
   345вЖТ   345вЖТ      : Promise.resolve({ data: [], error: null }),
   346вЖТ   346вЖТ    nivelIds.length > 0
   347вЖТ   347вЖТ      ? dataClient
   348вЖТ   348вЖТ          .from('piso_niveles')
   349вЖТ   349вЖТ          .select('id, codigo_ubicacion, posicion_id')
   350вЖТ   350вЖТ          .in('id', nivelIds)
   351вЖТ   351вЖТ      : Promise.resolve({ data: [], error: null }),
   352вЖТ   352вЖТ  ])
   353вЖТ   353вЖТ
   354вЖТ   354вЖТ  const bloqueMap = new Map<string, string>()
   355вЖТ   355вЖТ  ;((bloquesRes.data ?? []) as { id: string; codigo: string }[]).forEach(
   356вЖТ   356вЖТ    (b) => bloqueMap.set(b.id, b.codigo)
   357вЖТ   357вЖТ  )
   358вЖТ   358вЖТ  const nivelMap = new Map<string, string | null>()
   359вЖТ   359вЖТ  const nivelToPos = new Map<string, string>()
   360вЖТ   360вЖТ  ;(
   361вЖТ   361вЖТ    (nivelesRes.data ?? []) as { id: string; codigo_ubicacion: string | null; posicion_id: string }[]
   362вЖТ   362вЖТ  ).forEach((n) => {
   363вЖТ   363вЖТ    nivelMap.set(n.id, n.codigo_ubicacion)
   364вЖТ   364вЖТ    nivelToPos.set(n.id, n.posicion_id)
   365вЖТ   365вЖТ  })
   366вЖТ   366вЖТ
   367вЖТ   367вЖТ  // Resolve sector and position for each nivel

   428вЖТ   428вЖТ  }))
   429вЖТ   429вЖТ
   430вЖТ   430вЖТ  // Filter by sector/columna/bloque if specified
   431вЖТ   431вЖТ  if (sectorId || columnaId || bloqueId) {

   434вЖТ   434вЖТ        .from('piso_columnas')
   435вЖТ   435вЖТ        .select('id')
   436вЖТ   436вЖТ        .eq('sector_id', sectorId ?? columnaId)
   437вЖТ   437вЖТ      if (!colErr && cols) {
   438вЖТ   438вЖТ        const colIds = new Set(
   439вЖТ   439вЖТ          (cols as { id: string }[]).map((c) => c.id)
   440вЖТ   440вЖТ        )

   442вЖТ   442вЖТ          return []
   443вЖТ   443вЖТ        }
   444вЖТ   444вЖТ      }
   445вЖТ   445вЖТ    }
   446вЖТ   446вЖТ  }
   447вЖТ   447вЖТ
   448вЖТ   448вЖТ  return result
   449вЖТ   449вЖТ}

   508вЖТ   508вЖТ    .filter(([, qty]) => qty > 0)

   662вЖТ   662вЖТ
   663вЖТ   663вЖТ  // 5. Obtener stock por nivel y bloque
   664вЖТ   664вЖТ  let stockPorNivel: { nivel_id: string; bloque_id: string; cantidad: number; tipo: string }[] = []
   665вЖТ   665вЖТ  if (nivelIds.length > 0) {
   666вЖТ   666вЖТ    const { data: detData, error: detErr } = await dataClient

   672вЖТ   672вЖТ    const stockMap = new Map<string, { bloque_id: string; cantidad: number }[]>()

   682вЖТ   682вЖТ        ? qty : -qty

   686вЖТ   686вЖТ      if (!arr) { arr = []; stockMap.set(d.nivel_id, arr) }
   687вЖТ   687вЖТ      const existing = arr.find((e) => e.bloque_id === d.bloque_id)
   688вЖТ   688вЖТ      if (existing) {
   689вЖТ   689вЖТ        existing.cantidad += delta
   690вЖТ   690вЖТ      } else {
   691вЖТ   691вЖТ        arr.push({ bloque_id: d.bloque_id, cantidad: delta })
   692вЖТ   692вЖТ      }
   693вЖТ   693вЖТ    }

   707вЖТ   707вЖТ  let bloqueMap = new Map<string, string>()

   720вЖТ   720вЖТ  for (const id of bloqueIdsSet) {
   721вЖТ   721вЖТ    if (id.startsWith('cat_')) {

   724вЖТ   724вЖТ  }

   726вЖТ   726вЖТ  // 7. Construir resultado por posici√≥n
   727вЖТ   727вЖТ  const subMap = new Map(subcolumnas.map((s) => [s.id, s]))

   729вЖТ   729вЖТ

   731вЖТ   731вЖТ  const nivelesPorPosicion = new Map<string, string[]>()
   732вЖТ   732вЖТ  for (const n of niveles) {
   733вЖТ   733вЖТ    const arr = nivelesPorPosicion.get(n.posicion_id) ?? []
   734вЖТ   734вЖТ    arr.push(n.id)
   735вЖТ   735вЖТ    nivelesPorPosicion.set(n.posicion_id, arr)
   736вЖТ   736вЖТ  }
   737вЖТ   737вЖТ

   740вЖТ   740вЖТ  for (const s of stockPorNivel) {
   741вЖТ   741вЖТ    // Encontrar la posici√≥n de este nivel
   742вЖТ   742вЖТ    for (const pos of posiciones) {
   743вЖТ   743вЖТ      const posNivIds = nivelesPorPosicion.get(pos.id) ?? []
   744вЖТ   744вЖТ      if (posNivIds.includes(s.nivel_id)) {
   745вЖТ   745вЖТ        const arr = stockPorPosicion.get(pos.id) ?? []
   746вЖТ   746вЖТ        const existing = arr.find((e) => e.bloque_id === s.bloque_id)

   748вЖТ   748вЖТ          existing.cantidad += s.cantidad
   749вЖТ   749вЖТ        } else {
   750вЖТ   750вЖТ          arr.push({
   751вЖТ   751вЖТ            bloque_id: s.bloque_id,
   752вЖТ   752вЖТ            bloque_codigo: bloqueMap.get(s.bloque_id) ?? s.bloque_id,
   753вЖТ   753вЖТ            cantidad: s.cantidad,
   754вЖТ   754вЖТ          })
   755вЖТ   755вЖТ        }
   756вЖТ   756вЖТ        stockPorPosicion.set(pos.id, arr)
   757вЖТ   757вЖТ        break
   758вЖТ   758вЖТ      }
   759вЖТ   759вЖТ    }
   760вЖТ   760вЖТ  }
   761вЖТ   761вЖТ
   762вЖТ   762вЖТ  const result: PosicionConStock[] = posiciones.map((pos) => {
   763вЖТ   763вЖТ    const sub = subMap.get(pos.subcolumna_id)
   764вЖТ   764вЖТ    const bloques = (stockPorPosicion.get(pos.id) ?? []).filter((b) => b.cantidad > 0)
   765вЖТ   765вЖТ    const totalStock = bloques.reduce((sum, b) => sum + b.cantidad, 0)
   766вЖТ   766вЖТ    return {
   767вЖТ   767вЖТ      posicionId: pos.id,
   768вЖТ   768вЖТ      posicionNumero: pos.numero,
   769вЖТ   769вЖТ      subcolumnaCodigo: sub?.codigo ?? '',
   770вЖТ   770вЖТ      columnaLetra: sub ? (colMap.get(sub.columna_id) ?? '') : '',
   771вЖТ   771вЖТ      stock: totalStock,
   772вЖТ   772вЖТ      bloques,
   773вЖТ   773вЖТ    }
   774вЖТ   774вЖТ  })
   775вЖТ   775вЖТ
   776вЖТ   776вЖТ  return result
   777вЖТ   777вЖТ}

   779вЖТ   779вЖТ/**
   780вЖТ   780вЖТ * Obtiene el stock detallado de una posici√≥n espec√≠fica (todos los bloques y cantidades).

   785вЖТ   785вЖТ  posicionId: string
   786вЖТ   786вЖТ): Promise<{ bloque_id: string; bloque_codigo: string; bloque_descripcion: string; bloque_unidad: string; cantidad: number; fecha_vencimiento: string }[]> {
   787вЖТ   787вЖТ  // вХРвХРвХР M√ЙTODO PRINCIPAL: RPC server-side (JHIA-57b) вХРвХРвХР
   788вЖТ   788вЖТ  try {
   789вЖТ   789вЖТ    const { data, error } = await dataClient.rpc('piso_stock_detalle_posicion', {
   790вЖТ   790вЖТ      _posicion_id: posicionId,

   800вЖТ   800вЖТ        bloque_unidad: r.bloque_unidad || 'KG',

   802вЖТ   802вЖТ        fecha_vencimiento: r.fecha_vencimiento ?? '',
   803вЖТ   803вЖТ      }))
   804вЖТ   804вЖТ    }
   805вЖТ   805вЖТ    // Si el RPC no existe (error 428P01), caer al fallback
   806вЖТ   806вЖТ    console.warn('[Piso] RPC piso_stock_detalle_posicion no disponible, usando fallback client-side')
   807вЖТ   807вЖТ  } catch (rpcErr) {
   808вЖТ   808вЖТ    console.warn('[Piso] Error RPC piso_stock_detalle_posicion:', rpcErr)
   809вЖТ   809вЖТ  }
   810вЖТ   810вЖТ
   811вЖТ   811вЖТ  // вХРвХРвХР FALLBACK: C√°lculo client-side FEFO вХРвХРвХР
   812вЖТ   812вЖТ  const { data: nivData, error: nivErr } = await dataClient
   813вЖТ   813вЖТ    .from('piso_niveles')
   814вЖТ   814вЖТ    .select('id')
   815вЖТ   815вЖТ    .eq('posicion_id', posicionId)
   816вЖТ   816вЖТ  if (nivErr) throw nivErr
   817вЖТ   817вЖТ  const nivelIds = ((nivData ?? []) as { id: string }[]).map((n) => n.id)

   819вЖТ   819вЖТ
   820вЖТ   820вЖТ  // Intentar select con fecha_vencimiento primero; si falla (columna no existe), reintentar sin ella
   821вЖТ   821вЖТ  const getDetalles = async () => {
   822вЖТ   822вЖТ    // Intento 1: con fecha_vencimiento
   823вЖТ   823вЖТ    const { data: d1, error: e1 } = await dataClient
   824вЖТ   824вЖТ      .from('piso_movimiento_detalles')
   825вЖТ   825вЖТ      .select('bloque_id, cantidad, fecha_vencimiento, movimiento_id, piso_movimientos(tipo)')
   826вЖТ   826вЖТ      .in('nivel_id', nivelIds)
   827вЖТ   827вЖТ    if (!e1) return d1 as unknown[]

   830вЖТ   830вЖТ    const { data: d2, error: e2 } = await dataClient
   831вЖТ   831вЖТ      .from('piso_movimiento_detalles')
   832вЖТ   832вЖТ      .select('bloque_id, cantidad, movimiento_id, piso_movimientos(tipo)')
   833вЖТ   833вЖТ      .in('nivel_id', nivelIds)

   836вЖТ   836вЖТ  }
   837вЖТ   837вЖТ  const rawDetalles = await getDetalles()
   838вЖТ   838вЖТ
   839вЖТ   839вЖТ  type DetRow = { bloque_id: string; cantidad: unknown; fecha_vencimiento?: string | null; piso_movimientos: { tipo: string } | null | { tipo: string }[] }
   840вЖТ   840вЖТ  const detalles = (rawDetalles ?? []) as DetRow[]
   841вЖТ   841вЖТ
   842вЖТ   842вЖТ  const isIngresoType = (tipo: string) =>
   843вЖТ   843вЖТ    tipo === 'ingreso' || tipo === 'stock_inicial' || tipo === 'devolucion'
   844вЖТ   844вЖТ  const getTipo = (pm: DetRow['piso_movimientos']): string | null => {
   845вЖТ   845вЖТ    if (!pm) return null
   846вЖТ   846вЖТ    if (Array.isArray(pm)) return pm.length > 0 ? pm[0].tipo : null
   847вЖТ   847вЖТ    return pm.tipo
   848вЖТ   848вЖТ  }
   849вЖТ   849вЖТ
   850вЖТ   850вЖТ  // Pool de lotes por ingreso
   851вЖТ   851вЖТ  const ingresoPools = new Map<string, Map<string, number>>()
   852вЖТ   852вЖТ  for (const d of detalles) {

   854вЖТ   854вЖТ    if (!tipo || !isIngresoType(tipo)) continue
   855вЖТ   855вЖТ    const qty = typeof d.cantidad === 'number' ? d.cantidad : parseFloat(String(d.cantidad ?? '0')) || 0
   856вЖТ   856вЖТ    if (qty <= 0) continue
   857вЖТ   857вЖТ    const fv = (typeof d.fecha_vencimiento === 'string' && d.fecha_vencimiento) ? d.fecha_vencimiento : ''
   858вЖТ   858вЖТ    const pool = ingresoPools.get(d.bloque_id) ?? new Map<string, number>()
   859вЖТ   859вЖТ    pool.set(fv, (pool.get(fv) ?? 0) + qty)
   860вЖТ   860вЖТ    ingresoPools.set(d.bloque_id, pool)
   861вЖТ   861вЖТ  }

   875вЖТ   875вЖТ  for (const [bloqueId, pool] of ingresoPools) {

   877вЖТ   877вЖТ      if (!a && b) return 1

   880вЖТ   880вЖТ    })

   882вЖТ   882вЖТ    let pendiente = totalSalida
   883вЖТ   883вЖТ    const restantes: { fecha: string; qty: number }[] = []
   884вЖТ   884вЖТ    for (const [fecha, qty] of sortedLots) {
   885вЖТ   885вЖТ      if (pendiente <= 0) { restantes.push({ fecha, qty }) }
   886вЖТ   886вЖТ      else if (qty <= pendiente) { pendiente -= qty }
   887вЖТ   887вЖТ      else { restantes.push({ fecha, qty: qty - pendiente }); pendiente = 0 }
   888вЖТ   888вЖТ    }
   889вЖТ   889вЖТ    if (restantes.length > 0) lotesRestantes.set(bloqueId, restantes)
   890вЖТ   890вЖТ  }
   891вЖТ   891вЖТ
   892вЖТ   892вЖТ  if (lotesRestantes.size === 0) return []
   893вЖТ   893вЖТ
   894вЖТ   894вЖТ  // Info de bloques

   920вЖТ   920вЖТ      })
   921вЖТ   921вЖТ    }
   922вЖТ   922вЖТ  }
   923вЖТ   923вЖТ  results.sort((a, b) => {
   924вЖТ   924вЖТ    if (a.fecha_vencimiento && b.fecha_vencimiento) return a.fecha_vencimiento.localeCompare(b.fecha_vencimiento)
   925вЖТ   925вЖТ    if (a.fecha_vencimiento && !b.fecha_vencimiento) return -1
   926вЖТ   926вЖТ    if (!a.fecha_vencimiento && b.fecha_vencimiento) return 1
   927вЖТ   927вЖТ    return a.bloque_codigo.localeCompare(b.bloque_codigo)
   928вЖТ   928вЖТ  })
   929вЖТ   929вЖТ  return results
   930вЖТ   930вЖТ}

   933вЖТ   933вЖТ * Obtiene el stock detallado de un NIVEL espec√≠fico (para vista por niveles).
