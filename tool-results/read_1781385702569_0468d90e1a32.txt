    47→// Calcula días restantes hasta vencimiento (negativo si ya venció)
    48→function diasParaVencer(fVencimiento: string | undefined): number | null {
    49→  if (!fVencimiento) return null
    50→  const hoy = new Date(); hoy.setHours(0, 0, 0, 0)
    51→  const venc = new Date(fVencimiento + 'T00:00:00')
    52→  return Math.ceil((venc.getTime() - hoy.getTime()) / (1000 * 60 * 60 * 24))
    53→}

    55→// Detecta códigos duplicados (mismo artículo con distintas fechas de vencimiento)
    56→function codigosConMultiplesLotes(stock: StockEnUbicacion[]): Set<string> {
    57→  const counts = new Map<string, number>()
    58→  for (const s of stock) {
    59→    counts.set(s.codigo, (counts.get(s.codigo) ?? 0) + 1)
    60→  }
    61→  const multiples = new Set<string>()
    62→  for (const [code, count] of counts) {
    63→    if (count > 1) multiples.add(code)
    64→  }
    65→  return multiples
    66→}

    68→// Calcula ocupación desde movimientos — rastrea stock POR LOTE (código + vencimiento).
    69→// IMPORTANTE: Los movimientos INC se EXCLUYEN del stock normal.
    70→// La detección INC se hace por separado con fetchIncPorUbicacion().
    71→function calcularOcupacion(movs: Movimiento[]): OcupacionCelda[] {
    72→  // Mapa: ubicacion_key → (lote_key "codigo||fVencimiento" → stock)
    73→  const cellMap = new Map<string, Map<string, number>>()
    74→  for (const m of movs) {
    75→    // EXCLUIR movimientos INC del cálculo de ocupación normal
    76→    if (m.codigoInc) continue
    77→    const key = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}`
    78→    const code = m.codigo.trim().toUpperCase()
    79→    const lotKey = `${code}||${m.fVencimiento || ''}`
    80→    let lotMap = cellMap.get(key)
    81→    if (!lotMap) { lotMap = new Map(); cellMap.set(key, lotMap) }
    82→    const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? m.cantidad : -m.cantidad
    83→    const current = lotMap.get(lotKey) ?? 0
    84→    lotMap.set(lotKey, current + delta)
    85→  }
    86→  // Construir resultado: solo celdas con stock total > 0
    87→  const result: OcupacionCelda[] = []
    88→  for (const [key, lotMap] of cellMap) {
    89→    let totalStock = 0
    90→    const codigos = new Set<string>()
    91→    let lotes = 0
    92→    for (const [lotKey, stock] of lotMap) {
    93→      if (stock > 0) {
    94→        totalStock += stock
    95→        lotes++
    96→        codigos.add(lotKey.split('||')[0])
    97→      }
    98→    }
    99→    if (totalStock > 0) {
   100→      const [bloque, torre, piso, posicion] = key.split('-')
   101→      result.push({ bloque, torre, piso, posicion, stock: totalStock, codigos: Array.from(codigos), lotes, tieneInc: false, incItems: [] })
   102→    }
   103→  }
   104→  return result
   105→}

   286→        codigo: ingCodigo.trim().toUpperCase(), descripcion: ingDescripcion, un: ingUn, cantidad: q,

   349→        // PRESERVAR codigoInc para que la salida descuente del stock INC correctamente
   350→        codigoInc: item.codigoInc || undefined,
   351→      })
   352→      if (wasOffline) {
   353→        toast.success(`Salida de ${qty} ${item.un} (offline)`, { description: 'Se sincronizará al reconectarse', duration: 5000 })

   366→  async function doTransferir() {
   367→    if (!detail || !perfil || !trItem) return
   368→    if (!trDestBloque || !trDestTorre || !trDestPiso || !trDestPos) { toast.error('Completa destino'); return }
   369→    const qty = parseFloat(trCantidad)
   370→    if (isNaN(qty) || qty <= 0) { toast.error('Cantidad inválida'); return }
   371→    const origKey = `${detail.bloque}-${detail.torre}-${detail.piso}-${detail.posicion}`
   372→    const destKey = `${trDestBloque}-${trDestTorre}-${trDestPiso}-${trDestPos}`
   373→    if (origKey === destKey) { toast.error('Origen y destino no pueden ser iguales'); return }
   374→    setActionBusy(true)
   375→    try {
   376→      // Check destination occupancy
   377→      const destStock = await stockEnUbicacion(trDestBloque, trDestTorre, trDestPiso, trDestPos)
   378→      if (destStock.length > 0) {
   379→        toast.error(`Destino ocupado con ${destStock.length} artículo(s). Vacía primero el destino.`)
   380→        setActionBusy(false); return
   381→      }
   382→      const input: TrasladoInput = {
   383→        codigo: trItem.codigo, descripcion: trItem.descripcion, un: trItem.un, cantidad: qty,
   384→        origen: { bloque: detail.bloque, torre: detail.torre, piso: detail.piso, posicion: detail.posicion },
   385→        destino: { bloque: trDestBloque, torre: trDestTorre, piso: trDestPiso, posicion: trDestPos },
   386→        turno: calcularTurno(), usuarioId: perfil.id, usuarioNombre: perfil.nombre, usuarioCorreo: perfil.correo,
   387→        fVencimiento: trItem.fVencimiento ?? '',
   388→        cantidadAjuste: trTieneAjuste ? trDiferencia : 0,
   389→        // Preservar codigoInc para que el traslado mueva el INC correctamente
   390→        codigoInc: trItem.codigoInc || undefined,
   391→      }
   392→      const { wasOffline: wasTrOffline } = await SyncEngine.offlineAwareTraslado(input)
   393→      toast.success(wasTrOffline ? 'Traslado guardado (offline)' : 'Traslado registrado', {
   394→        ...(wasTrOffline ? { description: 'Se sincronizará al reconectarse', duration: 5000 } : {}),
   395→      })
   396→      if (mountedRef.current && !wasTrOffline) { await refreshDetail(); refreshData(); setDetailMode('view') }
   397→    } catch (err: unknown) {
   398→      if (isInsufficientStockError(err)) {
   399→        toast.error('Stock insuficiente en origen', { description: 'Otro usuario pudo haber modificado el stock. Los datos se han actualizado.', duration: 6000 })
   400→        refreshDetail(); refreshData()
   401→      } else { toast.error('Error al trasladar', { description: extractError(err) }) }
   402→    } finally { setActionBusy(false) }
   403→  }

   405→  async function handleExport() {
   406→    setBusyExport(true)
   407→    try {
