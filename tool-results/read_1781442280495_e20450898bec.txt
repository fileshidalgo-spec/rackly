    71→//  ANIMATED COUNTER HOOK
    72→// ═══════════════════════════════════════════════
    73→function useAnimatedCounter(target: number, duration = 800) {
    74→  const [count, setCount] = useState(0)
    75→  const prevTarget = useRef(0)
    76→
    77→  useEffect(() => {
    78→    if (target === prevTarget.current) return
    79→    const start = prevTarget.current
    80→    const diff = target - start
    81→    if (diff === 0) return
    82→    const startTime = performance.now()
    83→    let raf: number
    84→
    85→    function tick(now: number) {
    86→      const elapsed = now - startTime
    87→      const progress = Math.min(elapsed / duration, 1)
    88→      const eased = 1 - Math.pow(1 - progress, 3) // ease-out cubic
    89→      setCount(Math.round(start + diff * eased))
    90→      if (progress < 1) {
    91→        raf = requestAnimationFrame(tick)
    92→      }
    93→    }
    94→
    95→    raf = requestAnimationFrame(tick)
    96→    prevTarget.current = target
    97→    return () => cancelAnimationFrame(raf)
    98→  }, [target, duration])
    99→
   100→  return count
   101→}

   347→  // Click posicion (nivelId opcional para seleccionar nivel especifico desde la tabla)
   348→  async function handleClick(pos: PosicionConStock | null, nivelId?: string) {
   349→    if (!pos) return
   350→    try {
   351→      const [stock, nivs] = await Promise.all([
   352→        stockDetallePosicion(pos.posicionId),
   353→        obtenerNivelesPosicion(pos.posicionId),
   354→      ])
   355→      // Cargar stock por nivel (en paralelo)
   356→      const stockPerNivel: Record<string, DetailStock[]> = {}
   357→      if (nivs.length > 0) {
   358→        const nivelStocks = await Promise.all(
   359→          nivs.map((n) => stockDetalleNivel(n.id))
   360→        )
   361→        nivs.forEach((n, i) => { stockPerNivel[n.id] = nivelStocks[i] })
   362→      }
   363→      if (mountedRef.current) {
   364→        // Si se clickeó un nivel específico, filtrar stock solo a ese nivel
   365→        const filteredStock = nivelId ? stock.filter(s => {
   366→          // Buscar en qué nivel está este bloque
   367→          for (const niv of nivs) {
   368→            if (niv.id === nivelId) {
   369→              const nivelStock = stockPerNivel[niv.id] ?? []
   370→              if (nivelStock.some(ns => ns.bloque_id === s.bloque_id && ns.fecha_vencimiento === s.fecha_vencimiento)) {
   371→                return true
   372→              }
   373→            }
   374→          }
   375→          return false
   376→        }) : stock
   377→
   378→        setDetail({ posicionId: pos.posicionId, posicionNumero: pos.posicionNumero, subcolumnaCodigo: pos.subcolumnaCodigo, columnaLetra: pos.columnaLetra, stock: filteredStock })
   379→        setNiveles(nivs)
   380→        setSelectedNivelId(nivelId || (nivs.length > 0 ? nivs[0].id : ''))
   381→        setStockByNivel(stockPerNivel)
   382→        // Si viene de un clic en nivel específico, activar singleNivelMode
   383→        const isSingleNivel = !!nivelId
   384→        setSingleNivelMode(isSingleNivel)
   385→        setViewNivelTab(nivelId || 'all')
   386→        setMode('view')
   387→        // Inicializar salItems/trItems con el stock filtrado (por nivel o todos)
   388→        const itemsForActions = filteredStock.map((s) => ({
   389→          bloque_id: s.bloque_id,
   390→          bloque_codigo: s.bloque_codigo,
   391→          bloque_descripcion: s.bloque_descripcion,
   392→          bloque_unidad: s.bloque_unidad,
   393→          cantidad: String(s.cantidad),
   394→          stockActual: s.cantidad,
   395→          fecha_vencimiento: s.fecha_vencimiento || '',
   396→          selected: false,
   397→        }))
   398→        setSalItems(itemsForActions)
   399→        setTrItems(itemsForActions.map((s) => ({ ...s, saldoMode: 'saldo' as const })))
   400→      }
   401→    } catch { toast.error('Error al cargar detalle') }
   402→  }

   639→  // Auto-create manual_ bloques in piso_bloques before registering movement
   640→  async function ensureManualBloqueCreated(rows: RowEntry[]): Promise<RowEntry[]> {
   641→    const resolved: RowEntry[] = []
   642→    for (const r of rows) {
   643→      if (r.bloque_id.startsWith('manual_')) {
   644→        const code = r.bloque_id.replace('manual_', '')
   645→        try {
   646→          const created = await crearBloque(code, r.descripcion || '', r.unidad || 'KG')
   647→          if (created.length > 0) {
   648→            // Find the newly created one
   649→            const found = created.find((b) => b.codigo === code)
   650→            if (found) {
   651→              resolved.push({ ...r, bloque_id: found.id })
   652→              continue
   653→            }
   654→          }
   655→        } catch {
   656→          // If creation fails, try searching for it (maybe it was already created)
   657→        }
   658→        // Last resort: try buscarBloquePorCodigo
   659→        const existing = await buscarBloquePorCodigo(code)
   660→        if (existing) {
   661→          resolved.push({ ...r, bloque_id: existing.id })
   662→        } else {
   663→          resolved.push(r)
   664→        }
   665→      } else {
   666→        resolved.push(r)
   667→      }
   668→    }
   669→    return resolved
   670→  }

   790→  async function doTraslado() {
   791→    if (!detail || !perfil || !trDestPos) return
   792→    if (detail.posicionId === trDestPos.posicionId) { toast.error('Origen y destino no pueden ser iguales'); return }
   793→    const validRows = trItems.filter((r) => r.selected && r.bloque_id && r.cantidad && parseFloat(r.cantidad) > 0)
   794→    if (validRows.length === 0) { toast.error('No hay articulos para trasladar'); return }
   795→    if (!selectedNivelId || !trDestNivelId) { toast.error('Selecciona nivel de origen y destino'); return }
   796→    const origNivelId = selectedNivelId
   797→    const destNivelId = trDestNivelId
   798→    setBusy(true)
   799→    try {
   800→
   801→      // Separate items by discrepancy type
   802→      const exactItems = validRows.filter((r) => parseFloat(r.cantidad) === r.stockActual)
   803→      const deficitItems = validRows.filter((r) => parseFloat(r.cantidad) < r.stockActual)
   804→      const surplusItems = validRows.filter((r) => parseFloat(r.cantidad) > r.stockActual)
   805→
   806→      // 1) Base transfer for ALL selected items (move the entered amount)
   807→      const allDetSal = validRows.map((r) => ({ nivel_id: origNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   808→      const allDetIng = validRows.map((r) => ({ nivel_id: destNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   809→      await registrarTrasladoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', allDetSal, allDetIng)
   810→
   811→      // 2) For surplus items (qty > stock): create ingreso at destination for the excess
   812→      for (const item of surplusItems) {
   813→        const excess = parseFloat(item.cantidad) - item.stockActual
   814→        if (excess > 0) {
   815→          await registrarIngresoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', [
   816→            { nivel_id: destNivelId!, bloque_id: item.bloque_id, cantidad: excess, fecha_vencimiento: item.fecha_vencimiento || '' },
   817→          ])
   818→        }
   819→      }
   820→
   821→      // 3) For deficit items with saldoMode 'ajustar': create salida at origin for remaining balance
   822→      for (const item of deficitItems) {
   823→        if (item.saldoMode === 'ajustar') {
   824→          const remaining = item.stockActual - parseFloat(item.cantidad)
   825→          if (remaining > 0) {
   826→            await registrarSalidaPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', [
   827→              { nivel_id: origNivelId!, bloque_id: item.bloque_id, cantidad: remaining, fecha_vencimiento: item.fecha_vencimiento || null },
   828→            ])
   829→          }
   830→        }
   831→        // saldoMode === 'saldo': leave the balance as-is, no extra action needed
   832→      }
   833→
   834→      // Build result message
   835→      const parts: string[] = ['Traslado registrado']
   836→      if (surplusItems.length > 0) parts.push(`${surplusItems.length} ingreso(s) por excedente`)
   837→      const adjustedCount = deficitItems.filter((r) => r.saldoMode === 'ajustar').length
   838→      if (adjustedCount > 0) parts.push(`${adjustedCount} salida(s) por ajuste`)
   839→      toast.success(parts.join(' · '))
   840→      setTrConfirmOpen(false)
   841→      if (mountedRef.current) { setDetail(null); setTrDestPos(null); loadPosiciones(); if (selectedColumn) loadColumnDetail(selectedColumn) }
   842→    } catch (err: unknown) { toast.error('Error al trasladar', { description: extractError(err) }) } finally { setBusy(false) }
   843→  }

   938→  async function doMassSalida() {
   939→    if (!perfil || massData.size === 0) return
   940→    setMassBusy(true)
   941→    let successCount = 0
   942→    let errorCount = 0
   943→    try {
   944→      const entries = [...massData.entries()]
   945→      await Promise.all(entries.map(async ([posicionId, pd]) => {
   946→        try {
   947→          // Si no hay niveles seleccionados, salir de todos los niveles
   948→          const isAllLevels = pd.selectedNivelIds.size === 0
   949→          const detalles: { nivel_id: string; bloque_id: string; cantidad: number; fecha_vencimiento: string | null }[] = []
   950→          if (isAllLevels) {
   951→            // Salir de todos los niveles: agrupar por nivel_id real
   952→            for (const niv of pd.niveles) {
   953→              const items = pd.stockByNivel[niv.id] ?? []
   954→              for (const s of items) {
   955→                detalles.push({ nivel_id: niv.id, bloque_id: s.bloque_id, cantidad: s.cantidad, fecha_vencimiento: s.fecha_vencimiento || null })
   956→              }
   957→            }
   958→          } else {
   959→            // Salir solo de los niveles seleccionados
   960→            for (const nivelId of pd.selectedNivelIds) {
   961→              const items = pd.stockByNivel[nivelId] ?? []
   962→              for (const s of items) {
   963→                detalles.push({ nivel_id: nivelId, bloque_id: s.bloque_id, cantidad: s.cantidad, fecha_vencimiento: s.fecha_vencimiento || null })
   964→              }
   965→            }
   966→          }
   967→          if (detalles.length === 0) { return }
   968→          await registrarSalidaPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', detalles)
   969→          successCount++
   970→        } catch { errorCount++ }
   971→      }))
   972→      if (successCount > 0) toast.success(`Salida registrada en ${successCount} posicion(es)`)
   973→      if (errorCount > 0) toast.error(`${errorCount} posicion(es) con error`)
   974→      setMassDialogOpen(false)
   975→      setMassSelected(new Set())
   976→      setMassMode(false)
   977→      await loadPosiciones()
   978→      if (selectedColumn) loadColumnDetail(selectedColumn)
   979→    } catch (err: unknown) { toast.error('Error en salida masiva', { description: extractError(err) }) } finally { setMassBusy(false) }
   980→  }
