    71→//  ANIMATED COUNTER HOOK
    72→// ═══════════════════════════════════════════════
    73→function useAnimatedCounter(target: number, duration = 800) {
    74→  const [count, setCount] = useState(0)
    75→  const prevTarget = useRef(0)
    76→
    77→  useEffect(() => {
    78→    if (target === prevTarget.current) return
    79→    const start = prevTarget.current
    80→    const diff = target - start
    81→    if (diff === 0) return
    82→    const startTime = performance.now()
    83→    let raf: number
    84→
    85→    function tick(now: number) {
    86→      const elapsed = now - startTime
    87→      const progress = Math.min(elapsed / duration, 1)
    88→      const eased = 1 - Math.pow(1 - progress, 3) // ease-out cubic
    89→      setCount(Math.round(start + diff * eased))
    90→      if (progress < 1) {
    91→        raf = requestAnimationFrame(tick)
    92→      }
    93→    }
    94→
    95→    raf = requestAnimationFrame(tick)
    96→    prevTarget.current = target
    97→    return () => cancelAnimationFrame(raf)
    98→  }, [target, duration])
    99→
   100→  return count
   101→}

   349→  // Click posicion (nivelId opcional para seleccionar nivel especifico desde la tabla)
   350→  async function handleClick(pos: PosicionConStock | null, nivelId?: string) {
   351→    if (!pos) return
   352→    try {
   353→      const [stock, nivs] = await Promise.all([
   354→        stockDetallePosicion(pos.posicionId),
   355→        obtenerNivelesPosicion(pos.posicionId, pos.columnaLetra),
   356→      ])
   357→      // Cargar stock por nivel (en paralelo)
   358→      const stockPerNivel: Record<string, DetailStock[]> = {}
   359→      if (nivs.length > 0) {
   360→        const nivelStocks = await Promise.all(
   361→          nivs.map((n) => stockDetalleNivel(n.id))
   362→        )
   363→        nivs.forEach((n, i) => { stockPerNivel[n.id] = nivelStocks[i] })
   364→      }
   365→      if (mountedRef.current) {
   366→        // Si se clickeó un nivel específico, filtrar stock solo a ese nivel
   367→        const filteredStock = nivelId ? stock.filter(s => {
   368→          // Buscar en qué nivel está este bloque
   369→          for (const niv of nivs) {
   370→            if (niv.id === nivelId) {
   371→              const nivelStock = stockPerNivel[niv.id] ?? []
   372→              if (nivelStock.some(ns => ns.bloque_id === s.bloque_id && ns.fecha_vencimiento === s.fecha_vencimiento)) {
   373→                return true
   374→              }
   375→            }
   376→          }
   377→          return false
   378→        }) : stock
   379→
   380→        setDetail({ posicionId: pos.posicionId, posicionNumero: pos.posicionNumero, subcolumnaCodigo: pos.subcolumnaCodigo, columnaLetra: pos.columnaLetra, stock: filteredStock })
   381→        setNiveles(nivs)
   382→        setSelectedNivelId(nivelId || (nivs.length > 0 ? nivs[0].id : ''))
   383→        setStockByNivel(stockPerNivel)
   384→        // Si viene de un clic en nivel específico, activar singleNivelMode
   385→        const isSingleNivel = !!nivelId
   386→        setSingleNivelMode(isSingleNivel)
   387→        setViewNivelTab(nivelId || 'all')
   388→        setMode('view')
   389→        // Inicializar salItems/trItems con el stock filtrado (por nivel o todos)
   390→        const itemsForActions = filteredStock.map((s) => ({
   391→          bloque_id: s.bloque_id,
   392→          bloque_codigo: s.bloque_codigo,
   393→          bloque_descripcion: s.bloque_descripcion,
   394→          bloque_unidad: s.bloque_unidad,
   395→          cantidad: String(s.cantidad),
   396→          stockActual: s.cantidad,
   397→          fecha_vencimiento: s.fecha_vencimiento || '',
   398→          selected: false,
   399→        }))
   400→        setSalItems(itemsForActions)
   401→        setTrItems(itemsForActions.map((s) => ({ ...s, saldoMode: 'saldo' as const })))
   402→      }
   403→    } catch { toast.error('Error al cargar detalle') }
   404→  }

   641→  // Auto-create manual_ bloques in piso_bloques before registering movement
   642→  async function ensureManualBloqueCreated(rows: RowEntry[]): Promise<RowEntry[]> {
   643→    const resolved: RowEntry[] = []
   644→    for (const r of rows) {
   645→      if (r.bloque_id.startsWith('manual_')) {
   646→        const code = r.bloque_id.replace('manual_', '')
   647→        try {
   648→          const created = await crearBloque(code, r.descripcion || '', r.unidad || 'KG')
   649→          if (created.length > 0) {
   650→            // Find the newly created one
   651→            const found = created.find((b) => b.codigo === code)
   652→            if (found) {
   653→              resolved.push({ ...r, bloque_id: found.id })
   654→              continue
   655→            }
   656→          }
   657→        } catch {
   658→          // If creation fails, try searching for it (maybe it was already created)
   659→        }
   660→        // Last resort: try buscarBloquePorCodigo
   661→        const existing = await buscarBloquePorCodigo(code)
   662→        if (existing) {
   663→          resolved.push({ ...r, bloque_id: existing.id })
   664→        } else {
   665→          resolved.push(r)
   666→        }
   667→      } else {
   668→        resolved.push(r)
   669→      }
   670→    }
   671→    return resolved
   672→  }

   792→  async function doTraslado() {
   793→    if (!detail || !perfil || !trDestPos) return
   794→    if (detail.posicionId === trDestPos.posicionId) { toast.error('Origen y destino no pueden ser iguales'); return }
   795→    const validRows = trItems.filter((r) => r.selected && r.bloque_id && r.cantidad && parseFloat(r.cantidad) > 0)
   796→    if (validRows.length === 0) { toast.error('No hay articulos para trasladar'); return }
   797→    if (!selectedNivelId || !trDestNivelId) { toast.error('Selecciona nivel de origen y destino'); return }
   798→    const origNivelId = selectedNivelId
   799→    const destNivelId = trDestNivelId
   800→    setBusy(true)
   801→    try {
   802→
   803→      // Separate items by discrepancy type
   804→      const exactItems = validRows.filter((r) => parseFloat(r.cantidad) === r.stockActual)
   805→      const deficitItems = validRows.filter((r) => parseFloat(r.cantidad) < r.stockActual)
   806→      const surplusItems = validRows.filter((r) => parseFloat(r.cantidad) > r.stockActual)
   807→
   808→      // 1) Base transfer for ALL selected items (move the entered amount)
   809→      const allDetSal = validRows.map((r) => ({ nivel_id: origNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   810→      const allDetIng = validRows.map((r) => ({ nivel_id: destNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   811→      await registrarTrasladoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', allDetSal, allDetIng)
   812→
   813→      // 2) For surplus items (qty > stock): create ingreso at destination for the excess
   814→      for (const item of surplusItems) {
   815→        const excess = parseFloat(item.cantidad) - item.stockActual
   816→        if (excess > 0) {
   817→          await registrarIngresoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', [
   818→            { nivel_id: destNivelId!, bloque_id: item.bloque_id, cantidad: excess, fecha_vencimiento: item.fecha_vencimiento || '' },
   819→          ])
   820→        }
   821→      }
   822→
   823→      // 3) For deficit items with saldoMode 'ajustar': create salida at origin for remaining balance
   824→      for (const item of deficitItems) {
   825→        if (item.saldoMode === 'ajustar') {
   826→          const remaining = item.stockActual - parseFloat(item.cantidad)
   827→          if (remaining > 0) {
   828→            await registrarSalidaPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', [
   829→              { nivel_id: origNivelId!, bloque_id: item.bloque_id, cantidad: remaining, fecha_vencimiento: item.fecha_vencimiento || null },
   830→            ])
   831→          }
   832→        }
   833→        // saldoMode === 'saldo': leave the balance as-is, no extra action needed
   834→      }
   835→
   836→      // Build result message
   837→      const parts: string[] = ['Traslado registrado']
   838→      if (surplusItems.length > 0) parts.push(`${surplusItems.length} ingreso(s) por excedente`)
   839→      const adjustedCount = deficitItems.filter((r) => r.saldoMode === 'ajustar').length
   840→      if (adjustedCount > 0) parts.push(`${adjustedCount} salida(s) por ajuste`)
   841→      toast.success(parts.join(' · '))
   842→      setTrConfirmOpen(false)
   843→      if (mountedRef.current) { setDetail(null); setTrDestPos(null); loadPosiciones(); if (selectedColumn) loadColumnDetail(selectedColumn) }
   844→    } catch (err: unknown) { toast.error('Error al trasladar', { description: extractError(err) }) } finally { setBusy(false) }
   845→  }

   940→  async function doMassSalida() {
   941→    if (!perfil || massData.size === 0) return
   942→    setMassBusy(true)
   943→    let successCount = 0
   944→    let errorCount = 0
   945→    try {
   946→      const entries = [...massData.entries()]
   947→      await Promise.all(entries.map(async ([posicionId, pd]) => {
   948→        try {
   949→          // Si no hay niveles seleccionados, salir de todos los niveles
   950→          const isAllLevels = pd.selectedNivelIds.size === 0
   951→          const detalles: { nivel_id: string; bloque_id: string; cantidad: number; fecha_vencimiento: string | null }[] = []
   952→          if (isAllLevels) {
   953→            // Salir de todos los niveles: agrupar por nivel_id real
   954→            for (const niv of pd.niveles) {
   955→              const items = pd.stockByNivel[niv.id] ?? []
   956→              for (const s of items) {
   957→                detalles.push({ nivel_id: niv.id, bloque_id: s.bloque_id, cantidad: s.cantidad, fecha_vencimiento: s.fecha_vencimiento || null })
   958→              }
   959→            }
   960→          } else {
   961→            // Salir solo de los niveles seleccionados
   962→            for (const nivelId of pd.selectedNivelIds) {
   963→              const items = pd.stockByNivel[nivelId] ?? []
   964→              for (const s of items) {
   965→                detalles.push({ nivel_id: nivelId, bloque_id: s.bloque_id, cantidad: s.cantidad, fecha_vencimiento: s.fecha_vencimiento || null })
   966→              }
   967→            }
   968→          }
   969→          if (detalles.length === 0) { return }
   970→          await registrarSalidaPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', detalles)
   971→          successCount++
   972→        } catch { errorCount++ }
   973→      }))
   974→      if (successCount > 0) toast.success(`Salida registrada en ${successCount} posicion(es)`)
   975→      if (errorCount > 0) toast.error(`${errorCount} posicion(es) con error`)
   976→      setMassDialogOpen(false)
   977→      setMassSelected(new Set())
   978→      setMassMode(false)
   979→      await loadPosiciones()
   980→      if (selectedColumn) loadColumnDetail(selectedColumn)
   981→    } catch (err: unknown) { toast.error('Error en salida masiva', { description: extractError(err) }) } finally { setMassBusy(false) }
   982→  }
