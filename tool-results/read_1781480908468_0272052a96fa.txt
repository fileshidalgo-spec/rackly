    72→//  ANIMATED COUNTER HOOK
    73→// ═══════════════════════════════════════════════
    74→function useAnimatedCounter(target: number, duration = 800) {
    75→  const [count, setCount] = useState(0)
    76→  const prevTarget = useRef(0)
    77→
    78→  useEffect(() => {
    79→    if (target === prevTarget.current) return
    80→    const start = prevTarget.current
    81→    const diff = target - start
    82→    if (diff === 0) return
    83→    const startTime = performance.now()
    84→    let raf: number
    85→
    86→    function tick(now: number) {
    87→      const elapsed = now - startTime
    88→      const progress = Math.min(elapsed / duration, 1)
    89→      const eased = 1 - Math.pow(1 - progress, 3) // ease-out cubic
    90→      setCount(Math.round(start + diff * eased))
    91→      if (progress < 1) {
    92→        raf = requestAnimationFrame(tick)
    93→      }
    94→    }
    95→
    96→    raf = requestAnimationFrame(tick)
    97→    prevTarget.current = target
    98→    return () => cancelAnimationFrame(raf)
    99→  }, [target, duration])
   100→
   101→  return count
   102→}

   744→  // Auto-create manual_ bloques in piso_bloques before registering movement
   745→  async function ensureManualBloqueCreated(rows: RowEntry[]): Promise<RowEntry[]> {
   746→    const resolved: RowEntry[] = []
   747→    for (const r of rows) {
   748→      if (r.bloque_id.startsWith('manual_')) {
   749→        const code = r.bloque_id.replace('manual_', '')
   750→        try {
   751→          const created = await crearBloque(code, r.descripcion || '', r.unidad || 'KG')
   752→          if (created.length > 0) {
   753→            // Find the newly created one
   754→            const found = created.find((b) => b.codigo === code)
   755→            if (found) {
   756→              resolved.push({ ...r, bloque_id: found.id })
   757→              continue
   758→            }
   759→          }
   760→        } catch {
   761→          // If creation fails, try searching for it (maybe it was already created)
   762→        }
   763→        // Last resort: try buscarBloquePorCodigo
   764→        const existing = await buscarBloquePorCodigo(code)
   765→        if (existing) {
   766→          resolved.push({ ...r, bloque_id: existing.id })
   767→        } else {
   768→          resolved.push(r)
   769→        }
   770→      } else {
   771→        resolved.push(r)
   772→      }
   773→    }
   774→    return resolved
   775→  }

   895→  async function doTraslado() {
   896→    if (!detail || !perfil || !trDestPos) return
   897→    if (detail.posicionId === trDestPos.posicionId) { toast.error('Origen y destino no pueden ser iguales'); return }
   898→    const validRows = trItems.filter((r) => r.selected && r.bloque_id && r.cantidad && parseFloat(r.cantidad) > 0)
   899→    if (validRows.length === 0) { toast.error('No hay articulos para trasladar'); return }
   900→    if (!selectedNivelId || !trDestNivelId) { toast.error('Selecciona nivel de origen y destino'); return }
   901→    const origNivelId = selectedNivelId
   902→    const destNivelId = trDestNivelId
   903→    setBusy(true)
   904→    try {
   905→
   906→      // Separate items by discrepancy type
   907→      const exactItems = validRows.filter((r) => parseFloat(r.cantidad) === r.stockActual)
   908→      const deficitItems = validRows.filter((r) => parseFloat(r.cantidad) < r.stockActual)
   909→      const surplusItems = validRows.filter((r) => parseFloat(r.cantidad) > r.stockActual)
   910→
   911→      // 1) Base transfer for ALL selected items (move the entered amount)
   912→      const allDetSal = validRows.map((r) => ({ nivel_id: origNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   913→      const allDetIng = validRows.map((r) => ({ nivel_id: destNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   914→      await registrarTrasladoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', allDetSal, allDetIng)
   915→
   916→      // 2) For surplus items (qty > stock): create ingreso at destination for the excess
   917→      for (const item of surplusItems) {
   918→        const excess = parseFloat(item.cantidad) - item.stockActual
   919→        if (excess > 0) {
   920→          await registrarIngresoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', [
   921→            { nivel_id: destNivelId!, bloque_id: item.bloque_id, cantidad: excess, fecha_vencimiento: item.fecha_vencimiento || '' },
   922→          ])
   923→        }
   924→      }
   925→
   926→      // 3) For deficit items with saldoMode 'ajustar': create salida at origin for remaining balance
   927→      for (const item of deficitItems) {
   928→        if (item.saldoMode === 'ajustar') {
   929→          const remaining = item.stockActual - parseFloat(item.cantidad)

   932→              { nivel_id: origNivelId!, bloque_id: item.bloque_id, cantidad: remaining, fecha_vencimiento: item.fecha_vencimiento || null },
   933→            ])
   934→          }
   935→        }
   936→        // saldoMode === 'saldo': leave the balance as-is, no extra action needed
   937→      }
   938→
   939→      // Build result message
   940→      const parts: string[] = ['Traslado registrado']
   941→      if (surplusItems.length > 0) parts.push(`${surplusItems.length} ingreso(s) por excedente`)
   942→      const adjustedCount = deficitItems.filter((r) => r.saldoMode === 'ajustar').length
   943→      if (adjustedCount > 0) parts.push(`${adjustedCount} salida(s) por ajuste`)
   944→      toast.success(parts.join(' · '))
   945→      setTrConfirmOpen(false)
   946→      if (mountedRef.current) { setDetail(null); setTrDestPos(null); loadPosiciones(); if (selectedColumn) loadColumnDetail(selectedColumn) }
