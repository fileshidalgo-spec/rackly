    47→// Calcula días restantes hasta vencimiento (negativo si ya venció)
    48→function diasParaVencer(fVencimiento: string | undefined): number | null {
    49→  if (!fVencimiento) return null
    50→  const hoy = new Date(); hoy.setHours(0, 0, 0, 0)
    51→  const venc = new Date(fVencimiento + 'T00:00:00')
    52→  return Math.ceil((venc.getTime() - hoy.getTime()) / (1000 * 60 * 60 * 24))
    53→}

    55→// Detecta códigos duplicados (mismo artículo con distintas fechas de vencimiento)
    56→function codigosConMultiplesLotes(stock: StockEnUbicacion[]): Set<string> {
    57→  const counts = new Map<string, number>()
    58→  for (const s of stock) {
    59→    counts.set(s.codigo, (counts.get(s.codigo) ?? 0) + 1)
    60→  }
    61→  const multiples = new Set<string>()
    62→  for (const [code, count] of counts) {
    63→    if (count > 1) multiples.add(code)
    64→  }
    65→  return multiples
    66→}

    68→// Calcula ocupación desde movimientos — rastrea stock POR LOTE (código + vencimiento).
    69→// IMPORTANTE: Los movimientos INC se EXCLUYEN del stock normal.
    70→// La detección INC se hace por separado con fetchIncPorUbicacion().
    71→function calcularOcupacion(movs: Movimiento[]): OcupacionCelda[] {
    72→  // Mapa: ubicacion_key → (lote_key "codigo||fVencimiento" → stock)
    73→  const cellMap = new Map<string, Map<string, number>>()
    74→  for (const m of movs) {
    75→    // EXCLUIR movimientos INC del cálculo de ocupación normal
    76→    if (m.codigoInc) continue
    77→    const key = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}`
    78→    const code = m.codigo.trim().toUpperCase()
    79→    const lotKey = `${code}||${m.fVencimiento || ''}`
    80→    let lotMap = cellMap.get(key)
    81→    if (!lotMap) { lotMap = new Map(); cellMap.set(key, lotMap) }
    82→    const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? m.cantidad : -m.cantidad
    83→    const current = lotMap.get(lotKey) ?? 0
    84→    lotMap.set(lotKey, current + delta)
    85→  }
    86→  // Construir resultado: solo celdas con stock total > 0
    87→  const result: OcupacionCelda[] = []
    88→  for (const [key, lotMap] of cellMap) {
    89→    let totalStock = 0
    90→    const codigos = new Set<string>()
    91→    let lotes = 0
    92→    for (const [lotKey, stock] of lotMap) {
    93→      if (stock > 0) {
    94→        totalStock += stock
    95→        lotes++
    96→        codigos.add(lotKey.split('||')[0])
    97→      }
    98→    }
    99→    if (totalStock > 0) {
   100→      const [bloque, torre, piso, posicion] = key.split('-')
   101→      result.push({ bloque, torre, piso, posicion, stock: totalStock, codigos: Array.from(codigos), lotes, tieneInc: false, incItems: [] })
   102→    }
   103→  }
   104→  return result
   105→}

   411→  async function doSalida() {
   412→    if (!detail || !perfil) return
   413→    const item = detail.stock[salidaIdx]
   414→    if (!item) return
   415→    const qty = parseFloat(salidaCantidad)
   416→    if (isNaN(qty) || qty <= 0) { toast.error('Cantidad inválida'); return }
   417→    // Las salidas normales NO pueden exceder stock. Las salidas INC sí permiten exceder (autoajuste).
   418→    if (!item.codigoInc && qty > item.stock) { toast.error(`Cantidad excede stock (${item.stock})`); return }
   419→    setActionBusy(true)
   420→    try {
   421→      const { wasOffline } = await SyncEngine.offlineAwareAddMovimiento({
   422→        tipo: 'salida', bloque: detail.bloque, torre: detail.torre, piso: detail.piso, posicion: detail.posicion,
   423→        codigo: item.codigo, descripcion: item.descripcion, un: item.un, cantidad: qty,
   424→        fVencimiento: item.fVencimiento ?? '', turno: calcularTurno(), usuarioId: perfil.id, usuarioNombre: perfil.nombre, usuarioCorreo: perfil.correo,
   425→        // PRESERVAR codigoInc para que la salida descuente del stock INC correctamente
   426→        codigoInc: item.codigoInc || undefined,
   427→      })
   428→      if (wasOffline) {
   429→        toast.success(`Salida de ${qty} ${item.un} (offline)`, { description: 'Se sincronizará al reconectarse', duration: 5000 })
   430→      } else {
   431→        toast.success('Salida registrada')
   432→      }
   433→      if (mountedRef.current && !wasOffline) { await refreshDetail(); refreshData(); setDetailMode('view') }
   434→    } catch (err: unknown) {
   435→      if (isInsufficientStockError(err)) {

   442→  async function doTransferir() {
   443→    if (!detail || !perfil || !trItem) return
   444→    if (!trDestBloque || !trDestTorre || !trDestPiso || !trDestPos) { toast.error('Completa destino'); return }
   445→    const qty = parseFloat(trCantidad)
   446→    if (isNaN(qty) || qty <= 0) { toast.error('Cantidad inválida'); return }
   447→    const origKey = `${detail.bloque}-${detail.torre}-${detail.piso}-${detail.posicion}`
   448→    const destKey = `${trDestBloque}-${trDestTorre}-${trDestPiso}-${trDestPos}`
   449→    if (origKey === destKey) { toast.error('Origen y destino no pueden ser iguales'); return }
   450→    setActionBusy(true)
   451→    try {
   452→      // Check destination occupancy
   453→      const destStock = await stockEnUbicacion(trDestBloque, trDestTorre, trDestPiso, trDestPos)
   454→      if (destStock.length > 0) {
   455→        toast.error(`Destino ocupado con ${destStock.length} artículo(s). Vacía primero el destino.`)
   456→        setActionBusy(false); return
   457→      }
   458→      const input: TrasladoInput = {
   459→        codigo: trItem.codigo, descripcion: trItem.descripcion, un: trItem.un, cantidad: qty,
   460→        origen: { bloque: detail.bloque, torre: detail.torre, piso: detail.piso, posicion: detail.posicion },
   461→        destino: { bloque: trDestBloque, torre: trDestTorre, piso: trDestPiso, posicion: trDestPos },
   462→        turno: calcularTurno(), usuarioId: perfil.id, usuarioNombre: perfil.nombre, usuarioCorreo: perfil.correo,
   463→        fVencimiento: trItem.fVencimiento ?? '',
   464→        cantidadAjuste: trTieneAjuste ? trDiferencia : 0,
   465→        // Preservar codigoInc para que el traslado mueva el INC correctamente
   466→        codigoInc: trItem.codigoInc || undefined,
   467→      }
   468→      const { wasOffline: wasTrOffline } = await SyncEngine.offlineAwareTraslado(input)
   469→      toast.success(wasTrOffline ? 'Traslado guardado (offline)' : 'Traslado registrado', {
   470→        ...(wasTrOffline ? { description: 'Se sincronizará al reconectarse', duration: 5000 } : {}),
   471→      })
   472→      if (mountedRef.current && !wasTrOffline) { await refreshDetail(); refreshData(); setDetailMode('view') }
   473→    } catch (err: unknown) {
   474→      if (isInsufficientStockError(err)) {
   475→        toast.error('Stock insuficiente en origen', { description: 'Otro usuario pudo haber modificado el stock. Los datos se han actualizado.', duration: 6000 })
   476→        refreshDetail(); refreshData()
   477→      } else { toast.error('Error al trasladar', { description: extractError(err) }) }
   478→    } finally { setActionBusy(false) }
   479→  }
