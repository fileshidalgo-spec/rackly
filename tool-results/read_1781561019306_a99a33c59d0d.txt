    72→//  ANIMATED COUNTER HOOK
    73→// ═══════════════════════════════════════════════
    74→function useAnimatedCounter(target: number, duration = 800) {
    75→  const [count, setCount] = useState(0)
    76→  const prevTarget = useRef(0)
    77→
    78→  useEffect(() => {
    79→    if (target === prevTarget.current) return
    80→    const start = prevTarget.current
    81→    const diff = target - start
    82→    if (diff === 0) return
    83→    const startTime = performance.now()
    84→    let raf: number
    85→
    86→    function tick(now: number) {
    87→      const elapsed = now - startTime
    88→      const progress = Math.min(elapsed / duration, 1)
    89→      const eased = 1 - Math.pow(1 - progress, 3) // ease-out cubic
    90→      setCount(Math.round(start + diff * eased))
    91→      if (progress < 1) {
    92→        raf = requestAnimationFrame(tick)
    93→      }
    94→    }
    95→
    96→    raf = requestAnimationFrame(tick)
    97→    prevTarget.current = target
    98→    return () => cancelAnimationFrame(raf)
    99→  }, [target, duration])
   100→
   101→  return count
   102→}

   635→  async function doIngresoINC() {

   637→    if (!incCodigo.trim() || !incCantidad || !incCodigoInc.trim()) {
   638→      toast.error('Completa codigo, cantidad y codigo INC')
   639→      return
   640→    }
   641→    const qty = parseFloat(incCantidad)
   642→    if (isNaN(qty) || qty <= 0) { toast.error('Cantidad invalida'); return }
   643→    if (!selectedNivelId) { toast.error('Selecciona un nivel primero'); return }
   644→    const nivelId = selectedNivelId
   645→    setBusy(true)
   646→    try {
   647→      // Resolve bloque_id
   648→      const upper = incCodigo.trim().toUpperCase()

   650→      const existing = await buscarBloquePorCodigo(upper)
   651→      if (existing) {
   652→        bloqueId = existing.id
   653→      } else {
   654→        const created = await crearBloque(upper, incDescripcion || '', incUn || 'KG')
   655→        if (created.length > 0) {
   656→          const found = created.find((b) => b.codigo === upper)
   657→          if (found) bloqueId = found.id

   660→      if (!bloqueId) { toast.error('No se pudo crear/encontrar el articulo'); setBusy(false); return }
   661→
   662→      // Registrar ingreso INC con detalles de stock (usando registrarIngresoPosicion)
   663→      const detalles = [{
   664→        nivel_id: nivelId,
   665→        bloque_id: bloqueId,
   666→        cantidad: qty,
   667→        fecha_vencimiento: incSinVencimiento ? '' : incFechaVencimiento,
   668→      }]
   669→      await registrarIngresoPosicion(
   670→        calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '',
   671→        detalles,
   672→        { posicion_id: detail.posicionId, codigo_inc: incCodigoInc.trim() }

   676→      setMode('view')
   677→      setIncCodigo('')
   678→      setIncDescripcion('')
   679→      setIncUn('')
   680→      setIncCantidad('')
   681→      setIncCodigoInc('')
   682→      setIncFechaVencimiento('')
   683→      setIncSinVencimiento(false)

   686→        stockDetallePosicion(detail.posicionId),
   687→        niveles.length > 0 ? Promise.all(niveles.map((n) => stockDetalleNivel(n.id))) : Promise.resolve([]),
   688→      ])
   689→      const newSBN: Record<string, DetailStock[]> = {}

   701→      if (selectedColumn) loadColumnDetail(selectedColumn)

   764→  // Auto-create manual_ bloques in piso_bloques before registering movement
   765→  async function ensureManualBloqueCreated(rows: RowEntry[]): Promise<RowEntry[]> {
   766→    const resolved: RowEntry[] = []
   767→    for (const r of rows) {
   768→      if (r.bloque_id.startsWith('manual_')) {
   769→        const code = r.bloque_id.replace('manual_', '')
   770→        try {
   771→          const created = await crearBloque(code, r.descripcion || '', r.unidad || 'KG')
   772→          if (created.length > 0) {
   773→            // Find the newly created one
   774→            const found = created.find((b) => b.codigo === code)
   775→            if (found) {
   776→              resolved.push({ ...r, bloque_id: found.id })
   777→              continue
   778→            }
   779→          }
   780→        } catch {
   781→          // If creation fails, try searching for it (maybe it was already created)
   782→        }
   783→        // Last resort: try buscarBloquePorCodigo
   784→        const existing = await buscarBloquePorCodigo(code)
   785→        if (existing) {
   786→          resolved.push({ ...r, bloque_id: existing.id })
   787→        } else {
   788→          resolved.push(r)
   789→        }
   790→      } else {
   791→        resolved.push(r)
   792→      }
   793→    }
   794→    return resolved
   795→  }

   829→  async function doIngreso() {
   830→    if (!detail || !perfil) return
   831→    if (!selectedNivelId) { toast.error('No hay nivel seleccionado'); return }
   832→    const nivelId = selectedNivelId
   833→    const validRows = ingRows.filter((r) => r.bloque_id && r.cantidad)
   834→    if (validRows.length === 0) { toast.error('Agrega al menos un articulo con codigo y cantidad'); return }
   835→    for (const r of validRows) {
   836→      if (parseFloat(r.cantidad) <= 0 || isNaN(parseFloat(r.cantidad))) { toast.error('Cantidad invalida'); return }
   837→    }
   838→    setBusy(true)
   839→    try {
   840→      // Resolve manual_ IDs before registering
   841→      const resolved = await ensureManualBloqueCreated(validRows)
   842→      const detalles = resolved.map((r) => ({
   843→        nivel_id: nivelId,
   844→        bloque_id: r.bloque_id,
   845→        cantidad: parseFloat(r.cantidad),
   846→        fecha_vencimiento: r.sin_vencimiento ? '' : r.fecha_vencimiento,
   847→      }))
   848→      await registrarIngresoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', detalles)
   849→      toast.success('Ingreso registrado')
   850→      // Reload everything in parallel for real-time update
   851→      loadBloques()
   852→      const [stock, reloadedPerNivel] = await Promise.all([
   853→        stockDetallePosicion(detail.posicionId),
   854→        (async () => {
   855→          const perNivel: Record<string, DetailStock[]> = {}
   856→          for (const n of niveles) {
   857→            perNivel[n.id] = await stockDetalleNivel(n.id)
   858→          }
   859→          return perNivel
   860→        })(),
   861→      ])
   862→      if (mountedRef.current && detail) {
   863→        // Mantener filtro de nivel si estaba activo
   864→        const currentTab = viewNivelTab
   865→        const filteredStock = currentTab !== 'all' ? stock.filter(s => {
   866→          const nivelStock = reloadedPerNivel[currentTab] ?? []
   867→          return nivelStock.some(ns => ns.bloque_id === s.bloque_id && ns.fecha_vencimiento === s.fecha_vencimiento)
   868→        }) : stock
   869→        setDetail({ ...detail, stock: filteredStock })
   870→        setStockByNivel(reloadedPerNivel)
   871→        setMode('view')
   872→        setIngRows([{ ...EMPTY_ROW }])
   873→      }
   874→      // Reload positions grid
   875→      await loadPosiciones()
   876→      if (selectedColumn) loadColumnDetail(selectedColumn)
   877→    } catch (err: unknown) { toast.error('Error al registrar ingreso', { description: extractError(err) }) } finally { setBusy(false) }
   878→  }

   920→    if (!selectedNivelId || !trDestNivelId) { toast.error('Selecciona nivel de origen y destino'); return }
   921→    const origNivelId = selectedNivelId
   922→    const destNivelId = trDestNivelId
   923→    setBusy(true)
   924→    try {
   925→
   926→      // Separate items by discrepancy type
   927→      const exactItems = validRows.filter((r) => parseFloat(r.cantidad) === r.stockActual)
   928→      const deficitItems = validRows.filter((r) => parseFloat(r.cantidad) < r.stockActual)
   929→      const surplusItems = validRows.filter((r) => parseFloat(r.cantidad) > r.stockActual)
   930→
   931→      // 1) Base transfer for ALL selected items (move the entered amount)
   932→      const allDetSal = validRows.map((r) => ({ nivel_id: origNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   933→      const allDetIng = validRows.map((r) => ({ nivel_id: destNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   934→      await registrarTrasladoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', allDetSal, allDetIng)
   935→
   936→      // 2) For surplus items (qty > stock): create ingreso at destination for the excess
   937→      for (const item of surplusItems) {
   938→        const excess = parseFloat(item.cantidad) - item.stockActual
   939→        if (excess > 0) {
   940→          await registrarIngresoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', [
   941→            { nivel_id: destNivelId!, bloque_id: item.bloque_id, cantidad: excess, fecha_vencimiento: item.fecha_vencimiento || '' },
   942→          ])
   943→        }
   944→      }
   945→
   946→      // 3) For deficit items with saldoMode 'ajustar': create salida at origin for remaining balance
   947→      for (const item of deficitItems) {
   948→        if (item.saldoMode === 'ajustar') {
   949→          const remaining = item.stockActual - parseFloat(item.cantidad)
   950→          if (remaining > 0) {
   951→            await registrarSalidaPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', [
   952→              { nivel_id: origNivelId!, bloque_id: item.bloque_id, cantidad: remaining, fecha_vencimiento: item.fecha_vencimiento || null },
   953→            ])
   954→          }
   955→        }
   956→        // saldoMode === 'saldo': leave the balance as-is, no extra action needed
   957→      }
   958→
   959→      // Build result message
   960→      const parts: string[] = ['Traslado registrado']
   961→      if (surplusItems.length > 0) parts.push(`${surplusItems.length} ingreso(s) por excedente`)
   962→      const adjustedCount = deficitItems.filter((r) => r.saldoMode === 'ajustar').length
   963→      if (adjustedCount > 0) parts.push(`${adjustedCount} salida(s) por ajuste`)
   964→      toast.success(parts.join(' · '))
   965→      setTrConfirmOpen(false)
   966→      if (mountedRef.current) { setDetail(null); setTrDestPos(null); loadPosiciones(); if (selectedColumn) loadColumnDetail(selectedColumn) }
   967→    } catch (err: unknown) { toast.error('Error al trasladar', { description: extractError(err) }) } finally { setBusy(false) }
   968→  }
