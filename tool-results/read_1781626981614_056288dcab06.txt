    73тЖТ    73тЖТexport async function fetchMovimientos(): Promise<Movimiento[]> {
    74тЖТ    74тЖТ  const all: Record<string, unknown>[] = []
    75тЖТ    75тЖТ  let from = 0
    76тЖТ    76тЖТ  let iterations = 0
    77тЖТ    77тЖТ  while (iterations < MAX_ITERATIONS) {
    78тЖТ    78тЖТ    iterations++
    79тЖТ    79тЖТ    const to = from + PAGE_SIZE - 1
    80тЖТ    80тЖТ    const { data, error } = await dataClient
    81тЖТ    81тЖТ      .from('movimientos')
    82тЖТ    82тЖТ      .select('*')
    83тЖТ    83тЖТ      .order('f_modificacion', { ascending: false })
    84тЖТ    84тЖТ      .order('id', { ascending: true })
    85тЖТ    85тЖТ      .range(from, to)
    86тЖТ    86тЖТ    if (error) throw error
    87тЖТ    87тЖТ    const rows = data ?? []
    88тЖТ    88тЖТ    all.push(...rows)
    89тЖТ    89тЖТ    if (rows.length < PAGE_SIZE) break
    90тЖТ    90тЖТ    from += PAGE_SIZE
    91тЖТ    91тЖТ  }
    92тЖТ    92тЖТ  return all.map(fromRow)
    93тЖТ    93тЖТ}

   193тЖТ   193тЖТ  // Supabase lo ignora silenciosamente y el campo queda NULL.

   207тЖТ   207тЖТ      turno: m.turno,
   208тЖТ   208тЖТ      usuario_id: m.usuarioId,
   209тЖТ   209тЖТ      usuario_nombre: m.usuarioNombre ?? null,

   211тЖТ   211тЖТ      proveedor: m.proveedor ? m.proveedor : null,
   212тЖТ   212тЖТ      uuid_sync: uuidSync || null,
   213тЖТ   213тЖТ      codigo_inc: m.codigoInc,
   214тЖТ   214тЖТ    })
   215тЖТ   215тЖТ    if (error) throw error
   216тЖТ   216тЖТ    try {
   217тЖТ   217тЖТ      return await fetchMovimientos()
   218тЖТ   218тЖТ    } catch {
   219тЖТ   219тЖТ      return []

   221тЖТ   221тЖТ  }
   222тЖТ   222тЖТ
   223тЖТ   223тЖТ  // Usar RPC at├│mica con advisory lock para evitar race conditions
   224тЖТ   224тЖТ  try {
   225тЖТ   225тЖТ    const { data, error } = await dataClient.rpc('registrar_movimiento_kardex', {
   226тЖТ   226тЖТ      p_tipo: m.tipo,
   227тЖТ   227тЖТ      p_bloque: m.bloque,
   228тЖТ   228тЖТ      p_torre: m.torre,

   233тЖТ   233тЖТ      p_un: m.un,
   234тЖТ   234тЖТ      p_cantidad: m.cantidad,
   235тЖТ   235тЖТ      p_f_vencimiento: m.fVencimiento || null,
   236тЖТ   236тЖТ      p_turno: m.turno,
   237тЖТ   237тЖТ      p_usuario_id: m.usuarioId,
   238тЖТ   238тЖТ      p_usuario_nombre: m.usuarioNombre ?? null,
   239тЖТ   239тЖТ      p_usuario_correo: m.usuarioCorreo ?? null,

   247тЖТ   247тЖТ      const code = (error as unknown as Record<string, string>).code || ''

   266тЖТ   266тЖТ    } catch {

   270тЖТ   270тЖТ  } catch (err: unknown) {
   271тЖТ   271тЖТ    // Si la RPC no existe a├║n (SQL no ejecutado), fallback al insert directo

   276тЖТ   276тЖТ    }
   277тЖТ   277тЖТ    throw err
   278тЖТ   278тЖТ  }
   279тЖТ   279тЖТ}

   411тЖТ   411тЖТ      descripcion: g.descripcion,
   412тЖТ   412тЖТ      un: g.un,

   608тЖТ   608тЖТ    if (error) {

   610тЖТ   610тЖТ      const code = (error as unknown as Record<string, string>).code || ''
   611тЖТ   611тЖТ      if (msg.includes('INSUFFICIENT_STOCK')) {
   612тЖТ   612тЖТ        const parts = msg.split('|')
   613тЖТ   613тЖТ        const detail = parts.length > 1 ? parts[1] : 'Stock insuficiente en origen para este traslado'

   634тЖТ   634тЖТ    const errCode = err instanceof Error ? (err as unknown as Record<string, string>).code || '' : ''

   653тЖТ   653тЖТ  let from = 0
   654тЖТ   654тЖТ  let iterations = 0

   658тЖТ   658тЖТ      .from('movimientos')

   663тЖТ   663тЖТ    allIds.push(...data.map((r: { id: string }) => r.id))
   664тЖТ   664тЖТ    if (data.length < BATCH) break
   665тЖТ   665тЖТ    from += BATCH
   666тЖТ   666тЖТ  }
   667тЖТ   667тЖТ
   668тЖТ   668тЖТ  if (allIds.length === 0) return { deleted: true }
   669тЖТ   669тЖТ

   671тЖТ   671тЖТ  const DELETE_BATCH = 500
   672тЖТ   672тЖТ  for (let i = 0; i < allIds.length; i += DELETE_BATCH) {
   673тЖТ   673тЖТ    const batch = allIds.slice(i, i + DELETE_BATCH)

   675тЖТ   675тЖТ      .from('movimientos')

   677тЖТ   677тЖТ      .in('id', batch)

   679тЖТ   679тЖТ  }

   681тЖТ   681тЖТ  return { deleted: true }
