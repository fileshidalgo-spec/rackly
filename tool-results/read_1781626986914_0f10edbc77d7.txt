  1485тЖТ  1485тЖТ  1485тЖТ/**
  1486тЖТ  1486тЖТ  1486тЖТ * Obtiene todo el stock de Piso con FEFO y ubicaci├│n resuelta.

  1489тЖТ  1489тЖТ  1489тЖТ */
  1490тЖТ  1490тЖТ  1490тЖТexport async function stockPisoGlobal(): Promise<StockPisoItem[]> {
  1491тЖТ  1491тЖТ  1491тЖТ  // 1. Fetch ALL detalles with movement type
  1492тЖТ  1492тЖТ  1492тЖТ  const { data: detData, error: detErr } = await dataClient
  1493тЖТ  1493тЖТ  1493тЖТ    .from('piso_movimiento_detalles')
  1494тЖТ  1494тЖТ  1494тЖТ    .select('bloque_id, nivel_id, cantidad, fecha_vencimiento, movimiento_id, piso_movimientos(tipo, codigo_inc)')
  1495тЖТ  1495тЖТ  1495тЖТ  if (detErr) throw detErr
  1496тЖТ  1496тЖТ  1496тЖТ
  1497тЖТ  1497тЖТ  1497тЖТ  // 2. Group by (bloque_id, nivel_id)
  1498тЖТ  1498тЖТ  1498тЖТ  const isIngresoType = (tipo: string) =>
  1499тЖТ  1499тЖТ  1499тЖТ    tipo === 'ingreso' || tipo === 'stock_inicial' || tipo === 'devolucion'
  1500тЖТ  1500тЖТ  1500тЖТ  const getTipo = (pm: unknown): string => {
  1501тЖТ  1501тЖТ  1501тЖТ    if (!pm) return ''
  1502тЖТ  1502тЖТ  1502тЖТ    if (Array.isArray(pm)) return pm.length > 0 ? (pm[0] as { tipo: string }).tipo : ''
  1503тЖТ  1503тЖТ  1503тЖТ    return (pm as { tipo: string }).tipo ?? ''
  1504тЖТ  1504тЖТ  1504тЖТ  }
  1505тЖТ  1505тЖТ  1505тЖТ
  1506тЖТ  1506тЖТ  1506тЖТ  type RawDet = {
  1507тЖТ  1507тЖТ  1507тЖТ    bloque_id: string
  1508тЖТ  1508тЖТ  1508тЖТ    nivel_id: string
  1509тЖТ  1509тЖТ  1509тЖТ    cantidad: unknown
  1510тЖТ  1510тЖТ  1510тЖТ    fecha_vencimiento?: string | null
  1511тЖТ  1511тЖТ  1511тЖТ    movimiento_id: string
  1512тЖТ  1512тЖТ  1512тЖТ    piso_movimientos: { tipo: string; codigo_inc?: string | null } | null | { tipo: string; codigo_inc?: string | null }[]
  1513тЖТ  1513тЖТ  1513тЖТ  }
  1514тЖТ  1514тЖТ  1514тЖТ  const rawDets = (detData ?? []) as RawDet[]
  1515тЖТ  1515тЖТ  1515тЖТ
  1516тЖТ  1516тЖТ  1516тЖТ  const blockNivelMap = new Map<string, { bloque_id: string; nivel_id: string; tipo: string; cantidad: number; fecha_vencimiento: string; codigo_inc: string }[]>()
  1517тЖТ  1517тЖТ  1517тЖТ  for (const d of rawDets) {
  1518тЖТ  1518тЖТ  1518тЖТ    const qty = typeof d.cantidad === 'number' ? d.cantidad : parseFloat(String(d.cantidad ?? '0')) || 0
  1519тЖТ  1519тЖТ  1519тЖТ    if (qty <= 0) continue
  1520тЖТ  1520тЖТ  1520тЖТ    const tipo = getTipo(d.piso_movimientos)
  1521тЖТ  1521тЖТ  1521тЖТ    if (!tipo) continue
  1522тЖТ  1522тЖТ  1522тЖТ    const fv = (typeof d.fecha_vencimiento === 'string' && d.fecha_vencimiento) ? d.fecha_vencimiento : ''
  1523тЖТ  1523тЖТ  1523тЖТ    const getCodigoInc = (pm: unknown): string => {
  1524тЖТ  1524тЖТ  1524тЖТ      if (!pm) return ''
  1525тЖТ  1525тЖТ  1525тЖТ      if (Array.isArray(pm)) return pm.length > 0 ? (pm[0] as { codigo_inc?: string | null }).codigo_inc || '' : ''
  1526тЖТ  1526тЖТ  1526тЖТ      return (pm as { codigo_inc?: string | null }).codigo_inc || ''
  1527тЖТ  1527тЖТ  1527тЖТ    }
  1528тЖТ  1528тЖТ  1528тЖТ    const codigoInc = getCodigoInc(d.piso_movimientos)
  1529тЖТ  1529тЖТ  1529тЖТ    const key = `${d.bloque_id}__${d.nivel_id}`
  1530тЖТ  1530тЖТ  1530тЖТ    const arr = blockNivelMap.get(key) ?? []
  1531тЖТ  1531тЖТ  1531тЖТ    arr.push({ bloque_id: d.bloque_id, nivel_id: d.nivel_id, tipo, cantidad: qty, fecha_vencimiento: fv, codigo_inc: codigoInc })
  1532тЖТ  1532тЖТ  1532тЖТ    blockNivelMap.set(key, arr)
  1533тЖТ  1533тЖТ  1533тЖТ  }
  1534тЖТ  1534тЖТ  1534тЖТ
  1535тЖТ  1535тЖТ  1535тЖТ  // 3. FEFO per (bloque_id, nivel_id)
  1536тЖТ  1536тЖТ  1536тЖТ  const fefoResults: { bloque_id: string; nivel_id: string; cantidad: number; fecha_vencimiento: string; codigo_inc: string }[] = []
  1537тЖТ  1537тЖТ  1537тЖТ  for (const [key, details] of blockNivelMap) {
  1538тЖТ  1538тЖТ  1538тЖТ    const ingresoPool = new Map<string, { cantidad: number; codigo_inc: string }>()
  1539тЖТ  1539тЖТ  1539тЖТ    let totalExit = 0
  1540тЖТ  1540тЖТ  1540тЖТ    for (const d of details) {
  1541тЖТ  1541тЖТ  1541тЖТ      if (isIngresoType(d.tipo)) {
  1542тЖТ  1542тЖТ  1542тЖТ        const existing = ingresoPool.get(d.fecha_vencimiento)
  1543тЖТ  1543тЖТ  1543тЖТ        if (existing) {
  1544тЖТ  1544тЖТ  1544тЖТ          existing.cantidad += d.cantidad
  1545тЖТ  1545тЖТ  1545тЖТ          if (!existing.codigo_inc && d.codigo_inc) existing.codigo_inc = d.codigo_inc
  1546тЖТ  1546тЖТ  1546тЖТ        } else {
  1547тЖТ  1547тЖТ  1547тЖТ          ingresoPool.set(d.fecha_vencimiento, { cantidad: d.cantidad, codigo_inc: d.codigo_inc })
  1548тЖТ  1548тЖТ  1548тЖТ        }
  1549тЖТ  1549тЖТ  1549тЖТ      } else {
  1550тЖТ  1550тЖТ  1550тЖТ        totalExit += d.cantidad
  1551тЖТ  1551тЖТ  1551тЖТ      }
  1552тЖТ  1552тЖТ  1552тЖТ    }
  1553тЖТ  1553тЖТ  1553тЖТ    const sortedLots = [...ingresoPool.entries()].sort(([a], [b]) => {
  1554тЖТ  1554тЖТ  1554тЖТ      if (!a && b) return 1
  1555тЖТ  1555тЖТ  1555тЖТ      if (a && !b) return -1
  1556тЖТ  1556тЖТ  1556тЖТ      return a.localeCompare(b)
  1557тЖТ  1557тЖТ  1557тЖТ    })
  1558тЖТ  1558тЖТ  1558тЖТ    let pendiente = totalExit
  1559тЖТ  1559тЖТ  1559тЖТ    for (const [fecha, pool] of sortedLots) {
  1560тЖТ  1560тЖТ  1560тЖТ      if (pendiente <= 0) {
  1561тЖТ  1561тЖТ  1561тЖТ        fefoResults.push({ bloque_id: details[0].bloque_id, nivel_id: details[0].nivel_id, cantidad: pool.cantidad, fecha_vencimiento: fecha, codigo_inc: pool.codigo_inc })
  1562тЖТ  1562тЖТ  1562тЖТ      } else if (pool.cantidad <= pendiente) {
  1563тЖТ  1563тЖТ  1563тЖТ        pendiente -= pool.cantidad
  1564тЖТ  1564тЖТ  1564тЖТ      } else {
  1565тЖТ  1565тЖТ  1565тЖТ        fefoResults.push({ bloque_id: details[0].bloque_id, nivel_id: details[0].nivel_id, cantidad: pool.cantidad - pendiente, fecha_vencimiento: fecha, codigo_inc: pool.codigo_inc })
  1566тЖТ  1566тЖТ  1566тЖТ        pendiente = 0
  1567тЖТ  1567тЖТ  1567тЖТ      }
  1568тЖТ  1568тЖТ  1568тЖТ    }
  1569тЖТ  1569тЖТ  1569тЖТ  }
  1570тЖТ  1570тЖТ  1570тЖТ
  1571тЖТ  1571тЖТ  1571тЖТ  if (fefoResults.length === 0) return []
  1572тЖТ  1572тЖТ  1572тЖТ
  1573тЖТ  1573тЖТ  1573тЖТ  // 4. Resolve bloque info
  1574тЖТ  1574тЖТ  1574тЖТ  const bloqueIds = [...new Set(fefoResults.map(r => r.bloque_id))]
  1575тЖТ  1575тЖТ  1575тЖТ  const bloqueInfoMap = new Map<string, { codigo: string; descripcion: string; unidad: string }>()
  1576тЖТ  1576тЖТ  1576тЖТ  const realIds = bloqueIds.filter(id => !id.startsWith('cat_') && !id.startsWith('manual_'))
  1577тЖТ  1577тЖТ  1577тЖТ  if (realIds.length > 0) {
  1578тЖТ  1578тЖТ  1578тЖТ    const { data: bData } = await dataClient.from('piso_bloques').select('id, codigo, descripcion, unidad').in('id', realIds)
  1579тЖТ  1579тЖТ  1579тЖТ    for (const b of (bData ?? []) as { id: string; codigo: string; descripcion: string; unidad: string }[])
  1580тЖТ  1580тЖТ  1580тЖТ      bloqueInfoMap.set(b.id, { codigo: b.codigo, descripcion: b.descripcion ?? '', unidad: b.unidad ?? '' })
  1581тЖТ  1581тЖТ  1581тЖТ  }
  1582тЖТ  1582тЖТ  1582тЖТ  const catIds = bloqueIds.filter(id => id.startsWith('cat_'))

  1585тЖТ  1585тЖТ  1585тЖТ    const { data: cData } = await dataClient.from('catalogo').select('codigo, descripcion, un').in('codigo', codes)
  1586тЖТ  1586тЖТ  1586тЖТ    for (const c of (cData ?? []) as { codigo: string; descripcion: string; un: string }[])
  1587тЖТ  1587тЖТ  1587тЖТ      bloqueInfoMap.set(`cat_${c.codigo}`, { codigo: c.codigo, descripcion: c.descripcion ?? '', unidad: c.un ?? '' })
  1588тЖТ  1588тЖТ  1588тЖТ  }
  1589тЖТ  1589тЖТ  1589тЖТ  for (const id of bloqueIds) {
  1590тЖТ  1590тЖТ  1590тЖТ    if (id.startsWith('manual_') && !bloqueInfoMap.has(id))
  1591тЖТ  1591тЖТ  1591тЖТ      bloqueInfoMap.set(id, { codigo: id.replace('manual_', ''), descripcion: 'Articulo nuevo', unidad: 'KG' })
  1592тЖТ  1592тЖТ  1592тЖТ  }
  1593тЖТ  1593тЖТ  1593тЖТ
  1594тЖТ  1594тЖТ  1594тЖТ  // 5. Resolve location for each nivel_id
  1595тЖТ  1595тЖТ  1595тЖТ  const nivelIds = [...new Set(fefoResults.map(r => r.nivel_id))]

  1650тЖТ  1650тЖТ  1650тЖТ

  1655тЖТ  1655тЖТ  1655тЖТ      const info = bloqueInfoMap.get(r.bloque_id)

  1660тЖТ  1660тЖТ  1660тЖТ        bloque_unidad: info?.unidad ?? 'KG',

  1664тЖТ  1664тЖТ  1664тЖТ        fecha_vencimiento: r.fecha_vencimiento,
  1665тЖТ  1665тЖТ  1665тЖТ        codigo_inc: r.codigo_inc || undefined,
  1666тЖТ  1666тЖТ  1666тЖТ      }
  1667тЖТ  1667тЖТ  1667тЖТ    })
  1668тЖТ  1668тЖТ  1668тЖТ    .sort((a, b) => {

  1671тЖТ  1671тЖТ  1671тЖТ      if (a.fecha_vencimiento && !b.fecha_vencimiento) return -1
  1672тЖТ  1672тЖТ  1672тЖТ      if (!a.fecha_vencimiento && b.fecha_vencimiento) return 1
  1673тЖТ  1673тЖТ  1673тЖТ      return a.ubicacion.localeCompare(b.ubicacion)
  1674тЖТ  1674тЖТ  1674тЖТ    })
  1675тЖТ  1675тЖТ  1675тЖТ}

  1695тЖТ  1695тЖТ  1695тЖТ/**
  1696тЖТ  1696тЖТ  1696тЖТ * Carga toda la informaci├│n de una columna para la vista de tabla:
  1697тЖТ  1697тЖТ  1697тЖТ * posiciones тЖТ niveles тЖТ stock por nivel (con detecci├│n de INC).

  1701тЖТ  1701тЖТ  1701тЖТ  sectorId: string,
  1702тЖТ  1702тЖТ  1702тЖТ  columnaLetra: string

  1704тЖТ  1704тЖТ  1704тЖТ  // 1. Obtener columna ID
  1705тЖТ  1705тЖТ  1705тЖТ  const { data: colData, error: colErr } = await dataClient

  1710тЖТ  1710тЖТ  1710тЖТ    .single()
  1711тЖТ  1711тЖТ  1711тЖТ  if (colErr || !colData) return []
  1712тЖТ  1712тЖТ  1712тЖТ
  1713тЖТ  1713тЖТ  1713тЖТ  // 2. Obtener subcolumnas тЖТ posiciones тЖТ niveles en una sola query anidada

  1716тЖТ  1716тЖТ  1716тЖТ    .select(`
  1717тЖТ  1717тЖТ  1717тЖТ      id, codigo,
  1718тЖТ  1718тЖТ  1718тЖТ      piso_posiciones(id, numero, piso_niveles(id, numero))
  1719тЖТ  1719тЖТ  1719тЖТ    `)
  1720тЖТ  1720тЖТ  1720тЖТ    .eq('columna_id', colData.id)
  1721тЖТ  1721тЖТ  1721тЖТ    .order('codigo')

  1724тЖТ  1724тЖТ  1724тЖТ  // Build flat structure

  1727тЖТ  1727тЖТ  1727тЖТ  for (const sc of treeData as { id: string; codigo: string; piso_posiciones: { id: string; numero: number; piso_niveles: { id: string; numero: number }[] | null }[] }[]) {
  1728тЖТ  1728тЖТ  1728тЖТ    for (const pos of sc.piso_posiciones ?? []) {
  1729тЖТ  1729тЖТ  1729тЖТ      for (const niv of pos.piso_niveles ?? []) {
  1730тЖТ  1730тЖТ  1730тЖТ        niveles.push({
  1731тЖТ  1731тЖТ  1731тЖТ          id: niv.id,
  1732тЖТ  1732тЖТ  1732тЖТ          numero: niv.numero,
  1733тЖТ  1733тЖТ  1733тЖТ          posicionId: pos.id,
  1734тЖТ  1734тЖТ  1734тЖТ          posicionNumero: pos.numero,
  1735тЖТ  1735тЖТ  1735тЖТ          subcolCodigo: sc.codigo,
  1736тЖТ  1736тЖТ  1736тЖТ        })
  1737тЖТ  1737тЖТ  1737тЖТ      }
  1738тЖТ  1738тЖТ  1738тЖТ    }
  1739тЖТ  1739тЖТ  1739тЖТ  }
  1740тЖТ  1740тЖТ  1740тЖТ
  1741тЖТ  1741тЖТ  1741тЖТ  if (niveles.length === 0) {
  1742тЖТ  1742тЖТ  1742тЖТ    // Return positions without niveles
  1743тЖТ  1743тЖТ  1743тЖТ    const cleanSubcol = (code: string) => {
  1744тЖТ  1744тЖТ  1744тЖТ      const idx = code.indexOf(columnaLetra)
  1745тЖТ  1745тЖТ  1745тЖТ      return idx >= 0 ? code.substring(idx) : code
  1746тЖТ  1746тЖТ  1746тЖТ    }
  1747тЖТ  1747тЖТ  1747тЖТ    const result: VistaPosicion[] = []
  1748тЖТ  1748тЖТ  1748тЖТ    for (const sc of treeData as { id: string; codigo: string; piso_posiciones: { id: string; numero: number; piso_niveles: unknown[] | null }[] }[]) {
  1749тЖТ  1749тЖТ  1749тЖТ      for (const pos of (sc.piso_posiciones ?? [])) {
  1750тЖТ  1750тЖТ  1750тЖТ        result.push({
  1751тЖТ  1751тЖТ  1751тЖТ          posicionId: pos.id,
  1752тЖТ  1752тЖТ  1752тЖТ          posicionNumero: pos.numero,
  1753тЖТ  1753тЖТ  1753тЖТ          subcolumnaCodigo: cleanSubcol(sc.codigo),
  1754тЖТ  1754тЖТ  1754тЖТ          tieneInc: false,
  1755тЖТ  1755тЖТ  1755тЖТ          niveles: [],
  1756тЖТ  1756тЖТ  1756тЖТ        })
  1757тЖТ  1757тЖТ  1757тЖТ      }
  1758тЖТ  1758тЖТ  1758тЖТ    }
  1759тЖТ  1759тЖТ  1759тЖТ    return result.sort((a, b) => a.posicionNumero - b.posicionNumero)
  1760тЖТ  1760тЖТ  1760тЖТ  }
  1761тЖТ  1761тЖТ  1761тЖТ
  1762тЖТ  1762тЖТ  1762тЖТ  const nivelIds = niveles.map(n => n.id)
  1763тЖТ  1763тЖТ  1763тЖТ
  1764тЖТ  1764тЖТ  1764тЖТ  // 3. Obtener TODOS los detalles de movimiento para estos niveles (1 sola query)
  1765тЖТ  1765тЖТ  1765тЖТ  const { data: movData, error: movErr } = await dataClient
  1766тЖТ  1766тЖТ  1766тЖТ    .from('piso_movimiento_detalles')
  1767тЖТ  1767тЖТ  1767тЖТ    .select('nivel_id, bloque_id, cantidad, fecha_vencimiento, movimiento_id, piso_movimientos(tipo, codigo_inc)')
  1768тЖТ  1768тЖТ  1768тЖТ    .in('nivel_id', nivelIds)
  1769тЖТ  1769тЖТ  1769тЖТ  const movDetalles = (movData ?? []) as {
  1770тЖТ  1770тЖТ  1770тЖТ    nivel_id: string; bloque_id: string; cantidad: unknown; fecha_vencimiento: string | null;
  1771тЖТ  1771тЖТ  1771тЖТ    movimiento_id: string;
  1772тЖТ  1772тЖТ  1772тЖТ    piso_movimientos: { tipo: string; codigo_inc: string | null } | null | { tipo: string; codigo_inc: string | null }[]
  1773тЖТ  1773тЖТ  1773тЖТ  }[]
  1774тЖТ  1774тЖТ  1774тЖТ
  1775тЖТ  1775тЖТ  1775тЖТ  // 4. Calcular stock por nivel por bloque (neto: ingresos - salidas)
  1776тЖТ  1776тЖТ  1776тЖТ  const isIngreso = (tipo: string) => tipo === 'ingreso' || tipo === 'stock_inicial' || tipo === 'devolucion'
  1777тЖТ  1777тЖТ  1777тЖТ  const getMovInfo = (pm: typeof movDetalles[0]['piso_movimientos']): { tipo: string; codigo_inc: string } => {
  1778тЖТ  1778тЖТ  1778тЖТ    if (!pm) return { tipo: '', codigo_inc: '' }
  1779тЖТ  1779тЖТ  1779тЖТ    if (Array.isArray(pm)) return pm.length > 0 ? { tipo: pm[0].tipo, codigo_inc: pm[0].codigo_inc || '' } : { tipo: '', codigo_inc: '' }
  1780тЖТ  1780тЖТ  1780тЖТ    return { tipo: (pm as { tipo: string; codigo_inc: string | null }).tipo, codigo_inc: (pm as { tipo: string; codigo_inc: string | null }).codigo_inc || '' }
  1781тЖТ  1781тЖТ  1781тЖТ  }
  1782тЖТ  1782тЖТ  1782тЖТ
  1783тЖТ  1783тЖТ  1783тЖТ  // Stock neto por (nivel_id, bloque_id): cantidad y si tiene INC
  1784тЖТ  1784тЖТ  1784тЖТ  const stockMap = new Map<string, { cantidad: number; codigo_inc: string }>()
  1785тЖТ  1785тЖТ  1785тЖТ  const bloqueIds = new Set<string>()
  1786тЖТ  1786тЖТ  1786тЖТ
  1787тЖТ  1787тЖТ  1787тЖТ  for (const d of movDetalles) {
  1788тЖТ  1788тЖТ  1788тЖТ    const { tipo, codigo_inc } = getMovInfo(d.piso_movimientos)
  1789тЖТ  1789тЖТ  1789тЖТ    if (!tipo) continue
  1790тЖТ  1790тЖТ  1790тЖТ    const qty = typeof d.cantidad === 'number' ? d.cantidad : parseFloat(String(d.cantidad ?? '0')) || 0
  1791тЖТ  1791тЖТ  1791тЖТ    if (qty <= 0) continue
  1792тЖТ  1792тЖТ  1792тЖТ    bloqueIds.add(d.bloque_id)
  1793тЖТ  1793тЖТ  1793тЖТ    const key = `${d.nivel_id}::${d.bloque_id}`
  1794тЖТ  1794тЖТ  1794тЖТ    const existing = stockMap.get(key) || { cantidad: 0, codigo_inc: '' }
  1795тЖТ  1795тЖТ  1795тЖТ    if (isIngreso(tipo)) {
  1796тЖТ  1796тЖТ  1796тЖТ      existing.cantidad += qty
  1797тЖТ  1797тЖТ  1797тЖТ      if (codigo_inc && !existing.codigo_inc) existing.codigo_inc = codigo_inc
  1798тЖТ  1798тЖТ  1798тЖТ    } else {
  1799тЖТ  1799тЖТ  1799тЖТ      existing.cantidad -= qty
  1800тЖТ  1800тЖТ  1800тЖТ    }
  1801тЖТ  1801тЖТ  1801тЖТ    stockMap.set(key, existing)
  1802тЖТ  1802тЖТ  1802тЖТ  }
  1803тЖТ  1803тЖТ  1803тЖТ
  1804тЖТ  1804тЖТ  1804тЖТ  // 5. Obtener info de bloques (1 query)
  1805тЖТ  1805тЖТ  1805тЖТ  const bloqueInfoMap = new Map<string, { codigo: string; descripcion: string; unidad: string }>()
  1806тЖТ  1806тЖТ  1806тЖТ  const realIds = [...bloqueIds].filter(id => !id.startsWith('cat_') && !id.startsWith('manual_'))
  1807тЖТ  1807тЖТ  1807тЖТ  if (realIds.length > 0) {
  1808тЖТ  1808тЖТ  1808тЖТ    const { data: bData } = await dataClient.from('piso_bloques').select('id, codigo, descripcion, unidad').in('id', realIds)

  1810тЖТ  1810тЖТ  1810тЖТ      bloqueInfoMap.set(b.id, { codigo: b.codigo, descripcion: b.descripcion ?? '', unidad: b.unidad ?? '' })

  1812тЖТ  1812тЖТ  1812тЖТ  }
  1813тЖТ  1813тЖТ  1813тЖТ  const catIds = [...bloqueIds].filter(id => id.startsWith('cat_'))

  1818тЖТ  1818тЖТ  1818тЖТ      bloqueInfoMap.set(`cat_${c.codigo}`, { codigo: c.codigo, descripcion: c.descripcion ?? '', unidad: c.un ?? '' })
  1819тЖТ  1819тЖТ  1819тЖТ    }
  1820тЖТ  1820тЖТ  1820тЖТ  }
  1821тЖТ  1821тЖТ  1821тЖТ
  1822тЖТ  1822тЖТ  1822тЖТ  // 6. Build result: agrupar por posici├│n
  1823тЖТ  1823тЖТ  1823тЖТ  const posMap = new Map<string, VistaPosicion>()
  1824тЖТ  1824тЖТ  1824тЖТ  for (const n of niveles) {

  1827тЖТ  1827тЖТ  1827тЖТ      posMap.set(n.posicionId, {
  1828тЖТ  1828тЖТ  1828тЖТ        posicionId: n.posicionId,
  1829тЖТ  1829тЖТ  1829тЖТ        posicionNumero: n.posicionNumero,
  1830тЖТ  1830тЖТ  1830тЖТ        subcolumnaCodigo: cleanCode,
  1831тЖТ  1831тЖТ  1831тЖТ        tieneInc: false,
  1832тЖТ  1832тЖТ  1832тЖТ        niveles: [],
  1833тЖТ  1833тЖТ  1833тЖТ      })
  1834тЖТ  1834тЖТ  1834тЖТ    }
  1835тЖТ  1835тЖТ  1835тЖТ    const pos = posMap.get(n.posicionId)!
  1836тЖТ  1836тЖТ  1836тЖТ
  1837тЖТ  1837тЖТ  1837тЖТ    // Find stock items for this nivel
  1838тЖТ  1838тЖТ  1838тЖТ    const items: VistaNivelStock['bloques'] = []

  1843тЖТ  1843тЖТ  1843тЖТ      items.push({
  1844тЖТ  1844тЖТ  1844тЖТ        bloque_codigo: info.codigo,
  1845тЖТ  1845тЖТ  1845тЖТ        bloque_descripcion: info.descripcion,
  1846тЖТ  1846тЖТ  1846тЖТ        bloque_unidad: info.unidad,
  1847тЖТ  1847тЖТ  1847тЖТ        cantidad: Math.round(val.cantidad * 1000) / 1000,
  1848тЖТ  1848тЖТ  1848тЖТ        codigo_inc: val.codigo_inc,
  1849тЖТ  1849тЖТ  1849тЖТ      })
  1850тЖТ  1850тЖТ  1850тЖТ      if (val.codigo_inc) pos.tieneInc = true

  1854тЖТ  1854тЖТ  1854тЖТ    pos.niveles.push({
  1855тЖТ  1855тЖТ  1855тЖТ      nivelId: n.id,
  1856тЖТ  1856тЖТ  1856тЖТ      nivelNumero: n.numero,
