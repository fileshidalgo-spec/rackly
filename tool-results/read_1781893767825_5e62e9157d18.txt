    73→export async function fetchMovimientos(): Promise<Movimiento[]> {
    74→  const all: Record<string, unknown>[] = []
    75→  let from = 0
    76→  let iterations = 0
    77→  while (iterations < MAX_ITERATIONS) {
    78→    iterations++
    79→    const to = from + PAGE_SIZE - 1
    80→    const { data, error } = await dataClient
    81→      .from('movimientos')
    82→      .select('*')
    83→      .order('f_modificacion', { ascending: false })
    84→      .order('id', { ascending: true })
    85→      .range(from, to)
    86→    if (error) throw error
    87→    const rows = data ?? []
    88→    all.push(...rows)
    89→    if (rows.length < PAGE_SIZE) break
    90→    from += PAGE_SIZE
    91→  }
    92→  return all.map(fromRow)
    93→}

   120→/**
   121→ * Calcula el stock neto en Racks para un código dado.
   122→ * Usado por PisoStockTab para mostrar "También en Kardex Racks".
   123→ * Solo lectura, no modifica nada.
   124→ */
   125→export async function buscarStockRacksPorCodigo(codigo: string): Promise<StockRacksPorCodigoItem[]> {
   126→  const movs = await fetchMovimientosByCodigo(codigo)
   127→  const locMap = new Map<string, StockRacksPorCodigoItem>()
   128→
   129→  for (const m of movs) {
   130→    const key = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}||${m.fVencimiento || ''}||${m.codigoInc || ''}`
   131→    const current = locMap.get(key)
   132→    const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? m.cantidad : -m.cantidad
   133→    if (current) {
   134→      current.stock += delta
   135→    } else {
   136→      locMap.set(key, {
   137→        bloque: m.bloque,
   138→        torre: m.torre,
   139→        piso: m.piso,
   140→        posicion: m.posicion,
   141→        stock: delta,
   142→        descripcion: m.descripcion,
   143→        un: m.un,
   144→        fVencimiento: m.fVencimiento || '',
   145→        codigo_inc: m.codigoInc || undefined,
   146→      })
   147→    }
   148→  }
   149→
   150→  return Array.from(locMap.values())
   151→    .filter(l => l.stock > 0)
   152→    .sort((a, b) => {
   153→      // FEFO primero
   154→      if (a.fVencimiento && b.fVencimiento) return a.fVencimiento.localeCompare(b.fVencimiento)
   155→      if (a.fVencimiento && !b.fVencimiento) return -1
   156→      if (!a.fVencimiento && b.fVencimiento) return 1
   157→      return 0
   158→    })
   159→}

   161→/** Fallback: insert directo cuando la RPC no existe en Supabase */
   162→async function addMovimientoFallback(
   163→  m: Omit<Movimiento, 'id' | 'fModificacion'>,
   164→  uuidSync?: string,
   165→  skipValidation: boolean = false
   166→): Promise<Movimiento[]> {
   167→  console.warn('[addMovimiento] RPC no encontrada, usando insert directo como fallback')
   168→
   169→  // ── Validación de stock negativo para salidas NORMALES (no INC) ──
   170→  // Las salidas normales NUNCA deben generar stock negativo.
   171→  // Los traslados se manejan aparte con autoajuste.
   172→  // Stock = suma de TODOS los movimientos en la ubicacion para ese codigo.
   173→  // f_vencimiento es solo para FEFO, NO para calcular stock.
   174→  if (!skipValidation && m.tipo === 'salida' && !m.codigoInc) {
   175→    try {
   176→      // Stock real = movimientos en la ubicacion para ese codigo, EXCLUYENDO INC.
   177→      // f_vencimiento es solo para FEFO, NO para calcular stock.
   178→      const currentStock = await calcularStockUbicacion(
   179→        m.codigo, m.bloque, m.torre, m.piso, m.posicion, true // excluir INC
   180→      )
   181→      // Tolerancia de 0.001 para redondeo (display redondea a 3 decimales)
   182→      const stockRedondeado = Math.round(currentStock * 1000) / 1000
   183→      if (m.cantidad > stockRedondeado + 0.001) {
   184→        const err = new Error('INSUFFICIENT_STOCK')
   185→        ;(err as unknown as Record<string, string>).detail =
   186→          `Stock actual = ${stockRedondeado} ${m.un}, cantidad solicitada = ${m.cantidad} ${m.un}`
   187→        throw err
   188→      }
   189→    } catch (stockErr) {
   190→      // Si es error de stock insuficiente, propagarlo
   191→      if (stockErr instanceof Error && stockErr.message === 'INSUFFICIENT_STOCK') throw stockErr
   192→      // Si falla la consulta de stock, NO permitir la inserción sin validar.
   193→      // Esto previene salidas con stock negativo cuando la red es inestable.
   194→      console.error('[addMovimientoFallback] No se pudo verificar stock, BLOQUEANDO inserción:', stockErr)
   195→      throw new Error('STOCK_VALIDATION_FAILED|No se pudo verificar el stock. Reintente.')
   196→    }
   197→  }
   198→
   199→  const { error } = await dataClient.from('movimientos').insert({
   200→    tipo: m.tipo,
   201→    bloque: m.bloque,
   202→    torre: m.torre,
   203→    piso: m.piso,
   204→    posicion: m.posicion,
   205→    codigo: m.codigo.trim().toUpperCase(),
   206→    descripcion: m.descripcion,
   207→    un: m.un,
   208→    cantidad: m.cantidad,
   209→    f_vencimiento: m.fVencimiento || null,
   210→    turno: m.turno,
   211→    usuario_id: m.usuarioId,
   212→    usuario_nombre: m.usuarioNombre ?? null,
   213→    usuario_correo: m.usuarioCorreo ?? null,
   214→    proveedor: m.proveedor ? m.proveedor : null,
   215→    uuid_sync: uuidSync || null,
   216→    codigo_inc: m.codigoInc || null,
   217→  })
   218→  if (error) throw error
   219→  try {
   220→    return await fetchMovimientos()
   221→  } catch {
   222→    return []
   223→  }
   224→}

   226→/** Chequeo de idempotencia: verificar si un movimiento con este uuid_sync ya existe */
   227→async function checkExistingByUuidSync(uuidSync: string): Promise<boolean> {
   228→  const { data, error } = await dataClient
   229→    .from('movimientos')
   230→    .select('id')
   231→    .eq('uuid_sync', uuidSync)
   232→    .limit(1)
   233→  if (error) {
   234→    console.warn('[checkExistingByUuidSync] Error consultando:', error.message)
   235→    return false // Si falla la consulta, seguir con el insert normal

   237→  return (data ?? []).length > 0
   238→}

   240→export async function addMovimiento(
   241→  m: Omit<Movimiento, 'id' | 'fModificacion'>,
   242→  uuidSync?: string
   243→): Promise<Movimiento[]> {
   244→  // Idempotencia: si viene uuidSync, verificar si ya existe en el servidor
   245→  if (uuidSync) {
   246→    const exists = await checkExistingByUuidSync(uuidSync)
   247→    if (exists) {
   248→      console.log('[addMovimiento] Movimiento ya existe (uuid_sync):', uuidSync, '— saltando insert.')
   249→      // Ya existe: refrescar y retornar como si hubiera sido exitoso
   250→      try { return await fetchMovimientos() } catch { return [] }
   251→    }
   252→  }
   253→
   254→  // NOTA: Los movimientos INC ahora pasan por la RPC como cualquier otro movimiento.
   255→  // La RPC registrar_movimiento_kardex ya tiene p_codigo_inc (migration 20260611).
   256→  // Esto garantiza advisory lock y validación atómica para TODOS los movimientos.
   257→
   258→  // Usar RPC atómica con advisory lock para evitar race conditions.
   259→  // La RPC maneja TODOS los tipos de movimiento incluyendo INC.
   260→  // INC items son tipo 'ingreso' así que no pasan validación de stock negativo.
   261→  try {
   262→    const { data, error } = await dataClient.rpc('registrar_movimiento_kardex', {
   263→      p_tipo: m.tipo,
   264→      p_bloque: m.bloque,
   265→      p_torre: m.torre,
   266→      p_piso: m.piso,
   267→      p_posicion: m.posicion,
   268→      p_codigo: m.codigo.trim().toUpperCase(),
   269→      p_descripcion: m.descripcion,
   270→      p_un: m.un,
   271→      p_cantidad: m.cantidad,
   272→      p_f_vencimiento: m.fVencimiento || null,
   273→      p_turno: m.turno,
   274→      p_usuario_id: m.usuarioId,
   275→      p_usuario_nombre: m.usuarioNombre ?? null,
   276→      p_usuario_correo: m.usuarioCorreo ?? null,
   277→      p_proveedor: m.proveedor ? m.proveedor : null,
   278→      p_uuid_sync: uuidSync || null,
   279→      p_codigo_inc: m.codigoInc || null,
   280→    })
   281→    // Stock insuficiente es un error controlado — NUNCA bypassear la decisión del RPC.
   282→    // El RPC tiene advisory lock y calcula stock de forma atómica; es la fuente de verdad.
   283→    if (error) {
   284→      const msg = error.message || ''
   285→      const code = (error as unknown as Record<string, string>).code || ''
   286→      if (msg.includes('INSUFFICIENT_STOCK')) {
   287→        const parts = msg.split('|')
   288→        const rpcDetail = parts.length > 1 ? parts[1] : 'Stock insuficiente para esta operación'
   289→        const err = new Error('INSUFFICIENT_STOCK')
   290→        ;(err as unknown as Record<string, string>).detail = rpcDetail
   291→        throw err
   292→      }
   293→      // Si la RPC no existe (404 / 42883 / 'Could not find'), usar fallback
   294→      if (code === '42883' || code === 'PGRST202' || msg.includes('Could not find') || msg.includes('does not exist') || msg.includes('404')) {
   295→        return await addMovimientoFallback(m, uuidSync)
   296→      }
   297→      throw error
   298→    }
   299→    // RPC exitosa: el movimiento ya fue registrado en la DB.
   300→    // Intentar refrescar la lista de movimientos, pero si falla (timeout, red)
   301→    // NO propagar el error — el movimiento ya está guardado.
   302→    try {
   303→      return await fetchMovimientos()
   304→    } catch {
   305→      console.warn('[addMovimiento] RPC exitosa pero fetchMovimientos() falló. Movimiento ya registrado.')
   306→      return []
   307→    }
   308→  } catch (err: unknown) {
   309→    // Si la RPC no existe aún (SQL no ejecutado), fallback al insert directo
   310→    const errMsg = err instanceof Error ? err.message : ''
   311→    const errCode = err instanceof Error ? (err as unknown as Record<string, string>).code || '' : ''
   312→    if (errCode === '42883' || errCode === 'PGRST202' || errMsg.includes('Could not find') || errMsg.includes('does not exist')) {
   313→      return await addMovimientoFallback(m, uuidSync)
   314→    }
   315→    throw err
   316→  }
   317→}

   325→export async function calcularStockUbicacion(
   326→  codigo: string,
   327→  bloque: string,
   328→  torre: string,
   329→  piso: string,
   330→  posicion: string,
   331→  excluirInc: boolean = false
   332→): Promise<number> {
   333→  const target = codigo.trim().toUpperCase()
   334→  let query = dataClient
   335→    .from('movimientos')
   336→    .select('tipo, cantidad')
   337→    .eq('codigo', target)
   338→    .eq('bloque', bloque)
   339→    .eq('torre', torre)
   340→    .eq('piso', piso)
   341→    .eq('posicion', posicion)
   342→  // Excluir movimientos INC para validar stock normal
   343→  if (excluirInc) {
   344→    query = query.is('codigo_inc', null)
   345→  }
   346→  const { data, error } = await query
   347→  if (error) throw error
   348→  return (data ?? []).reduce(
   349→    (s: number, r: { tipo: string; cantidad: unknown }) => {
   350→      const qty = typeof r.cantidad === 'number' ? r.cantidad : parseFloat(String(r.cantidad ?? '0')) || 0
   351→      return s + impactoStock(r.tipo, qty)
   352→    },
   353→    0
   354→  )
   355→}

   357→export type StockEnUbicacion = {
   358→  codigo: string
   359→  descripcion: string
   360→  un: string

   368→export async function stockEnUbicacion(
   369→  bloque: string,
   370→  torre: string,
   371→  piso: string,
   372→  posicion: string
   373→): Promise<StockEnUbicacion[]> {
   374→  try {
   375→    // Método principal: consultar movimientos directamente con paginación
   376→    // f_vencimiento es SOLO para FEFO (ordenamiento), NO para particionar stock.
   377→    // Se agrupa por (codigo, codigo_inc) — el stock es la suma de TODOS los lotes.
   378→    const allRows: Record<string, unknown>[] = []
   379→    let from = 0
   380→    const BATCH = 1000
   381→    const MAX_PAGES = 10 // máximo 10,000 movimientos por ubicación
   382→
   383→    for (let page = 0; page < MAX_PAGES; page++) {
   384→      const { data, error } = await dataClient
   385→        .from('movimientos')
   386→        .select('*')
   387→        .eq('bloque', bloque)
   388→        .eq('torre', torre)
   389→        .eq('piso', piso)
   390→        .eq('posicion', posicion)
   391→        .order('f_modificacion', { ascending: true })
   392→        .range(from, from + BATCH - 1)
   393→
   394→      if (error) throw error
   395→      const rows = data ?? []
   396→      allRows.push(...rows)
   397→      if (rows.length < BATCH) break
   398→      from += BATCH
   399→    }
   400→
   401→    // Agrupar por (codigo, codigo_inc) — f_vencimiento NO participa en la agrupación.
   402→    // Se rastrea la fecha de vencimiento más próxima (FEFO) para cada grupo.
   403→    const groups = new Map<string, {
   404→      codigo: string; descripcion: string; un: string;
   405→      stock: number; fVencimientoMasProxima: string;
   406→      usuarioPrimerNombre: string; proveedor: string; codigoInc: string;
   407→    }>()
   408→
   409→    for (const r of allRows) {
   410→      const m = fromRow(r)
   411→      const incKey = m.codigoInc || ''
   412→      const key = `${m.codigo}||${incKey}`
   413→
   414→      let group = groups.get(key)
   415→      if (!group) {
   416→        group = {
   417→          codigo: m.codigo,
   418→          descripcion: m.descripcion,
   419→          un: m.un,
   420→          stock: 0,
   421→          fVencimientoMasProxima: m.fVencimiento || '',
   422→          usuarioPrimerNombre: m.usuarioNombre?.split(' ')[0] ?? '',
   423→          proveedor: m.proveedor ?? '',
   424→          codigoInc: incKey,
   425→        }
   426→        groups.set(key, group)
   427→      } else {
   428→        // Si el grupo no tiene descripción pero este movimiento sí, usarla
   429→        if (!group.descripcion && m.descripcion) group.descripcion = m.descripcion
   430→        // Rastrear la fecha de vencimiento más próxima (para FEFO)
   431→        if (m.fVencimiento) {
   432→          if (!group.fVencimientoMasProxima || m.fVencimiento < group.fVencimientoMasProxima) {
   433→            group.fVencimientoMasProxima = m.fVencimiento
   434→          }
   435→        }
   436→      }
   437→
   438→      // Calcular stock neto (ingreso/devolucion/traslado = +, salida = -)
   439→      const qty = typeof m.cantidad === 'number' ? m.cantidad : parseFloat(String(m.cantidad)) || 0
   440→      const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? qty : -qty
   441→      group.stock += delta
   442→    }
   443→
   444→    // Filtrar solo artículos con stock > 0
   445→    const results = Array.from(groups.values()).filter(g => g.stock > 0)
   446→
   447→    // Ordenar por FEFO: fecha de vencimiento más próxima primero, sin fecha al final
   448→    results.sort((a, b) => {
   449→      if (a.fVencimientoMasProxima && b.fVencimientoMasProxima) return a.fVencimientoMasProxima.localeCompare(b.fVencimientoMasProxima)
   450→      if (a.fVencimientoMasProxima && !b.fVencimientoMasProxima) return -1
   451→      if (!a.fVencimientoMasProxima && b.fVencimientoMasProxima) return 1
   452→      return 0
   453→    })
   454→
   455→    return results.map(g => ({
   456→      codigo: g.codigo,
   457→      descripcion: g.descripcion,
   458→      un: g.un,
   459→      stock: Math.round(g.stock * 1000) / 1000,
   460→      fVencimiento: g.fVencimientoMasProxima || undefined,
   461→      usuarioPrimerNombre: g.usuarioPrimerNombre || undefined,
   462→      proveedor: g.proveedor || undefined,
   463→      codigoInc: g.codigoInc || undefined,
   464→    }))
   465→  } catch {
   466→    // Fallback al RPC original si falla la consulta directa
   467→    try {
   468→      const { data, error } = await dataClient.rpc('stock_en_ubicacion', {
   469→        _bloque: bloque,
   470→        _torre: torre,
   471→        _piso: piso,
   472→        _posicion: posicion,
   473→      })
   474→      if (error) throw error
   475→      return ((data ?? []) as Record<string, unknown>[]).map((r) => ({
   476→        codigo: String(r.codigo ?? '').trim().toUpperCase(),
   477→        descripcion: (r.descripcion as string) ?? '',
   478→        un: (r.un as string) ?? '',
   479→        stock: Number(r.stock ?? 0),
   480→        fVencimiento: (r.f_vencimiento as string) ?? '',
   481→        usuarioPrimerNombre: (r.usuario_primer_nombre as string) ?? '',
   482→        proveedor: (r.proveedor as string) ?? '',
   483→        codigoInc: (r.codigo_inc as string) ?? undefined,
   484→      }))
   485→    } catch {
   486→      return []
   487→    }
   488→  }
   489→}

   491→/** Consulta dedicada: ubicaciones que tienen INC con stock > 0 */
   492→export async function fetchIncPorUbicacion(): Promise<Map<string, IncEnCelda[]>> {
   493→  try {
   494→    const allRows: Record<string, unknown>[] = []
   495→    let from = 0
   496→    const BATCH = 1000
   497→    for (let page = 0; page < 10; page++) {
   498→      const { data, error } = await dataClient
   499→        .from('movimientos')
   500→        .select('bloque, torre, piso, posicion, codigo, descripcion, un, codigo_inc, tipo, cantidad')
   501→        .not('codigo_inc', 'is', null)
   502→        .neq('codigo_inc', '')
   503→        .range(from, from + BATCH - 1)
   504→      if (error) return new Map()
   505→      const rows = data ?? []
   506→      allRows.push(...rows)
   507→      if (rows.length < BATCH) break
   508→      from += BATCH
   509→    }
   510→    // Calcular stock neto por ubicación + código INC
   511→    const map = new Map<string, Map<string, { codigo: string; descripcion: string; codigoInc: string; stock: number }>>()
   512→    for (const r of allRows) {
   513→      const key = `${r.bloque}-${r.torre}-${r.piso}-${r.posicion}`
   514→      const code = String(r.codigo ?? '').trim().toUpperCase()
   515→      const codeInc = String(r.codigo_inc ?? '').trim()
   516→      const incKey = `${code}||${codeInc}`
   517→      let locMap = map.get(key)
   518→      if (!locMap) { locMap = new Map(); map.set(key, locMap) }
   519→      const qty = typeof r.cantidad === 'number' ? r.cantidad : parseFloat(String(r.cantidad ?? '0')) || 0
   520→      const delta = ['ingreso', 'devolucion', 'traslado'].includes(String(r.tipo)) ? qty : -qty
   521→      const item = locMap.get(incKey)
   522→      if (item) { item.stock += delta } else {
   523→        locMap.set(incKey, { codigo: code, descripcion: String(r.descripcion ?? ''), codigoInc: codeInc, stock: delta })
   524→      }
   525→    }
   526→    const result = new Map<string, IncEnCelda[]>()
   527→    for (const [key, locMap] of map) {
   528→      const items: IncEnCelda[] = []
   529→      for (const [, item] of locMap) {
   530→        if (item.stock > 0) items.push({ codigo: item.codigo, descripcion: item.descripcion, codigoInc: item.codigoInc, cantidad: item.stock })
   531→      }
   532→      if (items.length > 0) result.set(key, items)
   533→    }
   534→    return result
   535→  } catch { return new Map() }
   536→}

   631→export async function trasladarMovimiento(t: TrasladoInput): Promise<Movimiento[]> {
   632→  // Usar RPC atómica con advisory locks en origen Y destino
   633→  try {
   634→    const { data, error } = await dataClient.rpc('registrar_traslado_kardex', {
   635→      p_codigo: t.codigo,
   636→      p_descripcion: t.descripcion,
   637→      p_un: t.un,
   638→      p_cantidad: t.cantidad,
   639→      p_orig_bloque: t.origen.bloque,
   640→      p_orig_torre: t.origen.torre,
   641→      p_orig_piso: t.origen.piso,
   642→      p_orig_pos: t.origen.posicion,
   643→      p_dest_bloque: t.destino.bloque,
   644→      p_dest_torre: t.destino.torre,
   645→      p_dest_piso: t.destino.piso,
   646→      p_dest_pos: t.destino.posicion,
   647→      p_turno: t.turno,
   648→      p_usuario_id: t.usuarioId,
   649→      p_usuario_nombre: t.usuarioNombre ?? null,
   650→      p_usuario_correo: t.usuarioCorreo ?? null,
   651→      p_f_vencimiento: t.fVencimiento || null,
   652→      p_proveedor: t.proveedor ? t.proveedor : null,
   653→      p_cantidad_ajuste: t.cantidadAjuste ?? 0,
   654→      p_codigo_inc: t.codigoInc || null,
   655→      p_uuid_sync: t.uuidSync || null,
   656→    })
   657→    // Stock insuficiente en origen — NUNCA bypassear la decisión del RPC.
   658→    // El RPC tiene advisory locks en origen Y destino; es la fuente de verdad.
   659→    if (error) {
   660→      const msg = error.message || ''
   661→      const code = (error as unknown as Record<string, string>).code || ''
   662→      if (msg.includes('INSUFFICIENT_STOCK')) {
   663→        const parts = msg.split('|')
   664→        const rpcDetail = parts.length > 1 ? parts[1] : 'Stock insuficiente en origen para este traslado'
   665→        const err = new Error('INSUFFICIENT_STOCK')
   666→        ;(err as unknown as Record<string, string>).detail = rpcDetail
   667→        throw err
   668→      }
   669→      // Si la RPC no existe (404 / 42883 / 'Could not find'), usar fallback
   670→      if (code === '42883' || code === 'PGRST202' || msg.includes('Could not find') || msg.includes('does not exist') || msg.includes('404')) {
   671→        return await trasladarMovimientoFallback(t)
   672→      }
   673→      throw error
   674→    }
   675→    // RPC exitosa: refrescar movimientos, pero no fallar si fetchMovimientos falla
   676→    try {
   677→      return await fetchMovimientos()
   678→    } catch {
   679→      console.warn('[trasladarMovimiento] RPC exitosa pero fetchMovimientos() falló. Traslado ya registrado.')
   680→      return []
   681→    }
   682→  } catch (err: unknown) {
   683→    // Fallback si la RPC no existe aún
   684→    const errMsg = err instanceof Error ? err.message : ''
   685→    const errCode = err instanceof Error ? (err as unknown as Record<string, string>).code || '' : ''
   686→    if (errCode === '42883' || errCode === 'PGRST202' || errMsg.includes('Could not find') || errMsg.includes('does not exist')) {
   687→      return await trasladarMovimientoFallback(t)
   688→    }
   689→    throw err
   690→  }
   691→}

   693→/**
   694→ * Elimina TODOS los movimientos de la tabla.
   695→ * Usa service_role para bypassear RLS.
   696→ * Borra en lotes de 1000 IDs para no exceder límites de URL.
   697→ */
   698→export async function deleteAllMovimientos(): Promise<{ deleted: boolean; error?: string }> {
   699→  // Usar dataClient (service role) ya configurado — sin credenciales hardcodeadas
   700→  const admin = dataClient
   701→
   702→  // Obtener todos los IDs
   703→  const allIds: string[] = []
   704→  let from = 0
   705→  let iterations = 0
   706→  const BATCH = 1000
   707→  while (iterations++ < MAX_ITERATIONS) {
   708→    const { data, error } = await admin
   709→      .from('movimientos')
   710→      .select('id')
   711→      .range(from, from + BATCH - 1)
   712→    if (error) return { deleted: false, error: error.message }
   713→    if (!data || data.length === 0) break
   714→    allIds.push(...data.map((r: { id: string }) => r.id))
   715→    if (data.length < BATCH) break
   716→    from += BATCH
   717→  }
   718→
   719→  if (allIds.length === 0) return { deleted: true }
   720→
   721→  // Borrar en lotes usando .in('id', [...])
   722→  const DELETE_BATCH = 500
   723→  for (let i = 0; i < allIds.length; i += DELETE_BATCH) {
   724→    const batch = allIds.slice(i, i + DELETE_BATCH)
   725→    const { error } = await admin
   726→      .from('movimientos')
   727→      .delete()
   728→      .in('id', batch)
   729→    if (error) return { deleted: false, error: `Lote ${Math.floor(i / DELETE_BATCH) + 1}: ${error.message}` }
   730→  }
   731→
   732→  return { deleted: true }
   733→}

   750→ * Usa service_role para bypassear RLS.
   751→ * Usa batch inserts de 1000 filas para no exceder límites de Supabase.
   752→ */
   753→export async function addMovimientosBatch(
   754→  rows: UploadStockRow[],
   755→  usuarioId: string,
   756→  usuarioNombre?: string,
   757→  usuarioCorreo?: string,
   758→): Promise<{ inserted: number; errors: string[] }> {
   759→  // Usar dataClient (service role) ya configurado — sin credenciales hardcodeadas
   760→  const admin = dataClient
   761→
   762→  const turno = (await import('./turno')).calcularTurno()
   763→  const errors: string[] = []
   764→  let inserted = 0
   765→  const BATCH_SIZE = 1000
   766→
   767→  for (let i = 0; i < rows.length; i += BATCH_SIZE) {
   768→    const batch = rows.slice(i, i + BATCH_SIZE)
   769→    const inserts = batch.map((r) => ({
   770→      tipo: 'ingreso' as const,
   771→      bloque: String(r.bloque),
   772→      torre: String(r.torre),
   773→      piso: String(r.piso),
   774→      posicion: String(r.posicion),
   775→      codigo: r.codigo.trim().toUpperCase(),
   776→      descripcion: r.descripcion,
   777→      un: r.un || 'KG',
   778→      cantidad: Number(r.cantidad),
   779→      f_vencimiento: r.fVencimiento || null,
   780→      turno,
   781→      usuario_id: usuarioId,
   782→      usuario_nombre: usuarioNombre ?? null,
   783→      usuario_correo: usuarioCorreo ?? null,
   784→      proveedor: r.proveedor ? r.proveedor : null,
   785→      codigo_inc: null,
   786→    }))
   787→
   788→    const { error } = await admin.from('movimientos').insert(inserts)
   789→    if (error) {
   790→      errors.push(`Lote ${Math.floor(i / BATCH_SIZE) + 1}: ${error.message}`)
   791→      // Intentar insertar uno por uno el lote fallido para identificar la fila
   792→      for (let j = 0; j < inserts.length; j++) {
   793→        const { error: singleErr } = await admin
   794→          .from('movimientos')
   795→          .insert(inserts[j])
   796→        if (singleErr) {
   797→          errors.push(`Fila ${i + j + 1}: ${singleErr.message}`)
   798→        } else {
   799→          inserted++
   800→        }
   801→      }
   802→    } else {
   803→      inserted += batch.length
   804→    }
   805→  }
   806→
   807→  return { inserted, errors }
   808→}
