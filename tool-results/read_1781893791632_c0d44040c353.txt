    67→    67→export function MovimientoForm({ tipo, onCreated }: Props) {
    68→    68→  const { perfil } = useAuth()
    69→    69→  const [turno, setTurno] = useState<Turno>(calcularTurno())
    70→    70→
    71→    71→  useEffect(() => {
    72→    72→    const interval = setInterval(() => setTurno(calcularTurno()), 60000)
    73→    73→    return () => clearInterval(interval)
    74→    74→  }, [])
    75→    75→
    76→    76→  if (!perfil) {
    77→    77→    return <div className='p-4 text-muted-foreground animate-pulse'>Cargando usuario...</div>
    78→    78→  }
    79→    79→  if (tipo === 'salida') {
    80→    80→    return <SalidaForm turno={turno} onCreated={onCreated} perfil={perfil} />
    81→    81→  }
    82→    82→  if (tipo === 'salida-inc') {
    83→    83→    return <SalidaIncForm turno={turno} onCreated={onCreated} perfil={perfil} />
    84→    84→  }
    85→    85→  if (tipo === 'inc') {
    86→    86→    return <IncForm turno={turno} onCreated={onCreated} perfil={perfil} />
    87→    87→  }
    88→    88→  // ingreso y devolucion usan el mismo formulario, solo cambia el tipo
    89→    89→  return <IngresoForm turno={turno} onCreated={onCreated} perfil={perfil} tipo={tipo} />
    90→    90→}

    95→    95→function IngresoForm({
    96→    96→  turno,
    97→    97→  onCreated,
    98→    98→  perfil,
    99→    99→  tipo = 'ingreso',
   100→   100→}: {
   101→   101→  turno: Turno
   102→   102→  onCreated: (movs: Movimiento[]) => void
   103→   103→  perfil: { id: string; nombre: string; correo: string }
   104→   104→  tipo?: TipoMovimiento
   105→   105→}) {
   106→   106→  const [bloque, setBloque] = useState('')
   107→   107→  const [torre, setTorre] = useState('')
   108→   108→  const [piso, setPiso] = useState('')
   109→   109→  const [posicion, setPosicion] = useState('')
   110→   110→  const [codigo, setCodigo] = useState('')
   111→   111→  const [descripcion, setDescripcion] = useState('')
   112→   112→  const [un, setUn] = useState('')
   113→   113→  const [cantidad, setCantidad] = useState('')
   114→   114→  const [fVencimiento, setFVencimiento] = useState('')
   115→   115→  const [sinVencimiento, setSinVencimiento] = useState(false)
   116→   116→  const [proveedor, setProveedor] = useState('')
   117→   117→  const [busy, setBusy] = useState(false)
   118→   118→  const [confirmData, setConfirmData] = useState<StockEnUbicacion[] | null>(null)
   119→   119→  const [salidaBusy, setSalidaBusy] = useState<string | null>(null)
   120→   120→
   121→   121→  const torres = torresDeBloque(bloque)
   122→   122→  const posiciones = posicionesDeBloque(bloque)
   123→   123→
   124→   124→  function handleCatalogoPick(item: CatalogoItem) {
   125→   125→    setCodigo(item.codigo)
   126→   126→    setDescripcion(item.descripcion)
   127→   127→    setUn(item.un)
   128→   128→  }
   129→   129→
   130→   130→  async function handleSubmit(e: React.FormEvent) {
   131→   131→    e.preventDefault()
   132→   132→    if (!bloque || !torre || !piso || !posicion || !codigo.trim() || !cantidad) {
   133→   133→      toast.error('Completa todos los campos requeridos')
   134→   134→      return
   135→   135→    }
   136→   136→    if (requiereProveedor(descripcion) && !proveedor) {
   137→   137→      toast.error('Selecciona un proveedor para este producto')
   138→   138→      return
   139→   139→    }
   140→   140→    const qty = parseFloat(cantidad)
   141→   141→    if (isNaN(qty) || qty <= 0) {
   142→   142→      toast.error('La cantidad debe ser mayor a 0')
   143→   143→      return
   144→   144→    }
   145→   145→    setBusy(true)
   146→   146→    try {
   147→   147→      // Verificar si CUALQUIER producto existe en esta posición
   148→   148→      const details = await stockEnUbicacion(bloque, torre, piso, posicion)
   149→   149→      if (details.length > 0) {
   150→   150→        setConfirmData(details)
   151→   151→        setBusy(false)
   152→   152→        return
   153→   153→      }
   154→   154→      await doInsert(qty)
   155→   155→    } catch (err: unknown) {
   156→   156→      const message = extractError(err)
   157→   157→      toast.error('Error al verificar stock', { description: message })
   158→   158→      setBusy(false)
   159→   159→    }
   160→   160→  }
   161→   161→
   162→   162→  async function doInsert(qty: number) {
   163→   163→    try {
   164→   164→      const { movs, wasOffline } = await SyncEngine.offlineAwareAddMovimiento({
   165→   165→        tipo,
   166→   166→        bloque,
   167→   167→        torre,
   168→   168→        piso,
   169→   169→        posicion,
   170→   170→        codigo,
   171→   171→        descripcion,
   172→   172→        un,
   173→   173→        cantidad: qty,
   174→   174→        fVencimiento: sinVencimiento ? '' : fVencimiento,
   175→   175→        turno,
   176→   176→        usuarioId: perfil.id,
   177→   177→        usuarioNombre: perfil.nombre,
   178→   178→        usuarioCorreo: perfil.correo,
   179→   179→        proveedor: proveedor || undefined,
   180→   180→      })
   181→   181→      if (wasOffline) {
   182→   182→        toast.success(tipo === 'devolucion' ? 'Devolución guardada (offline)' : 'Ingreso guardado (offline)', {
   183→   183→          description: 'Se sincronizará al reconectarse',
   184→   184→          duration: 5000,
   185→   185→        })
   186→   186→        // OFFLINE: Limpiar formulario para evitar duplicados
   187→   187→        setCodigo('')
   188→   188→        setDescripcion('')
   189→   189→        setUn('')
   190→   190→        setCantidad('')
   191→   191→        setFVencimiento('')
   192→   192→        setProveedor('')
   193→   193→        setConfirmData(null)
   194→   194→      } else {
   195→   195→        toast.success(tipo === 'devolucion' ? 'Devolución registrada' : 'Ingreso registrado')
   196→   196→        setCodigo('')
   197→   197→        setDescripcion('')
   198→   198→        setUn('')
   199→   199→        setCantidad('')
   200→   200→        setFVencimiento('')
   201→   201→        setProveedor('')
   202→   202→        onCreated(movs)
   203→   203→      }
   204→   204→    } catch (err: unknown) {
   205→   205→      const message = extractError(err)
   206→   206→      toast.error('Error al registrar ingreso', { description: message })
   207→   207→    } finally {
   208→   208→      setBusy(false)
   209→   209→      setConfirmData(null)
   210→   210→    }
   211→   211→  }

   213→   213→  // Dar salida a un producto desde el alerta de ubicación ocupada

   218→   218→        tipo: 'salida',
   219→   219→        bloque,
   220→   220→        torre,
   221→   221→        piso,

   227→   227→        fVencimiento: stockItem.fVencimiento ?? '',
   228→   228→        turno,
   229→   229→        usuarioId: perfil.id,
   230→   230→        usuarioNombre: perfil.nombre,
   231→   231→        usuarioCorreo: perfil.correo,
   232→   232→        proveedor: stockItem.proveedor,
   233→   233→        // PRESERVAR codigoInc para que la salida descuente del stock INC correctamente
   234→   234→        codigoInc: stockItem.codigoInc || undefined,
   235→   235→      })
   236→   236→      if (wasOffline) {
   237→   237→        toast.success(`Salida de ${stockItem.stock} ${stockItem.un} de ${stockItem.codigo} (offline)`, {
   238→   238→          description: 'Se sincronizará al reconectarse. La ubicación se actualizará cuando haya conexión.',
   239→   239→          duration: 5000,
   240→   240→        })
   241→   241→        // OFFLINE: Limpiar UI para evitar doble operación
   242→   242→        setConfirmData(null)
   243→   243→        setSalidaBusy(null)
   244→   244→      } else {
   245→   245→        toast.success(`Salida de ${stockItem.stock} ${stockItem.un} de ${stockItem.codigo}`)
   246→   246→        onCreated(movs)
   247→   247→        // Refrescar datos del alerta
   248→   248→        const updated = await stockEnUbicacion(bloque, torre, piso, posicion)
   249→   249→        if (updated.length > 0) {
   250→   250→          setConfirmData(updated)

   261→   261→          description: detail || 'Otro usuario pudo haber modificado el stock mientras tú operabas.',
   262→   262→          duration: 8000,
   263→   263→        })
   264→   264→      } else {
   265→   265→        const message = extractError(err)
   266→   266→        toast.error('Error al dar salida', { description: message })
   267→   267→      }
   268→   268→    } finally {
   269→   269→      setSalidaBusy(null)
   270→   270→    }
   271→   271→  }
   272→   272→
   273→   273→  return (
   274→   274→    <>
   275→   275→      <div className={`flex items-center gap-2 mb-4 p-3 rounded-lg border ${
   276→   276→        tipo === 'devolucion'
   277→   277→          ? 'bg-orange-50 dark:bg-orange-950/40 border-orange-200 dark:border-orange-800'
   278→   278→          : 'bg-green-50 dark:bg-green-950/40 border-green-200 dark:border-green-800'

   280→   280→        {tipo === 'devolucion' ? (

   282→   282→        ) : (

   285→   285→        <div className="min-w-0">
   286→   286→          <p className={`text-sm font-semibold ${
   287→   287→            tipo === 'devolucion'
   288→   288→              ? 'text-orange-700 dark:text-orange-300'

   290→   290→          }`}>
   291→   291→            {tipo === 'devolucion' ? 'Ingreso por devolución' : 'Ingreso de mercadería'}
   292→   292→          </p>
   293→   293→          <p className="text-xs text-muted-foreground">Turno: {turno}</p>
   294→   294→        </div>
   295→   295→      </div>
   296→   296→
   297→   297→      <form onSubmit={handleSubmit} className="space-y-4">
   298→   298→        <div className="space-y-1">
   299→   299→          <Label className="text-xs font-medium text-muted-foreground uppercase tracking-wider">Ubicación</Label>
   300→   300→          <div className="grid grid-cols-2 sm:grid-cols-5 gap-2">
   301→   301→            <div className="space-y-1">
   302→   302→              <Label className="text-xs text-muted-foreground">Bloque</Label>
   303→   303→              <Select value={bloque} onValueChange={(v) => { setBloque(v); setTorre(''); setPiso(''); setPosicion('') }}>
   304→   304→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Bloque" /></SelectTrigger>
   305→   305→                <SelectContent>
   306→   306→                  {BLOQUES.map((b) => <SelectItem key={b} value={b}>{b}</SelectItem>)}
   307→   307→                </SelectContent>
   308→   308→              </Select>
   309→   309→            </div>
   310→   310→            <div className="space-y-1">
   311→   311→              <Label className="text-xs text-muted-foreground">Torre</Label>

   314→   314→                <SelectContent>
   315→   315→                  {torres.map((t) => <SelectItem key={t} value={t}>{t}</SelectItem>)}
   316→   316→                </SelectContent>
   317→   317→              </Select>
   318→   318→            </div>
   319→   319→            <div className="space-y-1">
   320→   320→              <Label className="text-xs text-muted-foreground">Piso</Label>
   321→   321→              <Select value={piso} onValueChange={setPiso} disabled={!bloque}>
   322→   322→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Piso" /></SelectTrigger>

   324→   324→                  {PISOS.map((p) => <SelectItem key={p} value={p}>{p}</SelectItem>)}
   325→   325→                </SelectContent>
   326→   326→              </Select>

   358→   358→        </div>
   359→   359→
   360→   360→        {descripcion && (

   362→   362→            <div className="flex items-start gap-3 flex-wrap">
   363→   363→              <div className="min-w-0 flex-1">

   394→   394→                <SelectContent>

   402→   402→        </div>
   403→   403→
   404→   404→        <Button type="submit" disabled={busy} className={`w-full h-11 text-white text-sm font-medium sm:w-auto ${
   405→   405→          tipo === 'devolucion'

   417→   417→          {/* Header con gradiente */}

   683→   683→        const desc = totalProcessed > 0
   684→   684→          ? `${totalProcessed} de ${totalToProcess} correctas${stockNote}:\n${errorDetails.slice(0, 5).join('\n')}${errorDetails.length > 5 ? `\n...y ${errorDetails.length - 5} más` : ''}`
   685→   685→          : `Errores${stockNote}:\n${errorDetails.join('\n')}`
   686→   686→        toast.warning(`Salida completada con ${totalErrors} error${totalErrors > 1 ? 'es' : ''}`, { description: desc, duration: 10000 })

   755→   755→            }
   756→   756→          }

   761→   761→            piso: m.piso,

   770→   770→        }

   784→   784→        const aT = parseInt(a.torre, 10) || 0

   786→   786→        if (aT !== bT) return aT - bT
   787→   787→        const aP = parseInt(a.piso, 10) || 0
