    73→//  ANIMATED COUNTER HOOK
    74→// ═══════════════════════════════════════════════
    75→function useAnimatedCounter(target: number, duration = 800) {
    76→  const [count, setCount] = useState(0)
    77→  const prevTarget = useRef(0)
    78→
    79→  useEffect(() => {
    80→    if (target === prevTarget.current) return
    81→    const start = prevTarget.current
    82→    const diff = target - start
    83→    if (diff === 0) return
    84→    const startTime = performance.now()
    85→    let raf: number
    86→
    87→    function tick(now: number) {
    88→      const elapsed = now - startTime
    89→      const progress = Math.min(elapsed / duration, 1)
    90→      const eased = 1 - Math.pow(1 - progress, 3) // ease-out cubic
    91→      setCount(Math.round(start + diff * eased))
    92→      if (progress < 1) {
    93→        raf = requestAnimationFrame(tick)
    94→      }
    95→    }
    96→
    97→    raf = requestAnimationFrame(tick)
    98→    prevTarget.current = target
    99→    return () => cancelAnimationFrame(raf)
   100→  }, [target, duration])
   101→
   102→  return count
   103→}

   648→  async function doIngresoINC() {

   651→      toast.error('Completa codigo, cantidad y codigo INC')

   656→    if (!selectedNivelId) { toast.error('Selecciona un nivel primero'); return }

   664→      if (existing) {

   666→      } else {
   667→        const created = await crearBloque(upper, incDescripcion || '', incUn || 'KG')
   668→        if (created.length > 0) {
   669→          const found = created.find((b) => b.codigo === upper)
   670→          if (found) bloqueId = found.id
   671→        }

   673→      if (!bloqueId) { toast.error('No se pudo crear/encontrar el articulo'); setBusy(false); return }
   674→

   676→      const detalles = [{
   677→        nivel_id: nivelId,
   678→        bloque_id: bloqueId,
   679→        cantidad: qty,
   680→        fecha_vencimiento: incSinVencimiento ? '' : incFechaVencimiento,
   681→      }]
   682→      await registrarIngresoPosicion(
   683→        calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '',

   685→        { posicion_id: detail.posicionId, codigo_inc: incCodigoInc.trim() }

   689→      setMode('view')
   690→      setIncCodigo('')
   691→      setIncDescripcion('')
   692→      setIncUn('')
   693→      setIncCantidad('')
   694→      setIncCodigoInc('')
   695→      setIncFechaVencimiento('')
   696→      setIncSinVencimiento(false)

   699→        stockDetallePosicion(detail.posicionId),
   700→        niveles.length > 0 ? Promise.all(niveles.map((n) => stockDetalleNivel(n.id))) : Promise.resolve([]),
   701→      ])

   714→      if (selectedColumn) loadColumnDetail(selectedColumn)

   716→      toast.error('Error al registrar INC', { description: extractError(err) })

   719→    }

   777→  // Auto-create manual_ bloques in piso_bloques before registering movement
   778→  async function ensureManualBloqueCreated(rows: RowEntry[]): Promise<RowEntry[]> {
   779→    const resolved: RowEntry[] = []
   780→    for (const r of rows) {
   781→      if (r.bloque_id.startsWith('manual_')) {
   782→        const code = r.bloque_id.replace('manual_', '')
   783→        try {
   784→          const created = await crearBloque(code, r.descripcion || '', r.unidad || 'KG')
   785→          if (created.length > 0) {
   786→            // Find the newly created one
   787→            const found = created.find((b) => b.codigo === code)
   788→            if (found) {
   789→              resolved.push({ ...r, bloque_id: found.id })
   790→              continue
   791→            }
   792→          }
   793→        } catch {
   794→          // If creation fails, try searching for it (maybe it was already created)
   795→        }
   796→        // Last resort: try buscarBloquePorCodigo
   797→        const existing = await buscarBloquePorCodigo(code)
   798→        if (existing) {
   799→          resolved.push({ ...r, bloque_id: existing.id })
   800→        } else {
   801→          resolved.push(r)
   802→        }
   803→      } else {
   804→        resolved.push(r)
   805→      }
   806→    }
   807→    return resolved
   808→  }

   850→  async function doIngreso() {
   851→    if (!detail || !perfil) return
   852→    if (!selectedNivelId) { toast.error('No hay nivel seleccionado'); return }
   853→    const nivelId = selectedNivelId
   854→    const validRows = ingRows.filter((r) => r.bloque_id && r.cantidad)
   855→    if (validRows.length === 0) { toast.error('Agrega al menos un articulo con codigo y cantidad'); return }
   856→    for (const r of validRows) {
   857→      if (parseFloat(r.cantidad) <= 0 || isNaN(parseFloat(r.cantidad))) { toast.error('Cantidad invalida'); return }
   858→    }
   859→    setBusy(true)
   860→    try {
   861→      // Resolve manual_ IDs before registering
   862→      const resolved = await ensureManualBloqueCreated(validRows)
   863→      const detalles = resolved.map((r) => ({
   864→        nivel_id: nivelId,
   865→        bloque_id: r.bloque_id,
   866→        cantidad: parseFloat(r.cantidad),
   867→        fecha_vencimiento: r.sin_vencimiento ? '' : r.fecha_vencimiento,
   868→      }))
   869→      await registrarIngresoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', detalles)
   870→      toast.success('Ingreso registrado')
   871→      // Reload everything in parallel for real-time update
   872→      loadBloques()
   873→      const [stock, reloadedPerNivel] = await Promise.all([
   874→        stockDetallePosicion(detail.posicionId),
   875→        (async () => {
   876→          const perNivel: Record<string, DetailStock[]> = {}
   877→          for (const n of niveles) {
   878→            perNivel[n.id] = await stockDetalleNivel(n.id)
   879→          }
   880→          return perNivel
   881→        })(),
   882→      ])
   883→      if (mountedRef.current && detail) {
   884→        // Mantener filtro de nivel si estaba activo
   885→        const currentTab = viewNivelTab
   886→        const filteredStock = currentTab !== 'all' ? stock.filter(s => {
   887→          const nivelStock = reloadedPerNivel[currentTab] ?? []
   888→          return nivelStock.some(ns => ns.bloque_id === s.bloque_id && ns.fecha_vencimiento === s.fecha_vencimiento)
   889→        }) : stock
   890→        setDetail({ ...detail, stock: filteredStock })
   891→        setStockByNivel(reloadedPerNivel)
   892→        setMode('view')
   893→        setIngRows([{ ...EMPTY_ROW }])
   894→        setIngresoKey(k => k + 1) // forzar remount de campos de fecha
   895→      }
   896→      // Reload positions grid
   897→      await loadPosiciones()
   898→      if (selectedColumn) loadColumnDetail(selectedColumn)
   899→    } catch (err: unknown) { toast.error('Error al registrar ingreso', { description: extractError(err) }) } finally { setBusy(false) }
   900→  }

   902→  async function doSalida() {

   906→    const validRows = filteredItems.filter((r) => r.selected && r.bloque_id && r.cantidad && parseFloat(r.cantidad) > 0)

   908→    // Determinar nivel_id para la salida
   909→    const nivelId = salNivelTab === 'all' ? selectedNivelId : salNivelTab
   910→    if (!nivelId) { toast.error('No hay nivel seleccionado'); return }
   911→    setBusy(true)
   912→    try {
   913→      const detalles = validRows.map((r) => ({ nivel_id: nivelId, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   914→      await registrarSalidaPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', detalles)
   915→      toast.success('Salida registrada')
   916→      // Recargar stock y stock por nivel
   917→      const [stock, nivelStocks] = await Promise.all([
   918→        stockDetallePosicion(detail.posicionId),
   919→        niveles.length > 0 ? Promise.all(niveles.map((n) => stockDetalleNivel(n.id))) : Promise.resolve([]),

   937→  async function doTraslado() {
   938→    if (!detail || !perfil || !trDestPos) return
   939→    if (detail.posicionId === trDestPos.posicionId) { toast.error('Origen y destino no pueden ser iguales'); return }
   940→    const validRows = trItems.filter((r) => r.selected && r.bloque_id && r.cantidad && parseFloat(r.cantidad) > 0)
   941→    if (validRows.length === 0) { toast.error('No hay articulos para trasladar'); return }
   942→    if (!selectedNivelId || !trDestNivelId) { toast.error('Selecciona nivel de origen y destino'); return }
   943→    const origNivelId = selectedNivelId
   944→    const destNivelId = trDestNivelId
   945→    setBusy(true)
   946→    try {
   947→
   948→      // Separate items by discrepancy type
   949→      const exactItems = validRows.filter((r) => parseFloat(r.cantidad) === r.stockActual)
   950→      const deficitItems = validRows.filter((r) => parseFloat(r.cantidad) < r.stockActual)
   951→      const surplusItems = validRows.filter((r) => parseFloat(r.cantidad) > r.stockActual)
   952→
   953→      // 1) Base transfer for ALL selected items (move the entered amount)
   954→      const allDetSal = validRows.map((r) => ({ nivel_id: origNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   955→      const allDetIng = validRows.map((r) => ({ nivel_id: destNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   956→      await registrarTrasladoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', allDetSal, allDetIng)
   957→
   958→      // 2) For surplus items (qty > stock): create ingreso at destination for the excess
   959→      for (const item of surplusItems) {
   960→        const excess = parseFloat(item.cantidad) - item.stockActual
   961→        if (excess > 0) {
   962→          await registrarIngresoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', [
   963→            { nivel_id: destNivelId!, bloque_id: item.bloque_id, cantidad: excess, fecha_vencimiento: item.fecha_vencimiento || '' },
   964→          ])
   965→        }
   966→      }
   967→
   968→      // 3) For deficit items with saldoMode 'ajustar': create salida at origin for remaining balance
   969→      for (const item of deficitItems) {
   970→        if (item.saldoMode === 'ajustar') {
   971→          const remaining = item.stockActual - parseFloat(item.cantidad)
   972→          if (remaining > 0) {
   973→            await registrarSalidaPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', [
   974→              { nivel_id: origNivelId!, bloque_id: item.bloque_id, cantidad: remaining, fecha_vencimiento: item.fecha_vencimiento || null },
   975→            ])
   976→          }
   977→        }
   978→        // saldoMode === 'saldo': leave the balance as-is, no extra action needed
   979→      }
   980→
   981→      // Build result message
   982→      const parts: string[] = ['Traslado registrado']
   983→      if (surplusItems.length > 0) parts.push(`${surplusItems.length} ingreso(s) por excedente`)
   984→      const adjustedCount = deficitItems.filter((r) => r.saldoMode === 'ajustar').length
   985→      if (adjustedCount > 0) parts.push(`${adjustedCount} salida(s) por ajuste`)
   986→      toast.success(parts.join(' · '))
   987→      setTrConfirmOpen(false)
   988→      if (mountedRef.current) { setDetail(null); setTrDestPos(null); loadPosiciones(); if (selectedColumn) loadColumnDetail(selectedColumn) }
   989→    } catch (err: unknown) { toast.error('Error al trasladar', { description: extractError(err) }) } finally { setBusy(false) }
   990→  }
