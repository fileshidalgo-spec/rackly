    48→// Calcula días restantes hasta vencimiento (negativo si ya venció)
    49→function diasParaVencer(fVencimiento: string | undefined): number | null {
    50→  if (!fVencimiento) return null
    51→  const hoy = new Date(); hoy.setHours(0, 0, 0, 0)
    52→  const venc = new Date(fVencimiento + 'T00:00:00')
    53→  return Math.ceil((venc.getTime() - hoy.getTime()) / (1000 * 60 * 60 * 24))
    54→}

    56→// Detecta códigos duplicados (mismo artículo con distintas fechas de vencimiento)
    57→function codigosConMultiplesLotes(stock: StockEnUbicacion[]): Set<string> {
    58→  const counts = new Map<string, number>()
    59→  for (const s of stock) {
    60→    counts.set(s.codigo, (counts.get(s.codigo) ?? 0) + 1)
    61→  }
    62→  const multiples = new Set<string>()
    63→  for (const [code, count] of counts) {
    64→    if (count > 1) multiples.add(code)
    65→  }
    66→  return multiples
    67→}

    69→// Calcula ocupación desde movimientos — stock POR CÓDIGO (no por lote).
    70→// f_vencimiento es SOLO para FEFO (ordenamiento), NO para particionar stock.
    71→// IMPORTANTE: Los movimientos INC se EXCLUYEN del stock normal.
    72→// La detección INC se hace por separado con fetchIncPorUbicacion().
    73→function calcularOcupacion(movs: Movimiento[]): OcupacionCelda[] {
    74→  // Mapa: ubicacion_key → (codigo → stock)
    75→  const cellMap = new Map<string, Map<string, number>>()
    76→  for (const m of movs) {
    77→    // EXCLUIR movimientos INC del cálculo de ocupación normal
    78→    if (m.codigoInc) continue
    79→    const key = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}`
    80→    const code = m.codigo.trim().toUpperCase()
    81→    let codeMap = cellMap.get(key)
    82→    if (!codeMap) { codeMap = new Map(); cellMap.set(key, codeMap) }
    83→    const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? m.cantidad : -m.cantidad
    84→    const current = codeMap.get(code) ?? 0
    85→    codeMap.set(code, current + delta)
    86→  }
    87→  // Construir resultado: solo celdas con stock total > 0
    88→  const result: OcupacionCelda[] = []
    89→  for (const [key, codeMap] of cellMap) {
    90→    let totalStock = 0
    91→    const codigos = new Set<string>()
    92→    for (const [code, stock] of codeMap) {
    93→      if (stock > 0) {
    94→        totalStock += stock
    95→        codigos.add(code)
    96→      }
    97→    }
    98→    if (totalStock > 0) {
    99→      const [bloque, torre, piso, posicion] = key.split('-')
   100→      // lotes = número de códigos distintos (ya no se cuenta por f_vencimiento)
   101→      result.push({ bloque, torre, piso, posicion, stock: totalStock, codigos: Array.from(codigos), lotes: codigos.size, tieneInc: false, incItems: [] })
   102→    }
   103→  }
   104→  return result
   105→}

   413→  async function doSalida() {
   414→    if (!detail || !perfil) return
   415→    const item = detail.stock[salidaIdx]
   416→    if (!item) return
   417→    const qty = parseFloat(salidaCantidad)
   418→    if (isNaN(qty) || qty <= 0) { toast.error('Cantidad inválida'); return }
   419→    // Las salidas normales NO pueden exceder stock. Las salidas INC sí permiten exceder (autoajuste).
   420→    // La validacion debe usar el stock TOTAL del codigo en esta ubicacion (igual que la RPC
   421→    // registrar_movimiento_kardex), NO el stock por lote individual. La UI agrupa por lote
   422→    // (f_vencimiento) pero la RPC suma TODOS los movimientos del codigo sin distinguir lotes.
   423→    if (!item.codigoInc) {
   424→      const totalStockForCodigo = detail.stock
   425→        .filter(s => s.codigo === item.codigo)
   426→        .reduce((sum, s) => sum + s.stock, 0)
   427→      if (qty > totalStockForCodigo) { toast.error(`Stock insuficiente. Stock total disponible: ${totalStockForCodigo} ${item.un}`); return }
   428→    }
   429→    setActionBusy(true)
   430→    try {
   431→      const { wasOffline } = await SyncEngine.offlineAwareAddMovimiento({
   432→        tipo: 'salida', bloque: detail.bloque, torre: detail.torre, piso: detail.piso, posicion: detail.posicion,
   433→        codigo: item.codigo, descripcion: item.descripcion, un: item.un, cantidad: qty,

   453→  async function doTransferir() {
   454→    if (!detail || !perfil || !trItem) return
   455→    if (!trDestBloque || !trDestTorre || !trDestPiso || !trDestPos) { toast.error('Completa destino'); return }
   456→    const qty = parseFloat(trCantidad)
   457→    if (isNaN(qty) || qty <= 0) { toast.error('Cantidad inválida'); return }
   458→    const origKey = `${detail.bloque}-${detail.torre}-${detail.piso}-${detail.posicion}`
   459→    const destKey = `${trDestBloque}-${trDestTorre}-${trDestPiso}-${trDestPos}`
   460→    if (origKey === destKey) { toast.error('Origen y destino no pueden ser iguales'); return }
   461→    setActionBusy(true)
   462→    try {
   463→      // Check destination occupancy
   464→      const destStock = await stockEnUbicacion(trDestBloque, trDestTorre, trDestPiso, trDestPos)
   465→      if (destStock.length > 0) {
   466→        toast.error(`Destino ocupado con ${destStock.length} artículo(s). Vacía primero el destino.`)
   467→        setActionBusy(false); return
   468→      }
   469→      const input: TrasladoInput = {
   470→        codigo: trItem.codigo, descripcion: trItem.descripcion, un: trItem.un, cantidad: qty,
   471→        origen: { bloque: detail.bloque, torre: detail.torre, piso: detail.piso, posicion: detail.posicion },
   472→        destino: { bloque: trDestBloque, torre: trDestTorre, piso: trDestPiso, posicion: trDestPos },
   473→        turno: calcularTurno(), usuarioId: perfil.id, usuarioNombre: perfil.nombre, usuarioCorreo: perfil.correo,
   474→        fVencimiento: trItem.fVencimiento ?? '',
   475→        cantidadAjuste: trTieneAjuste ? trDiferencia : 0,
   476→        // Preservar codigoInc para que el traslado mueva el INC correctamente
   477→        codigoInc: trItem.codigoInc || undefined,
   478→      }
   479→      const { wasOffline: wasTrOffline } = await SyncEngine.offlineAwareTraslado(input)
   480→      toast.success(wasTrOffline ? 'Traslado guardado (offline)' : 'Traslado registrado', {
   481→        ...(wasTrOffline ? { description: 'Se sincronizará al reconectarse', duration: 5000 } : {}),
   482→      })
   483→      if (!wasTrOffline && trFalta && trCorregirDiferencia) {
   484→        toast.info(`Salida de ajuste: -${Math.abs(trDiferencia)} ${trItem.un} en origen`, { duration: 6000 })
   485→      }
   486→      if (!wasTrOffline && trExcede) {
   487→        toast.info(`Ajuste automático: +${Math.abs(trDiferencia)} ${trItem.un} ingreso en origen`, { duration: 6000 })
   488→      }
   489→      if (mountedRef.current && !wasTrOffline) { await refreshDetail(); refreshData(); setDetailMode('view') }
   490→    } catch (err: unknown) {
   491→      if (isInsufficientStockError(err)) {
   492→        const detail = (err as Record<string, string>).detail || ''
   493→        toast.error('Stock insuficiente en origen', { description: detail || 'Otro usuario pudo haber modificado el stock. Los datos se han actualizado.', duration: 8000 })
   494→        refreshDetail(); refreshData()
   495→      } else { toast.error('Error al trasladar', { description: extractError(err) }) }
   496→    } finally { setActionBusy(false) }
   497→  }
