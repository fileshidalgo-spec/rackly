    79→function useAnimatedCounter(target: number, duration = 800) {
    80→  const [count, setCount] = useState(0)
    81→  const prevTarget = useRef(0)
    82→
    83→  useEffect(() => {
    84→    if (target === prevTarget.current) return
    85→    const start = prevTarget.current
    86→    const diff = target - start
    87→    if (diff === 0) return
    88→    const startTime = performance.now()
    89→    let raf: number
    90→
    91→    function tick(now: number) {
    92→      const elapsed = now - startTime
    93→      const progress = Math.min(elapsed / duration, 1)
    94→      const eased = 1 - Math.pow(1 - progress, 3) // ease-out cubic
    95→      setCount(Math.round(start + diff * eased))
    96→      if (progress < 1) {
    97→        raf = requestAnimationFrame(tick)
    98→      }
    99→    }
   100→
   101→    raf = requestAnimationFrame(tick)
   102→    prevTarget.current = target
   103→    return () => cancelAnimationFrame(raf)
   104→  }, [target, duration])
   105→
   106→  return count
   107→}

   787→  // Auto-create manual_ bloques in piso_bloques before registering movement
   788→  async function ensureManualBloqueCreated(rows: RowEntry[]): Promise<RowEntry[]> {
   789→    const resolved: RowEntry[] = []
   790→    for (const r of rows) {
   791→      if (r.bloque_id.startsWith('manual_')) {
   792→        const code = r.bloque_id.replace('manual_', '')
   793→        try {
   794→          const created = await crearBloque(code, r.descripcion || '', r.unidad || 'KG')
   795→          if (created.length > 0) {
   796→            // Find the newly created one
   797→            const found = created.find((b) => b.codigo === code)
   798→            if (found) {
   799→              resolved.push({ ...r, bloque_id: found.id })
   800→              continue
   801→            }
   802→          }
   803→        } catch {
   804→          // If creation fails, try searching for it (maybe it was already created)
   805→        }
   806→        // Last resort: try buscarBloquePorCodigo
   807→        const existing = await buscarBloquePorCodigo(code)
   808→        if (existing) {
   809→          resolved.push({ ...r, bloque_id: existing.id })
   810→        } else {
   811→          resolved.push(r)
   812→        }
   813→      } else {
   814→        resolved.push(r)
   815→      }
   816→    }
   817→    return resolved
   818→  }

   974→  // Ejecutar traslado (se llama desde la alerta o desde confirmación normal)
   975→  async function ejecutarTrasladoPiso() {
   976→    if (!detail || !perfil || !trDestPos) return
   977→    const validRows = trItems.filter((r) => r.selected && r.bloque_id && r.cantidad && parseFloat(r.cantidad) > 0)
   978→    if (validRows.length === 0) { toast.error('No hay articulos para trasladar'); return }
   979→    if (!selectedNivelId || !trDestNivelId) { toast.error('Selecciona nivel de origen y destino'); return }
   980→    const origNivelId = selectedNivelId
   981→    const destNivelId = trDestNivelId
   982→    setBusy(true)
   983→    try {
   984→
   985→      // Separate items by discrepancy type
   986→      const exactItems = validRows.filter((r) => parseFloat(r.cantidad) === r.stockActual)
   987→      const deficitItems = validRows.filter((r) => parseFloat(r.cantidad) < r.stockActual)
   988→      const surplusItems = validRows.filter((r) => parseFloat(r.cantidad) > r.stockActual)
   989→
   990→      // 1) Base transfer for ALL selected items (move the entered amount)
   991→      const allDetSal = validRows.map((r) => ({ nivel_id: origNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   992→      const allDetIng = validRows.map((r) => ({ nivel_id: destNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   993→      await registrarTrasladoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', allDetSal, allDetIng)
   994→
   995→      // 2) For surplus items (qty > stock): create ingreso at destination for the excess
   996→      for (const item of surplusItems) {
   997→        const excess = parseFloat(item.cantidad) - item.stockActual
   998→        if (excess > 0) {
   999→          await registrarIngresoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', [
  1000→            { nivel_id: destNivelId!, bloque_id: item.bloque_id, cantidad: excess, fecha_vencimiento: item.fecha_vencimiento || '' },
  1001→          ])
  1002→        }
  1003→      }
  1004→
  1005→      // 3) For deficit items with saldoMode 'ajustar': create salida at origin for remaining balance
  1006→      for (const item of deficitItems) {
  1007→        if (item.saldoMode === 'ajustar') {
  1008→          const remaining = item.stockActual - parseFloat(item.cantidad)
  1009→          if (remaining > 0) {
  1010→            await registrarSalidaPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', [
  1011→              { nivel_id: origNivelId!, bloque_id: item.bloque_id, cantidad: remaining, fecha_vencimiento: item.fecha_vencimiento || null },
  1012→            ])
  1013→          }
  1014→        }
  1015→        // saldoMode === 'saldo': leave the balance as-is, no extra action needed
  1016→      }
  1017→
  1018→      // Build result message
  1019→      const parts: string[] = ['Traslado registrado']
  1020→      if (surplusItems.length > 0) parts.push(`${surplusItems.length} ingreso(s) por excedente`)
  1021→      const adjustedCount = deficitItems.filter((r) => r.saldoMode === 'ajustar').length
  1022→      if (adjustedCount > 0) parts.push(`${adjustedCount} salida(s) por ajuste`)
  1023→      toast.success(parts.join(' · '))
  1024→      setTrConfirmOpen(false)
  1025→      setTrDestAlertOpen(false)
  1026→      if (mountedRef.current) { setDetail(null); setTrDestPos(null); loadPosiciones(); if (selectedColumn) loadColumnDetail(selectedColumn) }
  1027→    } catch (err: unknown) { toast.error('Error al trasladar', { description: extractError(err) }) } finally { setBusy(false) }
  1028→  }

  1164→              for (const s of items) {

  1178→          await registrarSalidaPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', detalles)
  1179→          successCount++
  1180→        } catch { errorCount++ }
