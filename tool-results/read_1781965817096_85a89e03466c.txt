    73тЖТ    73тЖТexport async function fetchMovimientos(): Promise<Movimiento[]> {
    74тЖТ    74тЖТ  const all: Record<string, unknown>[] = []
    75тЖТ    75тЖТ  let from = 0
    76тЖТ    76тЖТ  let iterations = 0
    77тЖТ    77тЖТ  while (iterations < MAX_ITERATIONS) {
    78тЖТ    78тЖТ    iterations++
    79тЖТ    79тЖТ    const to = from + PAGE_SIZE - 1
    80тЖТ    80тЖТ    const { data, error } = await dataClient
    81тЖТ    81тЖТ      .from('movimientos')
    82тЖТ    82тЖТ      .select('*')
    83тЖТ    83тЖТ      .order('f_modificacion', { ascending: false })
    84тЖТ    84тЖТ      .order('id', { ascending: true })
    85тЖТ    85тЖТ      .range(from, to)
    86тЖТ    86тЖТ    if (error) throw error
    87тЖТ    87тЖТ    const rows = data ?? []
    88тЖТ    88тЖТ    all.push(...rows)
    89тЖТ    89тЖТ    if (rows.length < PAGE_SIZE) break
    90тЖТ    90тЖТ    from += PAGE_SIZE
    91тЖТ    91тЖТ  }
    92тЖТ    92тЖТ  return all.map(fromRow)
    93тЖТ    93тЖТ}

   122тЖТ   122тЖТ * Usado por PisoStockTab para mostrar "Tambi├йn en Kardex Racks".

   125тЖТ   125тЖТexport async function buscarStockRacksPorCodigo(codigo: string): Promise<StockRacksPorCodigoItem[]> {
   126тЖТ   126тЖТ  const movs = await fetchMovimientosByCodigo(codigo)
   127тЖТ   127тЖТ  const locMap = new Map<string, StockRacksPorCodigoItem>()
   128тЖТ   128тЖТ
   129тЖТ   129тЖТ  for (const m of movs) {
   130тЖТ   130тЖТ    const key = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}||${m.fVencimiento || ''}||${m.codigoInc || ''}`
   131тЖТ   131тЖТ    const current = locMap.get(key)
   132тЖТ   132тЖТ    const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? m.cantidad : -m.cantidad
   133тЖТ   133тЖТ    if (current) {
   134тЖТ   134тЖТ      current.stock += delta
   135тЖТ   135тЖТ    } else {
   136тЖТ   136тЖТ      locMap.set(key, {
   137тЖТ   137тЖТ        bloque: m.bloque,
   138тЖТ   138тЖТ        torre: m.torre,
   139тЖТ   139тЖТ        piso: m.piso,
   140тЖТ   140тЖТ        posicion: m.posicion,
   141тЖТ   141тЖТ        stock: delta,
   142тЖТ   142тЖТ        descripcion: m.descripcion,
   143тЖТ   143тЖТ        un: m.un,
   144тЖТ   144тЖТ        fVencimiento: m.fVencimiento || '',
   145тЖТ   145тЖТ        codigo_inc: m.codigoInc || undefined,
   146тЖТ   146тЖТ      })
   147тЖТ   147тЖТ    }
   148тЖТ   148тЖТ  }
   149тЖТ   149тЖТ
   150тЖТ   150тЖТ  return Array.from(locMap.values())
   151тЖТ   151тЖТ    .filter(l => l.stock > 0)
   152тЖТ   152тЖТ    .sort((a, b) => {
   153тЖТ   153тЖТ      // FEFO primero
   154тЖТ   154тЖТ      if (a.fVencimiento && b.fVencimiento) return a.fVencimiento.localeCompare(b.fVencimiento)
   155тЖТ   155тЖТ      if (a.fVencimiento && !b.fVencimiento) return -1
   156тЖТ   156тЖТ      if (!a.fVencimiento && b.fVencimiento) return 1
   157тЖТ   157тЖТ      return 0
   158тЖТ   158тЖТ    })
   159тЖТ   159тЖТ}

   164тЖТ   164тЖТ  uuidSync?: string,

   168тЖТ   168тЖТ

   183тЖТ   183тЖТ      if (m.cantidad > stockRedondeado + 0.001) {

   194тЖТ   194тЖТ      console.error('[addMovimientoFallback] No se pudo verificar stock, BLOQUEANDO inserci├│n:', stockErr)
   195тЖТ   195тЖТ      throw new Error('STOCK_VALIDATION_FAILED|No se pudo verificar el stock. Reintente.')
   196тЖТ   196тЖТ    }
   197тЖТ   197тЖТ  }
   198тЖТ   198тЖТ

   200тЖТ   200тЖТ    tipo: m.tipo,

   212тЖТ   212тЖТ    usuario_nombre: m.usuarioNombre ?? null,
   213тЖТ   213тЖТ    usuario_correo: m.usuarioCorreo ?? null,
   214тЖТ   214тЖТ    proveedor: m.proveedor ? m.proveedor : null,
   215тЖТ   215тЖТ    uuid_sync: uuidSync || null,
   216тЖТ   216тЖТ    codigo_inc: m.codigoInc || null,
   217тЖТ   217тЖТ  })
   218тЖТ   218тЖТ  if (error) throw error
   219тЖТ   219тЖТ  try {
   220тЖТ   220тЖТ    return await fetchMovimientos()

   240тЖТ   240тЖТexport async function addMovimiento(
   241тЖТ   241тЖТ  m: Omit<Movimiento, 'id' | 'fModificacion'>,
   242тЖТ   242тЖТ  uuidSync?: string
   243тЖТ   243тЖТ): Promise<Movimiento[]> {
   244тЖТ   244тЖТ  // Idempotencia: si viene uuidSync, verificar si ya existe en el servidor
   245тЖТ   245тЖТ  if (uuidSync) {
   246тЖТ   246тЖТ    const exists = await checkExistingByUuidSync(uuidSync)
   247тЖТ   247тЖТ    if (exists) {
   248тЖТ   248тЖТ      console.log('[addMovimiento] Movimiento ya existe (uuid_sync):', uuidSync, 'тАФ saltando insert.')
   249тЖТ   249тЖТ      // Ya existe: refrescar y retornar como si hubiera sido exitoso
   250тЖТ   250тЖТ      try { return await fetchMovimientos() } catch { return [] }
   251тЖТ   251тЖТ    }
   252тЖТ   252тЖТ  }
   253тЖТ   253тЖТ
   254тЖТ   254тЖТ  // NOTA: Los movimientos INC ahora pasan por la RPC como cualquier otro movimiento.
   255тЖТ   255тЖТ  // La RPC registrar_movimiento_kardex ya tiene p_codigo_inc (migration 20260611).
   256тЖТ   256тЖТ  // Esto garantiza advisory lock y validaci├│n at├│mica para TODOS los movimientos.
   257тЖТ   257тЖТ
   258тЖТ   258тЖТ  // Usar RPC at├│mica con advisory lock para evitar race conditions.
   259тЖТ   259тЖТ  // La RPC maneja TODOS los tipos de movimiento incluyendo INC.
   260тЖТ   260тЖТ  // INC items son tipo 'ingreso' as├н que no pasan validaci├│n de stock negativo.
   261тЖТ   261тЖТ  try {
   262тЖТ   262тЖТ    const { data, error } = await dataClient.rpc('registrar_movimiento_kardex', {
   263тЖТ   263тЖТ      p_tipo: m.tipo,
   264тЖТ   264тЖТ      p_bloque: m.bloque,
   265тЖТ   265тЖТ      p_torre: m.torre,
   266тЖТ   266тЖТ      p_piso: m.piso,
   267тЖТ   267тЖТ      p_posicion: m.posicion,
   268тЖТ   268тЖТ      p_codigo: m.codigo.trim().toUpperCase(),
   269тЖТ   269тЖТ      p_descripcion: m.descripcion,
   270тЖТ   270тЖТ      p_un: m.un,
   271тЖТ   271тЖТ      p_cantidad: m.cantidad,
   272тЖТ   272тЖТ      p_f_vencimiento: m.fVencimiento || null,
   273тЖТ   273тЖТ      p_turno: m.turno,
   274тЖТ   274тЖТ      p_usuario_id: m.usuarioId,
   275тЖТ   275тЖТ      p_usuario_nombre: m.usuarioNombre ?? null,
   276тЖТ   276тЖТ      p_usuario_correo: m.usuarioCorreo ?? null,
   277тЖТ   277тЖТ      p_proveedor: m.proveedor ? m.proveedor : null,
   278тЖТ   278тЖТ      p_uuid_sync: uuidSync || null,
   279тЖТ   279тЖТ      p_codigo_inc: m.codigoInc || null,
   280тЖТ   280тЖТ    })
   281тЖТ   281тЖТ    // Stock insuficiente es un error controlado тАФ NUNCA bypassear la decisi├│n del RPC.
   282тЖТ   282тЖТ    // El RPC tiene advisory lock y calcula stock de forma at├│mica; es la fuente de verdad.
   283тЖТ   283тЖТ    if (error) {
   284тЖТ   284тЖТ      const msg = error.message || ''
   285тЖТ   285тЖТ      const code = (error as unknown as Record<string, string>).code || ''
   286тЖТ   286тЖТ      if (msg.includes('INSUFFICIENT_STOCK')) {
   287тЖТ   287тЖТ        const parts = msg.split('|')
   288тЖТ   288тЖТ        const rpcDetail = parts.length > 1 ? parts[1] : 'Stock insuficiente para esta operaci├│n'
   289тЖТ   289тЖТ        const err = new Error('INSUFFICIENT_STOCK')
   290тЖТ   290тЖТ        ;(err as unknown as Record<string, string>).detail = rpcDetail
   291тЖТ   291тЖТ        throw err
   292тЖТ   292тЖТ      }
   293тЖТ   293тЖТ      // Si la RPC no existe (404 / 42883 / 'Could not find'), usar fallback
   294тЖТ   294тЖТ      if (code === '42883' || code === 'PGRST202' || msg.includes('Could not find') || msg.includes('does not exist') || msg.includes('404')) {
   295тЖТ   295тЖТ        return await addMovimientoFallback(m, uuidSync)
   296тЖТ   296тЖТ      }
   297тЖТ   297тЖТ      throw error
   298тЖТ   298тЖТ    }
   299тЖТ   299тЖТ    // RPC exitosa: el movimiento ya fue registrado en la DB.
   300тЖТ   300тЖТ    // Intentar refrescar la lista de movimientos, pero si falla (timeout, red)
   301тЖТ   301тЖТ    // NO propagar el error тАФ el movimiento ya est├б guardado.
   302тЖТ   302тЖТ    try {
   303тЖТ   303тЖТ      return await fetchMovimientos()
   304тЖТ   304тЖТ    } catch {
   305тЖТ   305тЖТ      console.warn('[addMovimiento] RPC exitosa pero fetchMovimientos() fall├│. Movimiento ya registrado.')
   306тЖТ   306тЖТ      return []
   307тЖТ   307тЖТ    }
   308тЖТ   308тЖТ  } catch (err: unknown) {
   309тЖТ   309тЖТ    // Si la RPC no existe a├║n (SQL no ejecutado), fallback al insert directo
   310тЖТ   310тЖТ    const errMsg = err instanceof Error ? err.message : ''
   311тЖТ   311тЖТ    const errCode = err instanceof Error ? (err as unknown as Record<string, string>).code || '' : ''
   312тЖТ   312тЖТ    if (errCode === '42883' || errCode === 'PGRST202' || errMsg.includes('Could not find') || errMsg.includes('does not exist')) {
   313тЖТ   313тЖТ      return await addMovimientoFallback(m, uuidSync)
   314тЖТ   314тЖТ    }
   315тЖТ   315тЖТ    throw err
   316тЖТ   316тЖТ  }
   317тЖТ   317тЖТ}

   368тЖТ   368тЖТexport async function stockEnUbicacion(
   369тЖТ   369тЖТ  bloque: string,
   370тЖТ   370тЖТ  torre: string,
   371тЖТ   371тЖТ  piso: string,
   372тЖТ   372тЖТ  posicion: string
   373тЖТ   373тЖТ): Promise<StockEnUbicacion[]> {
   374тЖТ   374тЖТ  try {
   375тЖТ   375тЖТ    // M├йtodo principal: consultar movimientos directamente con paginaci├│n
   376тЖТ   376тЖТ    // f_vencimiento es SOLO para FEFO (ordenamiento), NO para particionar stock.
   377тЖТ   377тЖТ    // Se agrupa por (codigo, codigo_inc) тАФ el stock es la suma de TODOS los lotes.
   378тЖТ   378тЖТ    const allRows: Record<string, unknown>[] = []
   379тЖТ   379тЖТ    let from = 0
   380тЖТ   380тЖТ    const BATCH = 1000
   381тЖТ   381тЖТ    const MAX_PAGES = 10 // m├бximo 10,000 movimientos por ubicaci├│n
   382тЖТ   382тЖТ
   383тЖТ   383тЖТ    for (let page = 0; page < MAX_PAGES; page++) {
   384тЖТ   384тЖТ      const { data, error } = await dataClient
   385тЖТ   385тЖТ        .from('movimientos')
   386тЖТ   386тЖТ        .select('*')
   387тЖТ   387тЖТ        .eq('bloque', bloque)
   388тЖТ   388тЖТ        .eq('torre', torre)
   389тЖТ   389тЖТ        .eq('piso', piso)
   390тЖТ   390тЖТ        .eq('posicion', posicion)
   391тЖТ   391тЖТ        .order('f_modificacion', { ascending: true })
   392тЖТ   392тЖТ        .range(from, from + BATCH - 1)
   393тЖТ   393тЖТ
   394тЖТ   394тЖТ      if (error) throw error
   395тЖТ   395тЖТ      const rows = data ?? []
   396тЖТ   396тЖТ      allRows.push(...rows)
   397тЖТ   397тЖТ      if (rows.length < BATCH) break
   398тЖТ   398тЖТ      from += BATCH
   399тЖТ   399тЖТ    }
   400тЖТ   400тЖТ
   401тЖТ   401тЖТ    // Agrupar por (codigo, codigo_inc) тАФ f_vencimiento NO participa en la agrupaci├│n.
   402тЖТ   402тЖТ    // Se rastrea la fecha de vencimiento m├бs pr├│xima (FEFO) para cada grupo.
   403тЖТ   403тЖТ    const groups = new Map<string, {
   404тЖТ   404тЖТ      codigo: string; descripcion: string; un: string;
   405тЖТ   405тЖТ      stock: number; fVencimientoMasProxima: string;
   406тЖТ   406тЖТ      usuarioPrimerNombre: string; proveedor: string; codigoInc: string;
   407тЖТ   407тЖТ    }>()
   408тЖТ   408тЖТ
   409тЖТ   409тЖТ    for (const r of allRows) {
   410тЖТ   410тЖТ      const m = fromRow(r)
   411тЖТ   411тЖТ      const incKey = m.codigoInc || ''
   412тЖТ   412тЖТ      const key = `${m.codigo}||${incKey}`
   413тЖТ   413тЖТ
   414тЖТ   414тЖТ      let group = groups.get(key)
   415тЖТ   415тЖТ      if (!group) {
   416тЖТ   416тЖТ        group = {
   417тЖТ   417тЖТ          codigo: m.codigo,
   418тЖТ   418тЖТ          descripcion: m.descripcion,
   419тЖТ   419тЖТ          un: m.un,
   420тЖТ   420тЖТ          stock: 0,
   421тЖТ   421тЖТ          fVencimientoMasProxima: m.fVencimiento || '',
   422тЖТ   422тЖТ          usuarioPrimerNombre: m.usuarioNombre?.split(' ')[0] ?? '',
   423тЖТ   423тЖТ          proveedor: m.proveedor ?? '',
   424тЖТ   424тЖТ          codigoInc: incKey,
   425тЖТ   425тЖТ        }
   426тЖТ   426тЖТ        groups.set(key, group)
   427тЖТ   427тЖТ      } else {
   428тЖТ   428тЖТ        // Si el grupo no tiene descripci├│n pero este movimiento s├н, usarla
   429тЖТ   429тЖТ        if (!group.descripcion && m.descripcion) group.descripcion = m.descripcion
   430тЖТ   430тЖТ        // Rastrear la fecha de vencimiento m├бs pr├│xima (para FEFO)
   431тЖТ   431тЖТ        if (m.fVencimiento) {
   432тЖТ   432тЖТ          if (!group.fVencimientoMasProxima || m.fVencimiento < group.fVencimientoMasProxima) {
   433тЖТ   433тЖТ            group.fVencimientoMasProxima = m.fVencimiento
   434тЖТ   434тЖТ          }
   435тЖТ   435тЖТ        }
   436тЖТ   436тЖТ      }
   437тЖТ   437тЖТ
   438тЖТ   438тЖТ      // Calcular stock neto (ingreso/devolucion/traslado = +, salida = -)
   439тЖТ   439тЖТ      const qty = typeof m.cantidad === 'number' ? m.cantidad : parseFloat(String(m.cantidad)) || 0
   440тЖТ   440тЖТ      const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? qty : -qty
   441тЖТ   441тЖТ      group.stock += delta
   442тЖТ   442тЖТ    }
   443тЖТ   443тЖТ
   444тЖТ   444тЖТ    // Filtrar solo art├нculos con stock > 0
   445тЖТ   445тЖТ    const results = Array.from(groups.values()).filter(g => g.stock > 0)
   446тЖТ   446тЖТ
   447тЖТ   447тЖТ    // Ordenar por FEFO: fecha de vencimiento m├бs pr├│xima primero, sin fecha al final
   448тЖТ   448тЖТ    results.sort((a, b) => {
   449тЖТ   449тЖТ      if (a.fVencimientoMasProxima && b.fVencimientoMasProxima) return a.fVencimientoMasProxima.localeCompare(b.fVencimientoMasProxima)
   450тЖТ   450тЖТ      if (a.fVencimientoMasProxima && !b.fVencimientoMasProxima) return -1
   451тЖТ   451тЖТ      if (!a.fVencimientoMasProxima && b.fVencimientoMasProxima) return 1
   452тЖТ   452тЖТ      return 0
   453тЖТ   453тЖТ    })
   454тЖТ   454тЖТ
   455тЖТ   455тЖТ    return results.map(g => ({
   456тЖТ   456тЖТ      codigo: g.codigo,
   457тЖТ   457тЖТ      descripcion: g.descripcion,
   458тЖТ   458тЖТ      un: g.un,
   459тЖТ   459тЖТ      stock: Math.round(g.stock * 1000) / 1000,
   460тЖТ   460тЖТ      fVencimiento: g.fVencimientoMasProxima || undefined,
   461тЖТ   461тЖТ      usuarioPrimerNombre: g.usuarioPrimerNombre || undefined,
   462тЖТ   462тЖТ      proveedor: g.proveedor || undefined,
   463тЖТ   463тЖТ      codigoInc: g.codigoInc || undefined,
   464тЖТ   464тЖТ    }))
   465тЖТ   465тЖТ  } catch {
   466тЖТ   466тЖТ    // Fallback al RPC original si falla la consulta directa
   467тЖТ   467тЖТ    try {
   468тЖТ   468тЖТ      const { data, error } = await dataClient.rpc('stock_en_ubicacion', {
   469тЖТ   469тЖТ        _bloque: bloque,
   470тЖТ   470тЖТ        _torre: torre,
   471тЖТ   471тЖТ        _piso: piso,
   472тЖТ   472тЖТ        _posicion: posicion,
   473тЖТ   473тЖТ      })
   474тЖТ   474тЖТ      if (error) throw error
   475тЖТ   475тЖТ      return ((data ?? []) as Record<string, unknown>[]).map((r) => ({
   476тЖТ   476тЖТ        codigo: String(r.codigo ?? '').trim().toUpperCase(),
   477тЖТ   477тЖТ        descripcion: (r.descripcion as string) ?? '',
   478тЖТ   478тЖТ        un: (r.un as string) ?? '',
   479тЖТ   479тЖТ        stock: Number(r.stock ?? 0),
   480тЖТ   480тЖТ        fVencimiento: (r.f_vencimiento as string) ?? '',
   481тЖТ   481тЖТ        usuarioPrimerNombre: (r.usuario_primer_nombre as string) ?? '',
   482тЖТ   482тЖТ        proveedor: (r.proveedor as string) ?? '',
   483тЖТ   483тЖТ        codigoInc: (r.codigo_inc as string) ?? undefined,
   484тЖТ   484тЖТ      }))
   485тЖТ   485тЖТ    } catch {
   486тЖТ   486тЖТ      return []
   487тЖТ   487тЖТ    }
   488тЖТ   488тЖТ  }
   489тЖТ   489тЖТ}

   662тЖТ   662тЖТexport async function trasladarMovimiento(t: TrasladoInput): Promise<Movimiento[]> {
   663тЖТ   663тЖТ  // Usar RPC at├│mica con advisory locks en origen Y destino
   664тЖТ   664тЖТ  try {
   665тЖТ   665тЖТ    const { data, error } = await dataClient.rpc('registrar_traslado_kardex', {
   666тЖТ   666тЖТ      p_codigo: t.codigo,
   667тЖТ   667тЖТ      p_descripcion: t.descripcion,
   668тЖТ   668тЖТ      p_un: t.un,
   669тЖТ   669тЖТ      p_cantidad: t.cantidad,
   670тЖТ   670тЖТ      p_orig_bloque: t.origen.bloque,
   671тЖТ   671тЖТ      p_orig_torre: t.origen.torre,
   672тЖТ   672тЖТ      p_orig_piso: t.origen.piso,
   673тЖТ   673тЖТ      p_orig_pos: t.origen.posicion,
   674тЖТ   674тЖТ      p_dest_bloque: t.destino.bloque,
   675тЖТ   675тЖТ      p_dest_torre: t.destino.torre,
   676тЖТ   676тЖТ      p_dest_piso: t.destino.piso,
   677тЖТ   677тЖТ      p_dest_pos: t.destino.posicion,
   678тЖТ   678тЖТ      p_turno: t.turno,
   679тЖТ   679тЖТ      p_usuario_id: t.usuarioId,
   680тЖТ   680тЖТ      p_usuario_nombre: t.usuarioNombre ?? null,
   681тЖТ   681тЖТ      p_usuario_correo: t.usuarioCorreo ?? null,
   682тЖТ   682тЖТ      p_f_vencimiento: t.fVencimiento || null,
   683тЖТ   683тЖТ      p_proveedor: t.proveedor ? t.proveedor : null,
   684тЖТ   684тЖТ      p_cantidad_ajuste: t.cantidadAjuste ?? 0,
   685тЖТ   685тЖТ      p_codigo_inc: t.codigoInc || null,
   686тЖТ   686тЖТ      p_uuid_sync: t.uuidSync || null,
   687тЖТ   687тЖТ    })
   688тЖТ   688тЖТ    // Stock insuficiente en origen тАФ NUNCA bypassear la decisi├│n del RPC.
   689тЖТ   689тЖТ    // El RPC tiene advisory locks en origen Y destino; es la fuente de verdad.
   690тЖТ   690тЖТ    if (error) {
   691тЖТ   691тЖТ      const msg = error.message || ''
   692тЖТ   692тЖТ      const code = (error as unknown as Record<string, string>).code || ''
   693тЖТ   693тЖТ      if (msg.includes('INSUFFICIENT_STOCK')) {
   694тЖТ   694тЖТ        const parts = msg.split('|')
   695тЖТ   695тЖТ        const rpcDetail = parts.length > 1 ? parts[1] : 'Stock insuficiente en origen para este traslado'
   696тЖТ   696тЖТ        const err = new Error('INSUFFICIENT_STOCK')
   697тЖТ   697тЖТ        ;(err as unknown as Record<string, string>).detail = rpcDetail
   698тЖТ   698тЖТ        throw err
   699тЖТ   699тЖТ      }
   700тЖТ   700тЖТ      // Si la RPC no existe (404 / 42883 / 'Could not find'), usar fallback
   701тЖТ   701тЖТ      if (code === '42883' || code === 'PGRST202' || msg.includes('Could not find') || msg.includes('does not exist') || msg.includes('404')) {
   702тЖТ   702тЖТ        return await trasladarMovimientoFallback(t)
   703тЖТ   703тЖТ      }
   704тЖТ   704тЖТ      // Origen = destino (server guard)
   705тЖТ   705тЖТ      if (msg.includes('SAME_ORIGIN_DESTINATION')) {
   706тЖТ   706тЖТ        const err = new Error('SAME_ORIGIN_DESTINATION')
   707тЖТ   707тЖТ        ;(err as unknown as Record<string, string>).detail = 'El destino no puede ser igual al origen'
   708тЖТ   708тЖТ        throw err
   709тЖТ   709тЖТ      }
   710тЖТ   710тЖТ      throw error
   711тЖТ   711тЖТ    }
   712тЖТ   712тЖТ    // RPC exitosa: refrescar movimientos, pero no fallar si fetchMovimientos falla
   713тЖТ   713тЖТ    try {
   714тЖТ   714тЖТ      return await fetchMovimientos()
   715тЖТ   715тЖТ    } catch {
   716тЖТ   716тЖТ      console.warn('[trasladarMovimiento] RPC exitosa pero fetchMovimientos() fall├│. Traslado ya registrado.')
   717тЖТ   717тЖТ      return []
   718тЖТ   718тЖТ    }
   719тЖТ   719тЖТ  } catch (err: unknown) {
   720тЖТ   720тЖТ    // Fallback si la RPC no existe a├║n
   721тЖТ   721тЖТ    const errMsg = err instanceof Error ? err.message : ''
   722тЖТ   722тЖТ    const errCode = err instanceof Error ? (err as unknown as Record<string, string>).code || '' : ''
   723тЖТ   723тЖТ    if (errCode === '42883' || errCode === 'PGRST202' || errMsg.includes('Could not find') || errMsg.includes('does not exist')) {
   724тЖТ   724тЖТ      return await trasladarMovimientoFallback(t)
   725тЖТ   725тЖТ    }
   726тЖТ   726тЖТ    throw err
   727тЖТ   727тЖТ  }
   728тЖТ   728тЖТ}

   730тЖТ   730тЖТ/**
   731тЖТ   731тЖТ * Elimina TODOS los movimientos de la tabla.
   732тЖТ   732тЖТ * Usa service_role para bypassear RLS.
   733тЖТ   733тЖТ * Borra en lotes de 1000 IDs para no exceder l├нmites de URL.
   734тЖТ   734тЖТ */
   735тЖТ   735тЖТexport async function deleteAllMovimientos(): Promise<{ deleted: boolean; error?: string }> {
   736тЖТ   736тЖТ  // Usar dataClient (service role) ya configurado тАФ sin credenciales hardcodeadas
   737тЖТ   737тЖТ  const admin = dataClient
   738тЖТ   738тЖТ
   739тЖТ   739тЖТ  // Obtener todos los IDs
   740тЖТ   740тЖТ  const allIds: string[] = []
   741тЖТ   741тЖТ  let from = 0
   742тЖТ   742тЖТ  let iterations = 0
   743тЖТ   743тЖТ  const BATCH = 1000
   744тЖТ   744тЖТ  while (iterations++ < MAX_ITERATIONS) {
   745тЖТ   745тЖТ    const { data, error } = await admin
   746тЖТ   746тЖТ      .from('movimientos')
   747тЖТ   747тЖТ      .select('id')
   748тЖТ   748тЖТ      .range(from, from + BATCH - 1)
   749тЖТ   749тЖТ    if (error) return { deleted: false, error: error.message }
   750тЖТ   750тЖТ    if (!data || data.length === 0) break
   751тЖТ   751тЖТ    allIds.push(...data.map((r: { id: string }) => r.id))
   752тЖТ   752тЖТ    if (data.length < BATCH) break
   753тЖТ   753тЖТ    from += BATCH
   754тЖТ   754тЖТ  }
   755тЖТ   755тЖТ
   756тЖТ   756тЖТ  if (allIds.length === 0) return { deleted: true }
   757тЖТ   757тЖТ
   758тЖТ   758тЖТ  // Borrar en lotes usando .in('id', [...])
   759тЖТ   759тЖТ  const DELETE_BATCH = 500
   760тЖТ   760тЖТ  for (let i = 0; i < allIds.length; i += DELETE_BATCH) {
   761тЖТ   761тЖТ    const batch = allIds.slice(i, i + DELETE_BATCH)
   762тЖТ   762тЖТ    const { error } = await admin
   763тЖТ   763тЖТ      .from('movimientos')
   764тЖТ   764тЖТ      .delete()
   765тЖТ   765тЖТ      .in('id', batch)
   766тЖТ   766тЖТ    if (error) return { deleted: false, error: `Lote ${Math.floor(i / DELETE_BATCH) + 1}: ${error.message}` }
   767тЖТ   767тЖТ  }
   768тЖТ   768тЖТ
   769тЖТ   769тЖТ  return { deleted: true }
   770тЖТ   770тЖТ}
