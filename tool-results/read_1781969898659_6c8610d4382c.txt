    79→function useAnimatedCounter(target: number, duration = 800) {
    80→  const [count, setCount] = useState(0)
    81→  const prevTarget = useRef(0)
    82→
    83→  useEffect(() => {
    84→    if (target === prevTarget.current) return
    85→    const start = prevTarget.current
    86→    const diff = target - start
    87→    if (diff === 0) return
    88→    const startTime = performance.now()
    89→    let raf: number
    90→
    91→    function tick(now: number) {
    92→      const elapsed = now - startTime
    93→      const progress = Math.min(elapsed / duration, 1)
    94→      const eased = 1 - Math.pow(1 - progress, 3) // ease-out cubic
    95→      setCount(Math.round(start + diff * eased))
    96→      if (progress < 1) {
    97→        raf = requestAnimationFrame(tick)
    98→      }
    99→    }
   100→
   101→    raf = requestAnimationFrame(tick)
   102→    prevTarget.current = target
   103→    return () => cancelAnimationFrame(raf)
   104→  }, [target, duration])
   105→
   106→  return count
   107→}

   787→  // Auto-create manual_ bloques in piso_bloques before registering movement
   788→  async function ensureManualBloqueCreated(rows: RowEntry[]): Promise<RowEntry[]> {
   789→    const resolved: RowEntry[] = []
   790→    for (const r of rows) {
   791→      if (r.bloque_id.startsWith('manual_')) {
   792→        const code = r.bloque_id.replace('manual_', '')
   793→        try {
   794→          const created = await crearBloque(code, r.descripcion || '', r.unidad || 'KG')
   795→          if (created.length > 0) {
   796→            // Find the newly created one
   797→            const found = created.find((b) => b.codigo === code)
   798→            if (found) {
   799→              resolved.push({ ...r, bloque_id: found.id })
   800→              continue
   801→            }
   802→          }
   803→        } catch {
   804→          // If creation fails, try searching for it (maybe it was already created)
   805→        }
   806→        // Last resort: try buscarBloquePorCodigo
   807→        const existing = await buscarBloquePorCodigo(code)
   808→        if (existing) {
   809→          resolved.push({ ...r, bloque_id: existing.id })
   810→        } else {
   811→          resolved.push(r)
   812→        }
   813→      } else {
   814→        resolved.push(r)
   815→      }
   816→    }
   817→    return resolved
   818→  }

   868→    }
   869→    setBusy(true)

   871→      // Resolve manual_ IDs before registering

   874→        nivel_id: nivelId,

   879→      await registrarIngresoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', detalles)
   880→      toast.success('Ingreso registrado')
   881→      // Reload everything in parallel for real-time update
   882→      loadBloques()
   883→      const [stock, reloadedPerNivel] = await Promise.all([
   884→        stockDetallePosicion(detail.posicionId),
   885→        (async () => {
   886→          const perNivel: Record<string, DetailStock[]> = {}

   893→      if (mountedRef.current && detail) {

   897→          const nivelStock = reloadedPerNivel[currentTab] ?? []
   898→          return nivelStock.some(ns => ns.bloque_id === s.bloque_id && ns.fecha_vencimiento === s.fecha_vencimiento)
   899→        }) : stock
   900→        setDetail({ ...detail, stock: filteredStock })
   901→        setStockByNivel(reloadedPerNivel)
   902→        setMode('view')
   903→        setIngRows([{ ...EMPTY_ROW }])
   904→        setIngresoKey(k => k + 1) // forzar remount de campos de fecha

   912→  async function doSalida() {
   913→    if (!detail || !perfil) return
   914→    // Filtrar items por nivel seleccionado
   915→    const filteredItems = salNivelTab === 'all' ? salItems : salItemsByNivel
   916→    const validRows = filteredItems.filter((r) => r.selected && r.bloque_id && r.cantidad && parseFloat(r.cantidad) > 0)
   917→    if (validRows.length === 0) { toast.error('No hay articulos para salir'); return }
   918→    // Validar que las cantidades no excedan el stock actual
   919→    for (const r of validRows) {
   920→      if (parseFloat(r.cantidad) > r.stockActual) {
   921→        toast.error(`Cantidad excede stock para ${r.bloque_codigo}: max ${r.stockActual} ${r.bloque_unidad}`)
   922→        return
   923→      }

   981→  // Ejecutar traslado (se llama desde la alerta o desde confirmación normal)
   982→  async function ejecutarTrasladoPiso() {
   983→    if (!detail || !perfil || !trDestPos) return
   984→    const validRows = trItems.filter((r) => r.selected && r.bloque_id && r.cantidad && parseFloat(r.cantidad) > 0)
   985→    if (validRows.length === 0) { toast.error('No hay articulos para trasladar'); return }
   986→    if (!selectedNivelId || !trDestNivelId) { toast.error('Selecciona nivel de origen y destino'); return }
   987→    const origNivelId = selectedNivelId
   988→    const destNivelId = trDestNivelId
   989→    setBusy(true)
   990→    try {
   991→
   992→      // Separate items by discrepancy type
   993→      const exactItems = validRows.filter((r) => parseFloat(r.cantidad) === r.stockActual)
   994→      const deficitItems = validRows.filter((r) => parseFloat(r.cantidad) < r.stockActual)
   995→      const surplusItems = validRows.filter((r) => parseFloat(r.cantidad) > r.stockActual)
   996→
   997→      // 1) Base transfer for ALL selected items (move the entered amount)
   998→      const allDetSal = validRows.map((r) => ({ nivel_id: origNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
   999→      const allDetIng = validRows.map((r) => ({ nivel_id: destNivelId!, bloque_id: r.bloque_id, cantidad: parseFloat(r.cantidad), fecha_vencimiento: r.fecha_vencimiento || null }))
  1000→      await registrarTrasladoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', allDetSal, allDetIng)
  1001→
  1002→      // 2) For surplus items (qty > stock): create ingreso at destination for the excess
  1003→      for (const item of surplusItems) {
  1004→        const excess = parseFloat(item.cantidad) - item.stockActual
  1005→        if (excess > 0) {
  1006→          await registrarIngresoPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', [
  1007→            { nivel_id: destNivelId!, bloque_id: item.bloque_id, cantidad: excess, fecha_vencimiento: item.fecha_vencimiento || '' },
  1008→          ])
  1009→        }
  1010→      }
  1011→
  1012→      // 3) For deficit items with saldoMode 'ajustar': create salida at origin for remaining balance
  1013→      for (const item of deficitItems) {
  1014→        if (item.saldoMode === 'ajustar') {
  1015→          const remaining = item.stockActual - parseFloat(item.cantidad)
  1016→          if (remaining > 0) {
  1017→            await registrarSalidaPosicion(calcularTurno(), perfil.id, perfil.nombre ?? '', perfil.correo ?? '', [
  1018→              { nivel_id: origNivelId!, bloque_id: item.bloque_id, cantidad: remaining, fecha_vencimiento: item.fecha_vencimiento || null },
  1019→            ])
  1020→          }
  1021→        }
  1022→        // saldoMode === 'saldo': leave the balance as-is, no extra action needed
  1023→      }
  1024→
  1025→      // Build result message
  1026→      const parts: string[] = ['Traslado registrado']
  1027→      if (surplusItems.length > 0) parts.push(`${surplusItems.length} ingreso(s) por excedente`)
  1028→      const adjustedCount = deficitItems.filter((r) => r.saldoMode === 'ajustar').length
  1029→      if (adjustedCount > 0) parts.push(`${adjustedCount} salida(s) por ajuste`)
  1030→      toast.success(parts.join(' · '))
  1031→      setTrConfirmOpen(false)
  1032→      setTrDestAlertOpen(false)
  1033→      if (mountedRef.current) { setDetail(null); setTrDestPos(null); loadPosiciones(); if (selectedColumn) loadColumnDetail(selectedColumn) }
  1034→    } catch (err: unknown) { toast.error('Error al trasladar', { description: extractError(err) }) } finally { setBusy(false) }
  1035→  }
