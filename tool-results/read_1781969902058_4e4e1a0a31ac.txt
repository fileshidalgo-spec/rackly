     5→     5→/** Limpia el prefijo del sector del código de subcolumna.
     6→     6→ *  BD guarda "BA1" (prefijo "B" + letra "A" + idx "1"), queremos solo "A1" */
     7→     7→function cleanSubcolCode(codigo: string, columnaLetra: string): string {
     8→     8→  const idx = codigo.indexOf(columnaLetra)
     9→     9→  return idx >= 0 ? codigo.substring(idx) : codigo
    10→    10→}

   294→   294→export async function listarMovimientos(
   295→   295→  sectorId?: string,
   296→   296→  columnaId?: string,
   297→   297→  bloqueId?: string,
   298→   298→  desde?: string,
   299→   299→  hasta?: string
   300→   300→): Promise<MovimientoConDetalles[]> {
   301→   301→  let query = dataClient
   302→   302→    .from('piso_movimientos')
   303→   303→    .select('*')
   304→   304→    .order('fecha', { ascending: false })
   305→   305→    .order('numero_operacion', { ascending: false })
   306→   306→
   307→   307→  if (desde) query = query.gte('fecha', desde)
   308→   308→  if (hasta) query = query.lte('fecha', hasta + 'T23:59:59')
   309→   309→  const PAGE_SIZE = 500
   310→   310→  const MAX_PAGES = 100
   311→   311→  let all: PisoMovimiento[] = []
   312→   312→  let from = 0
   313→   313→  let pages = 0
   314→   314→  while (pages++ < MAX_PAGES) {
   315→   315→    const to = from + PAGE_SIZE - 1
   316→   316→    const { data, error } = await query.range(from, to)
   317→   317→    if (error) throw error
   318→   318→    const rows = (data ?? []) as PisoMovimiento[]
   319→   319→    all.push(...rows)
   320→   320→    if (rows.length < PAGE_SIZE) break
   321→   321→    from += PAGE_SIZE
   322→   322→  }
   323→   323→
   324→   324→  if (all.length === 0) return []
   325→   325→
   326→   326→  const movIds = all.map((m) => m.id)
   327→   327→  const { data: detData, error: detErr } = await dataClient
   328→   328→    .from('piso_movimiento_detalles')
   329→   329→    .select('*')
   330→   330→    .in('movimiento_id', movIds)
   331→   331→  if (detErr) throw detErr
   332→   332→  const detalles = (detData ?? []) as MovimientoDetalle[]
   333→   333→
   334→   334→  const detalleMap = new Map<
   335→   335→    string,
   336→   336→    (MovimientoDetalle & { bloque_codigo?: string; nivel_codigo?: string })[]
   337→   337→  >()
   338→   338→  for (const d of detalles) {
   339→   339→    if (!detalleMap.has(d.movimiento_id)) detalleMap.set(d.movimiento_id, [])
   340→   340→    detalleMap.get(d.movimiento_id)!.push(d)
   341→   341→  }
   342→   342→
   343→   343→  const blockIds = [...new Set(detalles.map((d) => d.bloque_id))]
   344→   344→  const nivelIds = [...new Set(detalles.map((d) => d.nivel_id))]
   345→   345→
   346→   346→  const [bloquesRes, nivelesRes] = await Promise.all([
   347→   347→    blockIds.length > 0
   348→   348→      ? dataClient
   349→   349→          .from('piso_bloques')
   350→   350→          .select('id, codigo')
   351→   351→          .in('id', blockIds)
   352→   352→      : Promise.resolve({ data: [], error: null }),
   353→   353→    nivelIds.length > 0
   354→   354→      ? dataClient
   355→   355→          .from('piso_niveles')
   356→   356→          .select('id, codigo_ubicacion, posicion_id')
   357→   357→          .in('id', nivelIds)
   358→   358→      : Promise.resolve({ data: [], error: null }),
   359→   359→  ])
   360→   360→
   361→   361→  const bloqueMap = new Map<string, string>()
   362→   362→  ;((bloquesRes.data ?? []) as { id: string; codigo: string }[]).forEach(
   363→   363→    (b) => bloqueMap.set(b.id, b.codigo)
   364→   364→  )
   365→   365→  const nivelMap = new Map<string, string | null>()
   366→   366→  const nivelToPos = new Map<string, string>()
   367→   367→  ;(
   368→   368→    (nivelesRes.data ?? []) as { id: string; codigo_ubicacion: string | null; posicion_id: string }[]
   369→   369→  ).forEach((n) => {
   370→   370→    nivelMap.set(n.id, n.codigo_ubicacion)
   371→   371→    nivelToPos.set(n.id, n.posicion_id)
   372→   372→  })
   373→   373→
   374→   374→  // Resolve sector and position for each nivel
   375→   375→  const posIds = [...new Set(nivelToPos.values())]
   376→   376→  let sectorPosMap = new Map<string, { sector_nombre: string; posicion_label: string }>()
   377→   377→  if (posIds.length > 0) {
   378→   378→    const posRes = await dataClient.from('piso_posiciones').select('id, numero, subcolumna_id').in('id', posIds)
   379→   379→    const posMap = new Map<string, { numero: number; subcolumna_id: string }>()
   380→   380→    ;((posRes.data ?? []) as { id: string; numero: number; subcolumna_id: string }[]).forEach(
   381→   381→      (p) => posMap.set(p.id, { numero: p.numero, subcolumna_id: p.subcolumna_id })
   382→   382→    )
   383→   383→    const subIds = [...new Set([...posMap.values()].map((p) => p.subcolumna_id))]
   384→   384→    if (subIds.length > 0) {
   385→   385→      const subRes = await dataClient.from('piso_subcolumnas').select('id, codigo, columna_id').in('id', subIds)
   386→   386→      const subMap = new Map<string, { codigo: string; columna_id: string }>()
   387→   387→      ;((subRes.data ?? []) as { id: string; codigo: string; columna_id: string }[]).forEach(
   388→   388→        (s) => subMap.set(s.id, { codigo: s.codigo, columna_id: s.columna_id })
   389→   389→      )
   390→   390→      const colIds = [...new Set([...subMap.values()].map((s) => s.columna_id))]
   391→   391→      if (colIds.length > 0) {
   392→   392→        const colRes = await dataClient.from('piso_columnas').select('id, letra, sector_id').in('id', colIds)
   393→   393→        const colMap = new Map<string, { letra: string; sector_id: string }>()
   394→   394→        ;((colRes.data ?? []) as { id: string; letra: string; sector_id: string }[]).forEach(
   395→   395→          (c) => colMap.set(c.id, { letra: c.letra, sector_id: c.sector_id })
   396→   396→        )
   397→   397→        const sIds = [...new Set([...colMap.values()].map((c) => c.sector_id))]
   398→   398→        if (sIds.length > 0) {
   399→   399→          const secData = await dataClient.from('piso_sectores').select('id, nombre, n_columnas, n_subcolumnas').in('id', sIds)
   400→   400→          const secInfoMap = new Map<string, { nombre: string; nCol: number; nSub: number }>()
   401→   401→          ;((secData.data ?? []) as { id: string; nombre: string; n_columnas: number; n_subcolumnas: number }[]).forEach(
   402→   402→            (s) => secInfoMap.set(s.id, { nombre: s.nombre, nCol: s.n_columnas, nSub: s.n_subcolumnas })
   403→   403→          )
   404→   404→          // Build final map: nivel_id -> { sector_nombre, posicion_label }
   405→   405→          for (const [nivId, posId] of nivelToPos) {
   406→   406→            const pos = posMap.get(posId)
   407→   407→            if (!pos) continue
   408→   408→            const sub = subMap.get(pos.subcolumna_id)
   409→   409→            if (!sub) continue
   410→   410→            const col = colMap.get(sub.columna_id)
   411→   411→            if (!col) continue
   412→   412→            const secInfo = secInfoMap.get(col.sector_id)
   413→   413→            const secName = secInfo?.nombre ?? ''
   414→   414→            const label = !secInfo
   415→   415→              ? `${col.letra}-${cleanSubcolCode(sub.codigo, col.letra)}-Pos ${pos.numero}`
   416→   416→              : secInfo.nCol === 1 && secInfo.nSub === 1
   417→   417→                ? `Pos ${pos.numero}`
   418→   418→                : secInfo.nSub === 1
   419→   419→                  ? `${col.letra}-Pos ${pos.numero}`
   420→   420→                  : `${col.letra}-${cleanSubcolCode(sub.codigo, col.letra)}-Pos ${pos.numero}`
   421→   421→            sectorPosMap.set(nivId, {
   422→   422→              sector_nombre: secName,
   423→   423→              posicion_label: label,
   424→   424→            })
   425→   425→          }
   426→   426→        }
   427→   427→      }
   428→   428→    }
   429→   429→  }
   430→   430→
   431→   431→  let result = all.map((m) => ({
   432→   432→    ...m,
   433→   433→    detalles: (detalleMap.get(m.id) ?? []).map((d) => {
   434→   434→      const locInfo = sectorPosMap.get(d.nivel_id)
   435→   435→      return {
   436→   436→        ...d,
   437→   437→        bloque_codigo: bloqueMap.get(d.bloque_id),
   438→   438→        nivel_codigo: nivelMap.get(d.nivel_id) ?? undefined,
   439→   439→        sector_nombre: locInfo?.sector_nombre ?? '',
   440→   440→        posicion_label: locInfo?.posicion_label ?? (d.nivel_codigo || ''),
   441→   441→      }
   442→   442→    }),
   443→   443→  }))
   444→   444→
   445→   445→  // Filter by sector/columna/bloque if specified
   446→   446→  if (sectorId || columnaId || bloqueId) {
   447→   447→    if (columnaId) {
   448→   448→      const { data: cols, error: colErr } = await dataClient
   449→   449→        .from('piso_columnas')
   450→   450→        .select('id')
   451→   451→        .eq('sector_id', sectorId ?? columnaId)
   452→   452→      if (!colErr && cols) {
   453→   453→        const colIds = new Set(
   454→   454→          (cols as { id: string }[]).map((c) => c.id)
   455→   455→        )
   456→   456→        if (!colIds.has(columnaId)) {
   457→   457→          return []
   458→   458→        }
   459→   459→      }
   460→   460→    }
   461→   461→  }
   462→   462→

   466→   466→export async function calcularStockNivel(
   467→   467→  nivelId: string
   468→   468→): Promise<{ bloque_codigo: string; cantidad: number }[]> {
   469→   469→  const { data, error } = await dataClient
   470→   470→    .from('piso_movimiento_detalles')
   471→   471→    .select(
   472→   472→      'cantidad, bloque_id, movimiento_id, piso_movimientos!inner(tipo)'
   473→   473→    )
   474→   474→    .eq('nivel_id', nivelId)
   475→   475→  if (error) throw error
   476→   476→
   477→   477→  const bloquesIds = [
   478→   478→    ...new Set((data ?? []).map((d: Record<string, unknown>) => d.bloque_id as string)),
   479→   479→  ] as string[]
   480→   480→  if (bloquesIds.length === 0) return []
   481→   481→
   482→   482→  const bloqueMap = new Map<string, string>()
   483→   483→  // Buscar en piso_bloques (solo IDs reales, no virtuales)
   484→   484→  const realIds = bloquesIds.filter((id) => !id.startsWith('cat_'))
   485→   485→  if (realIds.length > 0) {
   486→   486→    const { data: bloques } = await dataClient
   487→   487→      .from('piso_bloques')
   488→   488→      .select('id, codigo')
   489→   489→      .in('id', realIds)
   490→   490→    ;((bloques ?? []) as { id: string; codigo: string }[]).forEach((b) =>
   491→   491→      bloqueMap.set(b.id, b.codigo)
   492→   492→    )
   493→   493→  }
   494→   494→  // Para IDs virtuales de catálogo, extraer código del ID
   495→   495→  for (const id of bloquesIds) {
   496→   496→    if (id.startsWith('cat_') && !bloqueMap.has(id)) {
   497→   497→      bloqueMap.set(id, id.replace('cat_', ''))
   498→   498→    }
   499→   499→  }
   500→   500→
   501→   501→  const stockMap = new Map<string, number>()
   502→   502→  for (const d of data ?? []) {
   503→   503→    const det = d as unknown as {
   504→   504→      cantidad: number
   505→   505→      bloque_id: string
   506→   506→      movimiento_id: string
   507→   507→      piso_movimientos: { tipo: string }
   508→   508→    }
   509→   509→    const bloque = bloqueMap.get(det.bloque_id) ?? det.bloque_id
   510→   510→    const current = stockMap.get(bloque) ?? 0
   511→   511→    stockMap.set(
   512→   512→      bloque,
   513→   513→      current +
   514→   514→        (det.piso_movimientos.tipo === 'ingreso' ||
   515→   515→        det.piso_movimientos.tipo === 'stock_inicial' ||
   516→   516→        det.piso_movimientos.tipo === 'devolucion'
   517→   517→          ? det.cantidad
   518→   518→          : -det.cantidad)
   519→   519→    )
   520→   520→  }
   521→   521→
   522→   522→  return Array.from(stockMap.entries())
   523→   523→    .filter(([, qty]) => qty > 0)
   524→   524→    .map(([bloque_codigo, cantidad]) => ({ bloque_codigo, cantidad }))
   525→   525→}

   792→   792→
   793→   793→  return result
   794→   794→}

  1532→  1532→    const key = `${d.bloque_id}__${d.nivel_id}`
  1533→  1533→    const arr = blockNivelMap.get(key) ?? []
  1534→  1534→    arr.push({ bloque_id: d.bloque_id, nivel_id: d.nivel_id, tipo, cantidad: qty, fecha_vencimiento: fv, codigo_inc: codigoInc })
  1535→  1535→    blockNivelMap.set(key, arr)
  1536→  1536→  }
  1537→  1537→
  1538→  1538→  // 3. FEFO per (bloque_id, nivel_id)
  1539→  1539→  const fefoResults: { bloque_id: string; nivel_id: string; cantidad: number; fecha_vencimiento: string; codigo_inc: string }[] = []
  1540→  1540→  for (const [key, details] of blockNivelMap) {
  1541→  1541→    const ingresoPool = new Map<string, { cantidad: number; codigo_inc: string }>()
  1542→  1542→    let totalExit = 0
  1543→  1543→    for (const d of details) {
  1544→  1544→      if (isIngresoType(d.tipo)) {
  1545→  1545→        const existing = ingresoPool.get(d.fecha_vencimiento)
  1546→  1546→        if (existing) {
  1547→  1547→          existing.cantidad += d.cantidad
  1548→  1548→          if (!existing.codigo_inc && d.codigo_inc) existing.codigo_inc = d.codigo_inc
  1549→  1549→        } else {
  1550→  1550→          ingresoPool.set(d.fecha_vencimiento, { cantidad: d.cantidad, codigo_inc: d.codigo_inc })
  1551→  1551→        }
  1552→  1552→      } else {
  1553→  1553→        totalExit += d.cantidad
  1554→  1554→      }
  1555→  1555→    }
  1556→  1556→    const sortedLots = [...ingresoPool.entries()].sort(([a], [b]) => {
  1557→  1557→      if (!a && b) return 1
  1558→  1558→      if (a && !b) return -1
  1559→  1559→      return a.localeCompare(b)
  1560→  1560→    })
  1561→  1561→    let pendiente = totalExit
  1562→  1562→    for (const [fecha, pool] of sortedLots) {
  1563→  1563→      if (pendiente <= 0) {
  1564→  1564→        fefoResults.push({ bloque_id: details[0].bloque_id, nivel_id: details[0].nivel_id, cantidad: pool.cantidad, fecha_vencimiento: fecha, codigo_inc: pool.codigo_inc })
  1565→  1565→      } else if (pool.cantidad <= pendiente) {
  1566→  1566→        pendiente -= pool.cantidad
  1567→  1567→      } else {
  1568→  1568→        fefoResults.push({ bloque_id: details[0].bloque_id, nivel_id: details[0].nivel_id, cantidad: pool.cantidad - pendiente, fecha_vencimiento: fecha, codigo_inc: pool.codigo_inc })
  1569→  1569→        pendiente = 0
  1570→  1570→      }
  1571→  1571→    }
  1572→  1572→  }
  1573→  1573→
  1574→  1574→  if (fefoResults.length === 0) return []
  1575→  1575→
  1576→  1576→  // 4. Resolve bloque info
  1577→  1577→  const bloqueIds = [...new Set(fefoResults.map(r => r.bloque_id))]
  1578→  1578→  const bloqueInfoMap = new Map<string, { codigo: string; descripcion: string; unidad: string }>()
  1579→  1579→  const realIds = bloqueIds.filter(id => !id.startsWith('cat_') && !id.startsWith('manual_'))
  1580→  1580→  if (realIds.length > 0) {
  1581→  1581→    const { data: bData } = await dataClient.from('piso_bloques').select('id, codigo, descripcion, unidad').in('id', realIds)
  1582→  1582→    for (const b of (bData ?? []) as { id: string; codigo: string; descripcion: string; unidad: string }[])
  1583→  1583→      bloqueInfoMap.set(b.id, { codigo: b.codigo, descripcion: b.descripcion ?? '', unidad: b.unidad ?? '' })
  1584→  1584→  }
  1585→  1585→  const catIds = bloqueIds.filter(id => id.startsWith('cat_'))
  1586→  1586→  if (catIds.length > 0) {
  1587→  1587→    const codes = catIds.map(id => id.replace('cat_', ''))
  1588→  1588→    const { data: cData } = await dataClient.from('catalogo').select('codigo, descripcion, un').in('codigo', codes)
  1589→  1589→    for (const c of (cData ?? []) as { codigo: string; descripcion: string; un: string }[])
  1590→  1590→      bloqueInfoMap.set(`cat_${c.codigo}`, { codigo: c.codigo, descripcion: c.descripcion ?? '', unidad: c.un ?? '' })
  1591→  1591→  }
  1592→  1592→  for (const id of bloqueIds) {
  1593→  1593→    if (id.startsWith('manual_') && !bloqueInfoMap.has(id))
  1594→  1594→      bloqueInfoMap.set(id, { codigo: id.replace('manual_', ''), descripcion: 'Articulo nuevo', unidad: 'KG' })
  1595→  1595→  }
  1596→  1596→
  1597→  1597→  // 5. Resolve location for each nivel_id
  1598→  1598→  const nivelIds = [...new Set(fefoResults.map(r => r.nivel_id))]
  1599→  1599→  const locMap = new Map<string, { ubicacion: string; sector_nombre: string }>()
  1600→  1600→  if (nivelIds.length > 0) {
  1601→  1601→    const { data: nivData } = await dataClient.from('piso_niveles').select('id, posicion_id').in('id', nivelIds)
  1602→  1602→    const nivToPos = new Map<string, string>()
  1603→  1603→    for (const n of (nivData ?? []) as { id: string; posicion_id: string }[]) nivToPos.set(n.id, n.posicion_id)
  1604→  1604→
  1605→  1605→    const posIds = [...nivToPos.values()]
  1606→  1606→    if (posIds.length > 0) {
  1607→  1607→      const { data: posData } = await dataClient.from('piso_posiciones').select('id, numero, subcolumna_id').in('id', posIds)
  1608→  1608→      const posMap = new Map<string, { numero: number; subcolumna_id: string }>()
  1609→  1609→      for (const p of (posData ?? []) as { id: string; numero: number; subcolumna_id: string }[])
  1610→  1610→        posMap.set(p.id, { numero: p.numero, subcolumna_id: p.subcolumna_id })
  1611→  1611→      const subIds = [...new Set([...posMap.values()].map(p => p.subcolumna_id))]
  1612→  1612→      if (subIds.length > 0) {
  1613→  1613→        const { data: subData } = await dataClient.from('piso_subcolumnas').select('id, codigo, columna_id').in('id', subIds)
  1614→  1614→        const subMap = new Map<string, { codigo: string; columna_id: string }>()
  1615→  1615→        for (const s of (subData ?? []) as { id: string; codigo: string; columna_id: string }[])
  1616→  1616→          subMap.set(s.id, { codigo: s.codigo, columna_id: s.columna_id })
  1617→  1617→        const colIds = [...new Set([...subMap.values()].map(s => s.columna_id))]
  1618→  1618→        if (colIds.length > 0) {
  1619→  1619→          const { data: colData } = await dataClient.from('piso_columnas').select('id, letra, sector_id').in('id', colIds)
  1620→  1620→          const colMap = new Map<string, { letra: string; sector_id: string }>()
  1621→  1621→          for (const c of (colData ?? []) as { id: string; letra: string; sector_id: string }[])
  1622→  1622→            colMap.set(c.id, { letra: c.letra, sector_id: c.sector_id })
  1623→  1623→          const secIds = [...new Set([...colMap.values()].map(c => c.sector_id))]
  1624→  1624→          if (secIds.length > 0) {
  1625→  1625→            const { data: secData } = await dataClient.from('piso_sectores').select('id, nombre, n_columnas, n_subcolumnas').in('id', secIds)
  1626→  1626→            const secMap = new Map<string, { nombre: string; nCol: number; nSub: number }>()
  1627→  1627→            for (const s of (secData ?? []) as { id: string; nombre: string; n_columnas: number; n_subcolumnas: number }[]) secMap.set(s.id, { nombre: s.nombre, nCol: s.n_columnas, nSub: s.n_subcolumnas })
  1628→  1628→            for (const [nivId, posId] of nivToPos) {
  1629→  1629→              const pos = posMap.get(posId)
  1630→  1630→              if (!pos) continue
  1631→  1631→              const sub = subMap.get(pos.subcolumna_id)
  1632→  1632→              if (!sub) continue
  1633→  1633→              const col = colMap.get(sub.columna_id)
  1634→  1634→              if (!col) continue
  1635→  1635→              const secInfo = secMap.get(col.sector_id)
  1636→  1636→              const ubicacion = !secInfo
  1637→  1637→                ? `${col.letra}-${cleanSubcolCode(sub.codigo, col.letra)}-Pos ${pos.numero}`
  1638→  1638→                : secInfo.nCol === 1 && secInfo.nSub === 1
  1639→  1639→                  ? `Pos ${pos.numero}`
  1640→  1640→                  : secInfo.nSub === 1
  1641→  1641→                    ? `${col.letra}-Pos ${pos.numero}`
  1642→  1642→                    : `${col.letra}-${cleanSubcolCode(sub.codigo, col.letra)}-Pos ${pos.numero}`
  1643→  1643→              locMap.set(nivId, {
  1644→  1644→                ubicacion,
  1645→  1645→                sector_nombre: secInfo?.nombre ?? '',
  1646→  1646→              })
  1647→  1647→            }
  1648→  1648→          }
  1649→  1649→        }
  1650→  1650→      }
  1651→  1651→    }
  1652→  1652→  }
  1653→  1653→
  1654→  1654→  // 6. Build results
  1655→  1655→  return fefoResults
  1656→  1656→    .filter(r => r.cantidad > 0)
  1657→  1657→    .map(r => {
  1658→  1658→      const info = bloqueInfoMap.get(r.bloque_id)
  1659→  1659→      const loc = locMap.get(r.nivel_id)
  1660→  1660→      return {
  1661→  1661→        bloque_codigo: info?.codigo ?? r.bloque_id,
  1662→  1662→        bloque_descripcion: info?.descripcion ?? '',
  1663→  1663→        bloque_unidad: info?.unidad ?? 'KG',
  1664→  1664→        ubicacion: loc?.ubicacion ?? '',
  1665→  1665→        sector_nombre: loc?.sector_nombre ?? '',
  1666→  1666→        cantidad: Math.round(r.cantidad * 1000) / 1000,
  1667→  1667→        fecha_vencimiento: r.fecha_vencimiento,
  1668→  1668→        codigo_inc: r.codigo_inc || undefined,
  1669→  1669→      }
  1670→  1670→    })
  1671→  1671→    .sort((a, b) => {
  1672→  1672→      if (a.bloque_codigo !== b.bloque_codigo) return a.bloque_codigo.localeCompare(b.bloque_codigo)
  1673→  1673→      if (a.fecha_vencimiento && b.fecha_vencimiento) return a.fecha_vencimiento.localeCompare(b.fecha_vencimiento)
  1674→  1674→      if (a.fecha_vencimiento && !b.fecha_vencimiento) return -1
  1675→  1675→      if (!a.fecha_vencimiento && b.fecha_vencimiento) return 1
  1676→  1676→      return a.ubicacion.localeCompare(b.ubicacion)
  1677→  1677→    })
  1678→  1678→  } catch (err) {
  1679→  1679→    console.error('[stockPisoGlobal] Error consultando stock de Piso:', err)
  1680→  1680→    return []
  1681→  1681→  }
  1682→  1682→}

  1684→  1684→/**

  1690→  1690→export async function stockPisoPorCodigo(codigo: string): Promise<StockPisoItem[]> {
  1691→  1691→  const code = codigo.trim().toUpperCase()

  1693→  1693→
  1694→  1694→  // ─── TIMEOUT: seguridad contra hang ───
  1695→  1695→  const TIMEOUT_MS = 8000
  1696→  1696→  const result = await Promise.race([
  1697→  1697→    _stockPisoPorCodigoImpl(code),
  1698→  1698→    new Promise<StockPisoItem[]>((resolve) =>
  1699→  1699→      setTimeout(() => {
  1700→  1700→        console.warn('[stockPisoPorCodigo] TIMEOUT', TIMEOUT_MS, 'ms')

  1704→  1704→  ])

  1708→  1708→async function _stockPisoPorCodigoImpl(code: string): Promise<StockPisoItem[]> {
  1709→  1709→  try {
  1710→  1710→    // ─── 1. Buscar bloque por código ───

  1712→  1712→      .from('piso_bloques')

  1723→  1723→
  1724→  1724→    const bloqueIds = bloques.map((b: any) => b.id)
  1725→  1725→    const bloqueInfoMap = new Map<string, { codigo: string; descripcion: string; unidad: string }>()

  1732→  1732→    }
  1733→  1733→
  1734→  1734→    // ─── 2. Detalles de movimiento con JOIN profundo: movimiento + nivel → posicion → subcolumna → columna → sector ───
  1735→  1735→    // Una sola query que resuelve toda la cadena de ubicación

  1737→  1737→      .from('piso_movimiento_detalles')
  1738→  1738→      .select(`

  1740→  1740→        nivel_id,

  1751→  1751→              columna_id,
  1752→  1752→              piso_columnas!inner(

  1754→  1754→                sector_id,
  1755→  1755→                piso_sectores!inner(

  1757→  1757→                  n_columnas,
  1758→  1758→                  n_subcolumnas
  1759→  1759→                )
  1760→  1760→              )
  1761→  1761→            )
  1762→  1762→          )
  1763→  1763→        )
  1764→  1764→      `)
  1765→  1765→      .in('bloque_id', bloqueIds)
  1766→  1766→    if (errD) {
  1767→  1767→      console.warn('[stockPisoPorCodigo] Error detalles:', errD.message)
  1768→  1768→      return []
  1769→  1769→    }
  1770→  1770→    if (!detalles || detalles.length === 0) {
  1771→  1771→      console.log('[stockPisoPorCodigo] Sin movimientos para los bloques')
  1772→  1772→      return []
  1773→  1773→    }
  1774→  1774→
  1775→  1775→    // ─── 3. FEFO calculation ───
  1776→  1776→    type DetRow = {
  1777→  1777→      bloque_id: string
  1778→  1778→      nivel_id: string
  1779→  1779→      cantidad: unknown
  1780→  1780→      fecha_vencimiento?: string | null
  1781→  1781→      piso_movimientos: any

  1783→  1783→        posicion_id: string
  1784→  1784→        piso_posiciones: {
  1785→  1785→          numero: number
  1786→  1786→          subcolumna_id: string
  1787→  1787→          piso_subcolumnas: {
  1788→  1788→            codigo: string
  1789→  1789→            columna_id: string
  1790→  1790→            piso_columnas: {
  1791→  1791→              letra: string
  1792→  1792→              sector_id: string
  1793→  1793→              piso_sectores: {
  1794→  1794→                nombre: string
  1795→  1795→                n_columnas: number
  1796→  1796→                n_subcolumnas: number
  1797→  1797→              }
  1798→  1798→            }
  1799→  1799→          }
  1800→  1800→        }
  1801→  1801→      }
  1802→  1802→    }
  1803→  1803→
  1804→  1804→    const getTipo = (pm: any): string => {
  1805→  1805→      if (!pm) return ''
  1806→  1806→      if (Array.isArray(pm)) return pm.length > 0 ? pm[0].tipo ?? '' : ''
  1807→  1807→      return pm.tipo ?? ''
  1808→  1808→    }
  1809→  1809→    const getCodigoInc = (pm: any): string => {
  1810→  1810→      if (!pm) return ''
  1811→  1811→      if (Array.isArray(pm)) return pm.length > 0 ? pm[0].codigo_inc || '' : ''
  1812→  1812→      return pm.codigo_inc || ''
  1813→  1813→    }
  1814→  1814→    const isIngreso = (t: string) => t === 'ingreso' || t === 'stock_inicial' || t === 'devolucion'
  1815→  1815→
  1816→  1816→    // Group by (bloque_id, nivel_id)
  1817→  1817→    const groups = new Map<string, {
  1818→  1818→      bloque_id: string
  1819→  1819→      nivel_id: string
  1820→  1820→      tipo: string
  1821→  1821→      cantidad: number
  1822→  1822→      fecha_vencimiento: string
  1823→  1823→      codigo_inc: string

  1825→  1825→    }[]>()
  1826→  1826→
  1827→  1827→    for (const d of detalles as DetRow[]) {
  1828→  1828→      const qty = typeof d.cantidad === 'number' ? d.cantidad : parseFloat(String(d.cantidad ?? '0')) || 0
  1829→  1829→      if (qty <= 0) continue
  1830→  1830→      const tipo = getTipo(d.piso_movimientos)
  1831→  1831→      if (!tipo) continue
  1832→  1832→      const fv = (typeof d.fecha_vencimiento === 'string' && d.fecha_vencimiento) ? d.fecha_vencimiento : ''
  1833→  1833→      const key = `${d.bloque_id}__${d.nivel_id}`
  1834→  1834→      const arr = groups.get(key) ?? []
  1835→  1835→      arr.push({
  1836→  1836→        bloque_id: d.bloque_id,
  1837→  1837→        nivel_id: d.nivel_id,
  1838→  1838→        tipo,
  1839→  1839→        cantidad: qty,
  1840→  1840→        fecha_vencimiento: fv,
  1841→  1841→        codigo_inc: getCodigoInc(d.piso_movimientos),
  1842→  1842→        loc: d.piso_niveles,
  1843→  1843→      })
  1844→  1844→      groups.set(key, arr)
  1845→  1845→    }
  1846→  1846→
  1847→  1847→    // FEFO per group
  1848→  1848→    const results: StockPisoItem[] = []
  1849→  1849→    for (const [, entries] of groups) {
  1850→  1850→      const ingresoPool = new Map<string, { cantidad: number; codigo_inc: string }>()
  1851→  1851→      let totalExit = 0
  1852→  1852→      for (const e of entries) {
  1853→  1853→        if (isIngreso(e.tipo)) {
  1854→  1854→          const existing = ingresoPool.get(e.fecha_vencimiento)
  1855→  1855→          if (existing) {
  1856→  1856→            existing.cantidad += e.cantidad
  1857→  1857→            if (!existing.codigo_inc && e.codigo_inc) existing.codigo_inc = e.codigo_inc
  1858→  1858→          } else {
  1859→  1859→            ingresoPool.set(e.fecha_vencimiento, { cantidad: e.cantidad, codigo_inc: e.codigo_inc })
  1860→  1860→          }
  1861→  1861→        } else {
  1862→  1862→          totalExit += e.cantidad
  1863→  1863→        }
  1864→  1864→      }
  1865→  1865→      const sortedLots = [...ingresoPool.entries()].sort(([a], [b]) => {
  1866→  1866→        if (!a && b) return 1
  1867→  1867→        if (a && !b) return -1
  1868→  1868→        return a.localeCompare(b)
  1869→  1869→      })
  1870→  1870→      let pendiente = totalExit
  1871→  1871→      for (const [fecha, pool] of sortedLots) {
  1872→  1872→        if (pendiente <= 0) {
  1873→  1873→          // Lote completo disponible
  1874→  1874→          const e = entries[0]
  1875→  1875→          results.push({
  1876→  1876→            bloque_codigo: bloqueInfoMap.get(e.bloque_id)?.codigo ?? e.bloque_id,
  1877→  1877→            bloque_descripcion: bloqueInfoMap.get(e.bloque_id)?.descripcion ?? '',
