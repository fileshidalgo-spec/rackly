    59→    59→// Calcula días restantes hasta vencimiento (negativo si ya venció)
    60→    60→function diasParaVencer(fVencimiento: string | undefined): number | null {
    61→    61→  if (!fVencimiento) return null
    62→    62→  const hoy = new Date(); hoy.setHours(0, 0, 0, 0)
    63→    63→  const venc = new Date(fVencimiento + 'T00:00:00')
    64→    64→  return Math.ceil((venc.getTime() - hoy.getTime()) / (1000 * 60 * 60 * 24))
    65→    65→}

    67→    67→// Detecta códigos duplicados (mismo artículo con distintas fechas de vencimiento)
    68→    68→function codigosConMultiplesLotes(stock: StockEnUbicacion[]): Set<string> {
    69→    69→  const counts = new Map<string, number>()
    70→    70→  for (const s of stock) {
    71→    71→    counts.set(s.codigo, (counts.get(s.codigo) ?? 0) + 1)
    72→    72→  }
    73→    73→  const multiples = new Set<string>()
    74→    74→  for (const [code, count] of counts) {
    75→    75→    if (count > 1) multiples.add(code)
    76→    76→  }
    77→    77→  return multiples
    78→    78→}

    80→    80→// Calcula ocupación desde movimientos — stock POR CÓDIGO (no por lote).
    81→    81→// f_vencimiento es SOLO para FEFO (ordenamiento), NO para particionar stock.
    82→    82→// IMPORTANTE: Los movimientos INC se EXCLUYEN del stock normal.
    83→    83→// La detección INC se hace por separado con fetchIncPorUbicacion().
    84→    84→function calcularOcupacion(movs: Movimiento[]): OcupacionCelda[] {
    85→    85→  // Mapa: ubicacion_key → (codigo → stock)
    86→    86→  const cellMap = new Map<string, Map<string, number>>()
    87→    87→  for (const m of movs) {
    88→    88→    // EXCLUIR movimientos INC del cálculo de ocupación normal
    89→    89→    if (m.codigoInc) continue
    90→    90→    const key = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}`
    91→    91→    const code = m.codigo.trim().toUpperCase()
    92→    92→    let codeMap = cellMap.get(key)
    93→    93→    if (!codeMap) { codeMap = new Map(); cellMap.set(key, codeMap) }
    94→    94→    const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? m.cantidad : -m.cantidad
    95→    95→    const current = codeMap.get(code) ?? 0
    96→    96→    codeMap.set(code, current + delta)
    97→    97→  }
    98→    98→  // Construir resultado: solo celdas con stock total > 0
    99→    99→  const result: OcupacionCelda[] = []
   100→   100→  for (const [key, codeMap] of cellMap) {
   101→   101→    let totalStock = 0
   102→   102→    const codigos = new Set<string>()
   103→   103→    for (const [code, stock] of codeMap) {
   104→   104→      if (stock > 0) {
   105→   105→        totalStock += stock
   106→   106→        codigos.add(code)
   107→   107→      }
   108→   108→    }
   109→   109→    if (totalStock > 0) {
   110→   110→      const [bloque, torre, piso, posicion] = key.split('-')
   111→   111→      // lotes = número de códigos distintos (ya no se cuenta por f_vencimiento)
   112→   112→      result.push({ bloque, torre, piso, posicion, stock: totalStock, codigos: Array.from(codigos), lotes: codigos.size, tieneInc: false, incItems: [] })
   113→   113→    }
   114→   114→  }
   115→   115→  return result
   116→   116→}

   373→   373→  async function doIngreso() {
   374→   374→    if (!detail || !perfil) return
   375→   375→    if (!ingCodigo.trim() || !ingCantidad) { toast.error('Completa código y cantidad'); return }
   376→   376→    const q = parseFloat(ingCantidad); if (isNaN(q) || q <= 0) { toast.error('Cantidad inválida'); return }
   377→   377→    if (showProveedor && !ingProveedor) { toast.error('Selecciona un proveedor para este artículo'); return }
   378→   378→    setActionBusy(true)
   379→   379→    try {
   380→   380→      const { wasOffline } = await SyncEngine.offlineAwareAddMovimiento({
   381→   381→        tipo: ingTipo, bloque: detail.bloque, torre: detail.torre, piso: detail.piso, posicion: detail.posicion,
   382→   382→        codigo: ingCodigo.trim().toUpperCase(), descripcion: ingDescripcion, un: ingUn, cantidad: q,
   383→   383→        fVencimiento: ingSinFecha ? '' : ingFVenc,
