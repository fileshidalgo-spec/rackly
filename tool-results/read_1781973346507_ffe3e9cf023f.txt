    62→type Props = {
    63→  tipo: TipoMovimiento | 'inc' | 'salida-inc'
    64→  onCreated: (movs: Movimiento[]) => void
    65→}
    66→
    67→export function MovimientoForm({ tipo, onCreated }: Props) {
    68→  const { perfil } = useAuth()
    69→  const [turno, setTurno] = useState<Turno>(calcularTurno())
    70→
    71→  useEffect(() => {
    72→    const interval = setInterval(() => setTurno(calcularTurno()), 60000)
    73→    return () => clearInterval(interval)
    74→  }, [])
    75→
    76→  if (!perfil) {
    77→    return <div className='p-4 text-muted-foreground animate-pulse'>Cargando usuario...</div>
    78→  }
    79→  if (tipo === 'salida') {
    80→    return <SalidaForm turno={turno} onCreated={onCreated} perfil={perfil} />
    81→  }
    82→  if (tipo === 'salida-inc') {
    83→    return <SalidaIncForm turno={turno} onCreated={onCreated} perfil={perfil} />
    84→  }
    85→  if (tipo === 'inc') {
    86→    return <IncForm turno={turno} onCreated={onCreated} perfil={perfil} />
    87→  }
    88→  // ingreso y devolucion usan el mismo formulario, solo cambia el tipo
    89→  return <IngresoForm turno={turno} onCreated={onCreated} perfil={perfil} tipo={tipo} />
    90→}

    95→function IngresoForm({
    96→  turno,
    97→  onCreated,
    98→  perfil,
    99→  tipo = 'ingreso',
   100→}: {
   101→  turno: Turno
   102→  onCreated: (movs: Movimiento[]) => void
   103→  perfil: { id: string; nombre: string; correo: string }
   104→  tipo?: TipoMovimiento
   105→}) {
   106→  const [bloque, setBloque] = useState('')
   107→  const [torre, setTorre] = useState('')
   108→  const [piso, setPiso] = useState('')
   109→  const [posicion, setPosicion] = useState('')
   110→  const [codigo, setCodigo] = useState('')
   111→  const [descripcion, setDescripcion] = useState('')
   112→  const [un, setUn] = useState('')
   113→  const [cantidad, setCantidad] = useState('')
   114→  const [fVencimiento, setFVencimiento] = useState('')
   115→  const [sinVencimiento, setSinVencimiento] = useState(false)
   116→  const [proveedor, setProveedor] = useState('')
   117→  const [busy, setBusy] = useState(false)
   118→  const [confirmData, setConfirmData] = useState<StockEnUbicacion[] | null>(null)
   119→  const [salidaBusy, setSalidaBusy] = useState<string | null>(null)
   120→
   121→  const torres = torresDeBloque(bloque)
   122→  const posiciones = posicionesDeBloque(bloque)
   123→
   124→  function handleCatalogoPick(item: CatalogoItem) {
   125→    setCodigo(item.codigo)
   126→    setDescripcion(item.descripcion)
   127→    setUn(item.un)
   128→  }
   129→
   130→  async function handleSubmit(e: React.FormEvent) {
   131→    e.preventDefault()
   132→    if (!bloque || !torre || !piso || !posicion || !codigo.trim() || !cantidad) {
   133→      toast.error('Completa todos los campos requeridos')
   134→      return
   135→    }
   136→    if (requiereProveedor(descripcion) && !proveedor) {
   137→      toast.error('Selecciona un proveedor para este producto')
   138→      return
   139→    }
   140→    const qty = parseFloat(cantidad)
   141→    if (isNaN(qty) || qty <= 0) {
   142→      toast.error('La cantidad debe ser mayor a 0')
   143→      return
   144→    }
   145→    setBusy(true)
   146→    try {
   147→      // Verificar si CUALQUIER producto existe en esta posición
   148→      const details = await stockEnUbicacion(bloque, torre, piso, posicion)
   149→      if (details.length > 0) {
   150→        setConfirmData(details)
   151→        setBusy(false)
   152→        return
   153→      }
   154→      await doInsert(qty)
   155→    } catch (err: unknown) {
   156→      const message = extractError(err)
   157→      toast.error('Error al verificar stock', { description: message })
   158→      setBusy(false)
   159→    }
   160→  }
   161→
   162→  async function doInsert(qty: number) {
   163→    try {
   164→      const { movs, wasOffline } = await SyncEngine.offlineAwareAddMovimiento({
   165→        tipo,
   166→        bloque,
   167→        torre,
   168→        piso,
   169→        posicion,
   170→        codigo,
   171→        descripcion,
   172→        un,
   173→        cantidad: qty,
   174→        fVencimiento: sinVencimiento ? '' : fVencimiento,
   175→        turno,
   176→        usuarioId: perfil.id,
   177→        usuarioNombre: perfil.nombre,
   178→        usuarioCorreo: perfil.correo,
   179→        proveedor: proveedor || undefined,
   180→      })
   181→      if (wasOffline) {
   182→        toast.success(tipo === 'devolucion' ? 'Devolución guardada (offline)' : 'Ingreso guardado (offline)', {
   183→          description: 'Se sincronizará al reconectarse',
   184→          duration: 5000,
   185→        })
   186→        // OFFLINE: Limpiar formulario para evitar duplicados
   187→        setCodigo('')
   188→        setDescripcion('')
   189→        setUn('')
   190→        setCantidad('')
   191→        setFVencimiento('')
   192→        setProveedor('')
   193→        setConfirmData(null)
   194→      } else {
   195→        toast.success(tipo === 'devolucion' ? 'Devolución registrada' : 'Ingreso registrado')
   196→        setCodigo('')
   197→        setDescripcion('')
   198→        setUn('')
   199→        setCantidad('')
   200→        setFVencimiento('')
   201→        setProveedor('')
   202→        onCreated(movs)
   203→      }
   204→    } catch (err: unknown) {
   205→      const message = extractError(err)
   206→      toast.error('Error al registrar ingreso', { description: message })
   207→    } finally {
   208→      setBusy(false)
   209→      setConfirmData(null)
   210→    }
   211→  }
   212→
   213→  // Dar salida a un producto desde el alerta de ubicación ocupada
   214→  async function handleSalidaDesdeAlerta(stockItem: StockEnUbicacion) {
   215→    setSalidaBusy(stockItem.codigo)
   216→    try {
   217→      const { movs, wasOffline } = await SyncEngine.offlineAwareAddMovimiento({
   218→        tipo: 'salida',
   219→        bloque,
   220→        torre,
   221→        piso,
   222→        posicion,
   223→        codigo: stockItem.codigo,
   224→        descripcion: stockItem.descripcion,
   225→        un: stockItem.un,
   226→        cantidad: stockItem.stock,
   227→        fVencimiento: stockItem.fVencimiento ?? '',
   228→        turno,
   229→        usuarioId: perfil.id,
   230→        usuarioNombre: perfil.nombre,
   231→        usuarioCorreo: perfil.correo,
   232→        proveedor: stockItem.proveedor,
   233→        // PRESERVAR codigoInc para que la salida descuente del stock INC correctamente
   234→        codigoInc: stockItem.codigoInc || undefined,
   235→      })
   236→      if (wasOffline) {
   237→        toast.success(`Salida de ${stockItem.stock} ${stockItem.un} de ${stockItem.codigo} (offline)`, {
   238→          description: 'Se sincronizará al reconectarse. La ubicación se actualizará cuando haya conexión.',
   239→          duration: 5000,
   240→        })
   241→        // OFFLINE: Limpiar UI para evitar doble operación
   242→        setConfirmData(null)
   243→        setSalidaBusy(null)
   244→      } else {
   245→        toast.success(`Salida de ${stockItem.stock} ${stockItem.un} de ${stockItem.codigo}`)
   246→        onCreated(movs)
   247→        // Refrescar datos del alerta
   248→        const updated = await stockEnUbicacion(bloque, torre, piso, posicion)
   249→        if (updated.length > 0) {
   250→          setConfirmData(updated)
   251→        } else {
   252→          setConfirmData(null)
   253→          // Ya no hay productos, hacer el ingreso directamente
   254→          await doInsert(parseFloat(cantidad))
   255→        }
   256→      }
   257→    } catch (err: unknown) {
   258→      if (isInsufficientStockError(err)) {
   259→        const detail = (err as Record<string, string>).detail || ''
   260→        toast.error('Stock insuficiente', {
   261→          description: detail || 'Otro usuario pudo haber modificado el stock mientras tú operabas.',
   262→          duration: 8000,
   263→        })
   264→      } else {
   265→        const message = extractError(err)
   266→        toast.error('Error al dar salida', { description: message })
   267→      }
   268→    } finally {
   269→      setSalidaBusy(null)
   270→    }
   271→  }
   272→
   273→  return (
   274→    <>
   275→      <div className={`flex items-center gap-2 mb-4 p-3 rounded-lg border ${
   276→        tipo === 'devolucion'
   277→          ? 'bg-orange-50 dark:bg-orange-950/40 border-orange-200 dark:border-orange-800'
   278→          : 'bg-green-50 dark:bg-green-950/40 border-green-200 dark:border-green-800'
   279→      }`}>
   280→        {tipo === 'devolucion' ? (
   281→          <ArrowRightLeft className="h-5 w-5 text-orange-600 dark:text-orange-400 flex-shrink-0" />
   282→        ) : (
   283→          <ArrowDownToLine className="h-5 w-5 text-green-600 dark:text-green-400 flex-shrink-0" />
   284→        )}
   285→        <div className="min-w-0">
   286→          <p className={`text-sm font-semibold ${
   287→            tipo === 'devolucion'
   288→              ? 'text-orange-700 dark:text-orange-300'
   289→              : 'text-green-700 dark:text-green-300'
   290→          }`}>
   291→            {tipo === 'devolucion' ? 'Ingreso por devolución' : 'Ingreso de mercadería'}
   292→          </p>
   293→          <p className="text-xs text-muted-foreground">Turno: {turno}</p>
   294→        </div>
   295→      </div>
   296→
   297→      <form onSubmit={handleSubmit} className="space-y-4">
   298→        <div className="space-y-1">
   299→          <Label className="text-xs font-medium text-muted-foreground uppercase tracking-wider">Ubicación</Label>
   300→          <div className="grid grid-cols-2 sm:grid-cols-5 gap-2">
   301→            <div className="space-y-1">
   302→              <Label className="text-xs text-muted-foreground">Bloque</Label>
   303→              <Select value={bloque} onValueChange={(v) => { setBloque(v); setTorre(''); setPiso(''); setPosicion('') }}>
   304→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Bloque" /></SelectTrigger>
   305→                <SelectContent>
   306→                  {BLOQUES.map((b) => <SelectItem key={b} value={b}>{b}</SelectItem>)}
   307→                </SelectContent>
   308→              </Select>
   309→            </div>
   310→            <div className="space-y-1">
   311→              <Label className="text-xs text-muted-foreground">Torre</Label>
   312→              <Select value={torre} onValueChange={setTorre} disabled={!bloque}>
   313→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Torre" /></SelectTrigger>
   314→                <SelectContent>
   315→                  {torres.map((t) => <SelectItem key={t} value={t}>{t}</SelectItem>)}
   316→                </SelectContent>
   317→              </Select>
   318→            </div>
   319→            <div className="space-y-1">
   320→              <Label className="text-xs text-muted-foreground">Piso</Label>
   321→              <Select value={piso} onValueChange={setPiso} disabled={!bloque}>
   322→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Piso" /></SelectTrigger>
   323→                <SelectContent>
   324→                  {PISOS.map((p) => <SelectItem key={p} value={p}>{p}</SelectItem>)}
   325→                </SelectContent>
   326→              </Select>
   327→            </div>
   328→            <div className="space-y-1">
   329→              <Label className="text-xs text-muted-foreground">Posición</Label>
   330→              <Select value={posicion} onValueChange={setPosicion} disabled={!bloque}>
   331→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Pos." /></SelectTrigger>
   332→                <SelectContent>
   333→                  {posiciones.map((p) => <SelectItem key={p} value={p}>{p}</SelectItem>)}
   334→                </SelectContent>
   335→              </Select>
   336→            </div>
   337→            <div className="space-y-1 col-span-2 sm:col-span-1">
   338→              <Label className="text-xs text-muted-foreground">Turno</Label>
   339→              <Input value={turno} readOnly className="h-10 bg-muted text-sm" />
   340→            </div>
   341→          </div>
   342→        </div>
   343→
   344→        <div className="space-y-1">
   345→          <Label className="text-xs font-medium text-muted-foreground uppercase tracking-wider">Producto</Label>
   346→          <CatalogoSearchInput
   347→            onPick={handleCatalogoPick}
   348→            value={codigo}
   349→            onChange={(v) => {
   350→              setCodigo(v)
   351→              const cat = findCatalogoByCodigo(v)
   352→              if (cat) {
   353→                setDescripcion(cat.descripcion)
   354→                setUn(cat.un)
   355→              }
   356→            }}
   357→          />
   358→        </div>
   359→
   360→        {descripcion && (
   361→          <div className="rounded-lg border border-green-200 dark:border-green-800 bg-green-50/50 dark:bg-green-950/20 p-3 space-y-2 transition-all">
   362→            <div className="flex items-start gap-3 flex-wrap">
   363→              <div className="min-w-0 flex-1">
   364→                <p className="text-xs text-muted-foreground">Descripción</p>
   365→                <p className="text-sm font-medium truncate">{descripcion}</p>
   366→              </div>
   367→              <div className="min-w-[60px]">
   368→                <p className="text-xs text-muted-foreground">UN</p>
   369→                <p className="text-sm font-medium">{un || '—'}</p>
   370→              </div>
   371→            </div>
   372→          </div>
   373→        )}
   374→
   375→        <div className="grid grid-cols-2 sm:grid-cols-4 gap-2">
   376→          <div className="space-y-1 col-span-2 sm:col-span-1">
   377→            <Label className="text-xs text-muted-foreground">Cantidad</Label>
   378→            <Input type="number" step="any" min="0.001" value={cantidad} onChange={(e) => setCantidad(e.target.value)} placeholder="0" className="h-10" />
   379→          </div>
   380→          <div className="space-y-1 col-span-2 sm:col-span-1">
   381→            <Label className="text-xs text-muted-foreground">F. Vencimiento</Label>
   382→            <div className="flex items-center gap-1.5">
   383→              <Input type="date" value={fVencimiento} onChange={(e) => setFVencimiento(e.target.value)} disabled={sinVencimiento} className="h-10 [color-scheme:dark]" />
   384→              <Checkbox checked={sinVencimiento} onCheckedChange={(v) => setSinVencimiento(!!v)} />
   385→            </div>
   386→          </div>
   387→          {requiereProveedor(descripcion) && (
   388→            <div className="space-y-1 col-span-2">
   389→              <Label className="text-xs text-muted-foreground font-medium">Proveedor <span className="text-red-500">*</span></Label>
   390→              <Select value={proveedor} onValueChange={setProveedor}>
   391→                <SelectTrigger className="h-10">
   392→                  <SelectValue placeholder="Selecciona proveedor" />
   393→                </SelectTrigger>
   394→                <SelectContent>
   395→                  {PROVEEDORES_FILM.map((p) => (
   396→                    <SelectItem key={p} value={p}>{p}</SelectItem>
   397→                  ))}
   398→                </SelectContent>
   399→              </Select>
   400→            </div>
   401→          )}
   402→        </div>
   403→
   404→        <Button type="submit" disabled={busy} className={`w-full h-11 text-white text-sm font-medium sm:w-auto ${
   405→          tipo === 'devolucion'
   406→            ? 'bg-orange-600 hover:bg-orange-700'
   407→            : 'bg-green-600 hover:bg-green-700'
   408→        }`}>
   409→          {busy && <Loader2 className="h-4 w-4 animate-spin mr-2" />}
   410→          {tipo === 'devolucion' ? 'Registrar Devolución' : 'Registrar Ingreso'}
   411→        </Button>
   412→      </form>
   413→
   414→      {/* ═══ DIÁLOGO DE CONFIRMACIÓN — Ubicación Ocupada ═══ */}
   415→      <AlertDialog open={!!confirmData} onOpenChange={() => setConfirmData(null)}>
   416→        <AlertDialogContent className="max-w-[calc(100vw-1rem)] max-w-lg p-0 max-h-[85vh] flex flex-col overflow-hidden">
   417→          {/* Header con gradiente */}
   418→          <div className="bg-gradient-to-r from-amber-500 via-orange-500 to-red-500 px-4 sm:px-6 py-5 text-white shrink-0">
   419→            <div className="flex items-center gap-3">
   420→              <div className="w-11 h-11 rounded-full bg-white/20 backdrop-blur-sm flex items-center justify-center flex-shrink-0">
   421→                <TriangleAlert className="h-6 w-6 text-white" />
   422→              </div>
   423→              <div>
   424→                <AlertDialogTitle className="text-lg font-bold text-white m-0">
   425→                  Ubicación Ocupada
   426→                </AlertDialogTitle>
   427→                <AlertDialogDescription className="text-amber-100 text-sm mt-0.5">
   428→                  Esta posición ya tiene stock registrado. Revisa el detalle.
   429→                </AlertDialogDescription>
   430→              </div>
   431→            </div>
   432→          </div>
   433→
   434→          {/* Contenido scrolleable */}
   435→          <div className="flex-1 min-h-0 overflow-y-auto overscroll-contain">
   436→            {/* Ubicación destino */}
   437→            <div className="px-4 sm:px-6 pt-4 pb-2">
   438→              <p className="text-xs font-semibold text-muted-foreground uppercase tracking-wider mb-1.5">Ubicación destino</p>
   439→              <div className="flex items-center gap-2 text-sm text-muted-foreground bg-slate-50 dark:bg-slate-900 rounded-lg px-3 py-2.5">
   440→                <MapPin className="h-4 w-4 text-orange-500 flex-shrink-0" />
   441→                <span className="font-mono font-medium text-slate-700 dark:text-slate-300">
   442→                  Bloque {bloque} · Torre {torre} · Piso {piso} · Pos {posicion}
   443→                </span>
   444→              </div>
   445→            </div>
   446→
   447→            {/* Producto que se desea ingresar */}
   448→            <div className="px-4 sm:px-6 pb-3">
   449→              <p className="text-xs font-semibold text-muted-foreground uppercase tracking-wider mb-1.5">Producto a {tipo === 'devolucion' ? 'devolver' : 'ingresar'}</p>
   450→              <div className="rounded-xl border-2 border-dashed border-green-300 dark:border-green-700 bg-green-50/60 dark:bg-green-950/20 p-3 space-y-1.5">
   451→                <div className="flex items-center justify-between">
   452→                  <div className="flex items-center gap-2 min-w-0">
   453→                    <span className="font-mono font-bold text-sm text-green-800 dark:text-green-300">{codigo}</span>
   454→                    <span className="text-[10px] text-green-700/60 dark:text-green-400/60 bg-green-100 dark:bg-green-900/40 px-1.5 py-0.5 rounded">{un || '—'}</span>
   455→                  </div>
   456→                  <div className="flex items-center gap-1 flex-shrink-0">
   457→                    <ArrowDownToLine className="h-3.5 w-3.5 text-green-600" />
   458→                    <span className="font-bold text-green-700 dark:text-green-300">+{cantidad}</span>
   459→                  </div>
   460→                </div>
   461→                {descripcion && (
   462→                  <p className="text-xs text-green-700/70 dark:text-green-400/70 truncate">{descripcion}</p>
   463→                )}
   464→                {fVencimiento && !sinVencimiento && (
   465→                  <p className="text-[10px] text-muted-foreground">Venc: {fVencimiento}</p>
   466→                )}
   467→              </div>
   468→            </div>
   469→
   470→            {/* Separador */}
   471→            {confirmData && confirmData.length > 0 && (
   472→              <div className="px-4 sm:px-6 pb-2">
   473→                <div className="flex items-center gap-2">
   474→                  <div className="flex-1 h-px bg-gradient-to-r from-transparent via-slate-300 dark:via-slate-600 to-transparent" />
   475→                  <span className="text-[10px] font-semibold text-muted-foreground uppercase tracking-widest">Stock actual en esta ubicación</span>

   585→  function toggleSelect(key: string) {
   586→    setSelected((prev) => {
   587→      const next = new Set(prev)
   588→      if (next.has(key)) next.delete(key)
   589→      else next.add(key)

   591→    })
   592→  }

   598→    }
   599→  }
   600→
   601→  // Reset selección al cambiar búsqueda

   605→

   610→  // ─── Salida en masa ───
   611→  const [massBusy, setMassBusy] = useState(false)

   616→  qtyMapRef.current = qtyMap

   620→  function openMassConfirm() {

   624→    }

   628→  async function doMassSalida() {
   629→    setMassBusy(true)
   630→    const totalToProcess = selectedRef.current.size
   631→    let totalProcessed = 0
   632→    let totalErrors = 0
   633→    let stockErrors = 0
   634→    const errorDetails: string[] = []
   635→    try {
   636→      // Usar los values actuales al momento de iniciar (no refs que cambian durante el loop)
   637→      const selectedKeys = Array.from(selectedRef.current)
   638→      const locsSnap = [...locationsRef.current]
   639→      const qtySnap = { ...qtyMapRef.current }
   640→
   641→      for (const key of selectedKeys) {
   642→        const loc = locsSnap.find((l) => `${l.bloque}-${l.torre}-${l.piso}-${l.posicion}` === key)
   643→        if (!loc) {
   644→          errorDetails.push(`Ubicación ${key} no encontrada`)
   645→          totalErrors++
   646→          continue
   647→        }
   648→        const qtyVal = qtySnap[key] || ''
   649→        const qtyNum = qtyVal ? parseFloat(qtyVal.trim()) : loc.stock
   650→        if (isNaN(qtyNum) || qtyNum <= 0) {
   651→          errorDetails.push(`Cantidad inválida en B${loc.bloque}/T${loc.torre}/P${loc.piso}/Pos${loc.posicion}`)
   652→          totalErrors++
   653→          continue
   654→        }
   655→        const finalQty = Math.min(qtyNum, loc.stock)
   656→        try {
   657→          const { wasOffline } = await SyncEngine.offlineAwareAddMovimiento({
   658→            tipo: 'salida',
   659→            bloque: loc.bloque, torre: loc.torre, piso: loc.piso, posicion: loc.posicion,
   660→            codigo: loc.codigo, descripcion: loc.descripcion, un: loc.un,
   661→            cantidad: finalQty,
   662→            fVencimiento: loc.fVencimiento ?? '', turno,
   663→            usuarioId: perfil.id, usuarioNombre: perfil.nombre, usuarioCorreo: perfil.correo,
   664→            proveedor: loc.proveedor,
   665→            // PRESERVAR codigoInc para consistencia (aunque SalidaForm excluye INC)
   666→            codigoInc: loc.codigoInc || undefined,
   667→          })
   668→          totalProcessed++
   669→        } catch (err) {
   670→          if (isInsufficientStockError(err)) {
   671→            stockErrors++
   672→            const detail = (err as Record<string, string>).detail || ''
   673→            errorDetails.push(`Stock insuficiente en B${loc.bloque}/T${loc.torre}/P${loc.piso}/Pos${loc.posicion}${detail ? ` (${detail})` : ''}`)
   674→          } else {
   675→            const msg = extractError(err)
   676→            errorDetails.push(`Error en B${loc.bloque}/T${loc.torre}/P${loc.piso}/Pos${loc.posicion}: ${msg}`)
   677→          }
   678→          totalErrors++
   679→        }
   680→      }
   681→      if (totalErrors > 0) {
   682→        const stockNote = stockErrors > 0 ? ` (${stockErrors} por stock insuficiente)` : ''
   683→        const desc = totalProcessed > 0
   684→          ? `${totalProcessed} de ${totalToProcess} correctas${stockNote}:\n${errorDetails.slice(0, 5).join('\n')}${errorDetails.length > 5 ? `\n...y ${errorDetails.length - 5} más` : ''}`
   685→          : `Errores${stockNote}:\n${errorDetails.join('\n')}`
   686→        toast.warning(`Salida completada con ${totalErrors} error${totalErrors > 1 ? 'es' : ''}`, { description: desc, duration: 10000 })
   687→      } else {
   688→        toast.success(`Salida registrada en ${totalProcessed} ubicacion${totalProcessed > 1 ? 'es' : ''}`)
   689→      }
   690→      setMassConfirmOpen(false)
   691→      setSelected(new Set())
   692→      setQtyMap({})
   693→      // Si hubo operaciones offline, limpiar UI para evitar doble salida
   694→      if (totalProcessed > 0 && SyncEngine.isOffline()) {
   695→        setLocations([])
   696→        setProductoDesc('')
   697→        setProductoUn('')
   698→      } else {
   699→        // Refrescar datos reales cuando estamos online
   700→        await refreshLocations()
   701→      }

   703→    } catch (err: unknown) {
   704→      const message = extractError(err)

   706→    } finally {
   707→      setMassBusy(false)
   708→    }
   709→  }
   710→
   711→  const allSelected = locations.length > 0 && selected.size === locations.length
   712→
   713→  // Función central para refrescar las ubicaciones
   714→  const refreshLocations = useCallback(async () => {
   715→    const code = searchCodeRef.current.trim()
   716→    if (!code) {
   717→      setLocations([])
   718→      setProductoDesc('')
   719→      setProductoUn('')
   720→      setQtyMap({})
   721→      return
   722→    }
   723→    try {
   724→      let movs: Array<{ tipo: string; bloque: string; torre: string; piso: string; posicion: string; codigo: string; descripcion: string; un: string; cantidad: number; fVencimiento?: string; proveedor?: string; codigoInc?: string }>
   725→      try {
   726→        // Intentar obtener del servidor - consulta optimizada por código
   727→        const { fetchMovimientosByCodigo } = await import('@/lib/rackly/kardex')
   728→        movs = await fetchMovimientosByCodigo(code)
   729→      } catch {
   730→        // Fallback: usar movimientos cacheados en IndexedDB y filtrar por código
   731→        const cached = await SyncEngine.getCachedMovimientosForStock()
   732→        movs = cached.filter((m) => m.codigo === code.toUpperCase())
   733→      }
   734→      const upperCode = code.toUpperCase()
   735→      const locMap = new Map<string, LocWithKey>()
   736→      // EXCLUIR movimientos INC de la vista de salidas normales
   737→      const relevant = movs.filter((m) => m.codigo === upperCode && !m.codigoInc)
   738→      let desc = ''
   739→      let un = ''
   740→      for (const m of relevant) {
   741→        if (!desc && m.descripcion) desc = m.descripcion
   742→        if (!un && m.un) un = m.un
   743→        const impact = impactoStock(m.tipo, m.cantidad)
   744→        if (isNaN(impact) || !isFinite(impact)) continue
   745→        // Agrupar por ubicación SOLAMENTE — f_vencimiento NO participa.
   746→        // Se rastrea la fecha más próxima (FEFO) para pasar en la salida.
   747→        const key = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}`
   748→        const current = locMap.get(key)
   749→        if (current) {
   750→          current.stock += impact
   751→          // Rastrear f_vencimiento más próxima para FEFO
   752→          if (m.fVencimiento) {
   753→            if (!current.fVencimiento || m.fVencimiento < current.fVencimiento) {
   754→              current.fVencimiento = m.fVencimiento
   755→            }
   756→          }
   757→        } else {
   758→          locMap.set(key, {
   759→            bloque: m.bloque,
   760→            torre: m.torre,
   761→            piso: m.piso,
   762→            posicion: m.posicion,
   763→            codigo: m.codigo,
   764→            descripcion: m.descripcion,
   765→            un: m.un,
   766→            stock: impact,
   767→            fVencimiento: m.fVencimiento || undefined,
   768→            proveedor: m.proveedor,
   769→          })
   770→        }
   771→      }
   772→      const results = Array.from(locMap.values()).filter((l) => l.stock > 0)
   773→      // Ordenar: FEFO primero (con fecha de vencimiento), luego sin fecha por bloque (1→7)
   774→      results.sort((a, b) => {
   775→        const aHasDate = !!a.fVencimiento
   776→        const bHasDate = !!b.fVencimiento
   777→        if (aHasDate && bHasDate) return a.fVencimiento!.localeCompare(b.fVencimiento!)
   778→        if (aHasDate && !bHasDate) return -1
   779→        if (!aHasDate && bHasDate) return 1
   780→        // Ambos sin fecha: ordenar por bloque, torre, piso, posición
   781→        const aB = parseInt(a.bloque, 10) || 0
   782→        const bB = parseInt(b.bloque, 10) || 0
   783→        if (aB !== bB) return aB - bB
   784→        const aT = parseInt(a.torre, 10) || 0
   785→        const bT = parseInt(b.torre, 10) || 0
   786→        if (aT !== bT) return aT - bT
   787→        const aP = parseInt(a.piso, 10) || 0
   788→        const bP = parseInt(b.piso, 10) || 0
   789→        if (aP !== bP) return aP - bP
   790→        const aPos = parseInt(a.posicion, 10) || 0
   791→        const bPos = parseInt(b.posicion, 10) || 0
   792→        return aPos - bPos
   793→      })
   794→      // Limpiar selecciones de ubicaciones que ya no tienen stock
   795→      const newKeys = new Set(results.map((l) => `${l.bloque}-${l.torre}-${l.piso}-${l.posicion}`))
   796→      setSelected((prev) => {
   797→        const cleaned = new Set<string>()
   798→        for (const k of prev) {
   799→          if (newKeys.has(k)) cleaned.add(k)
   800→        }
   801→        return cleaned.size === prev.size ? prev : cleaned
   802→      })
   803→      setLocations(results)
   804→      setProductoDesc(desc)
   805→      setProductoUn(un)
   806→    } catch (err) {
   807→      console.error('[MovimientoForm] refreshLocations error:', err)
   808→      toast.error('Error al buscar stock', { description: extractError(err) })
   809→    }
   810→  }, [])
   811→
   812→  // Búsqueda inicial con debounce
   813→  useEffect(() => {
   814→    if (!searchCode.trim()) {
   815→      setLocations([])
   816→      setProductoDesc('')
   817→      setProductoUn('')
   818→      setQtyMap({})
   819→      return
   820→    }
   821→    setLoading(true)
   822→    const timer = setTimeout(async () => {
   823→      await refreshLocations()
   824→      setLoading(false)
   825→    }, 300)
   826→    return () => clearTimeout(timer)
   827→  }, [searchCode, refreshLocations])
   828→
   829→  // POLLING: refrescar automáticamente cada 8 segundos
   830→  useEffect(() => {
   831→    if (!searchCode.trim()) return
   832→    const interval = setInterval(() => {
   833→      refreshLocations()
   834→    }, 8000)
   835→    return () => clearInterval(interval)
   836→  }, [searchCode, refreshLocations])
   837→
   838→  async function handleSalidaParcial(locKey: string) {
   839→    const loc = locations.find((l) => `${l.bloque}-${l.torre}-${l.piso}-${l.posicion}` === locKey)
   840→    if (!loc) return

   842→    const qtyNum = parseFloat(qtyVal)
   843→    if (isNaN(qtyNum) || qtyNum <= 0) {

   848→      toast.error('La cantidad excede el stock disponible')

   854→  function handleRetirarTodo(locKey: string) {
   855→    const loc = locations.find((l) => `${l.bloque}-${l.torre}-${l.piso}-${l.posicion}` === locKey)

   858→  }
   859→
   860→  const [confirmState, setConfirmState] = useState<{
   861→    loc: LocWithKey
   862→    qtyNum: number
   863→    full: boolean
   864→  } | null>(null)
   865→
   866→  async function doSalida() {
   867→    if (!confirmState) return
   868→    const { loc, qtyNum } = confirmState
   869→    setBusy(true)
   870→    try {
   871→      const { movs, wasOffline } = await SyncEngine.offlineAwareAddMovimiento({
   872→        tipo: 'salida',
   873→        bloque: loc.bloque,
   874→        torre: loc.torre,
   875→        piso: loc.piso,
   876→        posicion: loc.posicion,
   877→        codigo: loc.codigo,
   878→        descripcion: loc.descripcion,
   879→        un: loc.un,
   880→        cantidad: qtyNum,
   881→        fVencimiento: loc.fVencimiento ?? '',
   882→        turno,
   883→        usuarioId: perfil.id,
   884→        usuarioNombre: perfil.nombre,
   885→        usuarioCorreo: perfil.correo,
   886→        proveedor: loc.proveedor,
   887→        // PRESERVAR codigoInc para consistencia (aunque SalidaForm excluye INC)
   888→        codigoInc: loc.codigoInc || undefined,
   889→      })
   890→      if (wasOffline) {
   891→        toast.success(`Salida de ${qtyNum} ${loc.un} guardada (offline)`, {
   892→          description: 'Se sincronizará al reconectarse. La ubicación se actualizará cuando haya conexión.',
   893→          duration: 5000,
   894→        })
   895→        // OFFLINE: Limpiar UI inmediatamente para evitar doble salida
   896→        setConfirmState(null)
   897→        setSearchCode('')
   898→        setQtyMap({})
   899→        setLocations([])
   900→        setProductoDesc('')
   901→        setProductoUn('')
   902→      } else {
   903→        toast.success(`Salida de ${qtyNum} ${loc.un} registrada`)
   904→        setConfirmState(null)
   905→        setSearchCode('')
   906→        setQtyMap({})

   912→        toast.error('Stock insuficiente', {

   915→        })

   917→        refreshLocations() // refrescar datos reales
   918→      } else {
   919→        const message = extractError(err)
   920→        toast.error('Error al registrar salida', { description: message })
   921→      }
   922→    } finally {
   923→      setBusy(false)
   924→    }
   925→  }
   926→
   927→  return (
   928→    <div className="space-y-4">
   929→      {/* Búsqueda */}
   930→      <div className="space-y-1">
   931→        <Label className="text-sm text-muted-foreground">Buscar (código o descripción)</Label>

   937→      </div>

   939→      {/* Producto (solo lectura) */}
   940→      {productoDesc && (
   941→        <div className="space-y-1">
   942→          <Label className="text-sm text-muted-foreground">Producto</Label>

   947→          />
   948→        </div>
   949→      )}
   950→
