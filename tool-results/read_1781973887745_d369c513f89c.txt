    62→type Props = {
    63→  tipo: TipoMovimiento | 'inc' | 'salida-inc'
    64→  onCreated: (movs: Movimiento[]) => void
    65→}
    66→
    67→export function MovimientoForm({ tipo, onCreated }: Props) {
    68→  const { perfil } = useAuth()
    69→  const [turno, setTurno] = useState<Turno>(calcularTurno())
    70→
    71→  useEffect(() => {
    72→    const interval = setInterval(() => setTurno(calcularTurno()), 60000)
    73→    return () => clearInterval(interval)
    74→  }, [])
    75→
    76→  if (!perfil) {
    77→    return <div className='p-4 text-muted-foreground animate-pulse'>Cargando usuario...</div>
    78→  }
    79→  if (tipo === 'salida') {
    80→    return <SalidaForm turno={turno} onCreated={onCreated} perfil={perfil} />
    81→  }
    82→  if (tipo === 'salida-inc') {
    83→    return <SalidaIncForm turno={turno} onCreated={onCreated} perfil={perfil} />
    84→  }
    85→  if (tipo === 'inc') {
    86→    return <IncForm turno={turno} onCreated={onCreated} perfil={perfil} />
    87→  }
    88→  // ingreso y devolucion usan el mismo formulario, solo cambia el tipo
    89→  return <IngresoForm turno={turno} onCreated={onCreated} perfil={perfil} tipo={tipo} />
    90→}

    95→function IngresoForm({
    96→  turno,
    97→  onCreated,
    98→  perfil,
    99→  tipo = 'ingreso',
   100→}: {
   101→  turno: Turno
   102→  onCreated: (movs: Movimiento[]) => void
   103→  perfil: { id: string; nombre: string; correo: string }
   104→  tipo?: TipoMovimiento
   105→}) {
   106→  const [bloque, setBloque] = useState('')
   107→  const [torre, setTorre] = useState('')
   108→  const [piso, setPiso] = useState('')
   109→  const [posicion, setPosicion] = useState('')
   110→  const [codigo, setCodigo] = useState('')
   111→  const [descripcion, setDescripcion] = useState('')
   112→  const [un, setUn] = useState('')
   113→  const [cantidad, setCantidad] = useState('')
   114→  const [fVencimiento, setFVencimiento] = useState('')
   115→  const [sinVencimiento, setSinVencimiento] = useState(false)
   116→  const [proveedor, setProveedor] = useState('')
   117→  const [busy, setBusy] = useState(false)
   118→  const [confirmData, setConfirmData] = useState<StockEnUbicacion[] | null>(null)
   119→  const [salidaBusy, setSalidaBusy] = useState<string | null>(null)
   120→
   121→  const torres = torresDeBloque(bloque)
   122→  const posiciones = posicionesDeBloque(bloque)
   123→
   124→  function handleCatalogoPick(item: CatalogoItem) {
   125→    setCodigo(item.codigo)
   126→    setDescripcion(item.descripcion)
   127→    setUn(item.un)
   128→  }
   129→
   130→  async function handleSubmit(e: React.FormEvent) {
   131→    e.preventDefault()
   132→    if (!bloque || !torre || !piso || !posicion || !codigo.trim() || !cantidad) {
   133→      toast.error('Completa todos los campos requeridos')
   134→      return
   135→    }
   136→    if (requiereProveedor(descripcion) && !proveedor) {
   137→      toast.error('Selecciona un proveedor para este producto')
   138→      return
   139→    }
   140→    const qty = parseFloat(cantidad)
   141→    if (isNaN(qty) || qty <= 0) {
   142→      toast.error('La cantidad debe ser mayor a 0')
   143→      return
   144→    }
   145→    setBusy(true)
   146→    try {
   147→      // Verificar si CUALQUIER producto existe en esta posición
   148→      const details = await stockEnUbicacion(bloque, torre, piso, posicion)
   149→      if (details.length > 0) {
   150→        setConfirmData(details)
   151→        setBusy(false)
   152→        return
   153→      }
   154→      await doInsert(qty)
   155→    } catch (err: unknown) {
   156→      const message = extractError(err)
   157→      toast.error('Error al verificar stock', { description: message })
   158→      setBusy(false)
   159→    }
   160→  }
   161→
   162→  async function doInsert(qty: number) {
   163→    try {
   164→      const movs = await addMovimiento({
   165→        tipo,
   166→        bloque,
   167→        torre,
   168→        piso,
   169→        posicion,
   170→        codigo,
   171→        descripcion,
   172→        un,
   173→        cantidad: qty,
   174→        fVencimiento: sinVencimiento ? '' : fVencimiento,
   175→        turno,
   176→        usuarioId: perfil.id,
   177→        usuarioNombre: perfil.nombre,
   178→        usuarioCorreo: perfil.correo,
   179→        proveedor: proveedor || undefined,
   180→      })
   181→      toast.success(tipo === 'devolucion' ? 'Devolución registrada' : 'Ingreso registrado')
   182→      setCodigo('')
   183→      setDescripcion('')
   184→      setUn('')
   185→      setCantidad('')
   186→      setFVencimiento('')
   187→      setProveedor('')
   188→      onCreated(movs)
   189→    } catch (err: unknown) {
   190→      const message = extractError(err)
   191→      toast.error('Error al registrar ingreso', { description: message })
   192→    } finally {
   193→      setBusy(false)
   194→      setConfirmData(null)
   195→    }
   196→  }
   197→
   198→  // Dar salida a un producto desde el alerta de ubicación ocupada
   199→  async function handleSalidaDesdeAlerta(stockItem: StockEnUbicacion) {
   200→    setSalidaBusy(stockItem.codigo)
   201→    try {
   202→      const movs = await addMovimiento({
   203→        tipo: 'salida',
   204→        bloque,
   205→        torre,
   206→        piso,
   207→        posicion,
   208→        codigo: stockItem.codigo,
   209→        descripcion: stockItem.descripcion,
   210→        un: stockItem.un,
   211→        cantidad: stockItem.stock,
   212→        fVencimiento: stockItem.fVencimiento ?? '',
   213→        turno,
   214→        usuarioId: perfil.id,
   215→        usuarioNombre: perfil.nombre,
   216→        usuarioCorreo: perfil.correo,
   217→        proveedor: stockItem.proveedor,
   218→        // PRESERVAR codigoInc para que la salida descuente del stock INC correctamente
   219→        codigoInc: stockItem.codigoInc || undefined,
   220→      })
   221→      toast.success(`Salida de ${stockItem.stock} ${stockItem.un} de ${stockItem.codigo}`)
   222→      onCreated(movs)
   223→      // Refrescar datos del alerta
   224→      const updated = await stockEnUbicacion(bloque, torre, piso, posicion)
   225→      if (updated.length > 0) {
   226→        setConfirmData(updated)
   227→      } else {
   228→        setConfirmData(null)
   229→        // Ya no hay productos, hacer el ingreso directamente
   230→        await doInsert(parseFloat(cantidad))
   231→      }
   232→    } catch (err: unknown) {
   233→      if (isInsufficientStockError(err)) {
   234→        const detail = (err as Record<string, string>).detail || ''
   235→        toast.error('Stock insuficiente', {
   236→          description: detail || 'Otro usuario pudo haber modificado el stock mientras tú operabas.',
   237→          duration: 8000,
   238→        })
   239→      } else {
   240→        const message = extractError(err)
   241→        toast.error('Error al dar salida', { description: message })
   242→      }
   243→    } finally {
   244→      setSalidaBusy(null)
   245→    }
   246→  }
   247→
   248→  return (
   249→    <>
   250→      <div className={`flex items-center gap-2 mb-4 p-3 rounded-lg border ${
   251→        tipo === 'devolucion'
   252→          ? 'bg-orange-50 dark:bg-orange-950/40 border-orange-200 dark:border-orange-800'
   253→          : 'bg-green-50 dark:bg-green-950/40 border-green-200 dark:border-green-800'
   254→      }`}>
   255→        {tipo === 'devolucion' ? (
   256→          <ArrowRightLeft className="h-5 w-5 text-orange-600 dark:text-orange-400 flex-shrink-0" />
   257→        ) : (
   258→          <ArrowDownToLine className="h-5 w-5 text-green-600 dark:text-green-400 flex-shrink-0" />
   259→        )}
   260→        <div className="min-w-0">
   261→          <p className={`text-sm font-semibold ${
   262→            tipo === 'devolucion'
   263→              ? 'text-orange-700 dark:text-orange-300'
   264→              : 'text-green-700 dark:text-green-300'
   265→          }`}>
   266→            {tipo === 'devolucion' ? 'Ingreso por devolución' : 'Ingreso de mercadería'}
   267→          </p>
   268→          <p className="text-xs text-muted-foreground">Turno: {turno}</p>
   269→        </div>
   270→      </div>
   271→
   272→      <form onSubmit={handleSubmit} className="space-y-4">
   273→        <div className="space-y-1">
   274→          <Label className="text-xs font-medium text-muted-foreground uppercase tracking-wider">Ubicación</Label>
   275→          <div className="grid grid-cols-2 sm:grid-cols-5 gap-2">
   276→            <div className="space-y-1">
   277→              <Label className="text-xs text-muted-foreground">Bloque</Label>
   278→              <Select value={bloque} onValueChange={(v) => { setBloque(v); setTorre(''); setPiso(''); setPosicion('') }}>
   279→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Bloque" /></SelectTrigger>
   280→                <SelectContent>
   281→                  {BLOQUES.map((b) => <SelectItem key={b} value={b}>{b}</SelectItem>)}
   282→                </SelectContent>
   283→              </Select>
   284→            </div>
   285→            <div className="space-y-1">
   286→              <Label className="text-xs text-muted-foreground">Torre</Label>
   287→              <Select value={torre} onValueChange={setTorre} disabled={!bloque}>
   288→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Torre" /></SelectTrigger>
   289→                <SelectContent>
   290→                  {torres.map((t) => <SelectItem key={t} value={t}>{t}</SelectItem>)}
   291→                </SelectContent>
   292→              </Select>
   293→            </div>
   294→            <div className="space-y-1">
   295→              <Label className="text-xs text-muted-foreground">Piso</Label>
   296→              <Select value={piso} onValueChange={setPiso} disabled={!bloque}>
   297→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Piso" /></SelectTrigger>
   298→                <SelectContent>
   299→                  {PISOS.map((p) => <SelectItem key={p} value={p}>{p}</SelectItem>)}
   300→                </SelectContent>
   301→              </Select>
   302→            </div>
   303→            <div className="space-y-1">
   304→              <Label className="text-xs text-muted-foreground">Posición</Label>
   305→              <Select value={posicion} onValueChange={setPosicion} disabled={!bloque}>
   306→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Pos." /></SelectTrigger>
   307→                <SelectContent>
   308→                  {posiciones.map((p) => <SelectItem key={p} value={p}>{p}</SelectItem>)}
   309→                </SelectContent>
   310→              </Select>
   311→            </div>
   312→            <div className="space-y-1 col-span-2 sm:col-span-1">
   313→              <Label className="text-xs text-muted-foreground">Turno</Label>
   314→              <Input value={turno} readOnly className="h-10 bg-muted text-sm" />
   315→            </div>
   316→          </div>
   317→        </div>
   318→
   319→        <div className="space-y-1">
   320→          <Label className="text-xs font-medium text-muted-foreground uppercase tracking-wider">Producto</Label>
   321→          <CatalogoSearchInput
   322→            onPick={handleCatalogoPick}
   323→            value={codigo}
   324→            onChange={(v) => {
   325→              setCodigo(v)
   326→              const cat = findCatalogoByCodigo(v)
   327→              if (cat) {
   328→                setDescripcion(cat.descripcion)
   329→                setUn(cat.un)
   330→              }
   331→            }}
   332→          />
   333→        </div>
   334→
   335→        {descripcion && (
   336→          <div className="rounded-lg border border-green-200 dark:border-green-800 bg-green-50/50 dark:bg-green-950/20 p-3 space-y-2 transition-all">
   337→            <div className="flex items-start gap-3 flex-wrap">
   338→              <div className="min-w-0 flex-1">
   339→                <p className="text-xs text-muted-foreground">Descripción</p>
   340→                <p className="text-sm font-medium truncate">{descripcion}</p>
   341→              </div>
   342→              <div className="min-w-[60px]">
   343→                <p className="text-xs text-muted-foreground">UN</p>
   344→                <p className="text-sm font-medium">{un || '—'}</p>
   345→              </div>
   346→            </div>
   347→          </div>
   348→        )}
   349→
   350→        <div className="grid grid-cols-2 sm:grid-cols-4 gap-2">
   351→          <div className="space-y-1 col-span-2 sm:col-span-1">
   352→            <Label className="text-xs text-muted-foreground">Cantidad</Label>
   353→            <Input type="number" step="any" min="0.001" value={cantidad} onChange={(e) => setCantidad(e.target.value)} placeholder="0" className="h-10" />
   354→          </div>
   355→          <div className="space-y-1 col-span-2 sm:col-span-1">
   356→            <Label className="text-xs text-muted-foreground">F. Vencimiento</Label>
   357→            <div className="flex items-center gap-1.5">
   358→              <Input type="date" value={fVencimiento} onChange={(e) => setFVencimiento(e.target.value)} disabled={sinVencimiento} className="h-10 [color-scheme:dark]" />
   359→              <Checkbox checked={sinVencimiento} onCheckedChange={(v) => setSinVencimiento(!!v)} />
   360→            </div>
   361→          </div>
   362→          {requiereProveedor(descripcion) && (
   363→            <div className="space-y-1 col-span-2">
   364→              <Label className="text-xs text-muted-foreground font-medium">Proveedor <span className="text-red-500">*</span></Label>
   365→              <Select value={proveedor} onValueChange={setProveedor}>
   366→                <SelectTrigger className="h-10">
   367→                  <SelectValue placeholder="Selecciona proveedor" />
   368→                </SelectTrigger>
   369→                <SelectContent>
   370→                  {PROVEEDORES_FILM.map((p) => (
   371→                    <SelectItem key={p} value={p}>{p}</SelectItem>
   372→                  ))}
   373→                </SelectContent>
   374→              </Select>
   375→            </div>
   376→          )}
   377→        </div>
   378→
   379→        <Button type="submit" disabled={busy} className={`w-full h-11 text-white text-sm font-medium sm:w-auto ${
   380→          tipo === 'devolucion'
   381→            ? 'bg-orange-600 hover:bg-orange-700'
   382→            : 'bg-green-600 hover:bg-green-700'
   383→        }`}>
   384→          {busy && <Loader2 className="h-4 w-4 animate-spin mr-2" />}
   385→          {tipo === 'devolucion' ? 'Registrar Devolución' : 'Registrar Ingreso'}
   386→        </Button>
   387→      </form>
   388→
   389→      {/* ═══ DIÁLOGO DE CONFIRMACIÓN — Ubicación Ocupada ═══ */}
   390→      <AlertDialog open={!!confirmData} onOpenChange={() => setConfirmData(null)}>
   391→        <AlertDialogContent className="max-w-[calc(100vw-1rem)] max-w-lg p-0 max-h-[85vh] flex flex-col overflow-hidden">
   392→          {/* Header con gradiente */}
   393→          <div className="bg-gradient-to-r from-amber-500 via-orange-500 to-red-500 px-4 sm:px-6 py-5 text-white shrink-0">
   394→            <div className="flex items-center gap-3">
   395→              <div className="w-11 h-11 rounded-full bg-white/20 backdrop-blur-sm flex items-center justify-center flex-shrink-0">
   396→                <TriangleAlert className="h-6 w-6 text-white" />
   397→              </div>
   398→              <div>
   399→                <AlertDialogTitle className="text-lg font-bold text-white m-0">
   400→                  Ubicación Ocupada
   401→                </AlertDialogTitle>
   402→                <AlertDialogDescription className="text-amber-100 text-sm mt-0.5">
   403→                  Esta posición ya tiene stock registrado. Revisa el detalle.
   404→                </AlertDialogDescription>
   405→              </div>
   406→            </div>
   407→          </div>
   408→
   409→          {/* Contenido scrolleable */}
   410→          <div className="flex-1 min-h-0 overflow-y-auto overscroll-contain">
   411→            {/* Ubicación destino */}
   412→            <div className="px-4 sm:px-6 pt-4 pb-2">
   413→              <p className="text-xs font-semibold text-muted-foreground uppercase tracking-wider mb-1.5">Ubicación destino</p>
   414→              <div className="flex items-center gap-2 text-sm text-muted-foreground bg-slate-50 dark:bg-slate-900 rounded-lg px-3 py-2.5">
   415→                <MapPin className="h-4 w-4 text-orange-500 flex-shrink-0" />
   416→                <span className="font-mono font-medium text-slate-700 dark:text-slate-300">
   417→                  Bloque {bloque} · Torre {torre} · Piso {piso} · Pos {posicion}
   418→                </span>
   419→              </div>
   420→            </div>
   421→
   422→            {/* Producto que se desea ingresar */}
   423→            <div className="px-4 sm:px-6 pb-3">
   424→              <p className="text-xs font-semibold text-muted-foreground uppercase tracking-wider mb-1.5">Producto a {tipo === 'devolucion' ? 'devolver' : 'ingresar'}</p>
   425→              <div className="rounded-xl border-2 border-dashed border-green-300 dark:border-green-700 bg-green-50/60 dark:bg-green-950/20 p-3 space-y-1.5">
   426→                <div className="flex items-center justify-between">
   427→                  <div className="flex items-center gap-2 min-w-0">
   428→                    <span className="font-mono font-bold text-sm text-green-800 dark:text-green-300">{codigo}</span>
   429→                    <span className="text-[10px] text-green-700/60 dark:text-green-400/60 bg-green-100 dark:bg-green-900/40 px-1.5 py-0.5 rounded">{un || '—'}</span>
   430→                  </div>
   431→                  <div className="flex items-center gap-1 flex-shrink-0">

   436→                {descripcion && (
   437→                  <p className="text-xs text-green-700/70 dark:text-green-400/70 truncate">{descripcion}</p>
   438→                )}
   439→                {fVencimiento && !sinVencimiento && (
   440→                  <p className="text-[10px] text-muted-foreground">Venc: {fVencimiento}</p>
   441→                )}
   442→              </div>
   443→            </div>
   444→
   445→            {/* Separador */}
   446→            {confirmData && confirmData.length > 0 && (
   447→              <div className="px-4 sm:px-6 pb-2">
   448→                <div className="flex items-center gap-2">
   449→                  <div className="flex-1 h-px bg-gradient-to-r from-transparent via-slate-300 dark:via-slate-600 to-transparent" />
   450→                  <span className="text-[10px] font-semibold text-muted-foreground uppercase tracking-widest">Stock actual en esta ubicación</span>
   451→                  <div className="flex-1 h-px bg-gradient-to-r from-transparent via-slate-300 dark:via-slate-600 to-transparent" />
   452→                </div>
   453→              </div>
   454→            )}
   455→
   456→            {/* Productos existentes en la ubicación */}
   457→            {confirmData && confirmData.length > 0 && (
   458→              <div className="px-4 sm:px-6 pb-4">
   459→                <p className="text-xs font-semibold text-amber-600 dark:text-amber-400 uppercase tracking-wider mb-2">
   460→                  {confirmData.length} producto{confirmData.length !== 1 ? 's' : ''} encontrado{confirmData.length !== 1 ? 's' : ''}
   461→                </p>
   462→                <div className="space-y-2">
   463→                  {confirmData.map((s, i) => (
   464→                    <div
   465→                      key={`${s.codigo}-${i}`}
   466→                      className="rounded-xl border border-slate-200 dark:border-slate-700 bg-white dark:bg-slate-950 p-3"
   467→                    >
   468→                      <div className="flex items-start gap-3">
   469→                        <div className="w-9 h-9 rounded-lg bg-orange-100 dark:bg-orange-900/30 flex items-center justify-center flex-shrink-0 mt-0.5">

   560→  function toggleSelect(key: string) {

   603→  async function doMassSalida() {
   604→    setMassBusy(true)
   605→    const totalToProcess = selectedRef.current.size
   606→    let totalProcessed = 0
   607→    let totalErrors = 0
   608→    let stockErrors = 0
   609→    const errorDetails: string[] = []
   610→    try {
   611→      // Usar los values actuales al momento de iniciar (no refs que cambian durante el loop)
   612→      const selectedKeys = Array.from(selectedRef.current)
   613→      const locsSnap = [...locationsRef.current]
   614→      const qtySnap = { ...qtyMapRef.current }
   615→
   616→      for (const key of selectedKeys) {
   617→        const loc = locsSnap.find((l) => `${l.bloque}-${l.torre}-${l.piso}-${l.posicion}` === key)
   618→        if (!loc) {
   619→          errorDetails.push(`Ubicación ${key} no encontrada`)

   621→          continue
   622→        }

   625→        if (isNaN(qtyNum) || qtyNum <= 0) {
   626→          errorDetails.push(`Cantidad inválida en B${loc.bloque}/T${loc.torre}/P${loc.piso}/Pos${loc.posicion}`)
   627→          totalErrors++
   628→          continue
   629→        }
   630→        const finalQty = Math.min(qtyNum, loc.stock)
   631→        try {
   632→          await addMovimiento({
   633→            tipo: 'salida',
   634→            bloque: loc.bloque, torre: loc.torre, piso: loc.piso, posicion: loc.posicion,
   635→            codigo: loc.codigo, descripcion: loc.descripcion, un: loc.un,
   636→            cantidad: finalQty,
   637→            fVencimiento: loc.fVencimiento ?? '', turno,
   638→            usuarioId: perfil.id, usuarioNombre: perfil.nombre, usuarioCorreo: perfil.correo,
   639→            proveedor: loc.proveedor,
   640→            // PRESERVAR codigoInc para consistencia (aunque SalidaForm excluye INC)
   641→            codigoInc: loc.codigoInc || undefined,
   642→          })
   643→          totalProcessed++
   644→        } catch (err) {
   645→          if (isInsufficientStockError(err)) {
   646→            stockErrors++
   647→            const detail = (err as Record<string, string>).detail || ''
   648→            errorDetails.push(`Stock insuficiente en B${loc.bloque}/T${loc.torre}/P${loc.piso}/Pos${loc.posicion}${detail ? ` (${detail})` : ''}`)
   649→          } else {
   650→            const msg = extractError(err)
   651→            errorDetails.push(`Error en B${loc.bloque}/T${loc.torre}/P${loc.piso}/Pos${loc.posicion}: ${msg}`)
   652→          }
   653→          totalErrors++
   654→        }
   655→      }
   656→      if (totalErrors > 0) {
   657→        const stockNote = stockErrors > 0 ? ` (${stockErrors} por stock insuficiente)` : ''
   658→        const desc = totalProcessed > 0
   659→          ? `${totalProcessed} de ${totalToProcess} correctas${stockNote}:\n${errorDetails.slice(0, 5).join('\n')}${errorDetails.length > 5 ? `\n...y ${errorDetails.length - 5} más` : ''}`
   660→          : `Errores${stockNote}:\n${errorDetails.join('\n')}`
   661→        toast.warning(`Salida completada con ${totalErrors} error${totalErrors > 1 ? 'es' : ''}`, { description: desc, duration: 10000 })
   662→      } else {
   663→        toast.success(`Salida registrada en ${totalProcessed} ubicacion${totalProcessed > 1 ? 'es' : ''}`)
   664→      }
   665→      setMassConfirmOpen(false)
   666→      setSelected(new Set())
   667→      setQtyMap({})

   669→        await refreshLocations()
   670→      }
   671→      onCreated([]) // trigger re-render
   672→    } catch (err: unknown) {
   673→      const message = extractError(err)
   674→      toast.error('Error en salida masiva', { description: message })
   675→    } finally {
   676→      setMassBusy(false)
   677→    }
   678→  }
   679→
   680→  const allSelected = locations.length > 0 && selected.size === locations.length
   681→
   682→  // Función central para refrescar las ubicaciones
   683→  const refreshLocations = useCallback(async () => {
   684→    const code = searchCodeRef.current.trim()
   685→    if (!code) {
   686→      setLocations([])
   687→      setProductoDesc('')
   688→      setProductoUn('')
   689→      setQtyMap({})

   691→    }
   692→    try {
   693→      let movs: Array<{ tipo: string; bloque: string; torre: string; piso: string; posicion: string; codigo: string; descripcion: string; un: string; cantidad: number; fVencimiento?: string; proveedor?: string; codigoInc?: string }>
   694→      try {
   695→        // Intentar obtener del servidor - consulta optimizada por código
   696→        const { fetchMovimientosByCodigo } = await import('@/lib/rackly/kardex')
   697→        movs = await fetchMovimientosByCodigo(code)
   698→      } catch {
   699→        // Sin cache offline, usar lista vacia

   701→      }
   702→      const upperCode = code.toUpperCase()

   704→      // EXCLUIR movimientos INC de la vista de salidas normales
   705→      const relevant = movs.filter((m) => m.codigo === upperCode && !m.codigoInc)
   706→      let desc = ''
   707→      let un = ''
   708→      for (const m of relevant) {

   712→        if (isNaN(impact) || !isFinite(impact)) continue
   713→        // Agrupar por ubicación SOLAMENTE — f_vencimiento NO participa.
   714→        // Se rastrea la fecha más próxima (FEFO) para pasar en la salida.
   715→        const key = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}`
   716→        const current = locMap.get(key)
   717→        if (current) {
   718→          current.stock += impact
   719→          // Rastrear f_vencimiento más próxima para FEFO
   720→          if (m.fVencimiento) {
   721→            if (!current.fVencimiento || m.fVencimiento < current.fVencimiento) {
   722→              current.fVencimiento = m.fVencimiento
   723→            }
   724→          }
   725→        } else {
   726→          locMap.set(key, {
   727→            bloque: m.bloque,
   728→            torre: m.torre,
   729→            piso: m.piso,
   730→            posicion: m.posicion,
   731→            codigo: m.codigo,
   732→            descripcion: m.descripcion,
   733→            un: m.un,
   734→            stock: impact,
   735→            fVencimiento: m.fVencimiento || undefined,
   736→            proveedor: m.proveedor,
   737→          })
   738→        }
   739→      }
   740→      const results = Array.from(locMap.values()).filter((l) => l.stock > 0)
   741→      // Ordenar: FEFO primero (con fecha de vencimiento), luego sin fecha por bloque (1→7)
   742→      results.sort((a, b) => {
   743→        const aHasDate = !!a.fVencimiento
   744→        const bHasDate = !!b.fVencimiento
   745→        if (aHasDate && bHasDate) return a.fVencimiento!.localeCompare(b.fVencimiento!)
   746→        if (aHasDate && !bHasDate) return -1
   747→        if (!aHasDate && bHasDate) return 1
   748→        // Ambos sin fecha: ordenar por bloque, torre, piso, posición
   749→        const aB = parseInt(a.bloque, 10) || 0
   750→        const bB = parseInt(b.bloque, 10) || 0
   751→        if (aB !== bB) return aB - bB
   752→        const aT = parseInt(a.torre, 10) || 0
   753→        const bT = parseInt(b.torre, 10) || 0
   754→        if (aT !== bT) return aT - bT
   755→        const aP = parseInt(a.piso, 10) || 0
   756→        const bP = parseInt(b.piso, 10) || 0
   757→        if (aP !== bP) return aP - bP
   758→        const aPos = parseInt(a.posicion, 10) || 0
   759→        const bPos = parseInt(b.posicion, 10) || 0
   760→        return aPos - bPos
   761→      })
   762→      // Limpiar selecciones de ubicaciones que ya no tienen stock
   763→      const newKeys = new Set(results.map((l) => `${l.bloque}-${l.torre}-${l.piso}-${l.posicion}`))
   764→      setSelected((prev) => {
   765→        const cleaned = new Set<string>()
   766→        for (const k of prev) {
   767→          if (newKeys.has(k)) cleaned.add(k)
   768→        }
   769→        return cleaned.size === prev.size ? prev : cleaned
   770→      })
   771→      setLocations(results)
   772→      setProductoDesc(desc)
   773→      setProductoUn(un)
   774→    } catch (err) {
   775→      console.error('[MovimientoForm] refreshLocations error:', err)
   776→      toast.error('Error al buscar stock', { description: extractError(err) })
   777→    }
   778→  }, [])
   779→
   780→  // Búsqueda inicial con debounce
   781→  useEffect(() => {
   782→    if (!searchCode.trim()) {
   783→      setLocations([])
   784→      setProductoDesc('')
   785→      setProductoUn('')
   786→      setQtyMap({})
   787→      return
   788→    }
   789→    setLoading(true)
   790→    const timer = setTimeout(async () => {

   792→      setLoading(false)
   793→    }, 300)
   794→    return () => clearTimeout(timer)

   796→
   797→  // POLLING: refrescar automáticamente cada 8 segundos
   798→  useEffect(() => {

   800→    const interval = setInterval(() => {
   801→      refreshLocations()
   802→    }, 8000)
   803→    return () => clearInterval(interval)
   804→  }, [searchCode, refreshLocations])

   806→  async function handleSalidaParcial(locKey: string) {
   807→    const loc = locations.find((l) => `${l.bloque}-${l.torre}-${l.piso}-${l.posicion}` === locKey)

   822→  function handleRetirarTodo(locKey: string) {

   828→  const [confirmState, setConfirmState] = useState<{

   852→        usuarioNombre: perfil.nombre,
   853→        usuarioCorreo: perfil.correo,
   854→        proveedor: loc.proveedor,
   855→        // PRESERVAR codigoInc para consistencia (aunque SalidaForm excluye INC)
