    58→// Calcula días restantes hasta vencimiento (negativo si ya venció)
    59→function diasParaVencer(fVencimiento: string | undefined): number | null {
    60→  if (!fVencimiento) return null
    61→  const hoy = new Date(); hoy.setHours(0, 0, 0, 0)
    62→  const venc = new Date(fVencimiento + 'T00:00:00')
    63→  return Math.ceil((venc.getTime() - hoy.getTime()) / (1000 * 60 * 60 * 24))
    64→}

    66→// Detecta códigos duplicados (mismo artículo con distintas fechas de vencimiento)
    67→function codigosConMultiplesLotes(stock: StockEnUbicacion[]): Set<string> {
    68→  const counts = new Map<string, number>()
    69→  for (const s of stock) {
    70→    counts.set(s.codigo, (counts.get(s.codigo) ?? 0) + 1)
    71→  }
    72→  const multiples = new Set<string>()
    73→  for (const [code, count] of counts) {
    74→    if (count > 1) multiples.add(code)
    75→  }
    76→  return multiples
    77→}

    79→// Calcula ocupación desde movimientos — stock POR CÓDIGO (no por lote).
    80→// f_vencimiento es SOLO para FEFO (ordenamiento), NO para particionar stock.
    81→// IMPORTANTE: Los movimientos INC se EXCLUYEN del stock normal.
    82→// La detección INC se hace por separado con fetchIncPorUbicacion().
    83→function calcularOcupacion(movs: Movimiento[]): OcupacionCelda[] {
    84→  // Mapa: ubicacion_key → (codigo → stock)
    85→  const cellMap = new Map<string, Map<string, number>>()
    86→  for (const m of movs) {
    87→    // EXCLUIR movimientos INC del cálculo de ocupación normal
    88→    if (m.codigoInc) continue
    89→    const key = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}`
    90→    const code = m.codigo.trim().toUpperCase()
    91→    let codeMap = cellMap.get(key)
    92→    if (!codeMap) { codeMap = new Map(); cellMap.set(key, codeMap) }
    93→    const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? m.cantidad : -m.cantidad
    94→    const current = codeMap.get(code) ?? 0
    95→    codeMap.set(code, current + delta)
    96→  }
    97→  // Construir resultado: solo celdas con stock total > 0
    98→  const result: OcupacionCelda[] = []
    99→  for (const [key, codeMap] of cellMap) {
   100→    let totalStock = 0
   101→    const codigos = new Set<string>()
   102→    for (const [code, stock] of codeMap) {
   103→      if (stock > 0) {
   104→        totalStock += stock
   105→        codigos.add(code)
   106→      }
   107→    }
   108→    if (totalStock > 0) {
   109→      const [bloque, torre, piso, posicion] = key.split('-')
   110→      // lotes = número de códigos distintos (ya no se cuenta por f_vencimiento)
   111→      result.push({ bloque, torre, piso, posicion, stock: totalStock, codigos: Array.from(codigos), lotes: codigos.size, tieneInc: false, incItems: [] })
   112→    }
   113→  }
   114→  return result
   115→}

   434→  async function doSalida() {
   435→    if (!detail || !perfil) return
   436→    const item = detail.stock[salidaIdx]
   437→    if (!item) return
   438→    const qty = parseFloat(salidaCantidad)

   440→    // Las salidas normales NO pueden exceder stock. Las salidas INC sí permiten exceder (autoajuste).

   552→  async function doTransferir() {
   553→    if (!detail || !perfil || !trItem) return
   554→    if (!trDestBloque || !trDestTorre || !trDestPiso || !trDestPos) { toast.error('Completa destino'); return }
   555→    const qty = parseFloat(trCantidad)
   556→    if (isNaN(qty) || qty <= 0) { toast.error('Cantidad inválida'); return }
   557→    const origKey = `${detail.bloque}-${detail.torre}-${detail.piso}-${detail.posicion}`
   558→    const destKey = `${trDestBloque}-${trDestTorre}-${trDestPiso}-${trDestPos}`
   559→    if (origKey === destKey) { toast.error('Origen y destino no pueden ser iguales'); return }
   560→    setActionBusy(true)
   561→    try {
   562→      // Check destination occupancy
   563→      const destStock = await stockEnUbicacion(trDestBloque, trDestTorre, trDestPiso, trDestPos)
   564→      if (destStock.length > 0) {
   565→        // Mostrar alerta con opción de dar salida en vez de bloquear
   566→        setTrDestStock(destStock)
   567→        setTrSalidaCant({})
   568→        setTrSalidaTotalFlags({})
   569→        setTrDestAlertOpen(true)
   570→        setActionBusy(false)
   571→        return
   572→      }
   573→      // Destino vacío — ejecutar traslado directamente
   574→      await ejecutarTraslado()
   575→    } catch (err: unknown) {
   576→      toast.error('Error al verificar destino', { description: extractError(err) })
   577→    } finally { setActionBusy(false) }
   578→  }
