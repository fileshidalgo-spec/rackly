    32→export function StockTab() {
    33→  const { perfil } = useAuth()
    34→  const esAdmin = perfil?.rol === 'admin'
    35→  const [movs, setMovs] = useState<Movimiento[]>([])
    36→  const [query, setQuery] = useState('')
    37→  const [selectedCodigo, setSelectedCodigo] = useState('')
    38→  const [stockFilter, setStockFilter] = useState<'todos' | 'disponibles' | 'inc'>('todos')
    39→  const [stock, setStock] = useState<
    40→    {
    41→      bloque: string
    42→      torre: string
    43→      piso: string
    44→      posicion: string
    45→      stock: number
    46→      descripcion: string
    47→      un: string
    48→      proveedor?: string
    49→      fVencimiento: string
    50→      codigoInc?: string
    51→    }[]
    52→  >([])
    53→
    54→  useMovimientosRealtime(setMovs)
    55→
    56→  // Catálogo para búsqueda y lookup
    57→  const [catalogoResults, setCatalogoResults] = useState<CatalogoItem[]>([])
    58→  const [catalogoLoaded, setCatalogoLoaded] = useState(false)
    59→
    60→  // Cargar catálogo al montar
    61→  useEffect(() => {
    62→    async function load() {
    63→      if (!isCatalogoLoaded()) {
    64→        await fetchCatalogo().catch(() => {})
    65→      }
    66→      setCatalogoLoaded(true)
    67→    }
    68→    load()
    69→  }, [])
    70→
    71→  // Buscar sugerencias del catálogo cuando el usuario escribe
    72→  useEffect(() => {
    73→    if (!catalogoLoaded || !query.trim()) {
    74→      setCatalogoResults([])
    75→      return
    76→    }
    77→    if (!isCatalogoLoaded()) return
    78→    const results = searchCatalogo(query.trim(), 8)
    79→    setCatalogoResults(results)
    80→  }, [query, catalogoLoaded])
    81→
    82→  // Datos del artículo seleccionado del catálogo
    83→  const selectedItem = useMemo(() => {
    84→    if (!selectedCodigo) return null
    85→    return findCatalogoByCodigo(selectedCodigo)
    86→  }, [selectedCodigo])
    87→
    88→  // Calcular stock por ubicación para el código seleccionado.
    89→  // ESTRATEGIA: Calcular stock neto por posición (IGUAL que OcupaciónTab, sin fechas),
    90→  // luego repartir ese neto en lotes con fecha (FEFO visual).
    91→  // Lo que sobra se muestra como "Sin fecha".
    92→  // Esto garantiza que la SUMA de lotes = stock neto = OcupaciónTab, SIEMPRE.
    93→  const stockData = useMemo(() => {
    94→    if (!selectedCodigo || movs.length === 0) return []
    95→    const code = selectedCodigo.trim().toUpperCase()
    96→    const isIncMode = stockFilter === 'inc'
    97→
    98→    // FIX: Normalizar m.codigo igual que OcupaciónTab (trim + uppercase)
    99→    // para evitar perder movimientos por espacios o mayúsculas en la DB
   100→    const relevant = movs.filter((m) => {
   101→      if (m.codigo.trim().toUpperCase() !== code) return false
   102→      if (isIncMode) return !!m.codigoInc
   103→      return !m.codigoInc
   104→    })
   105→
   106→    const isIngreso = (tipo: string) => ['ingreso', 'devolucion', 'traslado'].includes(tipo)
   107→
   108→    // 1. Calcular stock neto por posición (idéntico a OcupaciónTab)
   109→    //    + recoger ingreso pool por vencimiento (solo para display FEFO)
   110→    type PosBucket = {
   111→      bloque: string; torre: string; piso: string; posicion: string
   112→      descripcion: string; un: string; proveedor?: string
   113→      netStock: number                    // stock neto real (igual que Ocupación)
   114→      ingresoPool: Map<string, number>    // fv → cantidad (solo ingresos con fecha)
   115→      ingresoTotal: number                // total ingresos con fecha
   116→    }
   117→    const posMap = new Map<string, PosBucket>()
   118→
   119→    for (const m of relevant) {
   120→      const posKey = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}`
   121→      let bucket = posMap.get(posKey)
   122→      if (!bucket) {
   123→        bucket = {
   124→          bloque: m.bloque, torre: m.torre, piso: m.piso, posicion: m.posicion,
   125→          descripcion: m.descripcion, un: m.un, proveedor: m.proveedor || undefined,
   126→          netStock: 0, ingresoPool: new Map(), ingresoTotal: 0,
   127→        }
   128→        posMap.set(posKey, bucket)
   129→      }
   130→      const delta = isIngreso(m.tipo) ? m.cantidad : -m.cantidad
   131→      bucket.netStock += delta
   132→
   133→      // Solo rastrear ingresos/devoluciones que tengan fecha para FEFO visual
   134→      if (isIngreso(m.tipo) && m.fVencimiento) {
   135→        bucket.ingresoPool.set(m.fVencimiento, (bucket.ingresoPool.get(m.fVencimiento) ?? 0) + m.cantidad)
   136→        bucket.ingresoTotal += m.cantidad
   137→      }
   138→    }
   139→
   140→    // 2. Repartir stock neto en lotes FEFO (solo informativo)
   141→    const result: {
   142→      bloque: string; torre: string; piso: string; posicion: string
   143→      stock: number; descripcion: string; un: string; proveedor?: string
   144→      fVencimiento: string; codigoInc?: string
   145→    }[] = []
   146→
   147→    for (const [, bucket] of posMap) {
   148→      if (bucket.netStock <= 0) continue
   149→
   150→      // Lotes con fecha ordenados por FEFO (más antiguo primero)
   151→      const datedLots = [...bucket.ingresoPool.entries()].sort(([a], [b]) => a.localeCompare(b))
   152→
   153→      // Repartir el stock neto proporcionalmente entre los lotes con fecha
   154→      let remaining = bucket.netStock
   155→
   156→      // Caso especial: no hay ingresos con fecha → todo es "Sin fecha"
   157→      if (bucket.ingresoTotal === 0 || datedLots.length === 0) {
   158→        result.push({
   159→          bloque: bucket.bloque, torre: bucket.torre, piso: bucket.piso,
   160→          posicion: bucket.posicion, stock: bucket.netStock, descripcion: bucket.descripcion,
   161→          un: bucket.un, proveedor: bucket.proveedor, fVencimiento: '',
   162→        })
   163→        continue
   164→      }
   165→
   166→      // Si stock neto > total ingresos con fecha, asignar primero todos los lotes con fecha
   167→      // y el excedente va a "Sin fecha" (solo puede pasar si hay ingresos sin fecha)
   168→      for (const [fv, lotQty] of datedLots) {
   169→        // Asignar proporcionalmente: lo que quedó del neto, pero no más del lote
   170→        const assigned = Math.min(lotQty, remaining)
   171→        if (assigned > 0) {
   172→          result.push({
   173→            bloque: bucket.bloque, torre: bucket.torre, piso: bucket.piso,
   174→            posicion: bucket.posicion, stock: assigned, descripcion: bucket.descripcion,
   175→            un: bucket.un, proveedor: bucket.proveedor, fVencimiento: fv,
   176→          })
   177→          remaining -= assigned
   178→        }
   179→      }
   180→
   181→      // Si sobra stock (hubo ingresos sin fecha), mostrar como "Sin fecha"
   182→      if (remaining > 0) {
   183→        result.push({
   184→          bloque: bucket.bloque, torre: bucket.torre, piso: bucket.piso,
   185→          posicion: bucket.posicion, stock: remaining, descripcion: bucket.descripcion,
   186→          un: bucket.un, proveedor: bucket.proveedor, fVencimiento: '',
   187→        })
   188→      }
   189→    }
   190→
   191→    return result.sort((a, b) => {
   192→        // FEFO primero (con fecha de vencimiento), luego sin fecha por bloque (1→7)
   193→        const aHasDate = !!a.fVencimiento
   194→        const bHasDate = !!b.fVencimiento
   195→        if (aHasDate && bHasDate) return a.fVencimiento.localeCompare(b.fVencimiento)
   196→        if (aHasDate && !bHasDate) return -1
   197→        if (!aHasDate && bHasDate) return 1
   198→        // Ambos sin fecha: ordenar por bloque, torre, piso, posición
   199→        const aB = parseInt(a.bloque, 10) || 0
   200→        const bB = parseInt(b.bloque, 10) || 0
   201→        if (aB !== bB) return aB - bB
   202→        const aT = parseInt(a.torre, 10) || 0
   203→        const bT = parseInt(b.torre, 10) || 0
   204→        if (aT !== bT) return aT - bT
   205→        const aP = parseInt(a.piso, 10) || 0
   206→        const bP = parseInt(b.piso, 10) || 0
   207→        if (aP !== bP) return aP - bP
   208→        const aPos = parseInt(a.posicion, 10) || 0
   209→        const bPos = parseInt(b.posicion, 10) || 0
   210→        return aPos - bPos
   211→      })
   212→  }, [selectedCodigo, movs, stockFilter])
   213→
   214→  useEffect(() => {
   215→    setStock(stockData)
   216→  }, [stockData])
   217→
   218→  // Manejar selección de un código
   219→  function selectCodigo(codigo: string) {
   220→    setSelectedCodigo(codigo.toUpperCase())
   221→    setQuery(codigo.toUpperCase())
   222→    setCatalogoResults([])
   223→  }

   225→  async function handleDelete(
   226→    bloque: string,
   227→    torre: string,
   228→    piso: string,
   229→    posicion: string,
   230→    fVencimiento?: string,
   231→    codigoInc?: string
   232→  ) {
   233→    if (!confirm('¿Eliminar los movimientos de este lote en esta ubicación?')) return
   234→    try {
   235→      const next = await eliminarUbicacion(selectedCodigo, bloque, torre, piso, posicion, fVencimiento, codigoInc)
   236→      setMovs(next)
   237→      toast.success('Ubicación eliminada')
   238→    } catch (err: unknown) {
   239→      const message = err instanceof Error ? err.message : 'Error'
   240→      toast.error('No se pudo eliminar', { description: message })
   241→    }
   242→  }

   244→  // Badge color según días para vencer
   245→  function getBadgeClass(fVencimiento: string) {
   246→    const hoy = new Date(new Date().toDateString())
   247→    const fVen = fVencimiento ? new Date(fVencimiento + 'T00:00:00') : null
   248→    const diffDias = fVen ? (fVen.getTime() - hoy.getTime()) / (1000 * 60 * 60 * 24) : null
   249→    const vencido = diffDias !== null && diffDias < 0
   250→    const naranja = !vencido && diffDias !== null && diffDias <= 15
   251→    const azul = !vencido && !naranja && diffDias !== null && diffDias <= 30
   252→
   253→    return vencido
   254→      ? 'bg-red-100 text-red-700 border-red-200 dark:bg-red-950/40 dark:text-red-300 dark:border-red-800'
   255→      : naranja
   256→      ? 'bg-orange-100 text-orange-700 border-orange-200 dark:bg-orange-950/40 dark:text-orange-300 dark:border-orange-800'
   257→      : azul
   258→      ? 'bg-blue-100 text-blue-700 border-blue-200 dark:bg-blue-950/40 dark:text-blue-300 dark:border-blue-800'
   259→      : 'bg-green-100 text-green-700 border-green-200 dark:bg-green-950/40 dark:text-green-300 dark:border-green-800'
   260→  }
   261→
   262→  // ── Sin código seleccionado: buscar por código o descripción ──
   263→  if (!selectedCodigo) {
   264→    return (
   265→      <div className="space-y-4">
   266→        {/* Barra de búsqueda */}
   267→        <div className="relative">
   268→          <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
   269→          <Input
   270→            value={query}
   271→            onChange={(e) => {
   272→              setQuery(e.target.value)
   273→              // Si el usuario escribió un código exacto, seleccionarlo directo
   274→              const val = e.target.value.trim().toUpperCase()
   275→              if (isCatalogoLoaded() && findCatalogoByCodigo(val)) {
   276→                // No seleccionar automáticamente, dejar que el usuario elija
   277→              }
   278→            }}
   279→            onKeyDown={(e) => {
   280→              if (e.key === 'Enter' && query.trim()) {
   281→                // Buscar el primer resultado y seleccionarlo
   282→                if (catalogoResults.length > 0) {
   283→                  selectCodigo(catalogoResults[0].codigo)
   284→                } else {
   285→                  // Si no hay resultados en catálogo, buscar de todas formas
   286→                  selectCodigo(query.trim())
   287→                }
   288→              }
   289→              if (e.key === 'Escape') {
   290→                setQuery('')
   291→                setCatalogoResults([])
   292→              }
   293→            }}
   294→            placeholder="Buscar por código o descripción..."
   295→            className="pl-9"
   296→          />
   297→          {query && (
   298→            <button
   299→              type="button"
   300→              className="absolute right-2 top-1/2 -translate-y-1/2 text-muted-foreground hover:text-foreground"
   301→              onClick={() => { setQuery(''); setCatalogoResults([]) }}
   302→            >
   303→              ✕
   304→            </button>
   305→          )}
   306→        </div>
   307→
   308→        {/* Sugerencias del catálogo */}
   309→        {catalogoResults.length > 0 && (
   310→          <div className="rounded-lg border bg-card shadow-sm overflow-hidden">
   311→            <div className="px-3 py-2 border-b bg-muted/30">
   312→              <p className="text-xs font-medium text-muted-foreground">
   313→                {catalogoResults.length} resultado{catalogoResults.length > 1 ? 's' : ''} en el catálogo
   314→              </p>
   315→            </div>
   316→            {catalogoResults.map((item) => (
   317→              <button
   318→                key={item.codigo}
   319→                type="button"
   320→                className="w-full flex items-center gap-3 px-3 py-2.5 hover:bg-accent/50 transition-colors text-left border-b last:border-b-0"
   321→                onClick={() => selectCodigo(item.codigo)}
   322→              >
   323→                <div className="min-w-0 flex-1">
   324→                  <div className="flex items-center gap-2">
   325→                    <span className="font-mono font-semibold text-sm">{item.codigo}</span>
   326→                    <Badge variant="outline" className="text-[10px] px-1.5 py-0 shrink-0">{item.un}</Badge>
   327→                  </div>
   328→                  <p className="text-xs text-muted-foreground truncate mt-0.5">{item.descripcion}</p>
   329→                </div>
   330→                <div className="text-right shrink-0">
   331→                  {item.stock_big_magic > 0 && (
   332→                    <div className="flex items-center gap-1">
   333→                      <span className="text-[10px] text-amber-600 dark:text-amber-400">BM</span>
   334→                      <span className="text-sm font-bold text-amber-700 dark:text-amber-300">{item.stock_big_magic}</span>
   335→                    </div>
   336→                  )}
   337→                </div>
   338→                <ArrowRight className="h-3.5 w-3.5 text-muted-foreground shrink-0" />
   339→              </button>
   340→            ))}
   341→          </div>
   342→        )}
   343→
   344→        {/* Sin búsqueda activa */}
   345→        {!query && (
   346→          <p className="text-muted-foreground text-center py-8">
   347→            Escribe un código o descripción para buscar stock.
   348→          </p>
   349→        )}
   350→
   351→        {/* Búsqueda sin resultados */}
   352→        {query.trim() && catalogoResults.length === 0 && catalogoLoaded && (
   353→          <div className="flex items-center gap-2 text-muted-foreground py-8 justify-center">
   354→            <PackageSearch className="h-5 w-5" />
   355→            <span>No se encontró &quot;{query}&quot; en el catálogo.</span>
   356→          </div>
   357→        )}
   358→      </div>
   359→    )
   360→  }
   361→
   362→  // ── Con código seleccionado: mostrar stock por ubicación o info de Big Magic ──
   363→  return (
   364→    <div className="space-y-4">
   365→      {/* Barra de búsqueda con código seleccionado */}
   366→      <div className="relative">
   367→        <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
   368→        <Input
   369→          value={query}
   370→          onChange={(e) => {
   371→            setQuery(e.target.value)
   372→            setSelectedCodigo('')
   373→          }}
   374→          onKeyDown={(e) => {
   375→            if (e.key === 'Escape') {
   376→              setSelectedCodigo('')
   377→              setQuery('')
   378→            }
   379→          }}
   380→          placeholder="Buscar por código o descripción..."
   381→          className="pl-9"
   382→        />
   383→        <button
   384→          type="button"
   385→          className="absolute right-2 top-1/2 -translate-y-1/2 text-xs text-muted-foreground hover:text-foreground px-1.5 py-0.5 rounded border hover:border-foreground/30 transition-colors"
   386→          onClick={() => { setSelectedCodigo(''); setQuery('') }}
   387→        >
   388→          × Cambiar
   389→        </button>
   390→      </div>
   391→
   392→      {/* Info del artículo seleccionado */}
   393→      {selectedItem && (
   394→        <div className="rounded-lg border bg-card p-3 space-y-2">
   395→          <div className="flex items-center gap-2">
   396→            <span className="font-mono font-bold text-sm">{selectedItem.codigo}</span>
   397→            <Badge variant="outline" className="text-xs">{selectedItem.un}</Badge>
   398→          </div>
   399→          <p className="text-sm">{selectedItem.descripcion}</p>
   400→        </div>
   401→      )}
   402→
   403→      {/* Card de Stock Big Magic */}
   404→      {selectedItem && (
   405→        <div className="rounded-lg border border-amber-200 dark:border-amber-800 bg-gradient-to-r from-amber-50 to-orange-50 dark:from-amber-950/40 dark:to-orange-950/40 p-3 flex flex-col sm:flex-row sm:items-center sm:justify-between gap-3">
   406→          <div className="flex items-center gap-2">
   407→            <div className="w-8 h-8 rounded-lg bg-amber-500/10 flex items-center justify-center">
   408→              <span className="text-amber-600 dark:text-amber-400 font-bold text-xs">BM</span>
   409→            </div>
   410→            <div>
   411→              <p className="text-xs text-amber-600/80 dark:text-amber-400/80 font-medium">Stock Big Magic</p>
   412→              <p className="text-xs text-muted-foreground">Stock disponible en sistema Big Magic</p>
   413→            </div>
   414→          </div>
   415→          <div className="text-right">
   416→            <p className="text-2xl font-bold text-amber-700 dark:text-amber-300">{selectedItem.stock_big_magic}</p>
   417→          </div>
   418→        </div>
   419→      )}
   420→
   421→      {/* ── INC Filter buttons (always visible when a code is selected) ── */}
   422→      <div className="flex items-center justify-between">
   423→        <p className="text-xs font-medium text-muted-foreground flex items-center gap-1.5">
   424→          <Warehouse className="h-3.5 w-3.5" />
   425→          {stock.length > 0 ? 'Stock por ubicación en RACKLY' : 'Sin stock en ubicaciones de RACKLY'}
   426→        </p>
   427→      </div>
   428→      <div className="flex gap-2">
   429→        {([['todos', 'Todos'], ['disponibles', 'Disponibles'], ['inc', 'Solo INC']] as const).map(([key, label]) => (
   430→          <button
   431→            key={key}
   432→            type="button"
   433→            onClick={() => setStockFilter(key)}
   434→            className={`flex items-center gap-1.5 px-3 py-1.5 rounded-lg text-xs font-bold transition-all border ${
   435→              stockFilter === key
   436→                ? key === 'inc'
   437→                  ? 'bg-amber-500/20 border-amber-500/50 text-amber-300'
   438→                  : key === 'disponibles'
   439→                    ? 'bg-emerald-500/20 border-emerald-500/50 text-emerald-300'
   440→                    : 'bg-sky-500/20 border-sky-500/50 text-sky-300'

   457→            {stock.map((s, i) => (
   458→              <div key={i} className="rounded-lg border bg-card p-3 space-y-2">
   459→                <div className="flex items-center justify-between gap-2">
   460→                  <div className="flex items-center gap-1.5 text-xs font-semibold">
   461→                    <span className="text-muted-foreground">Bloq</span>
   462→                    <span className="font-mono">{s.bloque}</span>
   463→                    <span className="text-muted-foreground mx-0.5">|</span>
   464→                    <span className="text-muted-foreground">Tor</span>
   465→                    <span className="font-mono">{s.torre}</span>
   466→                    <span className="text-muted-foreground mx-0.5">|</span>
   467→                    <span className="text-muted-foreground">Pis</span>
   468→                    <span className="font-mono">{s.piso}</span>
   469→                    <span className="text-muted-foreground mx-0.5">|</span>
   470→                    <span className="text-muted-foreground">Pos</span>
   471→                    <span className="font-mono">{s.posicion}</span>
   472→                  </div>
   473→                  <div className="flex items-center gap-2">
   474→                    <Badge variant="default" className="text-sm">{s.stock}</Badge>
   475→                    {esAdmin && (
