    73тЖТ    73тЖТexport async function fetchMovimientos(): Promise<Movimiento[]> {
    74тЖТ    74тЖТ  const all: Record<string, unknown>[] = []
    75тЖТ    75тЖТ  let from = 0
    76тЖТ    76тЖТ  let iterations = 0
    77тЖТ    77тЖТ  while (iterations < FETCH_MOV_MAX_PAGES) {
    78тЖТ    78тЖТ    iterations++
    79тЖТ    79тЖТ    const to = from + PAGE_SIZE - 1
    80тЖТ    80тЖТ    const { data, error } = await dataClient
    81тЖТ    81тЖТ      .from('movimientos')
    82тЖТ    82тЖТ      .select('*')
    83тЖТ    83тЖТ      .order('f_modificacion', { ascending: false })
    84тЖТ    84тЖТ      .order('id', { ascending: true })
    85тЖТ    85тЖТ      .range(from, to)
    86тЖТ    86тЖТ    if (error) throw error
    87тЖТ    87тЖТ    const rows = data ?? []
    88тЖТ    88тЖТ    all.push(...rows)
    89тЖТ    89тЖТ    if (rows.length < PAGE_SIZE) break
    90тЖТ    90тЖТ    from += PAGE_SIZE
    91тЖТ    91тЖТ  }
    92тЖТ    92тЖТ  return all.map(fromRow)
    93тЖТ    93тЖТ}

   121тЖТ   121тЖТ * Calcula el stock neto en Racks para un c├│digo dado.
   122тЖТ   122тЖТ * Usado por PisoStockTab para mostrar "Tambi├йn en Kardex Racks".

   125тЖТ   125тЖТexport async function buscarStockRacksPorCodigo(codigo: string): Promise<StockRacksPorCodigoItem[]> {
   126тЖТ   126тЖТ  const movs = await fetchMovimientosByCodigo(codigo)
   127тЖТ   127тЖТ  const locMap = new Map<string, StockRacksPorCodigoItem>()
   128тЖТ   128тЖТ
   129тЖТ   129тЖТ  for (const m of movs) {
   130тЖТ   130тЖТ    const key = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}||${m.fVencimiento || ''}||${m.codigoInc || ''}`
   131тЖТ   131тЖТ    const current = locMap.get(key)
   132тЖТ   132тЖТ    const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? m.cantidad : -m.cantidad
   133тЖТ   133тЖТ    if (current) {
   134тЖТ   134тЖТ      current.stock += delta
   135тЖТ   135тЖТ    } else {
   136тЖТ   136тЖТ      locMap.set(key, {
   137тЖТ   137тЖТ        bloque: m.bloque,
   138тЖТ   138тЖТ        torre: m.torre,
   139тЖТ   139тЖТ        piso: m.piso,
   140тЖТ   140тЖТ        posicion: m.posicion,
   141тЖТ   141тЖТ        stock: delta,
   142тЖТ   142тЖТ        descripcion: m.descripcion,
   143тЖТ   143тЖТ        un: m.un,
   144тЖТ   144тЖТ        fVencimiento: m.fVencimiento || '',
   145тЖТ   145тЖТ        codigo_inc: m.codigoInc || undefined,
   146тЖТ   146тЖТ      })
   147тЖТ   147тЖТ    }
   148тЖТ   148тЖТ  }
   149тЖТ   149тЖТ
   150тЖТ   150тЖТ  return Array.from(locMap.values())
   151тЖТ   151тЖТ    .filter(l => l.stock > 0)
   152тЖТ   152тЖТ    .sort((a, b) => {
   153тЖТ   153тЖТ      // FEFO primero
   154тЖТ   154тЖТ      if (a.fVencimiento && b.fVencimiento) return a.fVencimiento.localeCompare(b.fVencimiento)
   155тЖТ   155тЖТ      if (a.fVencimiento && !b.fVencimiento) return -1
   156тЖТ   156тЖТ      if (!a.fVencimiento && b.fVencimiento) return 1
   157тЖТ   157тЖТ      return 0
   158тЖТ   158тЖТ    })
   159тЖТ   159тЖТ}

   162тЖТ   162тЖТasync function addMovimientoFallback(
   163тЖТ   163тЖТ  m: Omit<Movimiento, 'id' | 'fModificacion'>,
   164тЖТ   164тЖТ  uuidSync?: string,
   165тЖТ   165тЖТ  skipValidation: boolean = false
   166тЖТ   166тЖТ): Promise<Movimiento[]> {
   167тЖТ   167тЖТ  console.warn('[addMovimiento] RPC no encontrada, usando insert directo como fallback')
   168тЖТ   168тЖТ
   169тЖТ   169тЖТ  // тФАтФА Validaci├│n de stock negativo para salidas NORMALES (no INC) тФАтФА
   170тЖТ   170тЖТ  // Las salidas normales NUNCA deben generar stock negativo.
   171тЖТ   171тЖТ  // Los traslados se manejan aparte con autoajuste.
   172тЖТ   172тЖТ  // Stock = suma de TODOS los movimientos en la ubicacion para ese codigo.
   173тЖТ   173тЖТ  // f_vencimiento es solo para FEFO, NO para calcular stock.

   175тЖТ   175тЖТ    try {
   176тЖТ   176тЖТ      // Stock real = movimientos en la ubicacion para ese codigo, EXCLUYENDO INC.
   177тЖТ   177тЖТ      // f_vencimiento es solo para FEFO, NO para calcular stock.

   179тЖТ   179тЖТ        m.codigo, m.bloque, m.torre, m.piso, m.posicion, true // excluir INC
   180тЖТ   180тЖТ      )
   181тЖТ   181тЖТ      // Tolerancia de 0.001 para redondeo (display redondea a 3 decimales)
   182тЖТ   182тЖТ      const stockRedondeado = Math.round(currentStock * 1000) / 1000
   183тЖТ   183тЖТ      if (m.cantidad > stockRedondeado + 0.001) {
   184тЖТ   184тЖТ        const err = new Error('INSUFFICIENT_STOCK')
   185тЖТ   185тЖТ        ;(err as unknown as Record<string, string>).detail =
   186тЖТ   186тЖТ          `Stock actual = ${stockRedondeado} ${m.un}, cantidad solicitada = ${m.cantidad} ${m.un}`

   188тЖТ   188тЖТ      }
   189тЖТ   189тЖТ    } catch (stockErr) {
   190тЖТ   190тЖТ      // Si es error de stock insuficiente, propagarlo
   191тЖТ   191тЖТ      if (stockErr instanceof Error && stockErr.message === 'INSUFFICIENT_STOCK') throw stockErr
   192тЖТ   192тЖТ      // Si falla la consulta de stock, NO permitir la inserci├│n sin validar.
   193тЖТ   193тЖТ      // Esto previene salidas con stock negativo cuando la red es inestable.
   194тЖТ   194тЖТ      console.error('[addMovimientoFallback] No se pudo verificar stock, BLOQUEANDO inserci├│n:', stockErr)
   195тЖТ   195тЖТ      throw new Error('STOCK_VALIDATION_FAILED|No se pudo verificar el stock. Reintente.')
   196тЖТ   196тЖТ    }
   197тЖТ   197тЖТ  }
   198тЖТ   198тЖТ
   199тЖТ   199тЖТ  const { error } = await dataClient.from('movimientos').insert({
   200тЖТ   200тЖТ    tipo: m.tipo,
   201тЖТ   201тЖТ    bloque: m.bloque,
   202тЖТ   202тЖТ    torre: m.torre,
   203тЖТ   203тЖТ    piso: m.piso,
   204тЖТ   204тЖТ    posicion: m.posicion,
   205тЖТ   205тЖТ    codigo: m.codigo.trim().toUpperCase(),
   206тЖТ   206тЖТ    descripcion: m.descripcion,
   207тЖТ   207тЖТ    un: m.un,
   208тЖТ   208тЖТ    cantidad: m.cantidad,
   209тЖТ   209тЖТ    f_vencimiento: m.fVencimiento || null,
   210тЖТ   210тЖТ    turno: m.turno,
   211тЖТ   211тЖТ    usuario_id: m.usuarioId,
   212тЖТ   212тЖТ    usuario_nombre: m.usuarioNombre ?? null,
   213тЖТ   213тЖТ    usuario_correo: m.usuarioCorreo ?? null,
   214тЖТ   214тЖТ    proveedor: m.proveedor ? m.proveedor : null,
   215тЖТ   215тЖТ    uuid_sync: uuidSync || null,
   216тЖТ   216тЖТ    codigo_inc: m.codigoInc || null,
   217тЖТ   217тЖТ  })
   218тЖТ   218тЖТ  if (error) throw error
   219тЖТ   219тЖТ  try {
   220тЖТ   220тЖТ    return await fetchMovimientos()
   221тЖТ   221тЖТ  } catch (fetchErr) {
   222тЖТ   222тЖТ    console.warn('[addMovimientoFallback] Insert exitoso pero fetchMovimientos() fall├│.', fetchErr)
   223тЖТ   223тЖТ    return []
   224тЖТ   224тЖТ  }
   225тЖТ   225тЖТ}

   244тЖТ   244тЖТexport async function addMovimiento(
   245тЖТ   245тЖТ  m: Omit<Movimiento, 'id' | 'fModificacion'>,
   246тЖТ   246тЖТ  uuidSync?: string
   247тЖТ   247тЖТ): Promise<Movimiento[]> {
   248тЖТ   248тЖТ  // Idempotencia: si viene uuidSync, verificar si ya existe en el servidor
   249тЖТ   249тЖТ  if (uuidSync) {
   250тЖТ   250тЖТ    const exists = await checkExistingByUuidSync(uuidSync)
   251тЖТ   251тЖТ    if (exists) {
   252тЖТ   252тЖТ      console.warn('[addMovimiento] Movimiento ya existe (uuid_sync), saltando insert:', uuidSync)
   253тЖТ   253тЖТ      // Ya existe: refrescar y retornar como si hubiera sido exitoso
   254тЖТ   254тЖТ      return await fetchMovimientos()
   255тЖТ   255тЖТ    }
   256тЖТ   256тЖТ  }
   257тЖТ   257тЖТ
   258тЖТ   258тЖТ  // NOTA: Los movimientos INC ahora pasan por la RPC como cualquier otro movimiento.
   259тЖТ   259тЖТ  // La RPC registrar_movimiento_kardex ya tiene p_codigo_inc (migration 20260611).
   260тЖТ   260тЖТ  // Esto garantiza advisory lock y validaci├│n at├│mica para TODOS los movimientos.
   261тЖТ   261тЖТ
   262тЖТ   262тЖТ  // Usar RPC at├│mica con advisory lock para evitar race conditions.
   263тЖТ   263тЖТ  // La RPC maneja TODOS los tipos de movimiento incluyendo INC.
   264тЖТ   264тЖТ  // INC items son tipo 'ingreso' as├н que no pasan validaci├│n de stock negativo.
   265тЖТ   265тЖТ  try {
   266тЖТ   266тЖТ    const { data, error } = await dataClient.rpc('registrar_movimiento_kardex', {
   267тЖТ   267тЖТ      p_tipo: m.tipo,
   268тЖТ   268тЖТ      p_bloque: m.bloque,
   269тЖТ   269тЖТ      p_torre: m.torre,
   270тЖТ   270тЖТ      p_piso: m.piso,
   271тЖТ   271тЖТ      p_posicion: m.posicion,
   272тЖТ   272тЖТ      p_codigo: m.codigo.trim().toUpperCase(),
   273тЖТ   273тЖТ      p_descripcion: m.descripcion,
   274тЖТ   274тЖТ      p_un: m.un,
   275тЖТ   275тЖТ      p_cantidad: m.cantidad,
   276тЖТ   276тЖТ      p_f_vencimiento: m.fVencimiento || null,
   277тЖТ   277тЖТ      p_turno: m.turno,
   278тЖТ   278тЖТ      p_usuario_id: m.usuarioId,
   279тЖТ   279тЖТ      p_usuario_nombre: m.usuarioNombre ?? null,
   280тЖТ   280тЖТ      p_usuario_correo: m.usuarioCorreo ?? null,
   281тЖТ   281тЖТ      p_proveedor: m.proveedor ? m.proveedor : null,
   282тЖТ   282тЖТ      p_uuid_sync: uuidSync || null,
   283тЖТ   283тЖТ      p_codigo_inc: m.codigoInc || null,
   284тЖТ   284тЖТ    })
   285тЖТ   285тЖТ    // Stock insuficiente es un error controlado тАФ NUNCA bypassear la decisi├│n del RPC.
   286тЖТ   286тЖТ    // El RPC tiene advisory lock y calcula stock de forma at├│mica; es la fuente de verdad.
   287тЖТ   287тЖТ    if (error) {
   288тЖТ   288тЖТ      const msg = error.message || ''
   289тЖТ   289тЖТ      const code = (error as unknown as Record<string, string>).code || ''
   290тЖТ   290тЖТ      if (msg.includes('INSUFFICIENT_STOCK')) {
   291тЖТ   291тЖТ        const parts = msg.split('|')
   292тЖТ   292тЖТ        const rpcDetail = parts.length > 1 ? parts[1] : 'Stock insuficiente para esta operaci├│n'
   293тЖТ   293тЖТ        const err = new Error('INSUFFICIENT_STOCK')
   294тЖТ   294тЖТ        ;(err as unknown as Record<string, string>).detail = rpcDetail
   295тЖТ   295тЖТ        throw err
   296тЖТ   296тЖТ      }
   297тЖТ   297тЖТ      // Si la RPC no existe (404 / 42883 / 'Could not find'), usar fallback
   298тЖТ   298тЖТ      if (code === '42883' || code === 'PGRST202' || msg.includes('Could not find') || msg.includes('does not exist') || msg.includes('404')) {
   299тЖТ   299тЖТ        return await addMovimientoFallback(m, uuidSync)
   300тЖТ   300тЖТ      }
   301тЖТ   301тЖТ      throw error
   302тЖТ   302тЖТ    }
   303тЖТ   303тЖТ    // RPC exitosa: el movimiento ya fue registrado en la DB.
   304тЖТ   304тЖТ    // Intentar refrescar la lista de movimientos, pero si falla (timeout, red)
   305тЖТ   305тЖТ    // NO propagar el error тАФ el movimiento ya est├б guardado. Retornar array vac├нo
   306тЖТ   306тЖТ    // para que el caller no reciba null (evita crashes en .map()).
   307тЖТ   307тЖТ    try {
   308тЖТ   308тЖТ      return await fetchMovimientos()
   309тЖТ   309тЖТ    } catch (fetchErr) {
   310тЖТ   310тЖТ      console.warn('[addMovimiento] RPC exitosa pero fetchMovimientos() fall├│. Movimiento ya registrado.', fetchErr)
   311тЖТ   311тЖТ      return []
   312тЖТ   312тЖТ    }
   313тЖТ   313тЖТ  } catch (err: unknown) {
   314тЖТ   314тЖТ    // Si la RPC no existe a├║n (SQL no ejecutado), fallback al insert directo
   315тЖТ   315тЖТ    const errMsg = err instanceof Error ? err.message : ''

   373тЖТ   373тЖТexport async function stockEnUbicacion(
   374тЖТ   374тЖТ  bloque: string,
   375тЖТ   375тЖТ  torre: string,
   376тЖТ   376тЖТ  piso: string,
   377тЖТ   377тЖТ  posicion: string
   378тЖТ   378тЖТ): Promise<StockEnUbicacion[]> {
   379тЖТ   379тЖТ  try {
   380тЖТ   380тЖТ    // M├йtodo principal: consultar movimientos directamente con paginaci├│n
   381тЖТ   381тЖТ    // f_vencimiento es SOLO para FEFO (ordenamiento), NO para particionar stock.
   382тЖТ   382тЖТ    // Se agrupa por (codigo, codigo_inc) тАФ el stock es la suma de TODOS los lotes.
   383тЖТ   383тЖТ    const allRows: Record<string, unknown>[] = []
   384тЖТ   384тЖТ    let from = 0
   385тЖТ   385тЖТ    const BATCH = 1000
   386тЖТ   386тЖТ    const MAX_PAGES = 10 // m├бximo 10,000 movimientos por ubicaci├│n
   387тЖТ   387тЖТ

   389тЖТ   389тЖТ      const { data, error } = await dataClient
   390тЖТ   390тЖТ        .from('movimientos')
   391тЖТ   391тЖТ        .select('*')
   392тЖТ   392тЖТ        .eq('bloque', bloque)
   393тЖТ   393тЖТ        .eq('torre', torre)
   394тЖТ   394тЖТ        .eq('piso', piso)
   395тЖТ   395тЖТ        .eq('posicion', posicion)
   396тЖТ   396тЖТ        .order('f_modificacion', { ascending: true })
   397тЖТ   397тЖТ        .range(from, from + BATCH - 1)
   398тЖТ   398тЖТ
   399тЖТ   399тЖТ      if (error) throw error
   400тЖТ   400тЖТ      const rows = data ?? []
   401тЖТ   401тЖТ      allRows.push(...rows)
   402тЖТ   402тЖТ      if (rows.length < BATCH) break
   403тЖТ   403тЖТ      from += BATCH
   404тЖТ   404тЖТ    }
   405тЖТ   405тЖТ
   406тЖТ   406тЖТ    // Agrupar por (codigo, codigo_inc) тАФ f_vencimiento NO participa en la agrupaci├│n.
   407тЖТ   407тЖТ    // Se rastrea la fecha de vencimiento m├бs pr├│xima (FEFO) para cada grupo.
   408тЖТ   408тЖТ    const groups = new Map<string, {
   409тЖТ   409тЖТ      codigo: string; descripcion: string; un: string;
   410тЖТ   410тЖТ      stock: number; fVencimientoMasProxima: string;
   411тЖТ   411тЖТ      usuarioPrimerNombre: string; proveedor: string; codigoInc: string;
   412тЖТ   412тЖТ    }>()
   413тЖТ   413тЖТ
   414тЖТ   414тЖТ    for (const r of allRows) {
   415тЖТ   415тЖТ      const m = fromRow(r)
   416тЖТ   416тЖТ      const incKey = m.codigoInc || ''
   417тЖТ   417тЖТ      const key = `${m.codigo}||${incKey}`
   418тЖТ   418тЖТ
   419тЖТ   419тЖТ      let group = groups.get(key)
   420тЖТ   420тЖТ      if (!group) {
   421тЖТ   421тЖТ        group = {
   422тЖТ   422тЖТ          codigo: m.codigo,
   423тЖТ   423тЖТ          descripcion: m.descripcion,
   424тЖТ   424тЖТ          un: m.un,
   425тЖТ   425тЖТ          stock: 0,
   426тЖТ   426тЖТ          fVencimientoMasProxima: m.fVencimiento || '',
   427тЖТ   427тЖТ          usuarioPrimerNombre: m.usuarioNombre?.split(' ')[0] ?? '',
   428тЖТ   428тЖТ          proveedor: m.proveedor ?? '',
   429тЖТ   429тЖТ          codigoInc: incKey,
   430тЖТ   430тЖТ        }
   431тЖТ   431тЖТ        groups.set(key, group)
   432тЖТ   432тЖТ      } else {
   433тЖТ   433тЖТ        // Si el grupo no tiene descripci├│n pero este movimiento s├н, usarla
   434тЖТ   434тЖТ        if (!group.descripcion && m.descripcion) group.descripcion = m.descripcion
   435тЖТ   435тЖТ        // Rastrear la fecha de vencimiento m├бs pr├│xima (para FEFO)
   436тЖТ   436тЖТ        if (m.fVencimiento) {
   437тЖТ   437тЖТ          if (!group.fVencimientoMasProxima || m.fVencimiento < group.fVencimientoMasProxima) {
   438тЖТ   438тЖТ            group.fVencimientoMasProxima = m.fVencimiento
   439тЖТ   439тЖТ          }
   440тЖТ   440тЖТ        }
   441тЖТ   441тЖТ      }
   442тЖТ   442тЖТ
   443тЖТ   443тЖТ      // Calcular stock neto (ingreso/devolucion/traslado = +, salida = -)
   444тЖТ   444тЖТ      const qty = typeof m.cantidad === 'number' ? m.cantidad : parseFloat(String(m.cantidad)) || 0
   445тЖТ   445тЖТ      const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? qty : -qty
   446тЖТ   446тЖТ      group.stock += delta
   447тЖТ   447тЖТ    }
   448тЖТ   448тЖТ
   449тЖТ   449тЖТ    // Filtrar solo art├нculos con stock > 0
   450тЖТ   450тЖТ    const results = Array.from(groups.values()).filter(g => g.stock > 0)
   451тЖТ   451тЖТ
   452тЖТ   452тЖТ    // Ordenar por FEFO: fecha de vencimiento m├бs pr├│xima primero, sin fecha al final
   453тЖТ   453тЖТ    results.sort((a, b) => {
   454тЖТ   454тЖТ      if (a.fVencimientoMasProxima && b.fVencimientoMasProxima) return a.fVencimientoMasProxima.localeCompare(b.fVencimientoMasProxima)
   455тЖТ   455тЖТ      if (a.fVencimientoMasProxima && !b.fVencimientoMasProxima) return -1
   456тЖТ   456тЖТ      if (!a.fVencimientoMasProxima && b.fVencimientoMasProxima) return 1
   457тЖТ   457тЖТ      return 0
   458тЖТ   458тЖТ    })
   459тЖТ   459тЖТ
   460тЖТ   460тЖТ    return results.map(g => ({
   461тЖТ   461тЖТ      codigo: g.codigo,
   462тЖТ   462тЖТ      descripcion: g.descripcion,
   463тЖТ   463тЖТ      un: g.un,
   464тЖТ   464тЖТ      stock: Math.round(g.stock * 1000) / 1000,
   465тЖТ   465тЖТ      fVencimiento: g.fVencimientoMasProxima || undefined,
   466тЖТ   466тЖТ      usuarioPrimerNombre: g.usuarioPrimerNombre || undefined,
   467тЖТ   467тЖТ      proveedor: g.proveedor || undefined,
   468тЖТ   468тЖТ      codigoInc: g.codigoInc || undefined,
   469тЖТ   469тЖТ    }))
   470тЖТ   470тЖТ  } catch {
   471тЖТ   471тЖТ    // Fallback al RPC original si falla la consulta directa
   472тЖТ   472тЖТ    try {
   473тЖТ   473тЖТ      const { data, error } = await dataClient.rpc('stock_en_ubicacion', {
   474тЖТ   474тЖТ        _bloque: bloque,
   475тЖТ   475тЖТ        _torre: torre,
   476тЖТ   476тЖТ        _piso: piso,
   477тЖТ   477тЖТ        _posicion: posicion,
   478тЖТ   478тЖТ      })
   479тЖТ   479тЖТ      if (error) throw error
   480тЖТ   480тЖТ      return ((data ?? []) as Record<string, unknown>[]).map((r) => ({
   481тЖТ   481тЖТ        codigo: String(r.codigo ?? '').trim().toUpperCase(),
   482тЖТ   482тЖТ        descripcion: (r.descripcion as string) ?? '',
   483тЖТ   483тЖТ        un: (r.un as string) ?? '',
   484тЖТ   484тЖТ        stock: Number(r.stock ?? 0),
   485тЖТ   485тЖТ        fVencimiento: (r.f_vencimiento as string) ?? '',
   486тЖТ   486тЖТ        usuarioPrimerNombre: (r.usuario_primer_nombre as string) ?? '',
   487тЖТ   487тЖТ        proveedor: (r.proveedor as string) ?? '',
   488тЖТ   488тЖТ        codigoInc: (r.codigo_inc as string) ?? undefined,
   489тЖТ   489тЖТ      }))
   490тЖТ   490тЖТ    } catch (err) {
   491тЖТ   491тЖТ      console.error('[stockEnUbicacion] Tanto c├бlculo directo como RPC fallback fallaron:', err)
   492тЖТ   492тЖТ      // Retornar array vac├нo pero con marker de error para que el UI distinga

   495тЖТ   495тЖТ  }

   791тЖТ   791тЖТ      throw error

   803тЖТ   803тЖТ    const errMsg = err instanceof Error ? err.message : ''
   804тЖТ   804тЖТ    const errCode = err instanceof Error ? (err as unknown as Record<string, string>).code || '' : ''

   818тЖТ   818тЖТ  // Usar dataClient (service role) ya configurado тАФ sin credenciales hardcodeadas

   820тЖТ   820тЖТ

   842тЖТ   842тЖТ  for (let i = 0; i < allIds.length; i += DELETE_BATCH) {

   849тЖТ   849тЖТ  }
   850тЖТ   850тЖТ
   851тЖТ   851тЖТ  return { deleted: true }
