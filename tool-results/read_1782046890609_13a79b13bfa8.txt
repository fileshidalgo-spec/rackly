    53→type LocStock = {
    54→  bloque: string
    55→  torre: string

    67→export function TrasladoTab() {
    68→  const { perfil } = useAuth()
    69→
    70→  const [step, setStep] = useState<1 | 2>(1)
    71→  const [codigo, setCodigo] = useState('')
    72→  const [descripcion, setDescripcion] = useState('')
    73→  const [un, setUn] = useState('')
    74→  // locations ahora se calcula con useMemo (más abajo), reactivo a movs y codigo
    75→  const [selectedOrigin, setSelectedOrigin] = useState<string | null>(null)
    76→  const [trasladoTotal, setTrasladoTotal] = useState(true)
    77→  const [destBloque, setDestBloque] = useState('')
    78→  const [destTorre, setDestTorre] = useState('')
    79→  const [destPiso, setDestPiso] = useState('')
    80→  const [destPos, setDestPos] = useState('')
    81→  const [qty, setQty] = useState('')
    82→  const [busy, setBusy] = useState(false)
    83→  const [confirm, setConfirm] = useState(false)
    84→  const [destinoOcupado, setDestinoOcupado] = useState<StockEnUbicacion[]>([])
    85→  const [salidaBusy, setSalidaBusy] = useState<string | null>(null)
    86→  const [salidaParcialCant, setSalidaParcialCant] = useState<Record<string, string>>({})
    87→  const [salidaParcialTotal, setSalidaParcialTotal] = useState<Record<string, boolean>>({})
    88→  const [corregirDiferencia, setCorregirDiferencia] = useState(false)
    89→
    90→  const [movs, setMovs] = useState<Movimiento[]>([])
    91→
    92→  useMovimientosRealtime(setMovs)
    93→
    94→  if (!perfil) return <div className="p-4 text-muted-foreground animate-pulse">Cargando...</div>
    95→
    96→  function handleCatalogoPick(item: CatalogoItem) {
    97→    setCodigo(item.codigo)
    98→    setDescripcion(item.descripcion)
    99→    setUn(item.un)
   100→    setStep(1)
   101→    setSelectedOrigin(null)
   102→  }
   103→
   104→  // Recalcular ubicaciones reactivamente cuando cambian movs o codigo.
   105→  // Esto corrige el bug donde ubicaciones quedaban "pegadas" después de una salida.
   106→  // EXCLUYE movimientos INC del cálculo (igual que OcupaciónTab).
   107→  const locations = useMemo(() => {
   108→    if (!codigo) return []
   109→    const code = codigo.toUpperCase()
   110→    const locMap = new Map<string, LocStock>()
   111→    const relevant = movs.filter((m) => m.codigo === code && !m.codigoInc)
   112→    for (const m of relevant) {
   113→      const key = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}||${m.fVencimiento || ''}`
   114→      const current = locMap.get(key)
   115→      if (current) {
   116→        current.stock += ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? m.cantidad : -m.cantidad
   117→      } else {
   118→        locMap.set(key, {
   119→          bloque: m.bloque,
   120→          torre: m.torre,
   121→          piso: m.piso,
   122→          posicion: m.posicion,
   123→          stock: ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? m.cantidad : -m.cantidad,
   124→          descripcion: m.descripcion,
   125→          un: m.un,
   126→          fVencimiento: m.fVencimiento || '',
   127→          codigo: m.codigo,
   128→          proveedor: m.proveedor,
   129→          codigoInc: m.codigoInc || undefined,
   130→        })
   131→      }
   132→    }
   133→    return Array.from(locMap.values())
   134→      .filter((l) => l.stock > 0)
   135→      .sort((a, b) => {
   136→        const aHasDate = !!a.fVencimiento
   137→        const bHasDate = !!b.fVencimiento
   138→        if (aHasDate && bHasDate) return a.fVencimiento.localeCompare(b.fVencimiento)
   139→        if (aHasDate && !bHasDate) return -1
   140→        if (!aHasDate && bHasDate) return 1
   141→        const aB = parseInt(a.bloque, 10) || 0
   142→        const bB = parseInt(b.bloque, 10) || 0
   143→        if (aB !== bB) return aB - bB
   144→        const aT = parseInt(a.torre, 10) || 0
   145→        const bT = parseInt(b.torre, 10) || 0
   146→        if (aT !== bT) return aT - bT
   147→        const aP = parseInt(a.piso, 10) || 0
   148→        const bP = parseInt(b.piso, 10) || 0
   149→        if (aP !== bP) return aP - bP
   150→        const aPos = parseInt(a.posicion, 10) || 0
   151→        const bPos = parseInt(b.posicion, 10) || 0
   152→        return aPos - bPos
   153→      })
   154→  }, [movs, codigo])
   155→
   156→  // Limpiar selectedOrigin si la ubicación ya no existe en locations
   157→  useEffect(() => {
   158→    if (selectedOrigin && !locations.find((l) => `${l.bloque}-${l.torre}-${l.piso}-${l.posicion}||${l.fVencimiento || ''}||${l.codigoInc || ''}` === selectedOrigin)) {
   159→      setSelectedOrigin(null)
   160→    }
   161→  }, [locations, selectedOrigin])
   162→
   163→  const origin = locations.find((l) => `${l.bloque}-${l.torre}-${l.piso}-${l.posicion}||${l.fVencimiento || ''}||${l.codigoInc || ''}` === selectedOrigin)
   164→
   165→  const qtyNum = parseFloat(qty) || 0
   166→  const saldoRestante = origin ? origin.stock - qtyNum : 0
   167→  const excedeStock = origin ? qtyNum > origin.stock : false
   168→  const faltaStock = origin ? qtyNum > 0 && qtyNum < origin.stock : false
   169→  const diferencia = origin ? qtyNum - origin.stock : 0
   170→  // Ajuste automático: se activa cuando qty > stock (siempre), o cuando qty < stock Y el usuario elige corregir
   171→  const ajusteActivo = excedeStock || (faltaStock && corregirDiferencia)
   172→
   173→  async function handleConfirm() {
   174→    if (!origin) return
   175→    // Comparar ubicación real (bloque, torre, piso, posicion), ignorando fVencimiento y codigoInc.
   176→    // selectedOrigin tiene formato "bloque-torre-piso-pos||fVenc||inc", destKey solo "bloque-torre-piso-pos".
   177→    const originKey = (selectedOrigin || '').split('||')[0]
   178→    const destKey = `${destBloque}-${destTorre}-${destPiso || '1'}-${destPos}`
   179→    if (originKey === destKey) {
   180→      toast.error('El destino no puede ser igual al origen')
   181→      return
   182→    }
   183→    if (qtyNum <= 0) {
   184→      toast.error('Cantidad inválida')
   185→      return
   186→    }
   187→    // Verificar si destino está ocupado
   188→    try {
   189→      const destStock = await stockEnUbicacion(destBloque, destTorre, destPiso || '1', destPos)
   190→      setDestinoOcupado(destStock)
   191→    } catch {
   192→      setDestinoOcupado([])
   193→    }
   194→    setConfirm(true)
   195→  }
   196→
   197→  // Dar salida a un producto desde el alerta de destino ocupado (soporta parcial/total)
   198→  async function handleSalidaDesdeAlerta(stockItem: StockEnUbicacion) {
   199→    if (!perfil) return
   200→    const itemKey = `${stockItem.codigo}-${stockItem.fVencimiento || ''}`
   201→    const isTotal = salidaParcialTotal[itemKey] === true
   202→    const cantStr = salidaParcialCant[itemKey] || ''
   203→    const cantNum = isTotal ? stockItem.stock : parseFloat(cantStr)
   204→    if (isNaN(cantNum) || cantNum <= 0) { toast.error('Cantidad inválida'); return }
   205→    if (cantNum > stockItem.stock) { toast.error(`Máximo: ${stockItem.stock} ${stockItem.un}`); return }
   206→    setSalidaBusy(itemKey)
   207→    try {
   208→      await addMovimiento({
   209→        tipo: 'salida',
   210→        bloque: destBloque,
   211→        torre: destTorre,
   212→        piso: destPiso || '1',
   213→        posicion: destPos,
   214→        codigo: stockItem.codigo,
   215→        descripcion: stockItem.descripcion,
   216→        un: stockItem.un,
   217→        cantidad: cantNum,
   218→        fVencimiento: stockItem.fVencimiento ?? '',
   219→        turno: calcularTurno(),
   220→        usuarioId: perfil.id,
   221→        usuarioNombre: perfil.nombre,
   222→        usuarioCorreo: perfil.correo,
   223→        proveedor: stockItem.proveedor,
   224→        codigoInc: stockItem.codigoInc || undefined,
   225→      })
   226→      toast.success(`Salida de ${cantNum} ${stockItem.un} de ${stockItem.codigo}`)
   227→      setMovs(await fetchMovimientos())
   228→      const updated = await stockEnUbicacion(destBloque, destTorre, destPiso || '1', destPos)
   229→      setDestinoOcupado(updated)
   230→    } catch (err: unknown) {
   231→      if (isInsufficientStockError(err)) {
   232→        toast.error('Stock insuficiente', {
   233→          description: 'Otro usuario pudo haber modificado el stock mientras tú operabas. Los datos se han actualizado.',
   234→          duration: 6000,
   235→        })
   236→      } else {
   237→        toast.error('Error al dar salida', { description: extractError(err) })
   238→      }
   239→    } finally {
   240→      setSalidaBusy(null)
   241→    }
   242→  }
   243→
   244→  async function doTraslado() {
   245→    if (!origin || !perfil) return
   246→    const cantidadFinal = qtyNum || origin.stock
   247→    setBusy(true)
   248→    try {
   249→      await trasladarMovimiento({
   250→        codigo: origin.codigo,
   251→        descripcion: origin.descripcion,
   252→        un: origin.un,
   253→        cantidad: cantidadFinal,
   254→        origen: {
   255→          bloque: origin.bloque,
   256→          torre: origin.torre,
   257→          piso: origin.piso,
   258→          posicion: origin.posicion,
   259→        },
   260→        destino: {
   261→          bloque: destBloque,
   262→          torre: destTorre,
   263→          piso: destPiso || '1',
   264→          posicion: destPos,
   265→        },
   266→        turno: calcularTurno(),
   267→        usuarioId: perfil.id,
   268→        usuarioNombre: perfil.nombre,
   269→        usuarioCorreo: perfil.correo,
   270→        fVencimiento: origin.fVencimiento,
   271→        proveedor: origin.proveedor,
   272→        codigoInc: origin.codigoInc,
   273→        // Se genera ajuste automático solo si aplica: qty > stock o (qty < stock y el usuario elige corregir)
   274→        cantidadAjuste: ajusteActivo ? diferencia : undefined,
   275→      })
   276→      toast.success('Traslado registrado')
   277→      if (ajusteActivo) {
   278→        if (excedeStock) {
   279→          toast.info(`Ajuste automático: +${Math.abs(diferencia)} ${origin.un} ingreso en origen (faltaba stock registrado)`, { duration: 6000 })
   280→        } else if (corregirDiferencia) {
   281→          toast.info(`Ajuste automático: -${Math.abs(diferencia)} ${origin.un} salida en origen (se deja posición en 0)`, { duration: 6000 })
   282→        }
   283→      }
   284→      if (faltaStock && !corregirDiferencia) {
   285→        toast.info(`Queda un saldo de ${saldoRestante} ${origin.un} en la ubicación de origen`, { duration: 6000 })
   286→      }
   287→      setMovs(await fetchMovimientos())
   288→      resetForm()
   289→    } catch (err: unknown) {
   290→      if (isInsufficientStockError(err)) {
   291→        toast.error('Stock insuficiente en origen', {
   292→          description: 'Otro usuario pudo haber modificado el stock mientras tú operabas. Los datos se han actualizado.',
   293→          duration: 6000,
   294→        })
   295→      } else {
   296→        toast.error('Error al trasladar', { description: extractError(err) })
   297→      }
   298→    } finally {
   299→      setBusy(false)
   300→      setConfirm(false)
   301→      setDestinoOcupado([])
   302→    }
   303→  }
   304→
   305→  function resetForm() {
   306→    setCodigo('')
   307→    setDescripcion('')
   308→    setUn('')
   309→    // locations se limpia automáticamente al poner codigo en '' (useMemo returns [])
   310→    setSelectedOrigin(null)
   311→    setTrasladoTotal(true)
   312→    setCorregirDiferencia(false)
   313→    setDestBloque('')
   314→    setDestTorre('')
   315→    setDestPiso('')
   316→    setDestPos('')
   317→    setQty('')
   318→    setStep(1)
   319→  }
   320→
   321→  const destTorres = torresDeBloque(destBloque)
   322→  const destPositions = posicionesDeBloque(destBloque)
   323→  const destPisoSeleccionado = destPiso || '1'
   324→
   325→  return (
   326→    <div className="space-y-4">
   327→      <p className="text-sm text-muted-foreground">
   328→        Busca un código, elige la ubicación de origen y destino. Confirma el traslado.
   329→      </p>
   330→
   331→      <CatalogoSearchInput
   332→        onPick={handleCatalogoPick}
   333→        value={codigo}
   334→        onChange={(v) => {
   335→          setCodigo(v)
   336→          const cat = findCatalogoByCodigo(v)
   337→          if (cat) handleCatalogoPick(cat)
   338→        }}
   339→      />
   340→
   341→      {/* Info del producto buscado */}
   342→      {codigo && descripcion && (
   343→        <div className="rounded-lg border border-blue-200 dark:border-blue-800 bg-blue-50/50 dark:bg-blue-950/20 p-3">
   344→          <div className="flex items-start gap-3">
   345→            <div className="w-9 h-9 rounded-lg bg-blue-100 dark:bg-blue-900/30 flex items-center justify-center flex-shrink-0">
   346→              <PackageSearch className="h-4 w-4 text-blue-600 dark:text-blue-400" />
   347→            </div>
   348→            <div className="flex-1 min-w-0">
   349→              <div className="flex items-center gap-2 flex-wrap">
   350→                <span className="font-mono font-bold text-sm text-blue-800 dark:text-blue-300">{codigo}</span>
   351→                <span className="text-[10px] text-muted-foreground bg-slate-100 dark:bg-slate-800 px-1.5 py-0.5 rounded">{un || '—'}</span>
   352→                {locations[0]?.proveedor && (
   353→                  <Badge variant="outline" className="bg-purple-50 text-purple-700 border-purple-200 dark:bg-purple-950/40 dark:text-purple-300 dark:border-purple-800 font-semibold text-[10px]">
   354→                    {locations[0].proveedor}
   355→                  </Badge>
   356→                )}
   357→              </div>
   358→              <p className="text-xs text-muted-foreground truncate mt-0.5">{descripcion}</p>
   359→            </div>
   360→            <Badge variant="outline" className="border-blue-300 bg-blue-100 text-blue-700 dark:border-blue-700 dark:bg-blue-900/40 dark:text-blue-300 font-semibold">
   361→              {locations.length} ubicacion{locations.length !== 1 ? 'es' : ''}
   362→            </Badge>
   363→          </div>
   364→        </div>
   365→      )}
   366→
   367→      {locations.length > 0 && (
   368→        <>
   369→          <div className="flex items-center gap-2">
   370→            <p className="text-sm font-semibold">1. Selecciona ubicación de origen:</p>
   371→            <p className="text-xs text-muted-foreground">y elige el tipo de traslado</p>
   372→          </div>
   373→          {/* ── Mobile: Cards con TODA la info visible ── */}
   374→          <div className="sm:hidden space-y-3">
   375→            {locations.map((loc) => {
   376→              const key = `${loc.bloque}-${loc.torre}-${loc.piso}-${loc.posicion}||${loc.fVencimiento || ''}||${loc.codigoInc || ''}`
   377→              const isSelected = selectedOrigin === key
   378→              return (
   379→                <div
   380→                  key={key}
   381→                  onClick={() => {
   382→                    setSelectedOrigin(key)
   383→                    setTrasladoTotal(true)
   384→                    setQty(String(loc.stock))
   385→                    setStep(2)
   386→                  }}
   387→                  className={`rounded-xl border-2 p-3 space-y-2.5 cursor-pointer transition-all duration-200 ${
   388→                    isSelected
   389→                      ? 'border-blue-500 bg-blue-50/60 dark:bg-blue-950/30 shadow-md shadow-blue-500/10'
   390→                      : 'border-slate-200 dark:border-slate-700 bg-white dark:bg-slate-800/60 hover:border-blue-300 dark:hover:border-blue-700/50 hover:shadow-sm'
   391→                  }`}
   392→                >
   393→                  {/* Fila 1: Ubicación completa + Stock */}
   394→                  <div className="flex items-center justify-between">
   395→                    <div className="flex items-center gap-1.5">
   396→                      <MapPin className="h-3.5 w-3.5 text-blue-500" />
   397→                      <span className="font-mono font-bold text-sm text-slate-800 dark:text-slate-200">
   398→                        B{loc.bloque} · T{loc.torre} · P{loc.piso} · Pos {loc.posicion}
   399→                      </span>
   400→                    </div>
   401→                    <div className="flex items-center gap-1.5">
   402→                      <span className="text-[10px] text-muted-foreground bg-slate-100 dark:bg-slate-700 px-1.5 py-0.5 rounded font-medium">{loc.un}</span>
   403→                      <span className="font-bold text-base text-slate-800 dark:text-slate-100">{loc.stock}</span>
   404→                    </div>
   405→                  </div>
   406→
   407→                  {/* Fila 2: Vencimiento + Proveedor */}
   408→                  <div className="flex items-center gap-2 flex-wrap">
   409→                    {loc.fVencimiento ? (
   410→                      <span className={`text-[10px] font-semibold px-2 py-0.5 rounded-full border ${
   411→                        isExpired(loc.fVencimiento)
   412→                          ? 'bg-red-50 text-red-600 border-red-200 dark:bg-red-950/40 dark:text-red-400 dark:border-red-800'
   413→                          : isExpiringSoon(loc.fVencimiento, 15)
   414→                            ? 'bg-orange-50 text-orange-600 border-orange-200 dark:bg-orange-950/40 dark:text-orange-400 dark:border-orange-800'
   415→                            : isExpiringSoon(loc.fVencimiento, 30)
   416→                              ? 'bg-sky-50 text-sky-600 border-sky-200 dark:bg-sky-950/40 dark:text-sky-400 dark:border-sky-800'
   417→                              : 'bg-slate-50 text-muted-foreground border-slate-200 dark:bg-slate-800 dark:border-slate-600'
   418→                      }`}>
   419→                        Venc: {formatDate(loc.fVencimiento)}
   420→                      </span>
   421→                    ) : null}
   422→                    {loc.proveedor ? (
   423→                      <span className="text-[10px] font-semibold px-2 py-0.5 rounded-full bg-purple-50 text-purple-700 border border-purple-200 dark:bg-purple-950/40 dark:text-purple-300 dark:border-purple-800">
   424→                        {loc.proveedor}
   425→                      </span>
   426→                    ) : null}
   427→                    {loc.codigoInc && (
   428→                      <span className="text-[10px] font-semibold px-2 py-0.5 rounded-full bg-amber-50 text-amber-700 border border-amber-200 dark:bg-amber-950/40 dark:text-amber-300 dark:border-amber-800 flex items-center gap-0.5">
   429→                        <AlertTriangle className="w-3 h-3" /> {loc.codigoInc}
   430→                      </span>
   431→                    )}
   432→                    {!loc.fVencimiento && !loc.proveedor && !loc.codigoInc && (
   433→                      <span className="text-[10px] text-slate-400">Sin vencimiento · Sin proveedor</span>
   434→                    )}
   435→                  </div>
   436→
   437→                  {/* Botones de acción */}
   438→                  {isSelected ? (
   439→                    <div className="flex items-center justify-center gap-1.5 pt-1 text-xs font-bold text-blue-700 dark:text-blue-400">
   440→                      <CheckCircle2 className="h-3.5 w-3.5" />
   441→                      <span>{trasladoTotal ? 'Traslado Total' : 'Traslado Parcial'}</span>
   442→                    </div>
   443→                  ) : (
   444→                    <div className="flex gap-2 pt-0.5">
   445→                      <button
   446→                        onClick={(e) => {
   447→                          e.stopPropagation()
   448→                          setSelectedOrigin(key)
   449→                          setTrasladoTotal(true)
   450→                          setQty(String(loc.stock))
