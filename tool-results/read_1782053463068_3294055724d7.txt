    62→type Props = {
    63→  tipo: TipoMovimiento | 'inc' | 'salida-inc'
    64→  onCreated: (movs: Movimiento[]) => void
    65→}
    66→
    67→export function MovimientoForm({ tipo, onCreated }: Props) {
    68→  const { perfil } = useAuth()
    69→  const [turno, setTurno] = useState<Turno>(calcularTurno())
    70→
    71→  useEffect(() => {
    72→    const interval = setInterval(() => setTurno(calcularTurno()), 60000)
    73→    return () => clearInterval(interval)
    74→  }, [])
    75→
    76→  if (!perfil) {
    77→    return <div className='p-4 text-muted-foreground animate-pulse'>Cargando usuario...</div>
    78→  }
    79→  if (tipo === 'salida') {
    80→    return <SalidaForm turno={turno} onCreated={onCreated} perfil={perfil} />
    81→  }
    82→  if (tipo === 'salida-inc') {
    83→    return <SalidaIncForm turno={turno} onCreated={onCreated} perfil={perfil} />
    84→  }
    85→  if (tipo === 'inc') {
    86→    return <IncForm turno={turno} onCreated={onCreated} perfil={perfil} />
    87→  }
    88→  // ingreso y devolucion usan el mismo formulario, solo cambia el tipo
    89→  return <IngresoForm turno={turno} onCreated={onCreated} perfil={perfil} tipo={tipo} />
    90→}

    95→function IngresoForm({
    96→  turno,
    97→  onCreated,
    98→  perfil,
    99→  tipo = 'ingreso',
   100→}: {
   101→  turno: Turno
   102→  onCreated: (movs: Movimiento[]) => void
   103→  perfil: { id: string; nombre: string; correo: string }
   104→  tipo?: TipoMovimiento
   105→}) {
   106→  const [bloque, setBloque] = useState('')
   107→  const [torre, setTorre] = useState('')
   108→  const [piso, setPiso] = useState('')
   109→  const [posicion, setPosicion] = useState('')
   110→  const [codigo, setCodigo] = useState('')
   111→  const [descripcion, setDescripcion] = useState('')
   112→  const [un, setUn] = useState('')
   113→  const [cantidad, setCantidad] = useState('')
   114→  const [fVencimiento, setFVencimiento] = useState('')
   115→  const [sinVencimiento, setSinVencimiento] = useState(false)
   116→  const [proveedor, setProveedor] = useState('')
   117→  const [busy, setBusy] = useState(false)
   118→  const [confirmData, setConfirmData] = useState<StockEnUbicacion[] | null>(null)
   119→  const [salidaBusy, setSalidaBusy] = useState<string | null>(null)
   120→
   121→  const torres = torresDeBloque(bloque)
   122→  const posiciones = posicionesDeBloque(bloque)
   123→
   124→  function handleCatalogoPick(item: CatalogoItem) {
   125→    setCodigo(item.codigo)
   126→    setDescripcion(item.descripcion)
   127→    setUn(item.un)
   128→  }
   129→
   130→  async function handleSubmit(e: React.FormEvent) {
   131→    e.preventDefault()
   132→    if (!bloque || !torre || !piso || !posicion || !codigo.trim() || !cantidad) {
   133→      toast.error('Completa todos los campos requeridos')
   134→      return
   135→    }
   136→    if (requiereProveedor(descripcion) && !proveedor) {
   137→      toast.error('Selecciona un proveedor para este producto')
   138→      return
   139→    }
   140→    const qty = parseFloat(cantidad)
   141→    if (isNaN(qty) || qty <= 0) {
   142→      toast.error('La cantidad debe ser mayor a 0')
   143→      return
   144→    }
   145→    setBusy(true)
   146→    try {
   147→      // Verificar si CUALQUIER producto existe en esta posición
   148→      const details = await stockEnUbicacion(bloque, torre, piso, posicion)
   149→      if (details.length > 0) {
   150→        setConfirmData(details)
   151→        setBusy(false)
   152→        return
   153→      }
   154→      await doInsert(qty)
   155→    } catch (err: unknown) {
   156→      const message = extractError(err)
   157→      toast.error('Error al verificar stock', { description: message })
   158→      setBusy(false)
   159→    }
   160→  }
   161→
   162→  async function doInsert(qty: number) {
   163→    setBusy(true)
   164→    try {
   165→      const movs = await addMovimiento({
   166→        tipo,
   167→        bloque,
   168→        torre,
   169→        piso,
   170→        posicion,
   171→        codigo,
   172→        descripcion,
   173→        un,
   174→        cantidad: qty,
   175→        fVencimiento: sinVencimiento ? '' : fVencimiento,
   176→        turno,
   177→        usuarioId: perfil.id,
   178→        usuarioNombre: perfil.nombre,
   179→        usuarioCorreo: perfil.correo,
   180→        proveedor: proveedor || undefined,
   181→      })
   182→      toast.success(tipo === 'devolucion' ? 'Devolución registrada' : 'Ingreso registrado')
   183→      setCodigo('')
   184→      setDescripcion('')
   185→      setUn('')
   186→      setCantidad('')
   187→      setFVencimiento('')
   188→      setProveedor('')
   189→      onCreated(movs)
   190→    } catch (err: unknown) {
   191→      const message = extractError(err)
   192→      toast.error('Error al registrar ingreso', { description: message })
   193→    } finally {
   194→      setBusy(false)
   195→      setConfirmData(null)
   196→    }
   197→  }
   198→
   199→  // Dar salida a un producto desde el alerta de ubicación ocupada
   200→  async function handleSalidaDesdeAlerta(stockItem: StockEnUbicacion) {
   201→    setSalidaBusy(stockItem.codigo)
   202→    try {
   203→      const movs = await addMovimiento({
   204→        tipo: 'salida',
   205→        bloque,
   206→        torre,
   207→        piso,
   208→        posicion,
   209→        codigo: stockItem.codigo,
   210→        descripcion: stockItem.descripcion,
   211→        un: stockItem.un,
   212→        cantidad: stockItem.stock,
   213→        fVencimiento: stockItem.fVencimiento ?? '',
   214→        turno,
   215→        usuarioId: perfil.id,
   216→        usuarioNombre: perfil.nombre,
   217→        usuarioCorreo: perfil.correo,
   218→        proveedor: stockItem.proveedor,
   219→        // PRESERVAR codigoInc para que la salida descuente del stock INC correctamente
   220→        codigoInc: stockItem.codigoInc || undefined,
   221→      })
   222→      toast.success(`Salida de ${stockItem.stock} ${stockItem.un} de ${stockItem.codigo}`)
   223→      onCreated(movs)
   224→      // Refrescar datos del alerta
   225→      const updated = await stockEnUbicacion(bloque, torre, piso, posicion)
   226→      if (updated.length > 0) {
   227→        setConfirmData(updated)
   228→      } else {
   229→        setConfirmData(null)
   230→        // Ya no hay productos, hacer el ingreso directamente
   231→        await doInsert(parseFloat(cantidad))
   232→      }
   233→    } catch (err: unknown) {
   234→      if (isInsufficientStockError(err)) {
   235→        const detail = (err as Record<string, string>).detail || ''
   236→        toast.error('Stock insuficiente', {
   237→          description: detail || 'Otro usuario pudo haber modificado el stock mientras tú operabas.',
   238→          duration: 8000,
   239→        })
   240→      } else {
   241→        const message = extractError(err)
   242→        toast.error('Error al dar salida', { description: message })
   243→      }
   244→    } finally {
   245→      setSalidaBusy(null)
   246→    }
   247→  }
   248→
   249→  return (
   250→    <>
   251→      <div className={`flex items-center gap-2 mb-4 p-3 rounded-lg border ${
   252→        tipo === 'devolucion'
   253→          ? 'bg-orange-50 dark:bg-orange-950/40 border-orange-200 dark:border-orange-800'
   254→          : 'bg-green-50 dark:bg-green-950/40 border-green-200 dark:border-green-800'
   255→      }`}>
   256→        {tipo === 'devolucion' ? (
   257→          <ArrowRightLeft className="h-5 w-5 text-orange-600 dark:text-orange-400 flex-shrink-0" />
   258→        ) : (
   259→          <ArrowDownToLine className="h-5 w-5 text-green-600 dark:text-green-400 flex-shrink-0" />
   260→        )}
   261→        <div className="min-w-0">
   262→          <p className={`text-sm font-semibold ${
   263→            tipo === 'devolucion'
   264→              ? 'text-orange-700 dark:text-orange-300'
   265→              : 'text-green-700 dark:text-green-300'
   266→          }`}>
   267→            {tipo === 'devolucion' ? 'Ingreso por devolución' : 'Ingreso de mercadería'}
   268→          </p>
   269→          <p className="text-xs text-muted-foreground">Turno: {turno}</p>
   270→        </div>
   271→      </div>
   272→
   273→      <form onSubmit={handleSubmit} className="space-y-4">
   274→        <div className="space-y-1">
   275→          <Label className="text-xs font-medium text-muted-foreground uppercase tracking-wider">Ubicación</Label>
   276→          <div className="grid grid-cols-2 sm:grid-cols-5 gap-2">
   277→            <div className="space-y-1">
   278→              <Label className="text-xs text-muted-foreground">Bloque</Label>
   279→              <Select value={bloque} onValueChange={(v) => { setBloque(v); setTorre(''); setPiso(''); setPosicion('') }}>
   280→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Bloque" /></SelectTrigger>
   281→                <SelectContent>
   282→                  {BLOQUES.map((b) => <SelectItem key={b} value={b}>{b}</SelectItem>)}
   283→                </SelectContent>
   284→              </Select>
   285→            </div>
   286→            <div className="space-y-1">
   287→              <Label className="text-xs text-muted-foreground">Torre</Label>
   288→              <Select value={torre} onValueChange={setTorre} disabled={!bloque}>
   289→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Torre" /></SelectTrigger>
   290→                <SelectContent>
   291→                  {torres.map((t) => <SelectItem key={t} value={t}>{t}</SelectItem>)}
   292→                </SelectContent>
   293→              </Select>
   294→            </div>
   295→            <div className="space-y-1">
   296→              <Label className="text-xs text-muted-foreground">Piso</Label>
   297→              <Select value={piso} onValueChange={setPiso} disabled={!bloque}>
   298→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Piso" /></SelectTrigger>
   299→                <SelectContent>
   300→                  {PISOS.map((p) => <SelectItem key={p} value={p}>{p}</SelectItem>)}
   301→                </SelectContent>
   302→              </Select>
   303→            </div>
   304→            <div className="space-y-1">
   305→              <Label className="text-xs text-muted-foreground">Posición</Label>
   306→              <Select value={posicion} onValueChange={setPosicion} disabled={!bloque}>
   307→                <SelectTrigger className="h-10 w-full"><SelectValue placeholder="Pos." /></SelectTrigger>
   308→                <SelectContent>
   309→                  {posiciones.map((p) => <SelectItem key={p} value={p}>{p}</SelectItem>)}
   310→                </SelectContent>
   311→              </Select>
   312→            </div>
   313→            <div className="space-y-1 col-span-2 sm:col-span-1">
   314→              <Label className="text-xs text-muted-foreground">Turno</Label>
   315→              <Input value={turno} readOnly className="h-10 bg-muted text-sm" />
   316→            </div>
   317→          </div>
   318→        </div>
   319→
   320→        <div className="space-y-1">
   321→          <Label className="text-xs font-medium text-muted-foreground uppercase tracking-wider">Producto</Label>
   322→          <CatalogoSearchInput
   323→            onPick={handleCatalogoPick}
   324→            value={codigo}
   325→            onChange={(v) => {
   326→              setCodigo(v)
   327→              const cat = findCatalogoByCodigo(v)
   328→              if (cat) {
   329→                setDescripcion(cat.descripcion)
   330→                setUn(cat.un)
   331→              }
   332→            }}
   333→          />
   334→        </div>
   335→
   336→        {descripcion && (
   337→          <div className="rounded-lg border border-green-200 dark:border-green-800 bg-green-50/50 dark:bg-green-950/20 p-3 space-y-2 transition-all">
   338→            <div className="flex items-start gap-3 flex-wrap">
   339→              <div className="min-w-0 flex-1">
   340→                <p className="text-xs text-muted-foreground">Descripción</p>
   341→                <p className="text-sm font-medium truncate">{descripcion}</p>
   342→              </div>
   343→              <div className="min-w-[60px]">
   344→                <p className="text-xs text-muted-foreground">UN</p>
   345→                <p className="text-sm font-medium">{un || '—'}</p>
   346→              </div>
   347→            </div>
   348→          </div>
   349→        )}
   350→
   351→        <div className="grid grid-cols-2 sm:grid-cols-4 gap-2">
   352→          <div className="space-y-1 col-span-2 sm:col-span-1">
   353→            <Label className="text-xs text-muted-foreground">Cantidad</Label>
   354→            <Input type="number" step="any" min="0.001" value={cantidad} onChange={(e) => setCantidad(e.target.value)} placeholder="0" className="h-10" />
   355→          </div>
   356→          <div className="space-y-1 col-span-2 sm:col-span-1">
   357→            <Label className="text-xs text-muted-foreground">F. Vencimiento</Label>
   358→            <div className="flex items-center gap-1.5">
   359→              <Input type="date" value={fVencimiento} onChange={(e) => setFVencimiento(e.target.value)} disabled={sinVencimiento} className="h-10 [color-scheme:dark]" />
   360→              <Checkbox checked={sinVencimiento} onCheckedChange={(v) => setSinVencimiento(!!v)} />
   361→            </div>
   362→          </div>
   363→          {requiereProveedor(descripcion) && (
   364→            <div className="space-y-1 col-span-2">
   365→              <Label className="text-xs text-muted-foreground font-medium">Proveedor <span className="text-red-500">*</span></Label>
   366→              <Select value={proveedor} onValueChange={setProveedor}>
   367→                <SelectTrigger className="h-10">
   368→                  <SelectValue placeholder="Selecciona proveedor" />
   369→                </SelectTrigger>
   370→                <SelectContent>
   371→                  {PROVEEDORES_FILM.map((p) => (
   372→                    <SelectItem key={p} value={p}>{p}</SelectItem>
   373→                  ))}
   374→                </SelectContent>
   375→              </Select>
   376→            </div>
   377→          )}
   378→        </div>
   379→
   380→        <Button type="submit" disabled={busy} className={`w-full h-11 text-white text-sm font-medium sm:w-auto ${
   381→          tipo === 'devolucion'
   382→            ? 'bg-orange-600 hover:bg-orange-700'
   383→            : 'bg-green-600 hover:bg-green-700'
   384→        }`}>
   385→          {busy && <Loader2 className="h-4 w-4 animate-spin mr-2" />}
   386→          {tipo === 'devolucion' ? 'Registrar Devolución' : 'Registrar Ingreso'}
   387→        </Button>
   388→      </form>
   389→
   390→      {/* ═══ DIÁLOGO DE CONFIRMACIÓN — Ubicación Ocupada ═══ */}
   391→      <AlertDialog open={!!confirmData} onOpenChange={() => setConfirmData(null)}>
   392→        <AlertDialogContent className="max-w-[calc(100vw-1rem)] max-w-lg p-0 max-h-[85vh] flex flex-col overflow-hidden">
   393→          {/* Header con gradiente */}
   394→          <div className="bg-gradient-to-r from-amber-500 via-orange-500 to-red-500 px-4 sm:px-6 py-5 text-white shrink-0">
   395→            <div className="flex items-center gap-3">
   396→              <div className="w-11 h-11 rounded-full bg-white/20 backdrop-blur-sm flex items-center justify-center flex-shrink-0">
   397→                <TriangleAlert className="h-6 w-6 text-white" />
   398→              </div>
   399→              <div>
   400→                <AlertDialogTitle className="text-lg font-bold text-white m-0">
   401→                  Ubicación Ocupada
   402→                </AlertDialogTitle>
   403→                <AlertDialogDescription className="text-amber-100 text-sm mt-0.5">
   404→                  Esta posición ya tiene stock registrado. Revisa el detalle.
   405→                </AlertDialogDescription>
   406→              </div>
   407→            </div>
   408→          </div>
   409→
   410→          {/* Contenido scrolleable */}
   411→          <div className="flex-1 min-h-0 overflow-y-auto overscroll-contain">
   412→            {/* Ubicación destino */}
   413→            <div className="px-4 sm:px-6 pt-4 pb-2">
   414→              <p className="text-xs font-semibold text-muted-foreground uppercase tracking-wider mb-1.5">Ubicación destino</p>
   415→              <div className="flex items-center gap-2 text-sm text-muted-foreground bg-slate-50 dark:bg-slate-900 rounded-lg px-3 py-2.5">
   416→                <MapPin className="h-4 w-4 text-orange-500 flex-shrink-0" />
   417→                <span className="font-mono font-medium text-slate-700 dark:text-slate-300">
   418→                  Bloque {bloque} · Torre {torre} · Piso {piso} · Pos {posicion}
   419→                </span>
   420→              </div>
   421→            </div>
   422→
   423→            {/* Producto que se desea ingresar */}
   424→            <div className="px-4 sm:px-6 pb-3">
   425→              <p className="text-xs font-semibold text-muted-foreground uppercase tracking-wider mb-1.5">Producto a {tipo === 'devolucion' ? 'devolver' : 'ingresar'}</p>
   426→              <div className="rounded-xl border-2 border-dashed border-green-300 dark:border-green-700 bg-green-50/60 dark:bg-green-950/20 p-3 space-y-1.5">
   427→                <div className="flex items-center justify-between">
   428→                  <div className="flex items-center gap-2 min-w-0">
   429→                    <span className="font-mono font-bold text-sm text-green-800 dark:text-green-300">{codigo}</span>
   430→                    <span className="text-[10px] text-green-700/60 dark:text-green-400/60 bg-green-100 dark:bg-green-900/40 px-1.5 py-0.5 rounded">{un || '—'}</span>
   431→                  </div>
   432→                  <div className="flex items-center gap-1 flex-shrink-0">
   433→                    <ArrowDownToLine className="h-3.5 w-3.5 text-green-600" />

   436→                </div>
   437→                {descripcion && (
   438→                  <p className="text-xs text-green-700/70 dark:text-green-400/70 truncate">{descripcion}</p>
   439→                )}
   440→                {fVencimiento && !sinVencimiento && (
   441→                  <p className="text-[10px] text-muted-foreground">Venc: {fVencimiento}</p>
   442→                )}
   443→              </div>
   444→            </div>
   445→
   446→            {/* Separador */}
   447→            {confirmData && confirmData.length > 0 && (
   448→              <div className="px-4 sm:px-6 pb-2">
   449→                <div className="flex items-center gap-2">
   450→                  <div className="flex-1 h-px bg-gradient-to-r from-transparent via-slate-300 dark:via-slate-600 to-transparent" />
   451→                  <span className="text-[10px] font-semibold text-muted-foreground uppercase tracking-widest">Stock actual en esta ubicación</span>
   452→                  <div className="flex-1 h-px bg-gradient-to-r from-transparent via-slate-300 dark:via-slate-600 to-transparent" />
   453→                </div>
   454→              </div>
   455→            )}
   456→
   457→            {/* Productos existentes en la ubicación */}
   458→            {confirmData && confirmData.length > 0 && (
   459→              <div className="px-4 sm:px-6 pb-4">
   460→                <p className="text-xs font-semibold text-amber-600 dark:text-amber-400 uppercase tracking-wider mb-2">
   461→                  {confirmData.length} producto{confirmData.length !== 1 ? 's' : ''} encontrado{confirmData.length !== 1 ? 's' : ''}
   462→                </p>
   463→                <div className="space-y-2">
   464→                  {confirmData.map((s, i) => (
   465→                    <div
   466→                      key={`${s.codigo}-${i}`}
   467→                      className="rounded-xl border border-slate-200 dark:border-slate-700 bg-white dark:bg-slate-950 p-3"
   468→                    >
   469→                      <div className="flex items-start gap-3">
   470→                        <div className="w-9 h-9 rounded-lg bg-orange-100 dark:bg-orange-900/30 flex items-center justify-center flex-shrink-0 mt-0.5">

   566→  function toggleSelect(key: string) {

   568→      const next = new Set(prev)
   569→      if (next.has(key)) next.delete(key)
   570→      else next.add(key)

   572→    })

   592→  const [massBusy, setMassBusy] = useState(false)

   601→  function openMassConfirm() {

   605→    }

   609→  async function doMassSalida() {
   610→    setMassBusy(true)
   611→    const totalToProcess = selectedRef.current.size
   612→    let totalProcessed = 0
   613→    let totalErrors = 0
   614→    let stockErrors = 0
   615→    const errorDetails: string[] = []
   616→    try {
   617→      // Usar los values actuales al momento de iniciar (no refs que cambian durante el loop)
   618→      const selectedKeys = Array.from(selectedRef.current)
   619→      const locsSnap = [...locationsRef.current]
   620→      const qtySnap = { ...qtyMapRef.current }
   621→
   622→      for (const key of selectedKeys) {
   623→        const loc = locsSnap.find((l) => `${l.bloque}-${l.torre}-${l.piso}-${l.posicion}` === key)
   624→        if (!loc) {
   625→          errorDetails.push(`Ubicación ${key} no encontrada`)
   626→          totalErrors++
   627→          continue
   628→        }
   629→        const qtyVal = qtySnap[key] || ''
   630→        const qtyNum = qtyVal ? parseFloat(qtyVal.trim()) : loc.stock
   631→        if (isNaN(qtyNum) || qtyNum <= 0) {
   632→          errorDetails.push(`Cantidad inválida en B${loc.bloque}/T${loc.torre}/P${loc.piso}/Pos${loc.posicion}`)
   633→          totalErrors++
   634→          continue
   635→        }
   636→        const finalQty = Math.min(qtyNum, loc.stock)
   637→        try {
   638→          await addMovimiento({
   639→            tipo: 'salida',
   640→            bloque: loc.bloque, torre: loc.torre, piso: loc.piso, posicion: loc.posicion,
   641→            codigo: loc.codigo, descripcion: loc.descripcion, un: loc.un,
   642→            cantidad: finalQty,
   643→            fVencimiento: loc.fVencimiento ?? '', turno,
   644→            usuarioId: perfil.id, usuarioNombre: perfil.nombre, usuarioCorreo: perfil.correo,
   645→            proveedor: loc.proveedor,
   646→            // PRESERVAR codigoInc para consistencia (aunque SalidaForm excluye INC)
   647→            codigoInc: loc.codigoInc || undefined,
   648→          })
   649→          totalProcessed++
   650→        } catch (err) {
   651→          if (isInsufficientStockError(err)) {
   652→            stockErrors++
   653→            const detail = (err as Record<string, string>).detail || ''
   654→            errorDetails.push(`Stock insuficiente en B${loc.bloque}/T${loc.torre}/P${loc.piso}/Pos${loc.posicion}${detail ? ` (${detail})` : ''}`)
   655→          } else {
   656→            const msg = extractError(err)
   657→            errorDetails.push(`Error en B${loc.bloque}/T${loc.torre}/P${loc.piso}/Pos${loc.posicion}: ${msg}`)
   658→          }
   659→          totalErrors++
   660→        }
   661→      }
   662→      if (totalErrors > 0) {
   663→        const stockNote = stockErrors > 0 ? ` (${stockErrors} por stock insuficiente)` : ''
   664→        const desc = totalProcessed > 0
   665→          ? `${totalProcessed} de ${totalToProcess} correctas${stockNote}:\n${errorDetails.slice(0, 5).join('\n')}${errorDetails.length > 5 ? `\n...y ${errorDetails.length - 5} más` : ''}`
   666→          : `Errores${stockNote}:\n${errorDetails.join('\n')}`
   667→        toast.warning(`Salida completada con ${totalErrors} error${totalErrors > 1 ? 'es' : ''}`, { description: desc, duration: 10000 })
   668→      } else {
   669→        toast.success(`Salida registrada en ${totalProcessed} ubicacion${totalProcessed > 1 ? 'es' : ''}`)
   670→      }
   671→      setMassConfirmOpen(false)
   672→      setSelected(new Set())
   673→      setQtyMap({})
   674→      if (totalProcessed > 0) {
   675→        await refreshLocations()
   676→      }
   677→      onCreated([]) // trigger re-render
   678→    } catch (err: unknown) {
   679→      const message = extractError(err)
   680→      toast.error('Error en salida masiva', { description: message })
   681→    } finally {
   682→      setMassBusy(false)
   683→    }
   684→  }
   685→
   686→  const allSelected = locations.length > 0 && selected.size === locations.length
   687→
   688→  // Función central para refrescar las ubicaciones
   689→  const refreshLocations = useCallback(async () => {
   690→    const code = searchCodeRef.current.trim()
   691→    if (!code) {
   692→      setLocations([])
   693→      setProductoDesc('')
   694→      setProductoUn('')
   695→      setQtyMap({})
   696→      return
   697→    }
   698→    try {
   699→      let movs: Array<{ tipo: string; bloque: string; torre: string; piso: string; posicion: string; codigo: string; descripcion: string; un: string; cantidad: number; fVencimiento?: string; proveedor?: string; codigoInc?: string }>
   700→      try {
   701→        // Intentar obtener del servidor - consulta optimizada por código
   702→        const { fetchMovimientosByCodigo } = await import('@/lib/rackly/kardex')
   703→        movs = await fetchMovimientosByCodigo(code)
   704→      } catch {
   705→        // Sin cache offline, usar lista vacia
   706→        movs = []
   707→      }
   708→      const upperCode = code.toUpperCase()
   709→      const locMap = new Map<string, LocWithKey>()
   710→      // EXCLUIR movimientos INC de la vista de salidas normales
   711→      const relevant = movs.filter((m) => m.codigo === upperCode && !m.codigoInc)
   712→      let desc = ''
   713→      let un = ''
   714→      for (const m of relevant) {
   715→        if (!desc && m.descripcion) desc = m.descripcion
   716→        if (!un && m.un) un = m.un
   717→        const impact = impactoStock(m.tipo, m.cantidad)
   718→        if (isNaN(impact) || !isFinite(impact)) continue
   719→        // Agrupar por ubicación SOLAMENTE — f_vencimiento NO participa.
   720→        // Se rastrea la fecha más próxima (FEFO) para pasar en la salida.
   721→        const key = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}`
   722→        const current = locMap.get(key)
   723→        if (current) {
   724→          current.stock += impact
   725→          // Rastrear f_vencimiento más próxima para FEFO
   726→          if (m.fVencimiento) {
   727→            if (!current.fVencimiento || m.fVencimiento < current.fVencimiento) {
   728→              current.fVencimiento = m.fVencimiento
   729→            }
   730→          }
   731→        } else {
   732→          locMap.set(key, {
   733→            bloque: m.bloque,
   734→            torre: m.torre,
   735→            piso: m.piso,
   736→            posicion: m.posicion,
   737→            codigo: m.codigo,
   738→            descripcion: m.descripcion,
   739→            un: m.un,
   740→            stock: impact,
   741→            fVencimiento: m.fVencimiento || undefined,
   742→            proveedor: m.proveedor,
   743→          })
   744→        }
   745→      }
   746→      const results = Array.from(locMap.values()).filter((l) => l.stock > 0)
   747→      // Ordenar: FEFO primero (con fecha de vencimiento), luego sin fecha por bloque (1→7)
   748→      results.sort((a, b) => {
   749→        const aHasDate = !!a.fVencimiento
   750→        const bHasDate = !!b.fVencimiento
   751→        if (aHasDate && bHasDate) return a.fVencimiento!.localeCompare(b.fVencimiento!)
   752→        if (aHasDate && !bHasDate) return -1
   753→        if (!aHasDate && bHasDate) return 1
   754→        // Ambos sin fecha: ordenar por bloque, torre, piso, posición
   755→        const aB = parseInt(a.bloque, 10) || 0
   756→        const bB = parseInt(b.bloque, 10) || 0
   757→        if (aB !== bB) return aB - bB
   758→        const aT = parseInt(a.torre, 10) || 0
   759→        const bT = parseInt(b.torre, 10) || 0
   760→        if (aT !== bT) return aT - bT
   761→        const aP = parseInt(a.piso, 10) || 0
   762→        const bP = parseInt(b.piso, 10) || 0
   763→        if (aP !== bP) return aP - bP
   764→        const aPos = parseInt(a.posicion, 10) || 0
   765→        const bPos = parseInt(b.posicion, 10) || 0
   766→        return aPos - bPos
   767→      })
   768→      // Limpiar selecciones de ubicaciones que ya no tienen stock
   769→      const newKeys = new Set(results.map((l) => `${l.bloque}-${l.torre}-${l.piso}-${l.posicion}`))
   770→      setSelected((prev) => {
   771→        const cleaned = new Set<string>()
   772→        for (const k of prev) {
   773→          if (newKeys.has(k)) cleaned.add(k)
   774→        }
   775→        return cleaned.size === prev.size ? prev : cleaned
   776→      })
   777→      setLocations(results)
   778→      setProductoDesc(desc)
   779→      setProductoUn(un)
   780→    } catch (err) {
   781→      console.error('[MovimientoForm] refreshLocations error:', err)
   782→      toast.error('Error al buscar stock', { description: extractError(err) })
   783→    }
   784→  }, [])
   785→
   786→  // Búsqueda inicial con debounce
   787→  useEffect(() => {
   788→    if (!searchCode.trim()) {
   789→      setLocations([])
   790→      setProductoDesc('')
   791→      setProductoUn('')
   792→      setQtyMap({})
   793→      return
   794→    }
   795→    setLoading(true)
   796→    const timer = setTimeout(async () => {
   797→      await refreshLocations()
   798→      setLoading(false)
   799→    }, 300)
   800→    return () => clearTimeout(timer)
   801→  }, [searchCode, refreshLocations])
   802→
   803→  // POLLING: refrescar automáticamente cada 8 segundos
   804→  useEffect(() => {
   805→    if (!searchCode.trim()) return
   806→    const interval = setInterval(() => {
   807→      refreshLocations()
   808→    }, 8000)
   809→    return () => clearInterval(interval)
   810→  }, [searchCode, refreshLocations])
   811→
   812→  async function handleSalidaParcial(locKey: string) {
   813→    const loc = locations.find((l) => `${l.bloque}-${l.torre}-${l.piso}-${l.posicion}` === locKey)
   814→    if (!loc) return

   828→  function handleRetirarTodo(locKey: string) {

   830→    if (!loc) return

   833→
   834→  const [confirmState, setConfirmState] = useState<{
   835→    loc: LocWithKey
   836→    qtyNum: number
   837→    full: boolean
   838→  } | null>(null)
   839→
   840→  async function doSalida() {
   841→    if (!confirmState) return
   842→    const { loc, qtyNum } = confirmState

   845→      const movs = await addMovimiento({

   856→        turno,
   857→        usuarioId: perfil.id,
   858→        usuarioNombre: perfil.nombre,
   859→        usuarioCorreo: perfil.correo,
   860→        proveedor: loc.proveedor,
   861→        // PRESERVAR codigoInc para consistencia (aunque SalidaForm excluye INC)
   862→        codigoInc: loc.codigoInc || undefined,
   863→      })
   864→      toast.success(`Salida de ${qtyNum} ${loc.un} registrada`)
   865→      setConfirmState(null)
   866→      setSearchCode('')
   867→      setQtyMap({})
   868→      onCreated(movs)
   869→    } catch (err: unknown) {
   870→      if (isInsufficientStockError(err)) {
   871→        const detail = (err as Record<string, string>).detail || ''
   872→        toast.error('Stock insuficiente', {
   873→          description: detail || 'Otro usuario pudo haber modificado el stock mientras tú operabas. Los datos se han actualizado.',
   874→          duration: 8000,
   875→        })

   877→        refreshLocations() // refrescar datos reales
   878→      } else {
   879→        const message = extractError(err)
   880→        toast.error('Error al registrar salida', { description: message })

   883→      setBusy(false)

   907→          />
   908→        </div>
   909→      )}
   910→
   911→      {/* Badges de metadata */}
   912→      {searchCode.trim() && (
   913→        <div className="flex flex-wrap gap-2">
   914→          <Badge variant="outline" className="bg-blue-50 text-blue-700 border-blue-200 dark:bg-blue-950/40 dark:text-blue-300 dark:border-blue-800">
   915→            Turno: {turno}
   916→          </Badge>
   917→          <Badge variant="outline" className="bg-blue-50 text-blue-700 border-blue-200 dark:bg-blue-950/40 dark:text-blue-300 dark:border-blue-800">
   918→            Usuario: {perfil.nombre}

   929→      {loading && (
