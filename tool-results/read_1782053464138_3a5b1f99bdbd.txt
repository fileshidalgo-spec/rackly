    59→// Calcula días restantes hasta vencimiento (negativo si ya venció)
    60→function diasParaVencer(fVencimiento: string | undefined): number | null {
    61→  if (!fVencimiento) return null
    62→  const hoy = new Date(); hoy.setHours(0, 0, 0, 0)
    63→  const venc = new Date(fVencimiento + 'T00:00:00')
    64→  return Math.ceil((venc.getTime() - hoy.getTime()) / (1000 * 60 * 60 * 24))
    65→}

    67→// Detecta códigos duplicados (mismo artículo con distintas fechas de vencimiento)
    68→function codigosConMultiplesLotes(stock: StockEnUbicacion[]): Set<string> {
    69→  const counts = new Map<string, number>()
    70→  for (const s of stock) {
    71→    counts.set(s.codigo, (counts.get(s.codigo) ?? 0) + 1)
    72→  }
    73→  const multiples = new Set<string>()
    74→  for (const [code, count] of counts) {
    75→    if (count > 1) multiples.add(code)
    76→  }
    77→  return multiples
    78→}

    80→// Calcula ocupación desde movimientos — stock POR CÓDIGO (no por lote).
    81→// f_vencimiento es SOLO para FEFO (ordenamiento), NO para particionar stock.
    82→// IMPORTANTE: Los movimientos INC se EXCLUYEN del stock normal.
    83→// La detección INC se hace por separado con fetchIncPorUbicacion().
    84→function calcularOcupacion(movs: Movimiento[]): OcupacionCelda[] {
    85→  // Mapa: ubicacion_key → (codigo → stock)
    86→  const cellMap = new Map<string, Map<string, number>>()
    87→  for (const m of movs) {
    88→    // EXCLUIR movimientos INC del cálculo de ocupación normal
    89→    if (m.codigoInc) continue
    90→    const key = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}`
    91→    const code = m.codigo.trim().toUpperCase()
    92→    let codeMap = cellMap.get(key)
    93→    if (!codeMap) { codeMap = new Map(); cellMap.set(key, codeMap) }
    94→    const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? m.cantidad : -m.cantidad
    95→    const current = codeMap.get(code) ?? 0
    96→    codeMap.set(code, current + delta)
    97→  }
    98→  // Construir resultado: solo celdas con stock total > 0
    99→  const result: OcupacionCelda[] = []
   100→  for (const [key, codeMap] of cellMap) {
   101→    let totalStock = 0
   102→    const codigos = new Set<string>()
   103→    for (const [code, stock] of codeMap) {
   104→      if (stock > 0) {
   105→        totalStock += stock
   106→        codigos.add(code)
   107→      }
   108→    }
   109→    if (totalStock > 0) {
   110→      const [bloque, torre, piso, posicion] = key.split('-')
   111→      // lotes = número de códigos distintos (ya no se cuenta por f_vencimiento)
   112→      result.push({ bloque, torre, piso, posicion, stock: totalStock, codigos: Array.from(codigos), lotes: codigos.size, tieneInc: false, incItems: [] })
   113→    }
   114→  }
   115→  return result
   116→}

   407→  async function doIngreso() {
   408→    if (!detail || !perfil) return
   409→    if (!ingCodigo.trim() || !ingCantidad) { toast.error('Completa código y cantidad'); return }
   410→    const q = parseFloat(ingCantidad); if (isNaN(q) || q <= 0) { toast.error('Cantidad inválida'); return }
   411→    if (showProveedor && !ingProveedor) { toast.error('Selecciona un proveedor para este artículo'); return }
   412→    setActionBusy(true)
   413→    try {
   414→      await addMovimiento({
   415→        tipo: ingTipo, bloque: detail.bloque, torre: detail.torre, piso: detail.piso, posicion: detail.posicion,
   416→        codigo: ingCodigo.trim().toUpperCase(), descripcion: ingDescripcion, un: ingUn, cantidad: q,
   417→        fVencimiento: ingSinFecha ? '' : ingFVenc,
   418→        turno: calcularTurno(), usuarioId: perfil.id, usuarioNombre: perfil.nombre, usuarioCorreo: perfil.correo,
   419→        proveedor: showProveedor ? ingProveedor : undefined,

   421→      toast.success(ingTipo === 'ingreso' ? 'Ingreso registrado' : 'Devolución registrada')

   423→    } catch (err: unknown) { toast.error('Error', { description: extractError(err) }) } finally { setActionBusy(false) }
   424→  }

   455→  async function doSalida() {
   456→    if (!detail || !perfil) return
   457→    const item = detail.stock[salidaIdx]
   458→    if (!item) return
   459→    const qty = parseFloat(salidaCantidad)
   460→    if (isNaN(qty) || qty <= 0) { toast.error('Cantidad inválida'); return }
   461→    // Las salidas normales NO pueden exceder stock. Las salidas INC sí permiten exceder (autoajuste).
   462→    // La validacion debe usar el stock TOTAL del codigo en esta ubicacion (igual que la RPC
   463→    // registrar_movimiento_kardex), NO el stock por lote individual. La UI agrupa por lote
   464→    // (f_vencimiento) pero la RPC suma TODOS los movimientos del codigo sin distinguir lotes.
   465→    if (!item.codigoInc) {
   466→      const totalStockForCodigo = detail.stock
   467→        .filter(s => s.codigo === item.codigo)
   468→        .reduce((sum, s) => sum + s.stock, 0)
   469→      if (qty > totalStockForCodigo) { toast.error(`Stock insuficiente. Stock total disponible: ${totalStockForCodigo} ${item.un}`); return }
   470→    }
   471→    setActionBusy(true)
   472→    try {
   473→      await addMovimiento({
   474→        tipo: 'salida', bloque: detail.bloque, torre: detail.torre, piso: detail.piso, posicion: detail.posicion,
   475→        codigo: item.codigo, descripcion: item.descripcion, un: item.un, cantidad: qty,
   476→        fVencimiento: item.fVencimiento ?? '', turno: calcularTurno(), usuarioId: perfil.id, usuarioNombre: perfil.nombre, usuarioCorreo: perfil.correo,
   477→        // PRESERVAR codigoInc para que la salida descuente del stock INC correctamente
   478→        codigoInc: item.codigoInc || undefined,
   479→      })
   480→      toast.success('Salida registrada')
   481→      if (mountedRef.current) { await refreshDetail(); refreshData(); setDetailMode('view') }
   482→    } catch (err: unknown) {
   483→      if (isInsufficientStockError(err)) {
   484→        const detail = (err as Record<string, string>).detail || ''

   486→        refreshDetail(); refreshData()
   487→      } else { toast.error('Error al registrar salida', { description: extractError(err) }) }
   488→    } finally { setActionBusy(false) }
