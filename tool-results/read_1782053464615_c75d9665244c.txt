    33→export function StockTab() {
    34→  const { perfil } = useAuth()
    35→  const esAdmin = perfil?.rol === 'admin'
    36→  const [movs, setMovs] = useState<Movimiento[]>([])
    37→  const [query, setQuery] = useState('')
    38→  const [selectedCodigo, setSelectedCodigo] = useState('')
    39→  const [stockFilter, setStockFilter] = useState<'todos' | 'disponibles' | 'inc'>('todos')
    40→  const [stock, setStock] = useState<
    41→    {
    42→      bloque: string
    43→      torre: string
    44→      piso: string
    45→      posicion: string
    46→      stock: number
    47→      descripcion: string
    48→      un: string
    49→      proveedor?: string
    50→      fVencimiento: string          // fecha más próxima (FEFO), solo informativo
    51→      lotesInfo: string            // resumen de lotes, solo informativo
    52→      codigoInc?: string
    53→    }[]
    54→  >([])
    55→
    56→  useMovimientosRealtime(setMovs)
    57→
    58→  // Catálogo para búsqueda y lookup
    59→  const [catalogoResults, setCatalogoResults] = useState<CatalogoItem[]>([])
    60→  const [catalogoLoaded, setCatalogoLoaded] = useState(false)
    61→
    62→  // Cargar catálogo al montar
    63→  useEffect(() => {
    64→    async function load() {
    65→      if (!isCatalogoLoaded()) {
    66→        await fetchCatalogo().catch(() => {})
    67→      }
    68→      setCatalogoLoaded(true)
    69→    }
    70→    load()
    71→  }, [])
    72→
    73→  // Buscar sugerencias del catálogo cuando el usuario escribe
    74→  useEffect(() => {
    75→    if (!catalogoLoaded || !query.trim()) {
    76→      setCatalogoResults([])
    77→      return
    78→    }
    79→    if (!isCatalogoLoaded()) return
    80→    const results = searchCatalogo(query.trim(), 8)
    81→    setCatalogoResults(results)
    82→  }, [query, catalogoLoaded])
    83→
    84→  // Datos del artículo seleccionado del catálogo
    85→  const selectedItem = useMemo(() => {
    86→    if (!selectedCodigo) return null
    87→    return findCatalogoByCodigo(selectedCodigo)
    88→  }, [selectedCodigo])
    89→
    90→  // Calcular stock por ubicación para el código seleccionado.
    91→  // LÓGICA IDÉNTICA a calcularOcupacion() en OcupacionTab:
    92→  //   - Agrupa por posición + código
    93→  //   - Excluye INC
    94→  //   - Suma delta (ingreso/devolucion/traslado = +, salida = -)
    95→  //   - Ignora f_vencimiento para el cálculo de stock
    96→  // Luego filtra por el código seleccionado y agrega info FEFO como datos extras.
    97→  const stockData = useMemo(() => {
    98→    if (!selectedCodigo || movs.length === 0) return []
    99→    const code = selectedCodigo.trim().toUpperCase()
   100→    const isIncMode = stockFilter === 'inc'
   101→
   102→    // PASO 1: Calcular stock neto por (posición, código) — IDÉNTICO a calcularOcupacion
   103→    const cellMap = new Map<string, Map<string, number>>() // posKey → (codigo → stock)
   104→    const fvMap = new Map<string, Map<string, Map<string, number>>>() // posKey → (codigo → (fv → qty))
   105→    const descMap = new Map<string, { descripcion: string; un: string; proveedor?: string }>()
   106→
   107→    for (const m of movs) {
   108→      if (m.codigoInc) continue // EXCLUIR INC
   109→      const posKey = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}`
   110→      const mCode = m.codigo.trim().toUpperCase()
   111→      let codeMap = cellMap.get(posKey)
   112→      if (!codeMap) { codeMap = new Map(); cellMap.set(posKey, codeMap) }
   113→      const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? m.cantidad : -m.cantidad
   114→      const current = codeMap.get(mCode) ?? 0
   115→      codeMap.set(mCode, current + delta)
   116→
   117→      // Rastrear fechas de vencimiento para info FEFO (solo informativo)
   118→      if (m.fVencimiento && ['ingreso', 'devolucion', 'traslado'].includes(m.tipo)) {
   119→        let fvCodeMap = fvMap.get(posKey)
   120→        if (!fvCodeMap) { fvCodeMap = new Map(); fvMap.set(posKey, fvCodeMap) }
   121→        let fvQtyMap = fvCodeMap.get(mCode)
   122→        if (!fvQtyMap) { fvQtyMap = new Map(); fvCodeMap.set(mCode, fvQtyMap) }
   123→        fvQtyMap.set(m.fVencimiento, (fvQtyMap.get(m.fVencimiento) ?? 0) + m.cantidad)
   124→      }
   125→
   126→      if (!descMap.has(posKey)) {
   127→        descMap.set(posKey, { descripcion: m.descripcion, un: m.un, proveedor: m.proveedor || undefined })
   128→      }
   129→    }
   130→
   131→    // PASO 2: Filtrar por código seleccionado y construir resultado
   132→    // En modo INC, buscamos movimientos con codigoInc (se calculan aparte)
   133→    const result: {
   134→      bloque: string; torre: string; piso: string; posicion: string
   135→      stock: number; descripcion: string; un: string; proveedor?: string
   136→      fVencimiento: string      // fecha más próxima (FEFO), solo informativo
   137→      lotesInfo: string         // resumen de lotes, solo informativo
   138→      codigoInc?: string
   139→    }[] = []
   140→
   141→    if (isIncMode) {
   142→      // Modo INC: calcular stock de movimientos INC para este código
   143→      const incCellMap = new Map<string, number>()
   144→      for (const m of movs) {
   145→        if (!m.codigoInc) continue
   146→        const mCode = m.codigo.trim().toUpperCase()
   147→        if (mCode !== code) continue
   148→        const posKey = `${m.bloque}-${m.torre}-${m.piso}-${m.posicion}`
   149→        const delta = ['ingreso', 'devolucion', 'traslado'].includes(m.tipo) ? m.cantidad : -m.cantidad
   150→        incCellMap.set(posKey, (incCellMap.get(posKey) ?? 0) + delta)
   151→        if (!descMap.has(posKey)) {
   152→          descMap.set(posKey, { descripcion: m.descripcion, un: m.un, proveedor: m.proveedor || undefined })
   153→        }
   154→      }
   155→      for (const [posKey, stock] of incCellMap) {
   156→        if (stock <= 0) continue
   157→        const desc = descMap.get(posKey)
   158→        if (!desc) continue
   159→        const [bloque, torre, piso, posicion] = posKey.split('-')
   160→        result.push({
   161→          bloque, torre, piso, posicion, stock,
   162→          descripcion: desc.descripcion, un: desc.un, proveedor: desc.proveedor,
   163→          fVencimiento: '', lotesInfo: '', codigoInc: code,
   164→        })
   165→      }
   166→    } else {
   167→      // Modo normal: filtrar celdas donde el código seleccionado tiene stock > 0
   168→      for (const [posKey, codeMap] of cellMap) {
   169→        const posStock = codeMap.get(code)
   170→        if (!posStock || posStock <= 0) continue
   171→        const desc = descMap.get(posKey)
   172→        if (!desc) continue
   173→        const [bloque, torre, piso, posicion] = posKey.split('-')
   174→
   175→        // Info FEFO: fecha más próxima del código en esta posición
   176→        const fvCodeMap = fvMap.get(posKey)?.get(code)
   177→        let fVencimiento = ''
   178→        let lotesInfo = ''
   179→        if (fvCodeMap && fvCodeMap.size > 0) {
   180→          const sorted = [...fvCodeMap.entries()].sort(([a], [b]) => a.localeCompare(b))
   181→          fVencimiento = sorted[0][0] // fecha más próxima
   182→          lotesInfo = sorted.map(([fv, qty]) => `${fv}: ${qty}`).join(' | ')
   183→        }
   184→
   185→        result.push({
   186→          bloque, torre, piso, posicion, stock: posStock,
   187→          descripcion: desc.descripcion, un: desc.un, proveedor: desc.proveedor,
   188→          fVencimiento, lotesInfo, codigoInc: undefined,
   189→        })
   190→      }
   191→    }
   192→
   193→    return result.sort((a, b) => {
   194→      const aB = parseInt(a.bloque, 10) || 0
   195→      const bB = parseInt(b.bloque, 10) || 0
   196→      if (aB !== bB) return aB - bB
   197→      const aT = parseInt(a.torre, 10) || 0
   198→      const bT = parseInt(b.torre, 10) || 0
   199→      if (aT !== bT) return aT - bT
   200→      const aP = parseInt(a.piso, 10) || 0
   201→      const bP = parseInt(b.piso, 10) || 0
   202→      if (aP !== bP) return aP - bP
   203→      const aPos = parseInt(a.posicion, 10) || 0
   204→      const bPos = parseInt(b.posicion, 10) || 0
   205→      return aPos - bPos
   206→      })
   207→  }, [selectedCodigo, movs, stockFilter])
   208→
   209→  useEffect(() => {
   210→    setStock(stockData)
   211→  }, [stockData])
   212→
   213→  // Server-side fetch: cuando cambia el código seleccionado, intentar
   214→  // obtener datos server-side (sin límite de filas). Si falla, usar client-side.
   215→  const [serverStock, setServerStock] = useState<typeof stock | null>(null)
   216→  const [serverLoading, setServerLoading] = useState(false)
   217→
   218→  useEffect(() => {
   219→    if (!selectedCodigo) { setServerStock(null); return }
   220→    const code = selectedCodigo.trim().toUpperCase()
   221→    const isInc = stockFilter === 'inc'
   222→
   223→    setServerLoading(true)
   224→    fetchStockPorCodigoRPC(code, isInc)
   225→      .then((rows) => {
   226→        if (!mountedRef.current) return
   227→        setServerLoading(false)
   228→        if (rows === null) {
   229→          // Server-side falló → usar client-side (stockData)
   230→          setServerStock(null)
   231→          return
   232→        }
   233→        // Convertir filas server-side al formato de stock
   234→        const mapped = rows.map(r => ({
   235→          bloque: r.bloque,
   236→          torre: r.torre,
   237→          piso: r.piso,
   238→          posicion: r.posicion,
   239→          stock: r.stock,
   240→          descripcion: r.descripcion || '',
   241→          un: r.un || '',
   242→          proveedor: r.proveedor || undefined,
   243→          fVencimiento: r.fVencimiento || '',
   244→          lotesInfo: '', // Server-side no calcula lotesInfo por ahora
   245→          codigoInc: r.codigoInc || undefined,
   246→        }))
   247→        // Ordenar por bloque, torre, piso, posición
   248→        mapped.sort((a, b) => {
   249→          const aB = parseInt(a.bloque, 10) || 0; const bB = parseInt(b.bloque, 10) || 0
   250→          if (aB !== bB) return aB - bB
   251→          const aT = parseInt(a.torre, 10) || 0; const bT = parseInt(b.torre, 10) || 0
   252→          if (aT !== bT) return aT - bT
   253→          const aP = parseInt(a.piso, 10) || 0; const bP = parseInt(b.piso, 10) || 0
   254→          if (aP !== bP) return aP - bP
   255→          const aPos = parseInt(a.posicion, 10) || 0; const bPos = parseInt(b.posicion, 10) || 0
   256→          return aPos - bPos
   257→        })
   258→        setServerStock(mapped)
   259→      })
   260→      .catch(() => {
   261→        if (mountedRef.current) { setServerLoading(false); setServerStock(null) }
   262→      })
   263→  }, [selectedCodigo, stockFilter])
   264→
   265→  // useRef para evitar setStock en componentes desmontados
   266→  const mountedRef = useRef(true)
   267→  useEffect(() => () => { mountedRef.current = false }, [])
   268→
   269→  // Stock final: server-side si disponible, si no client-side
   270→  const displayStock = serverStock ?? stock

   272→  // Manejar selección de un código

   279→  async function handleDelete(
   280→    bloque: string,
   281→    torre: string,
   282→    piso: string,
   283→    posicion: string,
   284→    fVencimiento?: string,
   285→    codigoInc?: string

   287→    if (!confirm('¿Eliminar TODOS los movimientos de esta ubicación?')) return

   289→      const next = await eliminarUbicacion(selectedCodigo, bloque, torre, piso, posicion, fVencimiento, codigoInc)

   294→      toast.error('No se pudo eliminar', { description: message })
   295→    }
   296→  }

   298→  // Badge color según días para vencer
   299→  function getBadgeClass(fVencimiento: string) {
   300→    const hoy = new Date(new Date().toDateString())

   303→    const vencido = diffDias !== null && diffDias < 0
   304→    const naranja = !vencido && diffDias !== null && diffDias <= 15
   305→    const azul = !vencido && !naranja && diffDias !== null && diffDias <= 30
   306→
   307→    return vencido
   308→      ? 'bg-red-100 text-red-700 border-red-200 dark:bg-red-950/40 dark:text-red-300 dark:border-red-800'
   309→      : naranja
   310→      ? 'bg-orange-100 text-orange-700 border-orange-200 dark:bg-orange-950/40 dark:text-orange-300 dark:border-orange-800'
   311→      : azul
   312→      ? 'bg-blue-100 text-blue-700 border-blue-200 dark:bg-blue-950/40 dark:text-blue-300 dark:border-blue-800'
   313→      : 'bg-green-100 text-green-700 border-green-200 dark:bg-green-950/40 dark:text-green-300 dark:border-green-800'

   428→          onKeyDown={(e) => {
